282 void globalRefine(int refCount) {
283 for (int i = 0; i < refCount; ++i) {
284 const auto& finestPatchData = patchGeometries_.back().patchData();
285 auto newfinestPatchData = finestPatchData;
286
287 auto& olduniqueKnotVector = patchGeometries_.back().uniqueKnotVector();
288 std::array<std::vector<ctype>, dim> newUniqueKnotVecs;
289 for (int refDirection = 0; refDirection < dim; ++refDirection) {
290 auto additionalKnots = Splines::generateRefinedKnots(finestPatchData.knotSpans, refDirection, 1);
291 newfinestPatchData = Splines::knotRefinement<dim>(newfinestPatchData, additionalKnots, refDirection);
292 auto& newKnotVec = newUniqueKnotVecs[refDirection];
293
294 // compute new unique Knot vectors from refinement
295 newKnotVec.resize(olduniqueKnotVector[refDirection].size() + additionalKnots.size());
296 std::ranges::merge(olduniqueKnotVector[refDirection], additionalKnots, std::begin(newKnotVec));
297 }
298
299 patchGeometries_.emplace_back(newfinestPatchData, newUniqueKnotVecs);
300 patchGeometriesUnElevated.emplace_back(patchGeometries_.back());
301 }
302 //Here it is exploited that the knot refinem above is in sync with the globaRefine of the trimmer
303 trimmer_->globalRefine(refCount);
304 }

321 void globalRefineInDirection(const std::array<int, dim>& refines) {
322 const auto& finestPatchData = patchGeometries_.back().patchData();
323 auto newfinestPatchData = finestPatchData;
324 for (int dir = 0; auto refinesInDirection : refines) {
325 if (refinesInDirection == 0) {
326 ++dir;
327 continue;
328 }
329 auto additionalKnots = Splines::generateRefinedKnots(finestPatchData.knotSpans, dir, refinesInDirection);
330 newfinestPatchData = Splines::knotRefinement<dim>(newfinestPatchData, additionalKnots, dir);
331 ++dir;
332 }
333 auto newGrid = PatchGrid(newfinestPatchData);
334 *this = std::move(newGrid);
335 }

342 void degreeElevate(const std::array<int, dim>& elevationFactors, int lvl) {
343 if (lvl > maxLevel() and lvl >= 0) DUNE_THROW(Dune : RangeError, "This level does not exist");
344 auto& patchData = patchGeometries_[lvl].patchData();
345 patchGeometriesUnElevated[lvl] = GeometryKernel::NURBSPatch<dim, dimworld, ctype>(patchData);
346 for (int dir = 0; auto elevatesInDirection : elevationFactors) {
347 if (elevatesInDirection == 0) {
348 ++dir;
349 continue;
350 }
351 patchData = Splines::degreeElevate(patchData, dir, elevatesInDirection);
352 ++dir;
353 }
354 patchGeometries_[lvl] = GeometryKernel::NURBSPatch<dim, dimworld, ctype>(patchData);
355 }
