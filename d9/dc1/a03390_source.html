13 template <int dim, int dimworld, typename PatchGrid>
14 requires(dim == 2) && (dimworld == 2 || dimworld == 3) && (dim <= dimworld)
15 class IbraReader {
16 using GridFamily = typename PatchGrid::GridFamily;
17 using PatchData = NURBSPatchData<dim, dimworld, typename GridFamily::ctype>;
18 using PatchTrimData = typename GridFamily::TrimmerTraits::PatchTrimData;
19 using TrimmingCurve = typename GridFamily::TrimmerTraits::TrimmingCurve;
20
21 using ControlPointType = typename PatchData::ControlPointType;
22 using ControlPointNetType = typename PatchData::ControlPointNetType;

24 public:

26 std::ifstream ibraInputFile;

31 template <typename InputStringType>
32 requires(not std::convertible_to<std::string, InputStringType>
33 and not std::convertible_to<InputStringType, const char*>)
34 static auto read(InputStringType& ibraInputFile, const bool trim = true, std::array<int, 2> preKnotRefine = {0, 0}) {
35 using json = nlohmann::json;
36
37 std::vector<Ibra::Surface<dimworld>> surfaces;
38 std::vector<Ibra::Curve2D> curves2D;
39
40 std::vector<Ibra::BrepRepresentation> brepRepresentations;
41 std::vector<Ibra::BrepLoopRepresentation> brepLoopRepresentations;
42 std::vector<Ibra::BrepTrimRepresentation> brepTrimRepresentations;
43
44 json ibraJson;
45 try {
46 ibraJson = json::parse(ibraInputFile);
47
48 for (auto& j : ibraJson) {
49 auto geo = j.get<Ibra::IbraBase>();
50
51 switch (geo.type) {
52 case Ibra::Type::NurbsSurfaceGeometry3D:
53 surfaces.push_back(j.get<Ibra::Surface<dimworld>>());
54 break;
55 case Ibra::Type::NurbsCurveGeometry2D:
56 curves2D.push_back(j.get<Ibra::Curve2D>());
57 break;
58 case Ibra::Type::BrepType:
59 brepRepresentations.push_back(j.get<Ibra::BrepRepresentation>());
60 break;
61 case Ibra::Type::BrepLoopType:
62 brepLoopRepresentations.push_back(j.get<Ibra::BrepLoopRepresentation>());
63 break;
64 case Ibra::Type::BrepTrimType:
65 brepTrimRepresentations.push_back(j.get<Ibra::BrepTrimRepresentation>());
66 break;
67 default:
68 // Do nothing -- keep the compiler happy
69 break;
70 }
71 }
72 } catch (json::parse_error& ex) {
73 DUNE_THROW(Dune::IOError, "Error in reading input stream: "
74 << ", parse error at byte " << ex.byte << " What: " << ex.what());
75 }
76
77 // Make Connections
78 auto brepRepresentation = brepRepresentations[0];
79
80 Ibra::Brep brep{brepRepresentation, curves2D, surfaces, brepLoopRepresentations, brepTrimRepresentations};
81 // Each surface in a brep is one Patch, as for now only one brep is supported in NURBSGrid
82 assert(brep.surfaces.size() == 1);
83
84 std::cout << "IbraReader successfully read in 1 patch with " << brep.trims.size() << " edges." << std::endl;
85
86 // Reader has done its job, now NURBSGrid can be constructed
87
88 Ibra::Surface<dimworld> _surface = brep.surfaces[0];
89 const std::array<std::vector<double>, dim> knotSpans = _surface.compileKnotVectors();
90 const std::vector<std::vector<ControlPointType>> controlPoints
91 = _surface.template transformControlPoints<ControlPointType>();
92 std::array<int, dim> dimsize = _surface.n_controlPoints;
93
94 auto controlNet = ControlPointNetType(dimsize, controlPoints);
95 PatchData _patchData{knotSpans, controlNet, _surface.degree};
96
97 // Optional preKnot refinement
98 for (const auto i : std::views::iota(0, dim)) {
99 if( preKnotRefine[i] > 0) {
100 auto newKnots = Splines::generateRefinedKnots(knotSpans, i, preKnotRefine[i]);
101 _patchData = Splines::knotRefinement(_patchData, newKnots, i);
102 }
103 }
104
105 PatchTrimData trimData{};
106 if (trim) constructTrimmingCurves(brep, trimData);
107
108 return std::make_tuple(_patchData, trimData);
109 }
110
111 private:
112 static void constructTrimmingCurves(const Ibra::Brep<dimworld>& brep, PatchTrimData& trimData) {
113 const std::vector<Ibra::BrepLoop> loops = brep.loops;
114 assert(!loops.empty() && "Only one boundary loop is currently supported");
115 for (int i = 0; const Ibra::BrepLoop& loop : loops) {
116 trimData.addLoop();
117 for (const Ibra::BrepTrim& trim : loop.trims)
118 trimData.insertTrimCurve(trim.asCurve<TrimmingCurve>(), i);
119 i++;
120 }
121 }
122 };
