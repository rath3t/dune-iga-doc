271 void globalRefine(int refCount) {
272 for (int i = 0; i < refCount; ++i) {
273 const auto& finestPatchData = patchGeometries_.back().patchData();
274 auto newfinestPatchData = finestPatchData;
275
276 auto& olduniqueKnotVector = patchGeometries_.back().uniqueKnotVector();
277 std::array<std::vector<ctype>, dim> newUniqueKnotVecs;
278 for (int refDirection = 0; refDirection < dim; ++refDirection) {
279 auto additionalKnots = Splines::generateRefinedKnots(finestPatchData.knotSpans, refDirection, 1);
280 newfinestPatchData = Splines::knotRefinement<dim>(newfinestPatchData, additionalKnots, refDirection);
281 auto& newKnotVec = newUniqueKnotVecs[refDirection];
282
283 // compute new unique Knot vectors from refinement
284 newKnotVec.resize(olduniqueKnotVector[refDirection].size() + additionalKnots.size());
285 std::ranges::merge(olduniqueKnotVector[refDirection], additionalKnots, std::begin(newKnotVec));
286 }
287
288 patchGeometries_.emplace_back(newfinestPatchData, newUniqueKnotVecs);
289 patchGeometriesUnElevated.emplace_back(patchGeometries_.back());
290 }
291 // Here it is exploited that the knot refinem above is in sync with the globaRefine of the trimmer
292 trimmer_->globalRefine(refCount);
293 }

310 void globalRefineInDirection(const std::array<int, dim>& refines) {
311 const auto& finestPatchData = patchGeometries_.back().patchData();
312 auto newfinestPatchData = finestPatchData;
313 for (int dir = 0; auto refinesInDirection : refines) {
314 if (refinesInDirection == 0) {
315 ++dir;
316 continue;
317 }
318 auto additionalKnots = Splines::generateRefinedKnots(finestPatchData.knotSpans, dir, refinesInDirection);
319 newfinestPatchData = Splines::knotRefinement<dim>(newfinestPatchData, additionalKnots, dir);
320 ++dir;
321 }
322 auto newGrid = PatchGrid(newfinestPatchData);
323 *this = std::move(newGrid);
324 }

331 void degreeElevate(const std::array<int, dim>& elevationFactors, int lvl) {
332 if (lvl > maxLevel() and lvl >= 0) DUNE_THROW(Dune : RangeError, "This level does not exist");
333 auto& patchData = patchGeometries_[lvl].patchData();
334 patchGeometriesUnElevated[lvl] = GeometryKernel::NURBSPatch<dim, dimworld, ctype>(patchData);
335 for (int dir = 0; auto elevatesInDirection : elevationFactors) {
336 if (elevatesInDirection == 0) {
337 ++dir;
338 continue;
339 }
340 patchData = Splines::degreeElevate(patchData, dir, elevatesInDirection);
341 ++dir;
342 }
343 patchGeometries_[lvl] = GeometryKernel::NURBSPatch<dim, dimworld, ctype>(patchData);
344 }
