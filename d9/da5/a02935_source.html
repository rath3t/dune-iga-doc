76 [[nodiscard]] double volume(int scaleOrder = 1) const {
77 const auto rule = Dune::QuadratureRules<ctype, mydimension>::rule(
78 this->type(), scaleOrder * mydimension * (*std::ranges::max_element(patchData_.degree)));
79 ctype vol = 0.0;
80 for (auto& gp : rule) {
81 localView_.bind(gp.position());
82 vol += localView_.integrationElement(gp.position()) * gp.weight();
83 }
84 return vol;
85 }

162 auto calculateNurbsAndControlPointNet(const LocalCoordinate& u) const {
163 auto subNetStart = IGA::findSpanCorrected(patchData_.degree, u, patchData_.knotSpans);
164
165 auto cpCoordinateNet
166 = netOfSpan(subNetStart, patchData_.degree, IGA::extractControlCoordinates(patchData_.controlPoints));
167 auto nurbsLocalView = nurbs_.localView();
168 nurbsLocalView.bind(subNetStart);
169 return std::make_tuple(nurbsLocalView, cpCoordinateNet, subNetStart);
170 }

172 [[nodiscard]] static FieldVector<ctype, worlddimension> globalImpl(
173 const LocalCoordinate& u, const NurbsLocalView& nurbsLocalView,
174 const ControlPointCoordinateNetType& localControlPointNet) {
175 auto basis = nurbsLocalView.basisFunctions(u);
176
177 return IGA::dot(basis, localControlPointNet);

180 [[nodiscard]] static JacobianTransposed jacobianTransposedImpl(
181 const LocalCoordinate& u, const NurbsLocalView& nurbsLocalView,
182 const ControlPointCoordinateNetType& localControlPointNet) {
183 JacobianTransposed result;
184
185 const auto basisFunctionDerivatives = nurbsLocalView.basisFunctionDerivatives(u, 1);
186
187 for (int dir = 0; dir < mydimension; ++dir) {
188 std::array<unsigned int, mydimension> ithVec{};
189 ithVec[dir] = 1;
190 result[dir] = dot(basisFunctionDerivatives.get(ithVec), localControlPointNet);
191 }
192 return result;
193 }
