143 auto zeroFirstAndSecondDerivativeOfPosition(const LocalCoordinate& u) const {
144 FieldVector<ctype, dimworld> pos;
145 JacobianTransposed J;
146 FieldMatrix<ctype, dim*(dim + 1) / 2, coorddimension> H;
147 std::array<unsigned int, dim> subDirs;
148 for (int subI = 0, i = 0; i < mydimension; ++i)
149 subDirs[subI++] = i;
150
151 const auto basisFunctionDerivatives = nurbs_.basisFunctionDerivatives(u, 2);
152 auto cpCoordinateNet = netOfSpan(u, patchData_.knotSpans, patchData_.degree,
153 extractControlCoordinates(patchData_.controlPoints));
154 std::array<unsigned int, mydimension> ithVecZero{};
155 pos = Dune::IGA::dot(basisFunctionDerivatives.get(ithVecZero), cpCoordinateNet);
156
157 for (int dir = 0; dir < mydimension; ++dir) {
158 std::array<unsigned int, mydimension> ithVec{};
159 ithVec[subDirs[dir]] = 1;
160 J[dir] = dot(basisFunctionDerivatives.get(ithVec), cpCoordinateNet);
161 }
162 for (int dir = 0; dir < dim; ++dir) {
163 std::array<unsigned int, mydimension> ithVec{};
164 ithVec[subDirs[dir]] = 2; // second derivative in dir direction
165 H[dir] = dot(basisFunctionDerivatives.get(ithVec), cpCoordinateNet);
166 }
167 if constexpr (dim > 1) {
168 std::array<int, dim> mixeDerivs;
169 std::ranges::fill_n(mixeDerivs.begin(), 2, 1); // first mixed derivatives
170 int mixedDireCounter = dim;
171 do {
172 H[mixedDireCounter++] = dot(basisFunctionDerivatives.get(mixeDerivs), cpCoordinateNet);
173 } while (std::ranges::next_permutation(mixeDerivs, std::greater()).found);
174 }
175 return std::make_tuple(pos, J, H);
176 }

183 [[nodiscard]] LocalCoordinate local(const GlobalCoordinate& global) const {
184 if constexpr (dim == 0)
185 return {};
186 else if constexpr (dim != dimworld) {
187 auto [u, Ru, fu, gap] = Dune::IGA::closestPointProjectionByTrustRegion(*this, global);
188 return u;
189 } else {
190 const ctype tolerance = ctype(16) * std::numeric_limits<ctype>::epsilon();
191 LocalCoordinate x = domainMidPoint();
192
193 LocalCoordinate dx{};
194 do { // from multilinearGeometry
195 const GlobalCoordinate dglobal = (*this).global(x) - global;
196 MatrixHelper::template xTRightInvA<mydimension, coorddimension>(jacobianTransposed(x), dglobal, dx);
197 const bool invertible
198 = MatrixHelper::template xTRightInvA<mydimension, coorddimension>(jacobianTransposed(x), dglobal, dx);
199
200 if (!invertible) return LocalCoordinate(std::numeric_limits<ctype>::max());
201 x -= dx;
202 // if local is outside the maximum knot vector span bound, thus we clamp it to it and hope for convergence
203 for (int i = 0; i < mydimension; ++i) {
204 if (Dune::FloatCmp::gt(x[i], patchData_.knotSpans[i].back())) x[i] = patchData_.knotSpans[i].back();
205 if (Dune::FloatCmp::lt(x[i], patchData_.knotSpans[i].front())) x[i] = patchData_.knotSpans[i].front();
206 }
207
208 } while (dx.two_norm2() > tolerance);
209 return x;
210 }
211 }

256 auto calculateNurbsAndControlPointNet(const LocalCoordinate& u) const {
257 auto subNetStart = findSpanCorrected(patchData_.degree, u, patchData_.knotSpans);

259 auto cpCoordinateNet = netOfSpan(subNetStart,patchData_.degree,
260 extractControlCoordinates(patchData_.controlPoints));

272 [[nodiscard]] static JacobianTransposed jacobianTransposedImpl(const LocalCoordinate& u,const NurbsLocalView& nurbsLocalView,
273 const MultiDimensionNet<dim, ControlPoint<FieldVector<ctype,dimworld>>>& localControlPointNet) {
274
275 JacobianTransposed result;
276
277 const auto basisFunctionDerivatives = nurbsLocalView.basisFunctionDerivatives(u, 1);
278
279 for (int dir = 0; dir < mydimension; ++dir) {
280 std::array<unsigned int, mydimension> ithVec{};
281 ithVec[dir] = 1;
282 result[dir] = dot(basisFunctionDerivatives.get(ithVec), localControlPointNet);
283 }
284 return result;
285 }
