9 /* \brief Create a surface of revolution from a generating curve, from Piegl and Tiller, Algo 8.1
10 * \tparam ScalarType the field type (use float, double, complex, etc)
11 * \param generatrix 1D nurbs curve, which should be revolved
12 * \param point the origin of revolution
13 * \param revolutionAxis the axis of revolution
14 * \param revolutionAngle how far the curve should be revolved in degrees
15 * \return NURBSPatchData representing the surface
16 */
17 template <typename ScalarType = double>
18 auto makeSurfaceOfRevolution(const NURBSPatchData<1, 3, ScalarType>& generatrix,
19 const Dune::FieldVector<ScalarType, 3>& point,
20 const Dune::FieldVector<ScalarType, 3>& revolutionAxis,
21 const ScalarType& revolutionAngle = 360.0) {
22 const auto& genCP = generatrix.controlPoints;
23 using ControlPoint = typename NURBSPatchData<2, 3, ScalarType>::ControlPointType;
24 using GlobalCoordinateType = typename NURBSPatchData<2, 3, ScalarType>::GlobalCoordinateType;
25 const auto pi = std::numbers::pi_v<ScalarType>;
26
27 const auto revolutionAxisN = revolutionAxis / revolutionAxis.two_norm();
28 auto newKnotsArray = std::array<std::vector<double>, 2UL>();
29 newKnotsArray[1] = generatrix.knotSpans[0];
30 auto& U = newKnotsArray[0];
31
32 const int narcs = std::ceil(revolutionAngle / 90);
33 U.resize(2 * (narcs + 2));
34 if (revolutionAngle <= 90.0) {
35 } else if (revolutionAngle <= 180.0) {
36 U[3] = U[4] = 0.5;
37 } else if (revolutionAngle <= 270.0) {
38 U[3] = U[4] = 1.0 / 3.0;
39 U[5] = U[6] = 2.0 / 3.0;
40 } else {
41 U[3] = U[4] = 0.25;
42 U[5] = U[6] = 0.5;
43 U[7] = U[8] = 0.75;
44 }
45 const ScalarType dtheta = revolutionAngle / narcs * pi / 180;
46 std::ranges::fill_n(U.begin(), 3, 0.0);
47 std::ranges::fill_n(std::ranges::reverse_view(U).begin(), 3, 1.0);
48
49 typename NURBSPatchData<2UL, 3UL, ScalarType>::ControlPointNetType surfaceCP(
50 2 * narcs + 1, generatrix.controlPoints.strideSizes()[0]);
51 using std::cos;
52 using std::sin;
53 const ScalarType wm = cos(dtheta / 2.0);
54 std::vector<ScalarType> cosines(narcs);
55 std::vector<ScalarType> sines(narcs);
56 ScalarType angle = 0.0;
57 for (int i = 0; i < narcs; i++) {
58 angle += dtheta;
59 cosines[i] = cos(angle);
60 sines[i] = sin(angle);
61 }
62 ControlPoint PO = genCP.directGet(0);
63 for (int j = 0; j < genCP.strideSizes()[0]; j++) {
64 const GlobalCoordinateType Om = projectPointOntoLine(point, revolutionAxisN, genCP.directGet(j).p);
65 GlobalCoordinateType X = genCP.directGet(j).p - Om;
66 const ScalarType r = X.two_norm();
67 X /= r;
68 const GlobalCoordinateType Y = cross(revolutionAxisN, X);
69 surfaceCP(0, j) = PO = genCP.directGet(j);
70 GlobalCoordinateType TO = Y;
71 for (int index = 0, i = 0; i < narcs; ++i) {
72 const GlobalCoordinateType P2 = Om + r * cosines[i] * X + r * sines[i] * Y;
73 surfaceCP(index + 2, j) = {.p = P2, .w = genCP.directGet(j).w};
74
75 const GlobalCoordinateType T2 = -sines[i] * X + cosines[i] * Y;
76 surfaceCP(index + 1, j).p = intersect3DLines(PO.p, TO, P2, T2);
77 surfaceCP(index + 1, j).w = wm * genCP.directGet(j).w;
78 index += 2;
79 if (i < narcs - 1) {
80 PO.p = P2;
81 TO = T2;
82 }
83 }
84 }
85 return NURBSPatchData<2UL, 3UL, ScalarType>(newKnotsArray, surfaceCP, {2, generatrix.degree[0]});
86 }
