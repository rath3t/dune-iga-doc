33 template <std::floating_point ScalarType = double>
34 NURBSPatchData<1, 3, ScalarType> makeCircularArc(const ScalarType radius = 1.0, const ScalarType startAngle = 0.0,
35 ScalarType endAngle = 360.0,
36 const Dune::FieldVector<ScalarType, 3>& origin = {0, 0, 0},
37 const Dune::FieldVector<ScalarType, 3>& X = {1, 0, 0},
38 const Dune::FieldVector<ScalarType, 3>& Y = {0, 1, 0}) {
39 using GlobalCoordinateType = typename NURBSPatchData<1, 3, ScalarType>::GlobalCoordinateType;
40 const auto pi = std::numbers::pi_v<ScalarType>;
41
42 if (endAngle < startAngle) endAngle += 360.0;
43 const ScalarType theta = endAngle - startAngle;
44 const int narcs = std::ceil(theta / 90);
45
46 typename NURBSPatchData<1, 3, ScalarType>::ControlPointNetType circleCPs(2 * narcs + 1);
47 const ScalarType dtheta = theta / narcs * pi / 180;
48 const int n = 2 * narcs;
49 const ScalarType w1 = cos(dtheta / 2.0);
50 GlobalCoordinateType PO = origin + radius * cos(startAngle) * X + radius * sin(startAngle) * Y;
51 GlobalCoordinateType TO = -sin(startAngle) * X + cos(startAngle) * Y;
52 circleCPs.directGet(0).p = PO;
53 ScalarType angle = startAngle;
54 for (int index = 0, i = 0; i < narcs; ++i) {
55 angle += dtheta;
56 const GlobalCoordinateType P2 = origin + radius * cos(angle) * X + radius * sin(angle) * Y;
57 circleCPs.directGet(index + 2).p = P2;
58 const GlobalCoordinateType T2 = -sin(angle) * X + cos(angle) * Y;
59 const GlobalCoordinateType P1 = intersect3DLines(PO, TO, P2, T2);
60 circleCPs.directGet(index + 1) = {.p = P1, .w = w1};
61 index += 2;
62 if (i < narcs - 1) {
63 PO = P2;
64 TO = T2;
65 }
66 }
67
68 auto knotVec = std::array<std::vector<double>, 1>{};
69 auto& U = knotVec[0];
70 U.resize(2 * (narcs + 2));
71 if (narcs == 1) {
72 } else if (narcs == 2) {
73 U[3] = U[4] = 1.0 / 2.0;
74 } else if (narcs == 3) {
75 U[3] = U[4] = 1.0 / 3.0;
76 U[5] = U[6] = 2.0 / 3.0;
77 } else {
78 U[3] = U[4] = 1.0 / 4.0;
79 U[5] = U[6] = 2.0 / 4.0;
80 U[7] = U[8] = 3.0 / 4.0;
81 }
82
83 std::ranges::fill_n(U.begin(), 3, 0.0);
84 std::ranges::fill_n(std::ranges::reverse_view(U).begin(), 3, 1.0);
85 return NURBSPatchData<1, 3, ScalarType>(knotVec, circleCPs, {2});
86 }
