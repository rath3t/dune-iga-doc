98 template <std::ranges::sized_range V>
99 explicit MultiDimensionNet(const std::array<V, netdim>& values) {
100 for (int i = 0; i < netdim; ++i)
101 dimSize_[i] = values[i].size();
102
103 const int localSize = std::accumulate(dimSize_.begin(), dimSize_.end(), 1, std::multiplies{});
104 values_.resize(localSize);
105 for (int i = 0; i < localSize; i += dimSize_[0])
106 std::copy(values[0].begin(), values[0].end(), values_.begin() + i);
107 // https://godbolt.org/z/TbTPszGqT
108 for (int curSize = dimSize_[0], k = 1; k < netdim; curSize *= (dimSize_[k]), ++k)
109 for (int ij = 0; ij < localSize; ij += curSize * (dimSize_[k]))
110 for (int j = 0; auto& v2 : values[k]) {
111 auto countedView = std::views::counted(values_.begin() + ij + (j++) * curSize, curSize);
112 std::ranges::transform(countedView, countedView.begin(), [&v2](auto& v) { return v * v2; });
113 }
114 }

203 auto subNet(const std::array<int, netdim>& start, const std::array<int, netdim>& size) const {
204 std::vector<ValueType> subValues;
205 subValues.reserve(std::accumulate(size.begin(), size.end(), 1, std::multiplies{}));
206 if constexpr (netdim == 1)
207 for (int i = 0; i < size[0]; ++i)
208 subValues.push_back(get({start[0] + i}));
209 else if constexpr (netdim == 2)
210 for (int j = 0; j < size[1]; ++j)
211 for (int i = 0; i < size[0]; ++i)
212 subValues.push_back(get({start[0] + i, start[1] + j}));
213 else if constexpr (netdim == 3)
214 for (int k = 0; k < size[2]; ++k) // TODO generalize
215 for (int j = 0; j < size[1]; ++j)
216 for (int i = 0; i < size[0]; ++i)
217 subValues.push_back(get({start[0] + i, start[1] + j, start[2] + k}));
218
219 return MultiDimensionNet<netdim, ValueType>(size, subValues);
220 }

239 template <typename ReturnType = std::array<int, netdim>>
240 static ReturnType directToMultiIndex(const std::array<int, netdim>& dimSize, const int index) {
241 ReturnType multiIndex;
242
243 int help = index;
244 for (int i = 0; i < netdim; ++i) {
245 int temp = help % (dimSize[i]);
246 multiIndex[i] = temp;
247 help -= temp;
248 help = help / dimSize[i];
249 }
250 return multiIndex;
251 }

323 template <typename ArrayType = std::array<int, netdim>>
324 size_t index(const ArrayType& multiIndex) const {
325 assert(!std::ranges::any_of(multiIndex, [](int i) { return i < 0; })
326 && "The passed multiIndex has negative values");
327 assert(!std::ranges::any_of(multiIndex, [id = 0, this](int i) mutable { return i > dimSize_[id++] - 1; })
328 && "The passed multiIndex has too large values");
329
330 size_t index = 0;
331 for (int i = 0; i < netdim; ++i) {
332 int help = 1;
333 for (int j = i - 1; j > -1; --j)
334 help *= dimSize_[j];
335
336 index += help * static_cast<size_t>(multiIndex[i]);
337 }
338 return index;
339 }

376 public:
377 HyperSurfaceIterator(MultiDimensionNet<netdim, ValueType>& net, const std::array<int, netdim - 1>& direction,
378 int at)
379 : net_{&net}, direction_{direction}, at_{at} {
380 std::array<int, netdim - 1> indicesSurface;
381 for (int i = 0; i < indicesSurface.size(); ++i)
382 indicesSurface[i] = net.strideSizes().at(direction[i]);
383
384 const int cpSize = std::accumulate(indicesSurface.begin(), indicesSurface.end(), 1, std::multiplies{});
385 std::function<std::array<int, netdim - 1>(int)> func = [indicesSurface](auto i) {
386 return MultiDimensionNet<netdim - 1, int>::template directToMultiIndex<std::array<int, netdim - 1>>(
387 indicesSurface, i);
388 };
389 viewOverIndices_ = std::ranges::iota_view(0, cpSize) | std::views::transform(func);
390 }
391
392 std::ranges::transform_view<std::ranges::iota_view<int, int>, std::function<std::array<int, netdim - 1>(int)>>
393 viewOverIndices_;

401 --at_;

424 auto operator*() {
425 std::array<int, netdim> multiIndex;
426 if constexpr (netdim != 1)
427 for (int dirI = 0, i = 0; i < netdim; ++i) {
428 if (dirI < direction_.size() && i == direction_.at(dirI++)) continue;
429 multiIndex[i] = at_;
430 break;
431 }
432 else
433 multiIndex[0] = at_;
434
435 auto objectExtractor = [ multiIndex, this ](auto mI) mutable -> auto& {
436 if constexpr (netdim != 1)
437 for (int i = 0; i < direction_.size(); ++i)
438 multiIndex[direction_[i]] = mI[i];
439
440 return net_->get(multiIndex);
441 };
442 return std::ranges::transform_view(viewOverIndices_, objectExtractor);
443 }
