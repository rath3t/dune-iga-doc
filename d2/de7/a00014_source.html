55 else if (codim == 1 && griddim == 3) // surface case
56 return gridView_->getPatch(0).getGlobalSurfaceIndexFromElementIndex(e.impl().getIndex(), i);
57 else
58 throw std::logic_error("subIndex only defined from element to vertices, edges and surfaces");
59 }
