42 auto callFindIntersection(const auto& curvePatchGeo, const int edgeIdx, const auto& ip, const auto& corners)
43 -> std::pair<double, FieldVector<double, 2>> {
44 // @todo gerneralize for more than one loop
45 auto pos = corners[edgeIdx];

47
48 auto guessTParam = FieldVector<double, 2>{findGoodStartingPoint(curvePatchGeo, ip), 0.5};
49
50 auto [success, tParam, curvePoint] = findIntersectionCurveAndLine(curvePatchGeo, pos, dir, guessTParam);
51 if (success == FindIntersectionCurveAndLineResult::sucess) return std::make_pair(tParam[0], curvePoint);
52 if (success == FindIntersectionCurveAndLineResult::linesParallel) {
53 DUNE_THROW(Dune::GridError, "Couldn't find intersection Point, lines are parallel");
54 }
55
56 assert(Util::approxSamePoint(ip, curvePoint, 1e-4) && "Intersection Point and Curve Point are not the same");
57
58 auto domain = curvePatchGeo.domain();
59 if (FloatCmp::eq(curvePoint, curvePatchGeo.global({domain[0].front()})))
60 return std::make_pair(domain[0][0], curvePoint);
61 if (FloatCmp::eq(curvePoint, curvePatchGeo.global({domain[0].back()})))
62 return std::make_pair(domain[0].back(), curvePoint);
63
64 DUNE_THROW(Dune::GridError, "Couldn't find intersection Point");
65 };
