132 [[nodiscard]] double volume() const {
133 const auto rule = Dune::QuadratureRules<ctype, mydimension>::rule(
134 this->type(), mydimension * (*std::ranges::max_element(patchData_.degree)));
135 ctype vol = 0.0;
136 for (auto& gp : rule) {
137 vol += integrationElement(gp.position()) * gp.weight();
138 }
139 return vol;
140 }

146 [[nodiscard]] int corners() const { return 1 << mydimension; }

153 [[nodiscard]] GlobalCoordinate corner(int k) const {
154 LocalCoordinate localcorner;
155 for (size_t i = 0; i < mydimension; i++) {
156 localcorner[i] = (k & (1 << i)) ? uniqueKnotVector()[i].back() : uniqueKnotVector()[i].front();
157 }
158 return global(localcorner);
159 }

166 [[nodiscard]] FieldVector<ctype, worlddimension> global(const LocalCoordinate& u) const {

244 [[nodiscard]] FieldVector<ctype, mydimension> domainMidPoint() const {
245 auto dom = domain();
246 Dune::FieldVector<ctype, mydimension> result{};
247 for (int i = 0; i < mydimension; ++i)
248 result[i] = dom[i].center();
249 return result;
250 }

256 auto numberOfControlPoints() const { return patchData_.controlPoints.strideSizes(); }

263 std::array<int, mydimension> spansPerDirection;
264 for (int i = 0; i < mydimension; ++i)
265 spansPerDirection[i] = uniqueKnotSpans_[i].size() - 1;
266 return spansPerDirection;
267 }

282 /* @brief Calculate NURBS and control point net for a given local coordinate. */
283 auto calculateNurbsAndControlPointNet(const LocalCoordinate& u) const {
284 auto subNetStart = Splines::findSpan(patchData_.degree, u, patchData_.knotSpans);
285 auto cpCoordinateNet = Splines::netOfSpan(subNetStart, patchData_.degree,
286 Splines::extractControlCoordinates(patchData_.controlPoints));
287 auto nurbsLocalView = nurbs_.localView();
288 nurbsLocalView.bind(subNetStart);
289 return std::make_tuple(nurbsLocalView, cpCoordinateNet, subNetStart);
290 }
291
292 /* @brief Calculate zero first and second derivatives of the position for a given local coordinate. */
293 static auto zeroFirstAndSecondDerivativeOfPositionImpl(const LocalCoordinate& u,
294 const NurbsLocalView& nurbsLocalView,
295 const ControlPointCoordinateNetType& localControlPointNet) {
296 return std::make_tuple(GeometryKernel::position(u, nurbsLocalView, localControlPointNet),
297 GeometryKernel::jacobianTransposed(u, nurbsLocalView, localControlPointNet),
298 GeometryKernel::hessian(u, nurbsLocalView, localControlPointNet));
299 }

303 Nurbs nurbs_;
