55 template <typename ScalarType, int dim, size_t dim2>
56 requires(dim == dim2) bool clampToBoundaryAndCheckIfIsAtAllBoundaries(
57 FieldVector<ScalarType, dim>& x, const std::array<Domain<ScalarType>, dim2>& domain = {}) {
58 int breakDueToBoundaryCounter = 0;
59
60 for (int j = 0; j < dim; ++j) {
61 x[j] = std::clamp(x[j], domain[j].left(), domain[j].right());
62 if (x[j] == domain[j].left() or x[j] == domain[j].right()) ++breakDueToBoundaryCounter;
63 }
64 if (dim == breakDueToBoundaryCounter) return true;
65 return false;
66 }

77 template <std::floating_point T>
78 auto linspace(T a, T b, unsigned int N) {
79 const T inc = (b - a) / static_cast<T>(N - 1);
80 auto val = [=](int i) -> T {
81 if (i != 0 and i != N - 1) return a + i * inc;
82 if (i == N - 1) return b;
83 return a;
84 };
85
86 return std::views::iota(0u, N) | std::views::transform(val);
87 }
