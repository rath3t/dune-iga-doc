79 inline int giveEdgeIdx(const std::size_t e1, const std::size_t e2) {
80 assert(e1 < 4 and e2 < 4);
81 if ((e1 == 0 and e2 == 1) or (e1 == 1 and e2 == 0)) return 0;
82 if ((e1 == 1 and e2 == 2) or (e1 == 2 and e2 == 1)) return 1;
83 if ((e1 == 2 and e2 == 3) or (e1 == 3 and e2 == 2)) return 2;
84 if ((e1 == 3 and e2 == 0) or (e1 == 0 and e2 == 3)) return 3;
85 assert(false);

96 return FloatCmp::eq(Area(clippedEdges), Area(eleRect));

99 // First determine if element is trimmed

101 if (intersectResult.empty())
