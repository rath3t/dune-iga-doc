28 auto isAlreadyThere(const auto& pt) {
29 const auto it = std::ranges::find_if(vertices_, [&](const VertexVariant& vertexVariant) {
30 auto isSame = false;
31 std::visit([&](auto&& vertex) { isSame = FloatCmp::eq(vertex.pt.x, pt.x) and FloatCmp::eq(vertex.pt.y, pt.y); },
32 vertexVariant);
33 return isSame;
34 });
35 return it != vertices_.end();
36 }

38 void addOriginalVertex(const Clipper2Lib::PointD& pt) {
39 if (not isAlreadyThere(pt)) vertices_.emplace_back(HostVertex(pt.z, pt));
40 }

42 void addNewVertex(const int edgeIdx, const Clipper2Lib::PointD& pt) {
43 if (not isAlreadyThere(pt)) vertices_.emplace_back(NewVertex(edgeIdx, pt));

46 void finish() {
47 // Sort the points in counter clockwise manner such that the first point is in the lower left (courtesy to
48 // ChatGPT)
49 auto getPoint = [](auto&& vertexVariant) { return vertexVariant.pt; };
50
51 const auto minVertex
52 = *std::ranges::min_element(vertices_, [&](const VertexVariant& aV, const VertexVariant& bV) {
53 const auto a = std::visit(getPoint, aV);
54 const auto b = std::visit(getPoint, bV);
55 return (a.y < b.y) || (a.y == b.y && a.x < b.x);
56 });
57
58 auto polarAngle
59 = [](const auto& p, const auto& reference) -> double { return atan2(p.y - reference.y, p.x - reference.x); };
60
61 // Sort the points based on polar angle with respect to the reference point
62 auto comparePoints
63 = [&](const VertexVariant& aV, const VertexVariant& bV, const VertexVariant& referenceV) -> bool {
64 const auto a = std::visit(getPoint, aV);
65 const auto b = std::visit(getPoint, bV);
66 const auto reference = std::visit(getPoint, referenceV);
67 const double angleA = polarAngle(a, reference);
68 const double angleB = polarAngle(b, reference);
69
70 if (angleA != angleB) return angleA < angleB;
71
72 // If two points have the same angle, sort based on distance to the reference point
73 return (a.x - reference.x) * (a.x - reference.x) + (a.y - reference.y) * (a.y - reference.y)
74 < (b.x - reference.x) * (b.x - reference.x) + (b.y - reference.y) * (b.y - reference.y);
75 };
76 std::ranges::sort(vertices_, [&](const VertexVariant& aV, const VertexVariant& bV) {
77 return comparePoints(aV, bV, minVertex);
78 });
79 }
80 std::vector<VertexVariant> vertices_{};
81 };

83 inline int giveEdgeIdx(const std::size_t e1, const std::size_t e2) {
84 assert(e1 < 4 and e2 < 4);
85 if ((e1 == 0 and e2 == 1) or (e1 == 1 and e2 == 0)) return 0;
86 if ((e1 == 1 and e2 == 2) or (e1 == 2 and e2 == 1)) return 1;
87 if ((e1 == 2 and e2 == 3) or (e1 == 3 and e2 == 2)) return 2;
88 if ((e1 == 3 and e2 == 0) or (e1 == 0 and e2 == 3)) return 3;
89 assert(false);
90 }

94 inline auto clipElementRectangle(Clipper2Lib::PathD& eleRect, const Clipper2Lib::PathsD& trimmingCurves)
95 -> std::tuple<ElementTrimFlag, ClippingResult> {
96 using namespace Clipper2Lib;
97
98 auto isFullElement = [&](const auto& clippedEdges) { return FloatCmp::eq(Area(clippedEdges), Area(eleRect)); };
99
100 // First determine if element is trimmed
101 const auto intersectResult = Intersect({eleRect}, trimmingCurves, FillRule::NonZero, 2);
102 if (intersectResult.empty()) return std::make_tuple(ElementTrimFlag::empty, ClippingResult{});
103 if (isFullElement(intersectResult.front())) {
104 matplot::rectangle(eleRect[0].x, eleRect[1].y, eleRect[1].x - eleRect[0].x, eleRect[3].y - eleRect[0].y);
105 return std::make_tuple(ElementTrimFlag::full, ClippingResult{});
106 }
107
108 // Element is trimmed
109 ClipperD clipper(8);
110 ClippingResult result{};
111
112 clipper.SetZCallback([&](const PointD& e1bot, const PointD& e1top, const PointD& e2bot, const PointD& e2top,
113 const PointD& pt) { result.addNewVertex(giveEdgeIdx(e1bot.z, e1top.z), pt); });
114
115 PathsD resultClosedPaths{};
116 PathsD resultOpenPaths{};
117
118 for (const auto i : std::views::iota(0, 4)) {
119 // std::cout << "Edge " << i << std::endl;
120
121 clipper.Clear();
122 clipper.AddClip(trimmingCurves);
123
124 clipper.AddOpenSubject(PathsD{{eleRect[i], eleRect[nextEntityIdx(i, 1)]}});
125 clipper.Execute(ClipType::Intersection, FillRule::NonZero, resultClosedPaths, resultOpenPaths);
126
127 if (not resultOpenPaths.empty())
128 for (const auto& p : resultOpenPaths.front())
129 result.addOriginalVertex(p);
130 }
131
132 auto isPointInTrimmingCurves = [&](const auto& pt1) -> bool {
133 return std::ranges::any_of(trimmingCurves, [&](const auto& curve) {
134 auto it = std::ranges::find_if(
135 curve, [&](const auto& pt2) { return FloatCmp::eq(pt1.x, pt2.x) and FloatCmp::eq(pt1.y, pt2.y); });
136 return it != curve.end();
137 });
138 };
139
140 // check if any element vertex is a starting point of one of the trimming curves as the algorihtm cannot find them
141 for (const auto& cP : eleRect)
142 if (isPointInTrimmingCurves(cP)) result.addOriginalVertex(cP);
143
144 result.finish();
145
146 std::cout << "Vertices found\n";

148 struct Visitor {
149 static void plotEllipse(const PointD& pt) {
150 constexpr auto w = 0.025;
151 const auto c = matplot::ellipse(pt.x - (w / 2), pt.y - (w / 2), w, w);
152 c->color("blue");
153 }
154
155 void operator()(const ClippingResult::HostVertex& v) const {
156 plotEllipse(v.pt);
157 std::cout << "Pt: " << v.pt << " Host Idx: " << v.hostIdx << std::endl;
158 }
159 void operator()(const ClippingResult::NewVertex& v) const {
160 plotEllipse(v.pt);
161 std::cout << "Edge: " << v.onEdgeIdx << " Pt: " << v.pt << " On BC: " << v.pt.z << std::endl;
162 }
163 };
164
165 for (auto& vV : result.vertices_)
166 std::visit(Visitor{}, vV);
167
168 matplot::hold("on");
169 matplot::axis(matplot::square);
170 matplot::rectangle(eleRect[0].x, eleRect[1].y, eleRect[1].x - eleRect[0].x, eleRect[3].y - eleRect[0].y);
171
172 return std::make_tuple(ElementTrimFlag::trimmed, result);
173 }
