66 inline int giveEdgeIdx(const std::size_t e1, const std::size_t e2) {
67 assert(e1 < 4 and e2 < 4);
68 if ((e1 == 0 and e2 == 1) or (e1 == 1 and e2 == 0)) return 0;
69 if ((e1 == 1 and e2 == 2) or (e1 == 2 and e2 == 1)) return 1;
70 if ((e1 == 2 and e2 == 3) or (e1 == 3 and e2 == 2)) return 2;
71 if ((e1 == 3 and e2 == 0) or (e1 == 0 and e2 == 3)) return 3;
72 __builtin_unreachable();
73 }

77 return e1 > e2;
78 return e1 < e2;
