32 if constexpr (dim == 1) {

78 // Coordinates in a single knot span differ from coordinates on the B-spline patch
79 // by an affine transformation. This transformation is stored in the diagonal entries.
80 // If trimming is disabled the Jacobian in the parameter space of subentities (edges or surfaces) is a
81 // DiagonalMatrixBlock but is treated as a FieldMatrix, since is also just cubes and an axis-aligned geometry, if
82 // trimming is enabled the Jacobian of the parameterspace for subentities is a potentially fully populated
83 // FieldMatrix, Think about an arbitrary curve in the axis-aligned cube parameter grid
84 // ------->
85 // `____________________C_______D_______`
86 // ::````````````````::``````````'|````::
87 // :: :: `| ::
88 // :: ^F:: `| ::
89 // :: | :: `| ::
90 // :: | :: `| ::
91 // :: |E:: `| ::
92 // :: :: .| x ::
93 // :: :: _: ::
94 // ::................::..........|:....::
95 // ::''''''''''''''''::''''''''''/'''''::
96 // :: :: _:B ::
97 // :: :: .| ::
98 // :: :: `::A ::
99 // :: '::::||::::::' ::
100 // :: ::` :: x ::
101 // :: |' :: ::
102 // :: | x :: ::
103 // ::````````|```````::````````````````::
104 // `____________________________________`
105 // at point A the tangent of the curve (the JacobianInParameterSpace) points into the oblique direction (to B)
106 // for the axis-aligned geometry these tangents are from E to F or from C to D, depending on the orientation of
107 // the subentity, which are aligned with the axis
108 using JacobianTransposedInParameterSpace = typename ParameterSpaceGeometry::JacobianTransposed;
109 using GlobalInParameterSpace = typename ParameterSpaceGeometry::GlobalCoordinate;
110
111 using Hessian = FieldMatrix<ctype, numberOfSecondDerivatives, worlddimension>;

301 Hessian hessian(const LocalCoordinate& local) const {
302 checkState();
303 const auto ouInPatch = globalInParameterSpace(local);
304
305 const auto [p, dfdg, dfdgdg] = patchGeometry_->zeroFirstAndSecondDerivativeOfPositionImpl(
306 ouInPatch, nurbsLocalView_, localControlPointNet);
307
308 const JacobianTransposedInParameterSpace dgdt = jacobianTransposedInParameterSpace(local);
309 static_assert(JacobianTransposedInParameterSpace::rows == mydimension);
310 static_assert(JacobianTransposedInParameterSpace::cols == gridDimension);
311 // dgdt (1x2, curve on surface), (2x2 surface in surface), (2x3 surface in 3D), (3x3 volume in 3D)
312 FieldMatrix<ctype, patchNumberOfSecondDerivatives, numberOfSecondDerivatives> dgdtSquared(0);
313 for (int patchIndex = 0; auto [patchRow, patchCol] : Impl::voigtIndices<gridDimension>()) {
314 for (int myIndex = 0; auto [myRow, myCol] : Impl::voigtIndices<mydimension>()) {
315 if constexpr (not std::is_same_v<JacobianTransposedInParameterSpace, DiagonalMatrix<ctype, gridDimension>>)
316 dgdtSquared[patchIndex][myIndex]
317 = (dgdt[myRow][patchRow] * dgdt[myCol][patchCol] + dgdt[myCol][patchRow] * dgdt[myRow][patchCol])
318 * ((patchRow == patchCol) ? 0.5 : 1.0);
319 else {
320 const auto dgdtmyRowpatchRow = (myRow == patchRow) ? dgdt[myRow][patchRow] : 0;
321 const auto dgdtmyColpatchCol = (myCol == patchCol) ? dgdt[myCol][patchCol] : 0;
322 const auto dgdtmyColpatchRow = (myCol == patchRow) ? dgdt[myCol][patchRow] : 0;
323 const auto dgdtmyRowpatchCol = (myRow == patchCol) ? dgdt[myRow][patchCol] : 0;
324 dgdtSquared[patchIndex][myIndex]
325 = (dgdtmyRowpatchRow * dgdtmyColpatchCol + dgdtmyColpatchRow * dgdtmyRowpatchCol)
326 * ((patchRow == patchCol) ? 0.5 : 1.0);
327 }
328 ++myIndex;
329 }
330 ++patchIndex;
331 }
332
333 // h = mydimension * (mydimension + 1) / 2 X worlddimension = 1x2 (curve in 2D), 1x3 (curve in 3D), (3x2 surface
334 // in 2D), (3x3 surface in 3D), (6x3 volume in 3D)
335 //
336 // dfdgdg = gridDimension*(gridDimension + 1) / 2 X worlddimension
337 // = 3x2 (curve on surface flat),3x3 (curve on surface in 3D), 6x3 (curve in volume), 6x3 (surface in
338 // volume),(3x3 surface in surface in 3D)
339 //
340 // dgdtSquared = gridDimension * (gridDimension + 1) / 2 X mydimension * (mydimension +
341 // 1) / 2 = 3x1 (curve on surface), 6x1 (curve in volume), 6x3 (surface in volume), 3x3 ( surface in surface),
342 // 6x6 ( volume in volume)
343 Hessian h = transposedView(dgdtSquared) * dfdgdg;
344
345 /* if trimming is enabled the parameter space geometry is potentially non-linear,
346 * the resutling Hessian has another contribution due to chain-rule, namely the second derivative of g */
347 if constexpr (not TrimmerType::template isLocalGeometryLinear<codim>) {
348 // const auto dgdtdt = parameterSpaceGeometry->hessian(ouInPatch);
349 //
350 // assert(TrimmerType::template isLocalGeometryLinear<
351 // codim> && "This can not be checked yet. Check if this works with trimming and then remove
352 // assert");
353 // h += transpose(transposedView(dfdg) * dgdtdt);
354 }
355 return h;
356 }
