56 inline int giveEdgeIdx(int e1, int e2) {
57 if ((e1 == 0 and e2 == 1) or (e1 == 1 and e2 == 0))
58 return 0;
59 if ((e1 == 1 and e2 == 2) or (e1 == 2 and e2 == 1))
60 return 1;
61 if ((e1 == 2 and e2 == 3) or (e1 == 3 and e2 == 2))
62 return 2;
63 if ((e1 == 3 and e2 == 0) or (e1 == 0 and e2 == 3))
64 return 3;
65 __builtin_unreachable();
66 }
