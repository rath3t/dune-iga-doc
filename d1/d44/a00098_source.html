12 template <typename GeoCurve, typename ScalarType, int dim>
13 requires(dim == 2)
14 auto findIntersectionLinearCurveAndLine(const GeoCurve& geoCurve, const FieldVector<ScalarType, dim>& pos,
15 const FieldVector<ScalarType, dim>& dir,
16 FieldVector<ScalarType, 2> tParameter, ScalarType tol = 1e-10) {
17 assert(geoCurve.degree()[0] == 1 && "Curve has to be linear to be calling this function");
18 assert(geoCurve.patchData().controlPoints.size() == 2 && "The linear curve should only have two control points");
19
20 const auto& curveP0 = geoCurve.patchData().controlPoints.directGet(0).p;
21 const auto& curveP1 = geoCurve.patchData().controlPoints.directGet(1).p;
22
23 const FieldMatrix<ScalarType, dim, 2> linearSystem({curveP1 - curveP0, -dir});
24 const FieldVector<ScalarType, dim> b = pos - curveP0;
25
26 if (not FloatCmp::eq(linearSystem.determinant(), 0.0, tol)) {
27 FieldVector<ScalarType, dim> sol;
28 transpose(linearSystem).solve(sol, b);
29 // Map the solution to the knotSpan, thus [0,1] to [domain.left(), domain.right()]
30 sol[0] = mapToRangeFromZeroToOne(sol[0], geoCurve.domain()[0]);
31
32 // Check if point is in domain
33 if (not geoCurve.domain()[0].checkInside(sol[0]))
34 return std::make_tuple(FindIntersectionCurveAndLineResult::disjoint, sol, curveP0);
35 return std::make_tuple(FindIntersectionCurveAndLineResult::intersect, sol, geoCurve.global(sol[0]));
36 }
37 // If system is not solvable, the lines are paralell and therefore have no intersection
38 return std::make_tuple(FindIntersectionCurveAndLineResult::parallel, tParameter, curveP0);
39 }

41 template <typename GeoCurve, typename ScalarType, int dim>
42 auto findIntersectionCurveAndLine(const GeoCurve& geoCurve, const FieldVector<ScalarType, dim>& pos,
43 const FieldVector<ScalarType, dim>& dir, FieldVector<ScalarType, 2> tParameter,
44 ScalarType tol = 1e-10) {
45 if (geoCurve.degree()[0] == 1) return findIntersectionLinearCurveAndLine(geoCurve, pos, dir, tParameter, tol);
46
47 struct Line {
48 FieldVector<ScalarType, dim> operator()(ScalarType t) { return pos + t * dir; }
49 const FieldVector<ScalarType, dim>& jacobian() { return dir; }
50
51 const FieldVector<ScalarType, dim>& pos;
52 const FieldVector<ScalarType, dim>& dir;
53 };
54
55 auto line = Line({pos, dir});
56
57 bool sucess = false;
58 const int maxIterations = 1000;
59 FieldVector<ScalarType, 2> curvePoint;
60 auto domain = geoCurve.domain();
61
62 const FieldVector<ScalarType, dim>& lineDerivative = line.jacobian();
63 for (int iter = 0; iter < maxIterations; ++iter) {
64 curvePoint = geoCurve.global(tParameter[0]);
65 const FieldVector curveDerivative = geoCurve.jacobianTransposed(tParameter[0])[0];
66 const FieldVector curvSecondeDerivative = geoCurve.hessian(tParameter[0])[0];
67 const FieldVector dist = curvePoint - line(tParameter[1]);
68
69 const ScalarType energy = 0.5 * dist.two_norm2();
70
71 FieldVector<ScalarType, 2> grad({curveDerivative * dist, -lineDerivative * dist});
72
73 FieldMatrix<ScalarType, 2> Hess;
74 Hess[0][0] = curveDerivative.two_norm2() + dist * curvSecondeDerivative;
75 Hess[1][1] = lineDerivative.two_norm2();
76 Hess[0][1] = Hess[1][0] = -curveDerivative * lineDerivative;
77
78 // Newton-Raphson update
79 FieldVector<ScalarType, 2> deltaT;
80 Hess.solve(deltaT, grad);
81 tParameter -= deltaT;
82
83 // Clamp
84 tParameter[0] = std::clamp(tParameter[0], domain[0].front(), domain[0].back());
85
86 // Check for convergence
87 if (deltaT.two_norm() < tol) {
88 sucess = true;
89 curvePoint = geoCurve.global(tParameter[0]);
90 break;
91 }
92 }
93
94 if (not domain[0].checkInside(tParameter[0])) sucess = false;
95 return std::make_tuple(
96 sucess ? FindIntersectionCurveAndLineResult::intersect : FindIntersectionCurveAndLineResult::disjoint,
97 tParameter, curvePoint);
98 }
