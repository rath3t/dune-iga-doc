12 template <typename GeoCurve, typename ScalarType = typename GeoCurve::ctype>
13 auto splitCurve(const GeoCurve& geoCurve, ScalarType u) -> std::pair<GeoCurve, GeoCurve> {
14 static_assert(GeoCurve::mydimension == 1);
15
16 auto knotMultiplicity = [](auto& knotVec, auto u_) {
17 return std::ranges::count_if(knotVec, [=](double val) { return FloatCmp::eq(u_, val, 1e-8); });
18 };
19
20 auto& patchData = geoCurve.patchData();
21 auto knots = patchData.knotSpans.front();
22 auto control_points = patchData.controlPoints.directGetAll();
23 auto domain = geoCurve.domain();
24 auto degree = patchData.degree[0];
25
26 auto span = Splines::findSpan(degree, u, knots);
27 int r = degree - knotMultiplicity(knots, u);
28
29 std::vector<ScalarType> newKnots(r);
30 std::ranges::fill(newKnots, u);
31 auto tmpPatchData = Splines::knotRefinement(patchData, newKnots, 0);
32
33 std::vector<ScalarType> tmpKnots{tmpPatchData.knotSpans.front()};
34 decltype(control_points) tmpCp{tmpPatchData.controlPoints.directGetAll()};
35
36 std::vector<ScalarType> leftKnots;
37 const int span_l = Splines::findSpan(degree, u, tmpKnots) + 1;
38 for (int i = 0; i < span_l; ++i)
39 leftKnots.push_back(tmpKnots[i]);
40 leftKnots.push_back(u);
41
42 std::vector<ScalarType> rightKnots;
43 for (int i = 0; i < degree + 1; ++i)
44 rightKnots.push_back(u);
45 for (int i = span_l; i < tmpKnots.size(); ++i)
46 rightKnots.push_back(tmpKnots[i]);
47
48 decltype(control_points) leftControlPoints;
49 const int ks = span - degree + 1;
50 for (int i = 0; i < ks + r; ++i)
51 leftControlPoints.push_back(tmpCp[i]);
52
53 decltype(control_points) rightControlPoints;
54 for (int i = ks + r - 1; i < tmpCp.size(); ++i)
55 rightControlPoints.push_back(tmpCp[i]);
56
57 typename GeoCurve::PatchData leftPatchData{
58 {leftKnots}, typename GeoCurve::ControlPointNetType{leftControlPoints}, patchData.degree};
59 typename GeoCurve::PatchData rightPatchData{
60 {rightKnots}, typename GeoCurve::ControlPointNetType{rightControlPoints}, patchData.degree};
61
62 return std::make_pair(GeoCurve{leftPatchData}, GeoCurve{rightPatchData});
63 }

65 template <typename GeoCurve, typename ScalarType = typename GeoCurve::ctype>
66 auto sliceCurve(const GeoCurve& geoCurve, std::array<ScalarType, 2> t) -> GeoCurve {
67 static_assert(GeoCurve::mydimension == 1);
68
69 auto domain = geoCurve.domain();
70 if (domain[0].front() == t[0] and domain[0].back() == t[1]) return geoCurve;
71 if (domain[0].front() == t[0]) return std::get<0>(splitCurve(geoCurve, t[1]));
72 if (domain[0].back() == t[1]) return std::get<1>(splitCurve(geoCurve, t[0]));
73
74 auto tmpCurve = std::get<1>(splitCurve(geoCurve, t[0]));
75 return std::get<0>(splitCurve(tmpCurve, t[1]));
76 }
