28 template <typename Geo> // @todo add concept
29 auto closestPointProjectionByTrustRegion(const Geo& geo, auto& point,
30 const std::optional<typename Geo::LocalCoordinate>& start = std::nullopt)
31 -> std::tuple<typename Geo::LocalCoordinate, typename Geo::ctype, typename Geo::ctype, typename Geo::ctype>
32 requires(Geo::mydimension == 1 or Geo::mydimension == 2) {
33 using ctype = typename Geo::ctype;
34 static constexpr int dim = Geo::mydimension;
35 auto energy = [&](const auto& uL) {
36 auto pointOnCurve = geo.global(uL);
37 auto dist = pointOnCurve - point;
38 return 0.5 * dist.two_norm2();
39 };
40
41 auto energyGradAndHess = [&](const auto& uL) {
42 auto [pointOnCurve, J, H] = geo.zeroFirstAndSecondDerivativeOfPosition(uL);
43 auto dist = pointOnCurve - point;
44 Dune::FieldVector<ctype, dim> g;
45 J.mv(dist, g);
46
47 Dune::FieldMatrix<ctype, dim*(dim + 1) / 2, 1> dfdxideta;
48 H.mv(dist, dfdxideta);
49
50 Dune::FieldMatrix<ctype, dim, dim> HV = J * Dune::transpose(J);
51 for (int i = 0; i < dim; ++i)
52 HV[i][i] += dfdxideta[i];
53
54 if constexpr (dim == 2) {
55 HV[0][1] += dfdxideta[2];
56 HV[1][0] += dfdxideta[2];
57 }
58
59 return std::make_tuple(0.5 * dist.two_norm2(), g, HV);
60 };
61
62 auto isPositiveDefiniteInDirection = [](auto&& H, auto&& du) {
63 std::remove_cvref_t<decltype(du)> tmp;
64 H.mv(du, tmp);
65 auto kappa = du * tmp;
66 return kappa > 0.0;
67 };
68
69 int maxiter = 100;
70
71 // start with mid-point if nothing else is given
72 Dune::FieldVector<ctype, dim> u = start ? start.value() : geo.domainMidPoint();
73
74 ctype tol = ctype(16) * std::numeric_limits<ctype>::epsilon();
75 ctype Rnorm;
76 auto domain = geo.domain();
77
78 double domainFractionFactor = 4;
79 ctype domainFraction = 1;
80 for (int j = 0; j < dim; ++j)
81 domainFraction *= domain[j].size();
82 domainFraction /= domainFractionFactor;
83 int i;
84 auto [oldEnergy, R, H] = energyGradAndHess(u);
85 ctype energyVal = oldEnergy;
86 ctype r = oldEnergy;
87 for (i = 0; i < maxiter; ++i) {
88 FieldVector<ctype, dim> du;
89 H.solve(du, -R);
90 // if the increment is too large, we tend to overshoot, thus, we limit it to a fraction of the total domain
91 ctype duNorm = du.two_norm();
92 if (duNorm > domainFraction) du *= domainFraction / duNorm;
93
94 const bool isPositiveDef = isPositiveDefiniteInDirection(H, du);
95 auto uold = u;
96 oldEnergy = energy(u);
97 u += du;
98 if (Utilities::clampToBoundaryAndCheckIfIsAtAllBoundaries(u, domain)) break;
99
100 int testSteps = 20;
101 for (int j = 0; j < testSteps; ++j) {
102 energyVal = energy(u);
103 if (energyVal > oldEnergy and duNorm > 1e-8) { // we directly accept very small steps
104 u = uold;
105 ctype scaleFactor = Dune::power(2.0, j + 1);
106 // If our new step didn't move in an energy-decreasing direction, we have to modify
107 if (isPositiveDef) {
108 // If our model m(du)= oldEnergy+ dot(R,du) + 0.5*(du^T*H*du) has a positive curvature du^T*Hessian*du, we
109 // got
110 // as du the minimizer of the parabola. Since we got energy increase, our model is bad in comparison to
111 // the real energy(u). Thus, we decrease our step size and hope for the best, at least in some (small)
112 // neighborhood our model is good
113 u += du / scaleFactor;
114 } else {
115 // If our model is not positive definite (a downward parabola), using du is useless, since it is at the
116 // maximizer of m(du)
117 // (classical Newton method results, since it only finds the stationary point)
118 // Thus, we go in the opposite direction, since we know that we go downward the model m(du) and again
119 // check for real energy decrease
120 u -= du / scaleFactor;
121 }
122 } else
123 break;
124
125 if (j == testSteps - 1) DUNE_THROW(Dune::MathError, "We couldn't find an energy decreasing direction");
126 }
127 std::tie(oldEnergy, R, H) = energyGradAndHess(u);
128
129 Rnorm = R.two_norm();
130 if (Rnorm < tol) break;
131 }
132
133 // clamp again to boundary
134 for (int j = 0; j < dim; ++j)
135 u[j] = clampToDomain(u[j], domain[j]);
136
137 return std::make_tuple(u, Rnorm, energyVal, (geo.global(u) - point).two_norm());
138 }
