11 template <typename GeoCurve, typename ScalarType, int dim>
12 auto findIntersectionLinearCurveAndLine(const GeoCurve& geoCurve, const FieldVector<ScalarType, dim>& pos,
13 const FieldVector<ScalarType, dim>& dir,
14 FieldVector<ScalarType, 2> tParameter, ScalarType tol = 1e-10) {
15 assert(geoCurve.degree()[0] == 1 && "Curve has to be linear to be calling this function");
16
17 const FieldVector curveDerivative = geoCurve.jacobianTransposed(tParameter[0])[0];
18 const FieldVector curvePos = geoCurve.global({geoCurve.domain()[0].front()});
19
20 FieldMatrix<ScalarType, dim, 2> linearSystem;
21 FieldVector<ScalarType, dim> b;
22
23 for (int i = 0; i < dim; i++) {
24 linearSystem[i][0] = curveDerivative[i];
25 linearSystem[i][1] = -dir[i];
26 b[i] = pos[i] - curvePos[i];
27 }
28
29 // If system is not solvable, the lines are paralell and therefore have no intersection
30 if (FloatCmp::eq(linearSystem.determinant(), 0.0, tol))
31 return std::make_tuple(FindIntersectionCurveAndLineResult::linesParallel, tParameter,
32 geoCurve.global(tParameter[0]));
33
34 FieldVector<ScalarType, dim> sol;
35 linearSystem.solve(sol, b);
36
37 // Map the solution to the knotSpan
38 sol[0] = geoCurve.domain()[0].front() + sol[0];
39
40 // Check if point is in domain
41 if (not geoCurve.domain()[0].checkInside(sol[0]))
42 return std::make_tuple(FindIntersectionCurveAndLineResult::noSucess, sol, geoCurve.global(sol[0]));
43 return std::make_tuple(FindIntersectionCurveAndLineResult::sucess, sol, geoCurve.global(sol[0]));
44 }

46 template <typename GeoCurve, typename ScalarType, int dim>
47 auto findIntersectionCurveAndLine(const GeoCurve& geoCurve, const FieldVector<ScalarType, dim>& pos,
48 const FieldVector<ScalarType, dim>& dir, FieldVector<ScalarType, 2> tParameter,
49 ScalarType tol = 1e-10) {
50 if (geoCurve.degree()[0] == 1) return findIntersectionLinearCurveAndLine(geoCurve, pos, dir, tParameter, tol);
51
52 struct Line {
53 FieldVector<ScalarType, dim> operator()(ScalarType t) { return pos + t * dir; }
54 const FieldVector<ScalarType, dim>& jacobian() { return dir; }
55
56 const FieldVector<ScalarType, dim>& pos;
57 const FieldVector<ScalarType, dim>& dir;
58 };
59
60 auto line = Line({pos, dir});
61
62 bool sucess = false;
63 const int maxIterations = 1000;
64 FieldVector<ScalarType, 2> curvePoint;
65 auto domain = geoCurve.domain();
66
67 const FieldVector<ScalarType, dim>& lineDerivative = line.jacobian();
68 for (int iter = 0; iter < maxIterations; ++iter) {
69 curvePoint = geoCurve.global(tParameter[0]);
70 const FieldVector curveDerivative = geoCurve.jacobianTransposed(tParameter[0])[0];
71 const FieldVector curvSecondeDerivative = geoCurve.hessian(tParameter[0])[0];
72 const FieldVector dist = curvePoint - line(tParameter[1]);
73
74 const ScalarType energy = 0.5 * dist.two_norm2();
75
76 FieldVector<ScalarType, 2> grad({curveDerivative * dist, -lineDerivative * dist});
77
78 FieldMatrix<ScalarType, 2> Hess;
79 Hess[0][0] = curveDerivative.two_norm2() + dist * curvSecondeDerivative;
80 Hess[1][1] = lineDerivative.two_norm2();
81 Hess[0][1] = Hess[1][0] = -curveDerivative * lineDerivative;
82
83 // Newton-Raphson update
84 FieldVector<ScalarType, 2> deltaT;
85 Hess.solve(deltaT, grad);
86 tParameter -= deltaT;
87
88 // Clamp
89 tParameter[0] = std::clamp(tParameter[0], domain[0].front(), domain[0].back());
90
91 // Check for convergence
92 if (deltaT.two_norm() < tol) {
93 sucess = true;
94 curvePoint = geoCurve.global(tParameter[0]);
95 break;
96 }
97 }
98
99 if (not domain[0].checkInside(tParameter[0])) sucess = false;
100 return std::make_tuple(
101 sucess ? FindIntersectionCurveAndLineResult::sucess : FindIntersectionCurveAndLineResult::noSucess, tParameter,
102 curvePoint);
103 }
