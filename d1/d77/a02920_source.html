298 void globalRefine(int refCount) {
299 for (int i = 0; i < refCount; ++i) {
300 const auto& finestPatchData = patchGeometries_.back().patchData();
301 auto newfinestPatchData = finestPatchData;
302
303 auto& olduniqueKnotVector = patchGeometries_.back().uniqueKnotVector();
304 std::array<std::vector<ctype>, dim> newUniqueKnotVecs;
305 for (int refDirection = 0; refDirection < dim; ++refDirection) {
306 auto additionalKnots = Splines::generateRefinedKnots(finestPatchData.knotSpans, refDirection, 1);
307 newfinestPatchData = Splines::knotRefinement<dim>(newfinestPatchData, additionalKnots, refDirection);
308 auto& newKnotVec = newUniqueKnotVecs[refDirection];
309
310 // compute new unique Knot vectors from refinement
311 newKnotVec.resize(olduniqueKnotVector[refDirection].size() + additionalKnots.size());
312 std::ranges::merge(olduniqueKnotVector[refDirection], additionalKnots, std::begin(newKnotVec));
313 }
314
315 patchGeometries_.emplace_back(newfinestPatchData, newUniqueKnotVecs);
316 patchGeometriesUnElevated.emplace_back(patchGeometries_.back());
317 }
318
319 trimmer_.parameterSpaceGrid().globalRefine(refCount);
320 setIndices();
321 }

338 void globalRefineInDirection(const std::array<int, dim>& refines) {
339 const auto& finestPatchData = patchGeometries_.back().patchData();
340 auto newfinestPatchData = finestPatchData;
341 for (int dir = 0; auto refinesInDirection : refines) {
342 if (refinesInDirection == 0) {
343 ++dir;
344 continue;
345 }
346 auto additionalKnots = Splines::generateRefinedKnots(finestPatchData.knotSpans, dir, refinesInDirection);
347 newfinestPatchData = Splines::knotRefinement<dim>(newfinestPatchData, additionalKnots, dir);
348 ++dir;
349 }
350 auto newGrid = PatchGrid(newfinestPatchData);
351 *this = std::move(newGrid);
352 }

359 void degreeElevate(const std::array<int, dim>& elevationFactors, int lvl) {
360 if (lvl > maxLevel() and lvl >= 0) DUNE_THROW(Dune : RangeError, "This level does not exist");
361 auto& patchData = patchGeometries_[lvl].patchData();
362 patchGeometriesUnElevated[lvl] = GeometryKernel::NURBSPatch<dim, dimworld, ctype>(patchData);
363 for (int dir = 0; auto elevatesInDirection : elevationFactors) {
364 if (elevatesInDirection == 0) {
365 ++dir;
366 continue;
367 }
368 patchData = Splines::degreeElevate(patchData, dir, elevatesInDirection);
369 ++dir;
370 }
371 patchGeometries_[lvl] = GeometryKernel::NURBSPatch<dim, dimworld, ctype>(patchData);
372 }

378 void degreeElevateOnAllLevels(const std::array<int, dim>& elevationFactors) {
379 for (int lvl = 0; lvl < maxLevel() + 1; ++lvl)
380 degreeElevate(elevationFactors, lvl);
