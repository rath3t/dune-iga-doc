20 template <int dim>
21 class TrimmedSubGrid {
22 public:
23 using Point = Dune::FieldVector<double, dim>;
24 using Element = MultiLinearGeometry<double, dim, dim>;
25 using Index = std::uint64_t;
26
27 std::vector<Element> elements_{};
28 std::vector<Point> vertices_{};
29 std::vector<Index> indices_{};
30
31 private:
32 std::vector<Boundary> outerBoundaries_;
33 std::optional<std::vector<std::vector<Boundary>>> innerBoundaries_;
34 TransformToSpan<dim> transformer;
35
37 static constexpr int maxOuterBoundaryDivisions{10};
38 static constexpr int maxInnerBoundaryDivisions{10};
39
40 static constexpr double outerTargetTolerance{1e-4};
41 static constexpr double innerTargetTolerance{1e-4};
42
43 public:
45 explicit TrimmedSubGrid(Trim::ElementBoundaries& _boundaries,
46 std::pair<std::array<double, dim>, std::array<double, dim>> scalingAndOffset)
47 : outerBoundaries_(_boundaries.outerBoundaries),
48 innerBoundaries_(_boundaries.innerBoundaries),
49 transformer(scalingAndOffset) {
50 assert(not outerBoundaries_.empty());
51
52 checkAndDivideSmallLoops();
53 constructSubGrid();
54 }
55
56 private:
57 void constructSubGrid() {
58 auto boundaries = splitBoundaries();
59 std::tie(indices_, vertices_) = triangulate<dim>(boundaries, transformer);
60
61 for (auto it = indices_.begin(); it < indices_.end(); it += 3)
62 elements_.emplace_back(
63 Dune::GeometryTypes::triangle,
64 std::vector<FieldVector<double, dim>>{vertices_[*it], vertices_[*(it + 1)], vertices_[*(it + 2)]});
65 }
66
68 void checkAndDivideSmallLoops() {
69 auto partsForSize = [](std::size_t size) -> int { return size == 1 ? 4 : 2; };
70
71 auto divideLoop = [](std::vector<Boundary>& loop, int parts) {
72 std::vector<Boundary> newLoop;
73 for (auto& boundary : loop) {
74 auto geometry = boundary.nurbsGeometry;
75 auto uVec = Utilities::linspace(boundary.domain, parts + 1);
76
77 for (auto i : std::views::iota(0u, static_cast<unsigned int>(parts)))
78 newLoop.emplace_back(geometry, Utilities::Domain<double>{uVec[i], uVec[i + 1]});
79 }
80 loop = newLoop;
81 };
82 if (outerBoundaries_.size() < 3) divideLoop(outerBoundaries_, partsForSize(outerBoundaries_.size()));
83
84 if (innerBoundaries_)
85 for (auto& innerLoop : innerBoundaries_.value())
86 if (innerLoop.size() < 3) divideLoop(innerLoop, partsForSize(innerLoop.size()));
87 }
88
89 public:
91 [[nodiscard]] double calculateArea() const {
92 return std::accumulate(elements_.begin(), elements_.end(), 0.0,
93 [](double rhs, const auto& element) { return rhs + element.volume(); });
94 }
95
96 private:
97 auto splitBoundaries() const { return std::make_pair(splitOuterBoundaries(), splitInnerBoundaryLoops()); }
98 auto splitOuterBoundaries(int maxDivisions = maxOuterBoundaryDivisions,
99 double targetTolerance = outerTargetTolerance) const {
100 return splitBoundariesImpl(outerBoundaries_, maxDivisions, targetTolerance, transformer);
101 }
102
103 auto splitInnerBoundaryLoops(int maxDivisions = maxInnerBoundaryDivisions,
104 double targetTolerance = innerTargetTolerance) const -> decltype(innerBoundaries_) {
105 if (innerBoundaries_) {
106 typename decltype(innerBoundaries_)::value_type newInnerBoundaries;
107 std::ranges::transform(innerBoundaries_.value(), std::back_inserter(newInnerBoundaries), [&](auto& loop) {
108 return splitBoundariesImpl(loop, maxDivisions, targetTolerance, transformer);
109 });
110 return std::make_optional<std::vector<std::vector<Boundary>>>(newInnerBoundaries);
111 } else
112 return std::nullopt;
113 }
114
115 public:
116 auto createRefinedGrid(int refinementSteps) const
117 -> std::tuple<decltype(elements_), decltype(vertices_), decltype(indices_)> {
118 if (refinementSteps == 0) return {elements_, vertices_, indices_};
119
120 decltype(elements_) refElements{};
121 decltype(vertices_) refVertices{};
122 decltype(indices_) refIndices{};
123
124 int splitSteps = std::ceil((double)refinementSteps / 2.0);
125 int refineSteps = refinementSteps - splitSteps;
126
127 auto boundaries = std::make_pair(splitOuterBoundaries(splitSteps, 0.0), splitInnerBoundaryLoops(splitSteps, 0.0));
128 auto [ind, vert] = triangulate<dim>(boundaries, transformer);
129
130 Dune::GridFactory<Dune::UGGrid<dim>> gridFactory;
131 for (auto& vertex : vert)
132 gridFactory.insertVertex(vertex);
133
134 // The element indices are stored as a flat vector, 3 indices always make 1 triangle
135 for (auto it = ind.begin(); it < ind.end(); it += 3)
136 gridFactory.insertElement(Dune::GeometryTypes::triangle, std::vector<unsigned int>(it, std::next(it, 3)));
137
138 // Attention: This may break down for other grid implementations, as the insertion Index doesn't have to be the
139 // "real" entity index
140
141 auto determineOuterBoundaryIndices = [nBoundaries = (unsigned int)boundaries.first.size()](unsigned int bIndex) {
142 return std::vector<unsigned int>{bIndex, (bIndex + 1) % nBoundaries};
143 };
144 auto determineInnerBoundaryIndices
145 = [](unsigned int bIndex, unsigned int nBoundariesInLoop, unsigned int innerIndex) {
146 auto increment = ((innerIndex + 1) % nBoundariesInLoop) - innerIndex;
147 return std::vector<unsigned int>{bIndex, bIndex + increment};
148 };
149
150 unsigned int bIndex = 0;
151 for (auto& boundary : boundaries.first)
152 gridFactory.insertBoundarySegment(determineOuterBoundaryIndices(bIndex++),
153 std::make_shared<GridBoundarySegment<dim>>(boundary, transformer));
154
155 if (boundaries.second)
156 for (auto& innerLoop : boundaries.second.value()) {
157 unsigned int innerIndex = 0;
158 for (auto& boundary : innerLoop)
159 gridFactory.insertBoundarySegment(determineInnerBoundaryIndices(bIndex++, innerLoop.size(), innerIndex++),
160 std::make_shared<GridBoundarySegment<dim>>(boundary, transformer));
161 }
162
163 // Create Grid
164 auto grid = gridFactory.createGrid();
165 grid->globalRefine(refineSteps);
166
167 auto gridView = grid->leafGridView();
168 auto& indexSet = gridView.indexSet();
169
170 // Collect vertices and indices and elements
171 auto eleInd = std::vector<unsigned int>();
172
173 for (auto& v : vertices(gridView))
174 refVertices.push_back(v.geometry().center());
175
176 for (auto& ele : elements(gridView)) {
177 eleInd.clear();
178 for (auto i : std::views::iota(0u, ele.subEntities(dim)))
179 eleInd.push_back(indexSet.template index<dim>(ele.template subEntity<dim>(i)));
180
181 refElements.emplace_back(Dune::GeometryTypes::triangle,
182 std::vector<FieldVector<double, dim>>{refVertices[eleInd[0]], refVertices[eleInd[1]],
183 refVertices[eleInd[2]]});
184 std::ranges::copy(eleInd, std::back_inserter(refIndices));
185 }
186
187 return {refElements, refVertices, refIndices};
188 }
189 };
