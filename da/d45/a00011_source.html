41 template <class Grid>
42 requires(IsSpecializationTwoNonTypesAndType<Dune::IGA::NURBSGrid, Grid>::value) inline static std::shared_ptr<
43 Grid> reader(const pybind11::dict& dict) {
44 std::string file_path;
45 Dune::Python::IGA::Reader reader = IGA::Reader::json;
46 bool trim = true;
47 std::array<int, 2> elevateDegree = {0, 0};
48 std::array<int, 2> preKnotRefine = {0, 0};
49 std::array<int, 2> postKnotRefine = {0, 0};
50 if (dict.contains("reader")) reader = dict["reader"].cast<Dune::Python::IGA::Reader>();
51
52 switch (reader) {
53 case IGA::Reader::json:
54
55 if (dict.contains("file_path"))
56 file_path = dict["file_path"].cast<std::string>();
57 else
58 DUNE_THROW(Dune::IOError, "No field in dict with name file_path. Unable to read grid");
59 if (dict.contains("trim")) trim = dict["trim"].cast<bool>();
60 if (dict.contains("elevate_degree")) elevateDegree = dict["elevate_degree"].cast<std::array<int, 2>>();
61 if (dict.contains("pre_knot_refine")) preKnotRefine = dict["pre_knot_refine"].cast<std::array<int, 2>>();
62 if (dict.contains("post_knot_refine")) postKnotRefine = dict["post_knot_refine"].cast<std::array<int, 2>>();
63
64 static constexpr std::integral auto dim = Grid::dimension;
65 static constexpr std::integral auto dimworld = Grid::dimensionworld;
66 using ScalarType = typename Grid::ctype;
67 return Dune::IGA::IbraReader<dim, dimworld, ScalarType>::read(file_path, trim, elevateDegree, preKnotRefine,
68 postKnotRefine);
69 default:
70 DUNE_THROW(Dune::NotImplemented, "Your requested reader is not implemented");
71 }
72 }
