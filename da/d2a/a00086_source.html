132 [[nodiscard]] double volume() const {
133 auto dom = domain();
134 double volParamterSpace = 1.0;
135 for (int i = 0; i < mydimension; ++i)
136 volParamterSpace *= dom[i].volume();
137 const auto rule = Dune::QuadratureRules<ctype, mydimension>::rule(
138 this->type(), mydimension * (*std::ranges::max_element(patchData_.degree)));
139 ctype vol = 0.0;
140 const std::array<Utilities::Domain<double>, mydimension> input{};
141 for (auto& gp : rule) {
142 vol += integrationElement(Utilities::mapToRange(gp.position(), input, domain())) * gp.weight()
143 * volParamterSpace;
144 }
145 return vol;
146 }

159 [[nodiscard]] GlobalCoordinate corner(int k) const {

325 /* @brief Calculate NURBS and control point net for a given local coordinate. */
326 auto calculateNurbsAndControlPointNet(const LocalCoordinate& u) const {
327 auto subNetStart = Splines::findSpan(patchData_.degree, u, patchData_.knotSpans);
328 auto cpCoordinateNet = Splines::netOfSpan(subNetStart, patchData_.degree,
329 Splines::extractControlCoordinates(patchData_.controlPoints));
330 auto nurbsLocalView = nurbs_.localView();
331 nurbsLocalView.bind(subNetStart);
332 return std::make_tuple(nurbsLocalView, cpCoordinateNet, subNetStart);
333 }

335 /* @brief Calculate zero first and second derivatives of the position for a given local coordinate. */

337 const NurbsLocalView& nurbsLocalView,
338 const ControlPointCoordinateNetType& localControlPointNet) {
339 return std::make_tuple(GeometryKernel::position(u, nurbsLocalView, localControlPointNet),
340 GeometryKernel::jacobianTransposed(u, nurbsLocalView, localControlPointNet),
341 GeometryKernel::hessian(u, nurbsLocalView, localControlPointNet));
342 }
