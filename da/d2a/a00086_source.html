79 [[nodiscard]] int originalSize(int codim) const {
80 assert((codim <= dim) and (codim <= 3) and (codim >= 0));
81
82 if (codim == 0)
83 return elementNet_->size();
84 else if (codim == dim)
85 return vertexNet_->size();
86 else if (dim - codim == 1) // edge case
87 {
88 int edgeSize = 0;
89 for (int j = 0; j < dim; ++j) {
90 int subs = 1;
91 for (int i = 0; i < dim; ++i)
92 subs *= (i == j) ? uniqueSpanSize_[i] : uniqueSpanSize_[i] + 1;
93 edgeSize += subs;
94 }
95 return edgeSize;
96 } else if (dim - codim == 2 && dim == 3) // surface case
97 {
98 int surfSize = 0;
99 for (int j = 0; j < dim; ++j) {
100 int subs = 1;
101 for (int i = 0; i < dim; ++i)
102 subs *= (i == j) ? uniqueSpanSize_[i] + 1 : uniqueSpanSize_[i];
103 surfSize += subs;
104 }
105 return surfSize;
106 }
107 __builtin_unreachable();
108 }

162 [[nodiscard]] int patchBoundaryIndex(const int ocdim1Id) const {
163 int index = ocdim1Id;
164
165 if constexpr (dim == 1) { // early exit curves only have a left or right boundary point
166 if (ocdim1Id == 0) return 0;
167 if (ocdim1Id == uniqueSpanSize_[0]) return 1;
168 } else {
169 const auto codimSizes = this->sizeOfCodim1PerDirection<1>();
170
171 int surfFixedDir = 0;
172 int edgesPerDir = codimSizes[0];
173 for (int j = 0; j < dim; ++j)
174 if (ocdim1Id < edgesPerDir) { // find out to which edge direction this edge index belongs
175 surfFixedDir = j;
176 break;
177 } else
178 edgesPerDir += codimSizes.at(j + 1);
179
180 // move index to correct subset, e.g. edges in y-dir or surfaces with fixed z-dir
181 index -= std::accumulate(codimSizes.begin(), codimSizes.begin() + surfFixedDir, 0);
182
183 if constexpr (dim == 2) {
184 if (surfFixedDir == 1) {
185 const double fac = std::ceil((double)index / (uniqueSpanSize_[0] + 1));
186 index -= fac * (uniqueSpanSize_[0] - 1);
187 }
188 if (surfFixedDir == 0) {
189 const int fac2 = std::floor((double)(index) / ((uniqueSpanSize_[1]) * uniqueSpanSize_[0]));
190 index -= fac2 * (uniqueSpanSize_[0] * (uniqueSpanSize_[1] - 1));
191 }
192
193 std::array<int, dim> boundaries;
194 std::ranges::transform(uniqueSpanSize_, boundaries.begin(), [](auto& i) { return i * 2; });
195
196 // move back to correct index, e.g. the first y - boundary index comes after all x-boundary indices
197 index += std::accumulate(boundaries.begin(), boundaries.begin() + surfFixedDir, 0);
198 } else if constexpr (dim == 3) {
199 if (surfFixedDir == 0) {
200 const int fac = (index + 1) / (uniqueSpanSize_[0] + 1);
201 index -= fac * (uniqueSpanSize_[0] - 1);
202 } else if (surfFixedDir == 1) {
203 const int fac = (index + uniqueSpanSize_[0]) / (uniqueSpanSize_[0] * (uniqueSpanSize_[1] + 1));
204 index -= fac * (uniqueSpanSize_[0] * (uniqueSpanSize_[1] - 1));
205 } else if (surfFixedDir == 2) {
206 const int fac = index / ((uniqueSpanSize_[0] * uniqueSpanSize_[1]) * (uniqueSpanSize_[2]));
207 index -= fac * (uniqueSpanSize_[1] * uniqueSpanSize_[0] * (uniqueSpanSize_[2] - 1));
208 }
209
210 std::array<int, dim> boundaries;
211 std::ranges::fill(boundaries, 2);
212 for (int i = 0; i < dim; ++i)
213 for (int k = 0; k < dim; ++k) {
214 if (i == k) continue;
215 boundaries[i] *= uniqueSpanSize_[k];
216 }
217
218 // move back to correct index, e.g. the first y - boundary index comes after all x-boundary indices
219 index += std::accumulate(boundaries.begin(), boundaries.begin() + surfFixedDir, 0);
220 }
221 assert(index >= 0);
222 return index;
223 }
224 __builtin_unreachable();
225 }

234 template <std::integral auto codim>
235 std::pair<std::array<int, dim>, std::array<Impl::FixedOrFree, dim>> spanAndDirectionFromDirectIndex(
236 const int directIndex) const {
237 std::array<int, dim> currentKnotSpan;
238 std::array<Impl::FixedOrFree, dim> fixedOrFreeDirection;
239 if constexpr (codim == 0) { // elements
240 std::ranges::fill(fixedOrFreeDirection, Impl::FixedOrFree::free);
241 const auto multiIndex = elementNet_->directToMultiIndex(directIndex);
242 for (size_t i = 0; i < dim; i++)
243 currentKnotSpan[i] = Dune::IGA::findSpanCorrected(patchData_->degree[i], uniqueKnotVector_[i][multiIndex[i]],
244 patchData_->knotSpans[i], multiIndex[i]);
245 } else if constexpr (codim == dim) { // vertex
246 std::ranges::fill(fixedOrFreeDirection, Impl::FixedOrFree::fixed);
247 const auto multiIndex = vertexNet_->directToMultiIndex(directIndex);
248
249 for (size_t i = 0; i < dim; ++i) {
250 if (Dune::FloatCmp::eq(uniqueKnotVector_[i][multiIndex[i]], patchData_->knotSpans[i].back())) {
251 // If we are the vertex on the rightmost end of the knotspan, we set the knot span index by hand to the last
252 // entry This is needed since findSpan return to use the end - degree -2 index, which is wrong for the
253 // rightmost end
254 currentKnotSpan[i] = patchData_->knotSpans[i].size() - 1;
255 } else
256 currentKnotSpan[i] = Dune::IGA::findSpanCorrected(
257 patchData_->degree[i], uniqueKnotVector_[i][multiIndex[i]], patchData_->knotSpans[i], multiIndex[i]);
258 }
259
260 } else if constexpr (dim - codim == 1 && dim > 1) // edge case
261 {
262 std::ranges::fill(fixedOrFreeDirection, Impl::FixedOrFree::fixed);
263 std::array<int, dim> edgeSizes = sizeOfCodim1PerDirection<codim>();
264
265 int edgeDir = 0, edgesPerDir = edgeSizes[0];
266 for (int j = 0; j < dim; ++j)
267 if (directIndex < edgesPerDir) { // find out to which edge direction this edge index belongs
268 edgeDir = j;
269 fixedOrFreeDirection[j] = Impl::FixedOrFree::free;
270 break;
271 } else
272 edgesPerDir += edgeSizes.at(j + 1);
273
274 assert(((edgeDir < 2 && dim == 2) || (edgeDir < 3 && dim == 3)) && edgeDir >= 0);
275 const int indexInDir
276 = (directIndex - std::accumulate(edgeSizes.begin(), edgeSizes.begin() + edgeDir, 0)) % edgeSizes[edgeDir];
277 std::array<int, dim> spanIndices{};
278 spanIndices[edgeDir]
279 = std::fmod(std::floor(indexInDir
280 / (std::accumulate(uniqueSpanSize_.begin(), uniqueSpanSize_.begin() + edgeDir, 1,
281 [](auto res, auto r) { return res * (r + 1); }))),
282 uniqueSpanSize_[edgeDir]);
283
284 if (edgeDir == 0) {
285 spanIndices[1] = std::fmod(std::floor(indexInDir / (uniqueSpanSize_[0])), uniqueSpanSize_[1] + 1);
286 if constexpr (dim == 3)
287 spanIndices[2] = std::fmod(std::floor(indexInDir / ((uniqueSpanSize_[0]) * (uniqueSpanSize_[1] + 1))),
288 uniqueSpanSize_[2] + 1);
289 } else {
290 spanIndices[0] = std::fmod(indexInDir, uniqueSpanSize_[0] + 1);
291 if (edgeDir == 1)
292 if constexpr (dim == 3) {
293 spanIndices[2] = std::fmod(std::floor(indexInDir / ((uniqueSpanSize_[1]) * uniqueSpanSize_[0] + 1)),
294 uniqueSpanSize_[2] + 1);
295 }
296 if constexpr (dim == 3)
297 if (edgeDir == 2)
298 spanIndices[1] = std::fmod(std::floor(indexInDir / ((uniqueSpanSize_[0] + 1))), uniqueSpanSize_[1] + 1);
299 }
300 for (size_t i = 0; i < dim; i++)
301 if (fixedOrFreeDirection[i] == Impl::FixedOrFree::free) {
302 currentKnotSpan[i] = Dune::IGA::findSpanCorrected(
303 patchData_->degree[i], uniqueKnotVector_[i][spanIndices[i]], patchData_->knotSpans[i], spanIndices[i]);
304 } else {
305 currentKnotSpan[i] = Dune::IGA::findSpanUncorrected(
306 patchData_->degree[i], uniqueKnotVector_[i][spanIndices[i]], patchData_->knotSpans[i], spanIndices[i]);
307 }
308 } else if constexpr (dim - codim == 2 && dim > 2) // surface case
309 {
310 std::ranges::fill(fixedOrFreeDirection, Impl::FixedOrFree::free);
311 std::array<int, dim> surfaceSizes = sizeOfCodim1PerDirection<codim>();
312
313 int surfFixedDir = 0;
314 int edgesPerDir = surfaceSizes[0];
315 for (int j = 0; j < dim; ++j)
316 if (directIndex < edgesPerDir) { // find out to which edge direction this edge index belongs
317 surfFixedDir = j;
318 fixedOrFreeDirection[j] = Impl::FixedOrFree::fixed;
319 break;
320 } else
321 edgesPerDir += surfaceSizes.at(j + 1);
322 assert((surfFixedDir < 3 && dim == 3) && surfFixedDir >= 0);
323 const int indexInDir
324 = (directIndex - std::accumulate(surfaceSizes.begin(), surfaceSizes.begin() + surfFixedDir, 0))
325 % surfaceSizes[surfFixedDir];
326 std::array<int, dim> spanIndices{};
327 spanIndices[surfFixedDir]
328 = std::fmod(std::floor(indexInDir
329 / (std::accumulate(uniqueSpanSize_.begin(), uniqueSpanSize_.begin() + surfFixedDir,
330 1, std::multiplies{}))),
331 uniqueSpanSize_[surfFixedDir] + 1);
332
333 if (surfFixedDir == 0) {
334 spanIndices[1] = std::fmod(std::floor(indexInDir / (uniqueSpanSize_[0] + 1)), uniqueSpanSize_[1]);
335 spanIndices[2]
336 = std::fmod(std::floor(indexInDir / ((uniqueSpanSize_[0] + 1) * uniqueSpanSize_[1])), uniqueSpanSize_[2]);
337 } else {
338 spanIndices[0] = std::fmod(indexInDir, uniqueSpanSize_[0]);
339 if (surfFixedDir == 1)
340 spanIndices[2] = std::fmod(std::floor(indexInDir / ((uniqueSpanSize_[1] + 1) * uniqueSpanSize_[0])),
341 uniqueSpanSize_[2]);
342 else if (surfFixedDir == 2)
343 spanIndices[1] = std::fmod(std::floor(indexInDir / ((uniqueSpanSize_[0]))), uniqueSpanSize_[1]);
344 }
345 for (size_t i = 0; i < dim; i++)
346 currentKnotSpan[i] = Dune::IGA::findSpanCorrected(patchData_->degree[i], uniqueKnotVector_[i][spanIndices[i]],
347 patchData_->knotSpans[i], spanIndices[i]);
348 for (size_t i = 0; i < dim; i++)
349 if (fixedOrFreeDirection[i] == Impl::FixedOrFree::free) {
350 currentKnotSpan[i] = Dune::IGA::findSpanCorrected(
351 patchData_->degree[i], uniqueKnotVector_[i][spanIndices[i]], patchData_->knotSpans[i], spanIndices[i]);
352 } else {
353 currentKnotSpan[i] = Dune::IGA::findSpanUncorrected(
354 patchData_->degree[i], uniqueKnotVector_[i][spanIndices[i]], patchData_->knotSpans[i], spanIndices[i]);
355 }
356 }
357 return std::make_pair(currentKnotSpan, fixedOrFreeDirection);
358 };

376 currentKnotSpan);
377 return typename GridImpl::template Codim<codim>::Geometry(geo);
378 }

380 std::pair<std::array<double, dim>, std::array<double, dim>> scalingAndOffset(const DirectIndex directIndex) {
381 // This is a mirroring of nurbsGeometryCode
382 std::array<double, dim> scaling;
383 std::array<double, dim> offset;
384
385 auto [thisSpanIndices, fixedOrFreeDirection] = spanAndDirectionFromDirectIndex<0>(directIndex);
386 for (int i = 0; i < dim; ++i) {
387 if (thisSpanIndices[i] + 1 < patchData_->knotSpans[i].size())
388 scaling[i] = patchData_->knotSpans[i][thisSpanIndices[i] + 1] - patchData_->knotSpans[i][thisSpanIndices[i]];
389 offset[i] = patchData_->knotSpans[i][thisSpanIndices[i]];
390 }
391 return std::make_pair(scaling, offset);
392 }

426 template <int codim>
427 std::array<int, dim> sizeOfCodim1PerDirection() const {
428 std::array<int, dim> codim1Sizes;
429 std::ranges::fill(codim1Sizes, 1);
430 if constexpr (dim > 1) {
431 for (int i = 0; i < dim; ++i)
432 for (int k = 0; k < dim; ++k)
433 if constexpr (dim - codim == 2)
434 codim1Sizes[i] *= (i == k) ? uniqueSpanSize_[k] + 1 : uniqueSpanSize_[k]; // surfaces
435 else
436 codim1Sizes[i] *= (i == k) ? uniqueSpanSize_[k] : uniqueSpanSize_[k] + 1; // edges
437 } else
438 codim1Sizes[0] = uniqueSpanSize_[0] + 1; // vertices
439
440 return codim1Sizes;
441 }

443 [[nodiscard]] int getGlobalEdgeIndexFromElementIndex(const int realIndex, const int eI,
444 const bool returnOriginal = false) const {
445 DirectIndex elementDirectIndex = getDirectIndex<0>(realIndex);
446 const auto eleI = elementNet_->directToMultiIndex(elementDirectIndex);
447 std::array<int, dim> edgeSizes;
448 std::ranges::fill(edgeSizes, 1);
449 for (int i = 0; i < dim; ++i)
450 for (int k = 0; k < dim; ++k)
451 edgeSizes[i] *= (i == k) ? uniqueSpanSize_[k] : uniqueSpanSize_[k] + 1;
452 assert((dim == 2 && eI < 4) || (dim == 3 && eI < 12));
453 int edgeDir;
454 if constexpr (dim == 2)
455 edgeDir = (eI == 0 || eI == 1) ? 1 : 0; // 2,3 edgeDir =0
456 else
457 edgeDir
458 = (eI >= 0 && eI < 4) ? 2 : ((eI == 4 || eI == 5 || eI == 8 || eI == 9) ? 1 : 0); // 6,7,10,11 edgeDir=0
459
460 assert((edgeDir < 2 && dim == 2) || (edgeDir < 3 && dim == 3));
461 int dIndex
462 = std::accumulate(edgeSizes.begin(), edgeSizes.begin() + edgeDir, 0); // move index to correct subset
463 // directIndex += eleI[edgeDir];
464 if constexpr (dim == 3) {
465 if (edgeDir == 0)
466 dIndex += eleI[0] + (uniqueSpanSize_[0]) * (eleI[1])
467 + (uniqueSpanSize_[0]) * (uniqueSpanSize_[1] + 1) * (eleI[2]);
468 else if (edgeDir == 1)
469 dIndex += eleI[0] + ((uniqueSpanSize_[0] + 1) * eleI[1])
470 + ((uniqueSpanSize_[1]) * (uniqueSpanSize_[0] + 1)) * eleI[2];
471 else if (edgeDir == 2)
472 dIndex += eleI[0] + ((uniqueSpanSize_[0] + 1) * eleI[1])
473 + ((uniqueSpanSize_[1] + 1) * (uniqueSpanSize_[0] + 1)) * eleI[2];
474 } else if constexpr (dim == 2) {
475 if (edgeDir == 0)
476 dIndex += eleI[0] + (uniqueSpanSize_[0]) * (eleI[1]);
477 else
478 dIndex += eleI[0] + ((uniqueSpanSize_[0] + 1) * eleI[1]);
479 }
480
481 switch (dim) {
482 case 2:
483 dIndex += (eI == 1) ? 1 : (eI == 3) ? uniqueSpanSize_[0] : 0;
484 break;
485 case 3:
486 assert(uniqueSpanSize_.size() == dim);
487 switch (eI) {
488 case 1:
489 dIndex += 1;
490 break;
491 case 2:
492 dIndex += uniqueSpanSize_[0] + 1;
493 break;
494 case 3:
495 dIndex += uniqueSpanSize_[0] + 2;
496 break;
497 case 5:
498 dIndex += 1;
499 break;
500 case 8:
501 dIndex += (uniqueSpanSize_[0] + 1) * uniqueSpanSize_[1];
502 break;
503 case 9:
504 dIndex += (uniqueSpanSize_[0] + 1) * uniqueSpanSize_[1] + 1;
505 break;
506 case 7:
507 dIndex += uniqueSpanSize_[0];
508 break;
509 case 10:
510 dIndex += uniqueSpanSize_[0] * (uniqueSpanSize_[1] + 1);
511 break;
512 case 11:
513 dIndex += uniqueSpanSize_[0] * (uniqueSpanSize_[1] + 2);
514 break;
515 default: // edges 0,4,6
516 break;
517 }
518 default: // dim==1
519 break;
520 }
521 if (returnOriginal) return dIndex;
522 return getRealIndex<1>(dIndex);
523 }

525 [[nodiscard]] int getGlobalSurfaceIndexFromElementIndex(const int realIndex, const int eI) const {
526 DirectIndex elementDirectIndex = getDirectIndex<0>(realIndex);
527 auto eleI = elementNet_->directToMultiIndex(elementDirectIndex);
528
529 std::array<int, dim> edgeSizes;
530 std::ranges::fill(edgeSizes, 1);
531 for (int i = 0; i < dim; ++i)
532 for (int k = 0; k < dim; ++k)
533 edgeSizes[i] *= (i == k) ? uniqueSpanSize_[k] + 1 : uniqueSpanSize_[k];
534 assert((dim == 2 && eI < 4) || (dim == 3 && eI < 12));
535 const int surfaceFixedDir = std::floor(eI / 2);
536 int dIndex
537 = std::accumulate(edgeSizes.begin(), edgeSizes.begin() + surfaceFixedDir, 0); // move index to correct subset
538
539 assert(surfaceFixedDir < 3 && dim == 3);
540
541 if (surfaceFixedDir == 0)
542 dIndex += eleI[0] + (uniqueSpanSize_[0] + 1) * (eleI[1])
543 + (uniqueSpanSize_[0] + 1) * (uniqueSpanSize_[1]) * (eleI[2]);
544 else if (surfaceFixedDir == 1)
545 dIndex += eleI[0] + (uniqueSpanSize_[0] * eleI[1]) + ((uniqueSpanSize_[1] + 1) * uniqueSpanSize_[0]) * eleI[2];
546 else if (surfaceFixedDir == 2)
547 dIndex += eleI[0] + (uniqueSpanSize_[0] * eleI[1]) + (uniqueSpanSize_[1] * uniqueSpanSize_[0]) * eleI[2];
548
549 switch (eI) {
550 case 1:
551 ++dIndex;
552 break;
553 case 3:
554 dIndex += uniqueSpanSize_[0];
555 break;
556 case 5:
557 dIndex += uniqueSpanSize_[1] * uniqueSpanSize_[0];
558 break;
559 default: // surface local indices 0,2,4
560 break;
561 }
562 return dIndex;
563 }

566 template <typename ctype = double>
567 auto parameterSpaceGrid(int scale = 0) const {
568 using TensorSpaceCoordinates = TensorProductCoordinates<ctype, dim>;
569 using ParameterSpaceGrid = YaspGrid<dim, TensorSpaceCoordinates>;
570
571 std::array<std::vector<double>, dim> uniqueKnots;
572 for (int i = 0; i < dim; ++i)
573 std::ranges::unique_copy(patchData_->knotSpans[i], std::back_inserter(uniqueKnots[i]));
574
575 if constexpr (std::is_same<double, ctype>::value) return std::make_unique<ParameterSpaceGrid>(uniqueKnots);
576
577 std::array<std::vector<ctype>, dim> uniqueKnotsTransformed;

666 // Trim Info gets computed for dim == 2 && dimworld == 2
667 void computeElementTrimInfo() requires(dim == 2) {
668 if (!trimData_.has_value()) {
669 prepareForNoTrim();
670 return;
671 }
672 // Prepare Results
673 n_fullElement = 0;
674 n_trimmedElement = 0;
675 trimFlags_.resize(originalSize(0));
676
677 using Trimmer = Trim::NURBSPatchTrimmer<>;
678
679 auto paraGrid = parameterSpaceGrid<Trimmer::intType>(Trimmer::scale);
680 auto parameterSpaceGridView = paraGrid->leafGridView();
681
682 Trimmer trimmer(parameterSpaceGridView, trimData_.value().get());
683
684 const auto& indexSet = parameterSpaceGridView.indexSet();
685 for (auto& element : elements(parameterSpaceGridView)) {
686 DirectIndex directIndex = indexSet.index(element);
687 RealIndex realIndex = n_fullElement + n_trimmedElement;
688
689 auto [trimFlag, boundaries, errorFlag] = trimmer.trimElement(directIndex);
690 trimFlags_[directIndex] = trimFlag;
691 if (errorFlag) trimErrorFlag_ = true;
692
693 if (trimFlag == ElementTrimFlag::trimmed) {
694 elementIndexMap.push_back(directIndex);
695 ++n_trimmedElement;
696 subGridInfoMap.emplace(
697 directIndex, ElementSubGridInfo{.realIndex = realIndex,
698 .subgrid = std::make_unique<SubGridType>(boundaries.value(),
699 scalingAndOffset(directIndex))});
700 } else if (trimFlag == ElementTrimFlag::full) {
701 elementIndexMap.push_back(directIndex);
702 ++n_fullElement;
703 }
704 }
705
706 constructSubEntityMaps<2>();
707 constructSubEntityMaps<1>();
708 }
709 template <unsigned int codim>
710 requires(codim == 2 || codim == 1) && (dim == 2) void constructSubEntityMaps() {
711 int n_entities_original = originalSize(codim);
712 std::set<DirectIndex> indicesOfEntityInTrim;
713
714 for (auto realIndex : std::views::iota(0u, n_fullElement + n_trimmedElement)) {
715 for (int i = 0; i < 4; ++i) {
716 if constexpr (codim == 2)
717 indicesOfEntityInTrim.insert(getGlobalVertexIndexFromElementIndex(realIndex, i, true));
718 else
719 indicesOfEntityInTrim.insert(getGlobalEdgeIndexFromElementIndex(realIndex, i, true));
720 }
721 }
722
723 unsigned int realIndexCounter = 0;
724 auto& map = getEntityMap<codim>();
725
726 for (auto i : std::views::iota(0, n_entities_original)) {
727 auto it = std::ranges::find(indicesOfEntityInTrim, i);
728 if (it != indicesOfEntityInTrim.end()) {
729 map.push_back(i);
730 ++realIndexCounter;
731 }
732 }
