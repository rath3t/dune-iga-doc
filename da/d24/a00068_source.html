98 inline void partial(const typename std::array<unsigned int, dim>& order, const typename Traits::DomainType& in,

102 preBasis_.partial(order, globalIn, out, lFE_.currentKnotSpan_);

105 for (size_t i = 0; i < out.size(); i++)

107 out[i][0] *= scaling_[d][d];
108 }

149 template <int dim>
150 class NurbsLocalCoefficients {
151 // Return i as a d-digit number in the (k+1)-nary system
152 std::array<unsigned int, dim> multiindex(unsigned int i) const {
153 std::array<unsigned int, dim> alpha;
154 for (int j = 0; j < dim; j++) {
155 alpha[j] = i % sizes_[j];
156 i = i / sizes_[j];
157 }
158 return alpha;
159 }
160
162 void setup1d(std::vector<unsigned int>& subEntity) {
163 if (sizes_[0] == 1) {
164 subEntity[0] = 0;
165 return;
166 }
167
168 /* edge and vertex numbering
169 0----0----1
170 */
171 unsigned lastIndex = 0;
172 subEntity[lastIndex++] = 0; // corner 0
173 for (unsigned i = 0; i < sizes_[0] - 2; ++i)
174 subEntity[lastIndex++] = 0; // inner dofs of element (0)
175
176 subEntity[lastIndex++] = 1; // corner 1
177
178 assert(size() == lastIndex);
179 }
180
181 void setup2d(std::vector<unsigned int>& subEntity) {
182 unsigned lastIndex = 0;
183
184 // LocalKey: entity number , entity codim, dof indices within each entity
185 /* edge and vertex numbering
186 2----3----3
187 | |
188 | |
189 0 1
190 | |
191 | |
192 0----2----1
193 */
194
195 // lower edge (2)
196 subEntity[lastIndex++] = 0; // corner 0
197 for (unsigned i = 0; i < sizes_[0] - 2; ++i)
198 subEntity[lastIndex++] = 2; // inner dofs of lower edge (2)
199
200 subEntity[lastIndex++] = 1; // corner 1
201
202 // iterate from bottom to top over inner edge dofs
203 for (unsigned e = 0; e < sizes_[1] - 2; ++e) {
204 subEntity[lastIndex++] = 0; // left edge (0)
205 for (unsigned i = 0; i < sizes_[0] - 2; ++i)
206 subEntity[lastIndex++] = 0; // face dofs
207 subEntity[lastIndex++] = 1; // right edge (1)
208 }
209
210 // upper edge (3)
211 subEntity[lastIndex++] = 2; // corner 2
212 for (unsigned i = 0; i < sizes_[0] - 2; ++i)
213 subEntity[lastIndex++] = 3; // inner dofs of upper edge (3)
214
215 subEntity[lastIndex++] = 3; // corner 3
216
217 assert(size() == lastIndex);
218 }
219
220 public:
221 void init(const std::array<unsigned, dim>& sizes) {
222 sizes_ = sizes;
223
224 li_.resize(size());
225
226 // Set up array of codimension-per-dof-number
227 std::vector<unsigned int> codim(li_.size());
228
229 for (std::size_t i = 0; i < codim.size(); i++) {
230 codim[i] = 0;
231 // Codimension gets increased by 1 for each coordinate direction
232 // where dof is on boundary
233 std::array<unsigned int, dim> mIdx = multiindex(i);
234 for (int j = 0; j < dim; j++)
235 if (mIdx[j] == 0 or mIdx[j] == sizes[j] - 1) codim[i]++;
236 }
237
238 // Set up index vector (the index of the dof in the set of dofs of a given subentity)
239 // Algorithm: the 'index' has the same ordering as the dof number 'i'.
240 // To make it consecutive we interpret 'i' in the (k+1)-adic system, omit all digits
241 // that correspond to axes where the dof is on the element boundary, and transform the
242 // rest to the (k-1)-adic system.
243 std::vector<unsigned int> index(size());
244
245 for (std::size_t i = 0; i < index.size(); i++) {
246 index[i] = 0;
247
248 std::array<unsigned int, dim> mIdx = multiindex(i);
249
250 for (int j = dim - 1; j >= 0; j--)
251 if (mIdx[j] > 0 and mIdx[j] < sizes[j] - 1) index[i] = (sizes[j] - 1) * index[i] + (mIdx[j] - 1);
252 }
253
254 // Set up entity and dof numbers for each (supported) dimension separately
255 std::vector<unsigned int> subEntity(li_.size());
256
257 if (!subEntity.empty()) {
258 if (dim == 1) {
259 setup1d(subEntity);
260
261 } else if (dim == 2 and sizes_[0] > 1 and sizes_[1] > 1) {
262 setup2d(subEntity);
263 }
264 }
265
266 for (size_t i = 0; i < li_.size(); i++)
267 li_[i] = LocalKey(subEntity[i], codim[i], index[i]);
268 }
269
271 [[nodiscard]] std::size_t size() const {
272 return std::accumulate(sizes_.begin(), sizes_.end(), 1, std::multiplies<>());
273 }
274
276 [[nodiscard]] const LocalKey& localKey(std::size_t i) const { return li_[i]; }
277
278 private:
279 // Number of shape functions on this element per coordinate direction
280 std::array<unsigned, dim> sizes_;
281
282 std::vector<LocalKey> li_;
283 };

324 explicit NurbsLocalFiniteElement(const NurbsPreBasis<GV, R>& preBasis)
325 : preBasis_(preBasis), localBasis_(preBasis, *this) {}
326
329 NurbsLocalFiniteElement(const NurbsLocalFiniteElement& other)
330 : preBasis_(other.preBasis_), localBasis_(preBasis_, *this) {
331 if (other.isBound) {
332 this->bind(other.elementIdx_);
333 isBound = true;
334 }
335 }
336
344 // @todo save correct element info
345 void bind(const std::array<int, dim>& elementIdx) {
346 const auto& patchData = preBasis_.patchData_;
347 for (size_t i = 0; i < elementIdx.size(); i++) {
348 currentKnotSpan_[i] = Dune::IGANEW::Splines::findSpan(patchData.degree[i],
349 *(preBasis_.uniqueKnotVector_[i].begin() + elementIdx[i]),
350 patchData.knotSpans[i], elementIdx[i]);
351
352 // Compute the geometric transformation from knotspan-local to global coordinates
353 localBasis_.offset_[i] = preBasis_.patchData_.knotSpans[i][currentKnotSpan_[i]];
354 localBasis_.scaling_[i][i] = preBasis_.patchData_.knotSpans[i][currentKnotSpan_[i] + 1]
355 - preBasis_.patchData_.knotSpans[i][currentKnotSpan_[i]];
356 }
357
358 // Set up the LocalCoefficients object
359 std::array<unsigned int, dim> sizes;
360 for (size_t i = 0; i < dim; i++)
361 sizes[i] = oneDimensionalSize(i);
362 localCoefficients_.init(sizes);
363 isBound = true;
364 elementIdx_ = elementIdx;
365 }
366
368 const NurbsLocalBasis<GV, R>& localBasis() const {
369 assert(isBound && "This element is not bound!");
370 return localBasis_;
371 }
372
374 const NurbsLocalCoefficients<dim>& localCoefficients() const {
375 assert(isBound && "This element is not bound!");
376 return localCoefficients_;

381 assert(isBound && "This element is not bound!");
382 return localInterpolation_;
383 }
384
386 [[nodiscard]] unsigned size() const {
387 std::size_t r = 1;
388 for (int i = 0; i < dim; i++)
389 r *= oneDimensionalSize(i);
390 return r;
391 }
392
395 [[nodiscard]] GeometryType type() const { return GeometryTypes::cube(dim); }
396
398 [[nodiscard]] unsigned int oneDimensionalSize(int i) const { return preBasis_.patchData_.degree[i] + 1; }
399
400 const NurbsPreBasis<GV, R>& preBasis_;
401
402 NurbsLocalCoefficients<dim> localCoefficients_;
403 NurbsLocalInterpolation<dim, NurbsLocalBasis<GV, R>> localInterpolation_;
404
405 // The knot span we are bound to
406 std::array<int, dim> currentKnotSpan_;
407 NurbsLocalBasis<GV, R> localBasis_;
408 std::array<int, dim> elementIdx_{};
409 bool isBound{false};
410 };

547 void createUntrimmedNodeIndices() {
548 std::array<int, dim> localSizes;
549 size_type sizeOfShapeFunctions = 1;
550 for (int i = 0; i < dim; i++) {
551 localSizes[i] = patchData_.degree[i] + 1;
552 sizeOfShapeFunctions *= localSizes[i];
553 }
554 const auto order = patchData_.degree;
555
556 auto spanSize = gridView_.impl().untrimmedElementNumbers();
557 auto extractIndex = std::views::transform([&](const auto& ele) { return gridView_.indexSet().index(ele); });
558 for (auto directIndex : elements(gridView_) | extractIndex) {
559 auto elementIdx = getIJK(directIndex, spanSize);
560
561 std::array<int, dim> currentKnotSpan;
562 for (size_t i = 0; i < elementIdx.size(); i++)
563 currentKnotSpan[i]
564 = IGANEW::Splines::findSpan(patchData_.degree[i], *(uniqueKnotVector_[i].begin() + elementIdx[i]),
565 patchData_.knotSpans[i], elementIdx[i]);
566
567 // Here magic is happening
568 for (size_type i = 0; i < sizeOfShapeFunctions; ++i) {
569 std::array<int, dim> localIJK = getIJK(i, localSizes);
570 std::array<unsigned int, dim> globalIJK;
571 for (int j = 0; j < dim; j++)
572 globalIJK[j] = std::max((int)currentKnotSpan[j] - (int)order[j], 0) + localIJK[j];
573
574 // Make one global flat index from the globalIJK tuple
575 size_type globalIdx = globalIJK[dim - 1];
576
577 for (int j = dim - 2; j >= 0; j--)
578 globalIdx = globalIdx * sizePerDirection(j) + globalIJK[j];
579
580 originalIndices_[directIndex].push_back(globalIdx);
581 }
582 }

648 void evaluateJacobian(const FieldVector<typename GV::ctype, dim>& in, std::vector<FieldMatrix<R, 1, dim>>& out,
649 const std::array<int, dim>& currentKnotSpan) const {
650 const auto dN = IGANEW::Splines::Nurbs<dim, ScalarType>::basisFunctionDerivatives(
651 in, patchData_.knotSpans, patchData_.degree, extractWeights(patchData_.controlPoints), 1, false,
652 currentKnotSpan);
653 out.resize(dN.get(std::array<int, dim>{}).size());
654 for (int j = 0; j < dim; ++j) {
655 std::array<int, dim> multiIndex{};
656 multiIndex[j] = 1;
657 const auto& dNcurrent = dN.get(multiIndex);
658 for (int i = 0; i < dNcurrent.size(); ++i)
659 out[i][0][j] = dNcurrent.directGet(i);
660 }
