51 static Geometry geometryFromTrim(Ibra::Curve2D& _curve) {
52 const auto _cp = _curve.transformControlPoints()[0];
53 std::array<int, dim> dimSize{static_cast<int>(_cp.size())};
54
55 // Construct patch Data
56 std::array<std::vector<double>, dim> knotSpans{_curve.compileKnotVectors()};
57 auto controlNet{ControlPointNetType(dimSize, _cp)};
58
59 return Geometry(PatchData(knotSpans, controlNet, std::array<int, 1>{_curve.degree}));
60 }

63 static Geometry lineGeometryFromPoints(const Point& a, const Point b) {
64 std::vector<Point> _controlPoints{a, b};
65 std::array<ControlPointType, worldDim> _cp{ControlPointType{.p{_controlPoints.front()}, .w = 1},
66 {.p{_controlPoints.back()}, .w = 1}};
67 std::array<int, 1> dimSize{2};
68
69 // Construct patch Data
70 std::array<std::vector<double>, dim> knotSpans{std::vector<double>{0.0, 0.0, 1.0, 1.0}};
71 auto controlNet{ControlPointNetType(dimSize, _cp)};
72
73 return Geometry(PatchData(knotSpans, controlNet, std::array<int, 1>{1}));
74 }

80 [[nodiscard]] EdgeOrientation getEdgeOrientation(const double tolerance = 1e-6) const {
81 if (Dune::FloatCmp::eq(endPoints[0][0], endPoints[1][0], tolerance))

89 template <typename ctype = double>
90 [[nodiscard]] Clipper2Lib::Path<ctype> path(unsigned int samples = 200, const int intScale = 0) const {
91 Clipper2Lib::Path<ctype> path;
92
93 auto scaler = [intScale](const auto x) -> ctype { return x * static_cast<ctype>(std::pow(10, intScale)); };
94
95 if (degree() == 1 && endPoints.size() == 2) {
96 path.emplace_back(scaler(endPoints.front()[0]), scaler(endPoints.front()[1]));
97 path.emplace_back(scaler(endPoints.back()[0]), scaler(endPoints.back()[1]));
98 return path;
99 }
100
101 path.reserve(samples);
102 auto linS = Utilities::linspace(domain, samples);
103 for (auto u : linS) {
104 auto vertex = nurbsGeometry(u);
105 path.emplace_back(scaler(vertex[0]), scaler(vertex[1]));
106 }
107 return path;
108 }
