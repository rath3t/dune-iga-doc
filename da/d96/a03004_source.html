76 [[nodiscard]] double volume(int scaleOrder = 1) const {
77 const auto rule = Dune::QuadratureRules<ctype, patchDimension>::rule(
78 this->type(), scaleOrder * patchDimension * (*std::ranges::max_element(patchData_.degree)));
79 ctype vol = 0.0;
80 for (auto& gp : rule) {
81 localView_.bind(gp.position());
82 vol += localView_.integrationElement(gp.position()) * gp.weight();
83 }
84 return vol;
85 }

113 auto zeroFirstAndSecondDerivativeOfPosition(const LocalCoordinate& u) const {
114 FieldVector<ctype, dimworld> pos;
115 JacobianTransposed J;
116 FieldMatrix<ctype, dim*(dim + 1) / 2, coorddimension> H;
117 std::array<unsigned int, dim> subDirs;
118 for (int subI = 0, i = 0; i < patchDimension; ++i)
119 subDirs[subI++] = i;
120
121 const auto basisFunctionDerivatives = nurbs_.basisFunctionDerivatives(u, 2);
122 auto cpCoordinateNet = netOfSpan(u, patchData_.knotSpans, patchData_.degree,
123 IGA::extractControlCoordinates(patchData_.controlPoints));
124 std::array<unsigned int, patchDimension> ithVecZero{};
125 pos = Dune::IGA::dot(basisFunctionDerivatives.get(ithVecZero), cpCoordinateNet);
126
127 for (int dir = 0; dir < patchDimension; ++dir) {
128 std::array<unsigned int, patchDimension> ithVec{};
129 ithVec[subDirs[dir]] = 1;
130 J[dir] = dot(basisFunctionDerivatives.get(ithVec), cpCoordinateNet);
131 }
132 for (int dir = 0; dir < dim; ++dir) {
133 std::array<unsigned int, patchDimension> ithVec{};
134 ithVec[subDirs[dir]] = 2; // second derivative in dir direction
135 H[dir] = dot(basisFunctionDerivatives.get(ithVec), cpCoordinateNet);
136 }
137 if constexpr (dim > 1) {
138 std::array<int, dim> mixeDerivs;
139 std::ranges::fill_n(mixeDerivs.begin(), 2, 1); // first mixed derivatives
140 int mixedDireCounter = dim;
141 do {
142 H[mixedDireCounter++] = dot(basisFunctionDerivatives.get(mixeDerivs), cpCoordinateNet);
143 } while (std::ranges::next_permutation(mixeDerivs, std::greater()).found);
144 }
145 return std::make_tuple(pos, J, H);
146 }

192 auto calculateNurbsAndControlPointNet(const LocalCoordinate& u) const {
193 auto subNetStart = IGA::findSpanCorrected(patchData_.degree, u, patchData_.knotSpans);
194
195 auto cpCoordinateNet = netOfSpan(subNetStart,patchData_.degree,
196 IGA::extractControlCoordinates(patchData_.controlPoints));
197 auto nurbsLocalView = nurbs_.localView();
198 nurbsLocalView.bind(subNetStart);
199 return std::make_tuple(nurbsLocalView,cpCoordinateNet,subNetStart);
200 }

202 [[nodiscard]] static FieldVector<ctype, dimworld> global(const LocalCoordinate& u,const NurbsLocalView& nurbsLocalView,
203 const ControlPointCoordinateNetType& localControlPointNet) {
204 auto basis = nurbsLocalView.basisFunctions(u);
205
206 return IGA::dot(basis, localControlPointNet);
207 }

209 [[nodiscard]] static JacobianTransposed jacobianTransposedImpl(const LocalCoordinate& u,const NurbsLocalView& nurbsLocalView,
210 const ControlPointCoordinateNetType& localControlPointNet) {
211
212 JacobianTransposed result;
213
214 const auto basisFunctionDerivatives = nurbsLocalView.basisFunctionDerivatives(u, 1);
215
216 for (int dir = 0; dir < patchDimension; ++dir) {
217 std::array<unsigned int, patchDimension> ithVec{};
218 ithVec[dir] = 1;
219 result[dir] = dot(basisFunctionDerivatives.get(ithVec), localControlPointNet);
220 }
221 return result;
222 }
