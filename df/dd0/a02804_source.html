275 void globalRefine(int refCount) {
276 for (int i = 0; i < refCount; ++i) {
277 const auto& finestPatchData = patchGeometries_.back().patchData();
278 auto newfinestPatchData = finestPatchData;
279
280 auto& olduniqueKnotVector = patchGeometries_.back().uniqueKnotVector();
281 std::array<std::vector<ctype>, dim> newUniqueKnotVecs;
282 for (int refDirection = 0; refDirection < dim; ++refDirection) {
283 auto additionalKnots = Splines::generateRefinedKnots(finestPatchData.knotSpans, refDirection, 1);
284 newfinestPatchData = Splines::knotRefinement<dim>(newfinestPatchData, additionalKnots, refDirection);
285 auto& newKnotVec = newUniqueKnotVecs[refDirection];
286
287 // compute new unique Knot vectors from refinement
288 newKnotVec.resize(olduniqueKnotVector[refDirection].size() + additionalKnots.size());
289 std::ranges::merge(olduniqueKnotVector[refDirection], additionalKnots, std::begin(newKnotVec));
290 }
291
292 patchGeometries_.emplace_back(newfinestPatchData, newUniqueKnotVecs);
293 patchGeometriesUnElevated.emplace_back(patchGeometries_.back());
294 }
295
296 trimmer_.parameterSpaceGrid().globalRefine(refCount);
297 setIndices();
298 }

315 void globalRefineInDirection(const std::array<int, dim>& refines) {
316 const auto& finestPatchData = patchGeometries_.back().patchData();
317 auto newfinestPatchData = finestPatchData;
318 for (int dir = 0; auto refinesInDirection : refines) {
319 if (refinesInDirection == 0) {
320 ++dir;
321 continue;
322 }
323 auto additionalKnots = Splines::generateRefinedKnots(finestPatchData.knotSpans, dir, refinesInDirection);
324 newfinestPatchData = Splines::knotRefinement<dim>(newfinestPatchData, additionalKnots, dir);
325 ++dir;
326 }
327 auto newGrid = PatchGrid(newfinestPatchData);
328 *this = std::move(newGrid);
329 }

336 void degreeElevate(const std::array<int, dim>& elevationFactors, int lvl) {
337 if (lvl > maxLevel() and lvl >= 0) DUNE_THROW(Dune : RangeError, "This level does not exist");
338 auto& patchData = patchGeometries_[lvl].patchData();
339 patchGeometriesUnElevated[lvl] = GeometryKernel::NURBSPatch<dim, dimworld, ctype>(patchData);
340 for (int dir = 0; auto elevatesInDirection : elevationFactors) {
341 if (elevatesInDirection == 0) {
342 ++dir;
343 continue;
344 }
345 patchData = Splines::degreeElevate(patchData, dir, elevatesInDirection);
346 ++dir;
347 }
348 patchGeometries_[lvl] = GeometryKernel::NURBSPatch<dim, dimworld, ctype>(patchData);
349 }

355 void degreeElevateOnAllLevels(const std::array<int, dim>& elevationFactors) {

357 degreeElevate(elevationFactors, lvl);
