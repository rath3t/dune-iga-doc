75 [[nodiscard]] double volume(int scaleOrder = 1) const {
76 const auto rule = Dune::QuadratureRules<ctype, patchDimension>::rule(
77 this->type(), scaleOrder * patchDimension * (*std::ranges::max_element(patchData_.degree)));
78 ctype vol = 0.0;
79 for (auto& gp : rule) {
80 localView_.bind(gp.position());
81 vol += localView_.integrationElement(gp.position()) * gp.weight();
82 }
83 return vol;
84 }

156 auto calculateNurbsAndControlPointNet(const LocalCoordinate& u) const {
157 auto subNetStart = IGA::findSpanCorrected(patchData_.degree, u, patchData_.knotSpans);
158
159 auto cpCoordinateNet
160 = netOfSpan(subNetStart, patchData_.degree, IGA::extractControlCoordinates(patchData_.controlPoints));
161 auto nurbsLocalView = nurbs_.localView();
162 nurbsLocalView.bind(subNetStart);
163 return std::make_tuple(nurbsLocalView, cpCoordinateNet, subNetStart);
164 }

166 [[nodiscard]] static FieldVector<ctype, worlddimension> global(
167 const LocalCoordinate& u, const NurbsLocalView& nurbsLocalView,
168 const ControlPointCoordinateNetType& localControlPointNet) {
169 auto basis = nurbsLocalView.basisFunctions(u);
170
171 return IGA::dot(basis, localControlPointNet);
172 }

174 [[nodiscard]] static JacobianTransposed jacobianTransposedImpl(
175 const LocalCoordinate& u, const NurbsLocalView& nurbsLocalView,
176 const ControlPointCoordinateNetType& localControlPointNet) {
177 JacobianTransposed result;
178
179 const auto basisFunctionDerivatives = nurbsLocalView.basisFunctionDerivatives(u, 1);
180
181 for (int dir = 0; dir < patchDimension; ++dir) {
182 std::array<unsigned int, patchDimension> ithVec{};
183 ithVec[dir] = 1;
184 result[dir] = dot(basisFunctionDerivatives.get(ithVec), localControlPointNet);
185 }
186 return result;
187 }
188
189 static auto zeroFirstAndSecondDerivativeOfPositionImpl(const LocalCoordinate& u,
190 const NurbsLocalView& nurbsLocalView,
191 const ControlPointCoordinateNetType& localControlPointNet) {
192 GlobalCoordinate pos;
193
194 Hessian H;
195 const auto basisFunctionDerivatives = nurbsLocalView.basisFunctionDerivatives(u, 2);
196
197 std::array<unsigned int, patchDimension> ithVecZero{};
198 pos = Dune::IGA::dot(basisFunctionDerivatives.get(ithVecZero), localControlPointNet);
199 JacobianTransposed J;
200 for (int dir = 0; dir < patchDimension; ++dir) {
201 std::array<unsigned int, patchDimension> ithVec{};
202 ithVec[dir] = 1;
203 J[dir] = dot(basisFunctionDerivatives.get(ithVec), localControlPointNet);
204 }
205
206 for (int dir = 0; dir < patchDimension; ++dir) {
207 std::array<unsigned int, patchDimension> ithVec{};
208 ithVec[dir] = 2; // second derivative in subDirs[dir] direction
209 H[dir] = dot(basisFunctionDerivatives.get(ithVec), localControlPointNet);
210 }
211 if constexpr (patchDimension > 1) {
212 std::array<int, patchDimension> mixeDerivs;
213 std::ranges::fill(mixeDerivs, 0); // first mixed derivatives
214 if constexpr (patchDimension == 2)
215 for (int dir = 0; dir < patchDimension; ++dir) {
216 mixeDerivs[dir] = 1;
217 }
218 else
219 std::ranges::fill_n(mixeDerivs.begin()+1, 2, 1); // first mixed derivatives
220 int mixedDireCounter = patchDimension;
221 do {
222 H[mixedDireCounter++] = dot(basisFunctionDerivatives.get(mixeDerivs), localControlPointNet);
223
224 if constexpr (patchDimension == 2) break;
225
226 } while (std::ranges::next_permutation(mixeDerivs, std::less()).found);
227 }
228
229 return std::make_tuple(pos, J, H);
230 }
