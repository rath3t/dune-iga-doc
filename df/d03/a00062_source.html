20 Type typeForTypeString(const std::string& typeString) {
21 if (typeString == "NurbsCurveGeometry2D")
22 return Type::NurbsCurveGeometry2D;
23 else if (typeString == "NurbsSurfaceGeometry3D")
24 return Type::NurbsSurfaceGeometry3D;
25 else if (typeString == "BrepLoop")
26 return Type::BrepLoopType;
27 else if (typeString == "BrepTrim")
28 return Type::BrepTrimType;
29 else if (typeString == "Brep")
30 return Type::BrepType;
31 else

46 template <int dim, int worldDim>
47 requires(dim < 3) && (worldDim >= dim) struct IbraNURBSData : public IbraBase {
48 std::array<int, dim> degree{};
49 std::array<int, dim> n_controlPoints{};
50 std::array<std::vector<double>, dim> knots;
51 Dune::DynamicMatrix<double> controlPoints;
52 std::vector<double> weights;
53
54 using ControlPointType = typename NURBSPatchData<dim, worldDim>::ControlPointType;
55
56 [[nodiscard]] std::vector<std::vector<ControlPointType>> transformControlPoints() const {
57 if constexpr (dim == 1) {
58 std::vector<ControlPointType> cps;
59 for (int i = 0; i < n_controlPoints[0]; ++i) {
60 cps.push_back(controlPointAt({i}));
61 }
62 return {cps};
63 } else if constexpr (dim == 2) {
64 std::vector<std::vector<ControlPointType>> vec;
65 std::vector<ControlPointType> cpTemp(n_controlPoints[1]);
66 for (int i = 0; i < n_controlPoints[0]; ++i) {
67 cpTemp.clear();
68 for (int j = 0; j < n_controlPoints[1]; ++j) {
69 cpTemp.push_back(controlPointAt({i, j}));
70 }
71 vec.push_back(cpTemp);
72 }
73 return vec;
74 }
75 __builtin_unreachable();
76 }
77
78 [[nodiscard]] std::array<std::vector<double>, dim> compileKnotVectors() const {
79 std::array<std::vector<double>, dim> knotVec;
80
81 // Insert first and last knot repeatedly
82 for (int i = 0; i < dim; ++i) {
83 knotVec[i].insert(knotVec[i].end(), knots[i][0]);
84 knotVec[i].insert(knotVec[i].end(), knots[i].begin(), knots[i].end());
85 knotVec[i].insert(knotVec[i].end(), knots[i].back());
86 }
87 return knotVec;
88 }
89
90 private:
91 [[nodiscard]] ControlPointType controlPointAt(std::array<int, dim> idx) const {
92 int row = (dim == 1) ? idx[0] : idx[0] * n_controlPoints[1] + idx[1];
93
94 Dune::FieldVector<double, worldDim> p;
95 for (int k = 0; k < worldDim; ++k)
96 p[k] = controlPoints[row][k];
97
98 double w = weights[row];
99
100 return {p, w};
101 }
102 };

161 template <int worldDim>
162 struct Brep : IbraBase {
163 std::vector<Curve2D> curves2D;
164 using SurfaceType = Surface<worldDim>;
165 std::vector<SurfaceType> surfaces;
166 std::vector<BrepLoop> loops;
167 std::vector<BrepTrim> trims;
168
169 Brep(const BrepRepresentation& representation, std::vector<Curve2D>& allCurves,
170 std::vector<SurfaceType>& allSurfaces, std::vector<Ibra::BrepLoopRepresentation>& allBrepLoopRepresentations,
171 std::vector<Ibra::BrepTrimRepresentation>& allBrepTrimRepresentations)
172 : IbraBase{static_cast<const BrepRepresentation&>(representation)} {
173 // Lambda to check if a given key string is in a list of strings
174 auto objectIsInList = [](const std::vector<std::string>& strRepr) {
175 return [&](const auto& keyString) -> bool { return std::ranges::find(strRepr, keyString) != strRepr.end(); };
176 };
177
178 // Find in all Curves
179 std::ranges::copy_if(allCurves, std::back_inserter(curves2D), objectIsInList(representation.curve_geometries_2d),
180 &Curve2D::key);
181 // and all Surfaces
182 std::ranges::copy_if(allSurfaces, std::back_inserter(surfaces),
183 objectIsInList(representation.surface_geometries_3d), &SurfaceType::key);
184
185 // Find all Trims
186 for (auto& trimRepr : allBrepTrimRepresentations) {
187 if (objectIsInList(representation.trims)(trimRepr.key)) trims.emplace_back(trimRepr, curves2D);
188 }
189 // All Loops
190 for (auto& loopRepr : allBrepLoopRepresentations) {
191 if (objectIsInList(representation.loops)(loopRepr.key)) loops.emplace_back(loopRepr, trims);
192 }
193
194 // Asserts if all geometries were found
195 assert(curves2D.size() == representation.curve_geometries_2d.size());
196 assert(surfaces.size() == representation.surface_geometries_3d.size());
197 assert(loops.size() == representation.loops.size());
198 assert(trims.size() == representation.trims.size());
199 }
200 };

216 template <int worldDim>
217 void from_json(const json& j, Curve<worldDim>& curve) {
218 getGenerics(j, curve);
219
220 // Degree
221 j.at("degree").get_to(curve.degree[0]);
222
223 // Knots (u)
224 std::ranges::copy(j.at("knots"), std::back_inserter(curve.knots[0]));
225
226 // Poles aka Control Points
227 j.at("nb_poles").get_to(curve.n_controlPoints[0]);
228
229 curve.controlPoints.resize(curve.n_controlPoints[0], worldDim);
230
231 for (int i = 0; i < curve.n_controlPoints[0]; ++i)
232 for (int k = 0; k < worldDim; ++k)
233 curve.controlPoints[i][k] = j.at("poles")[i][k];
234
235 if (j.contains("weights"))
236 std::ranges::copy(j.at("weights"), std::back_inserter(curve.weights));
237 else {
238 curve.weights.resize(curve.n_controlPoints[0]);
239 std::ranges::fill(curve.weights, 1.0);
240 }
241 }

243 template <int worldDim>
244 void from_json(const json& j, Surface<worldDim>& surface) {
245 getGenerics(j, surface);
246
247 // Degree
248 j.at("degree_u").get_to(surface.degree[0]);
249 j.at("degree_v").get_to(surface.degree[1]);
250
251 // Knots (u,v)
252 std::ranges::copy(j.at("knots_u"), std::back_inserter(surface.knots[0]));
253 std::ranges::copy(j.at("knots_v"), std::back_inserter(surface.knots[1]));
254
255 // Poles aka Control Points
256 j.at("nb_poles_u").get_to(surface.n_controlPoints[0]);
257 j.at("nb_poles_v").get_to(surface.n_controlPoints[1]);
258
259 int total_controlPoints = surface.n_controlPoints[0] * surface.n_controlPoints[1];
260 surface.controlPoints.resize(total_controlPoints, worldDim);
261
262 for (int i = 0; i < total_controlPoints; ++i)
263 for (int k = 0; k < worldDim; ++k)
264 surface.controlPoints[i][k] = j.at("poles")[i][k];
265
266 if (j.contains("weights"))
267 std::ranges::copy(j.at("weights"), std::back_inserter(surface.weights));
268 else {
269 surface.weights.resize(total_controlPoints);
270 std::ranges::fill(surface.weights, 1.0);
271 }
272 }
