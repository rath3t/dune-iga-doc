145 template <int dim>
146 class NurbsLocalCoefficients {
147 // Return i as a d-digit number in the (k+1)-nary system
148 std::array<unsigned int, dim> multiindex(unsigned int i) const {
149 std::array<unsigned int, dim> alpha;
150 for (int j = 0; j < dim; j++) {
151 alpha[j] = i % sizes_[j];
152 i = i / sizes_[j];
153 }
154 return alpha;
155 }
156
158 void setup1d(std::vector<unsigned int>& subEntity) {
159 if (sizes_[0] == 1) {
160 subEntity[0] = 0;
161 return;
162 }
163
164 /* edge and vertex numbering
165 0----0----1
166 */
167 unsigned lastIndex = 0;
168 subEntity[lastIndex++] = 0; // corner 0
169 for (unsigned i = 0; i < sizes_[0] - 2; ++i)
170 subEntity[lastIndex++] = 0; // inner dofs of element (0)
171
172 subEntity[lastIndex++] = 1; // corner 1
173
174 assert(size() == lastIndex);
175 }
176
177 void setup2d(std::vector<unsigned int>& subEntity) {
178 unsigned lastIndex = 0;
179
180 // LocalKey: entity number , entity codim, dof indices within each entity
181 /* edge and vertex numbering
182 2----3----3
183 | |
184 | |
185 0 1
186 | |
187 | |
188 0----2----1
189 */
190
191 // lower edge (2)
192 subEntity[lastIndex++] = 0; // corner 0
193 for (unsigned i = 0; i < sizes_[0] - 2; ++i)
194 subEntity[lastIndex++] = 2; // inner dofs of lower edge (2)
195
196 subEntity[lastIndex++] = 1; // corner 1
197
198 // iterate from bottom to top over inner edge dofs
199 for (unsigned e = 0; e < sizes_[1] - 2; ++e) {
200 subEntity[lastIndex++] = 0; // left edge (0)
201 for (unsigned i = 0; i < sizes_[0] - 2; ++i)
202 subEntity[lastIndex++] = 0; // face dofs
203 subEntity[lastIndex++] = 1; // right edge (1)
204 }
205
206 // upper edge (3)
207 subEntity[lastIndex++] = 2; // corner 2
208 for (unsigned i = 0; i < sizes_[0] - 2; ++i)
209 subEntity[lastIndex++] = 3; // inner dofs of upper edge (3)
210
211 subEntity[lastIndex++] = 3; // corner 3
212
213 assert(size() == lastIndex);
214 }
215
216 public:
217 void init(const std::array<unsigned, dim>& sizes) {
218 sizes_ = sizes;
219
220 li_.resize(size());
221
222 // Set up array of codimension-per-dof-number
223 std::vector<unsigned int> codim(li_.size());
224
225 for (std::size_t i = 0; i < codim.size(); i++) {
226 codim[i] = 0;
227 // Codimension gets increased by 1 for each coordinate direction
228 // where dof is on boundary
229 std::array<unsigned int, dim> mIdx = multiindex(i);
230 for (int j = 0; j < dim; j++)
231 if (mIdx[j] == 0 or mIdx[j] == sizes[j] - 1) codim[i]++;
232 }
233
234 // Set up index vector (the index of the dof in the set of dofs of a given subentity)
235 // Algorithm: the 'index' has the same ordering as the dof number 'i'.
236 // To make it consecutive we interpret 'i' in the (k+1)-adic system, omit all digits
237 // that correspond to axes where the dof is on the element boundary, and transform the
238 // rest to the (k-1)-adic system.
239 std::vector<unsigned int> index(size());
240
241 for (std::size_t i = 0; i < index.size(); i++) {
242 index[i] = 0;
243
244 std::array<unsigned int, dim> mIdx = multiindex(i);
245
246 for (int j = dim - 1; j >= 0; j--)
247 if (mIdx[j] > 0 and mIdx[j] < sizes[j] - 1) index[i] = (sizes[j] - 1) * index[i] + (mIdx[j] - 1);
248 }
249
250 // Set up entity and dof numbers for each (supported) dimension separately
251 std::vector<unsigned int> subEntity(li_.size());
252
253 if (!subEntity.empty()) {
254 if (dim == 1) {
255 setup1d(subEntity);
256
257 } else if (dim == 2 and sizes_[0] > 1 and sizes_[1] > 1) {
258 setup2d(subEntity);
259 }
260 }
261
262 for (size_t i = 0; i < li_.size(); i++)
263 li_[i] = LocalKey(subEntity[i], codim[i], index[i]);
264 }
265
267 [[nodiscard]] std::size_t size() const {
268 return std::accumulate(sizes_.begin(), sizes_.end(), 1, std::multiplies<>());
269 }
270
272 [[nodiscard]] const LocalKey& localKey(std::size_t i) const { return li_[i]; }
273
274 private:
275 // Number of shape functions on this element per coordinate direction
276 std::array<unsigned, dim> sizes_;
277
278 std::vector<LocalKey> li_;
279 };

341 void bind(const std::array<unsigned, dim>& elementIdx) {
342 const auto& patchData = preBasis_.patchData_;
343 for (size_t i = 0; i < elementIdx.size(); i++) {
344 currentKnotSpan_[i] = Dune::IGA::findSpanCorrected(patchData.degree[i],

356 for (size_t i = 0; i < dim; i++)

360 elementIdx_ = elementIdx;

534 void createOriginalNodeIndices() {
535 std::array<unsigned int, dim> localSizes;
536 size_type sizeOfShapeFunctions = 1;
537 for (int i = 0; i < dim; i++) {
538 localSizes[i] = patchData_.degree[i] + 1;
539 sizeOfShapeFunctions *= localSizes[i];
540 }
541 const auto order = patchData_.degree;
542
543 for (auto directIndex : std::views::iota(0, gridView_.impl().getPatch().originalSize(0))) {
544 auto spanSize = gridView_.impl().getPatch().template originalGridSize<0, unsigned int>();
545 auto elementIdx = getIJK(directIndex, spanSize);
546
547 std::array<int, dim> currentKnotSpan;
548 for (size_t i = 0; i < elementIdx.size(); i++)
549 currentKnotSpan[i]
550 = Dune::IGA::findSpanCorrected(patchData_.degree[i], *(uniqueKnotVector_[i].begin() + elementIdx[i]),
551 patchData_.knotSpans[i], elementIdx[i]);
552
553 // Here magic is happening
554 for (size_type i = 0; i < sizeOfShapeFunctions; ++i) {
555 std::array<unsigned int, dim> localIJK = getIJK(i, localSizes);
556 std::array<unsigned int, dim> globalIJK;
557 for (int j = 0; j < dim; j++)
558 globalIJK[j] = std::max((int)currentKnotSpan[j] - (int)order[j], 0) + localIJK[j];
559
560 // Make one global flat index from the globalIJK tuple
561 size_type globalIdx = globalIJK[dim - 1];
562
563 for (int j = dim - 2; j >= 0; j--)
564 globalIdx = globalIdx * sizePerDirection(j) + globalIJK[j];
565
566 originalIndices_[directIndex].push_back(globalIdx);
567 }
568 }

581 void prepareForTrim() {
582 unsigned int n_ind_original = cachedSize_;
583
584 std::set<size_type> indicesInTrim;
585 for (auto directIndex : std::views::iota(0, gridView_.impl().getPatch().originalSize(0))) {
586 IGA::ElementTrimFlag trimFlag = gridView_.impl().getPatch().getTrimFlagForDirectIndex(directIndex);
587 if (trimFlag != IGA::ElementTrimFlag::empty)
588 std::ranges::copy(originalIndices_.at(directIndex), std::inserter(indicesInTrim, indicesInTrim.begin()));
589 }
590
591 unsigned int realIndexCounter = 0;
592 for (unsigned int i = 0; i < n_ind_original; ++i) {
593 if (std::ranges::find(indicesInTrim, i) != indicesInTrim.end()) indexMap_.emplace(i, realIndexCounter++);
594 }
595 cachedSize_ = realIndexCounter;
596 }

631 void evaluateJacobian(const FieldVector<typename GV::ctype, dim>& in, std::vector<FieldMatrix<R, 1, dim>>& out,
632 const std::array<int, dim>& currentKnotSpan) const {
633 const auto dN = IGA::Nurbs<dim, ScalarType>::basisFunctionDerivatives(in, patchData_.knotSpans, patchData_.degree,
634 extractWeights(patchData_.controlPoints), 1,

640 const auto& dNcurrent = dN.get(multiIndex);
641 for (int i = 0; i < dNcurrent.size(); ++i)
642 out[i][0][j] = dNcurrent.directGet(i);
643 }
