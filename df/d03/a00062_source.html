16 auto clipElementRectangle(const auto& element, const auto& patchTrimData)
17 -> std::tuple<ElementTrimFlag, ClippingResult> {
18 using namespace Clipper2Lib;
19
20 auto eleGeo = element.geometry();
21 static constexpr int numberOfCorners = 4;
22
23 PathD eleRect;
24 for (const auto vertexIndex : Dune::range(numberOfCorners)) {
25 auto corner = eleGeo.corner(vertexIndexMapping[vertexIndex]);
26 eleRect.emplace_back(corner[0], corner[1], vertexIndex);
27 }
28
29 const auto& trimmingCurves = patchTrimData.clipperLoops();
30
31 // Element is trimmed
32 ClipperD clipper(8);
33 ClippingResult result(eleRect);
34
35 auto checkParallel = [&](const auto& curve, const int edgeIndex) -> bool {
36 return curve.affine()
37 and std::get<0>(findIntersectionLinearCurveAndLine(curve, eleGeo.corner(vertexIndexMapping[edgeIndex]),
38 edgeDirections[edgeIndex],
39 {curve.domainMidPoint()[0], 0.5}))
40 == IntersectionCurveAndLine::parallel;
41 };
42
43 clipper.SetZCallback([&](const PointD& rectangleFirstPoint, const PointD& rectangleSecondPoint,
44 const PointD& firstTrimmingCurvePoint, const PointD& secondTrimmingCurvePoint,
45 const PointD& intersectionPoint) {
46 assert(rectangleFirstPoint.z < 4 && "Intersection should only occur with the rectangle edges");
47 assert(rectangleSecondPoint.z < 4 && "Intersection should only occur with the rectangle edges");
48
49 const auto [isHost, idx] = isCornerVertex(intersectionPoint, eleRect);
50 if (isHost) {
51 result.addOriginalVertex(idx);
52 return;
53 }
54
55 const auto edgeIdx = giveEdgeIdx(rectangleFirstPoint.z, rectangleSecondPoint.z);
56 assert(patchTrimData.getIndices(firstTrimmingCurvePoint.z).first
57 == patchTrimData.getIndices(secondTrimmingCurvePoint.z).first
58 && "The points of the trimming curves should be on the same loop");
59 assert(firstTrimmingCurvePoint.z==secondTrimmingCurvePoint.z or secondTrimmingCurvePoint.z== intersectionPoint.z or firstTrimmingCurvePoint.z== intersectionPoint.z &&
60 "The indices of the trimming curves should be the same or the intersection point should be on one of the two second curves");
61 const auto vertexZValue = secondTrimmingCurvePoint.z;
62
63 // Now check that we don't have a parallel intersection
64 if (checkParallel(patchTrimData.getCurve(vertexZValue), edgeIdx)) {
65 return;
66 }
67
68 result.addNewVertex(edgeIdx, intersectionPoint, vertexZValue);
69 });
70
71 PathsD resultClosedPaths{};
72 PathsD resultOpenPaths{};
73
74 clipper.AddClip(trimmingCurves);
75 clipper.AddSubject({eleRect});
76
77 clipper.Execute(ClipType::Intersection, FillRule::NonZero, resultClosedPaths, resultOpenPaths);
78 assert(resultOpenPaths.empty() && "There should be no open paths in the clipping result");
79
80 auto isFullElement
81 = [&](const auto& clippedEdges) {
82 return clippedEdges.size()==4 and (clippedEdges[0].z < 4 and clippedEdges[1].z <4 and clippedEdges[2].z <4 and clippedEdges[3].z <4);
83 };
84
85 if (resultClosedPaths.empty()) return std::make_tuple(ElementTrimFlag::empty, ClippingResult{eleRect});
86 if (isFullElement(resultClosedPaths.front())) {
87 assert(resultClosedPaths.size() == 1 && "If the element is full, there should be only one path");
88 return std::make_tuple(ElementTrimFlag::full, ClippingResult{eleRect});
89 }
90
91 for (const auto& pt : resultClosedPaths.front()) {
92 if (auto [isHost, idx] = isCornerVertex(pt, eleRect); isHost) result.addOriginalVertex(idx);
93 }
94
95 // Now we have to check for points inside the elements
96 for (auto i : Dune::range(patchTrimData.loops().size())) {
97 for (const auto& pt : patchTrimData.getPointsInPatch(i)) {
98 if (const PointD cp{pt.pt[0], pt.pt[1]}; PointInPolygon(cp, eleRect) == PointInPolygonResult::IsInside) {
99 result.addInsideVertex(cp, pt.curveIdxI, pt.curveIdxJ, i);
100 }
101 }
102 }
103
104 // Add startpoints of trimming curves to the vertices if they are on one of the element edges
105 for (auto cI = 0; cI < patchTrimData.loops().front().curves().size(); ++cI) {
106 const auto& curve = patchTrimData.loops().front().curves()[cI];
107
108 for (int i = 0; i < 2; ++i) {
109 auto pt = curve.corner(i);
110 PointD ptClipper{pt[0], pt[1]};
111
112 auto [isHost, idx] = isCornerVertex(ptClipper, eleRect);
113 if (isHost) {
114 result.addOriginalVertex(idx);
115 continue;
116 }
117
118 for (auto e = 0; e < edgeLookUp.size(); ++e) {
119 if (const auto& edgeIdx = edgeLookUp[e];
120 isPointOnLine(pt, eleGeo.corner(edgeIdx.front()), eleGeo.corner(edgeIdx.back()))
121 && !checkParallel(curve, e)) {
122 result.addNewVertex(e, ptClipper, patchTrimData.getZValue(cI,0));
123 break;
124 }
125 }
126 }
127 }
128
129 result.finish(patchTrimData);
130
131 // While developing
132 result.report();
133
134 return std::make_tuple(ElementTrimFlag::trimmed, result);
135 }
