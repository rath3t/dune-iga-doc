10template <class GridView, int ncomp = 1>
11class BoundaryPatchEnclosingVerticesPropertyTrimmed {
12 typedef typename GridView::IndexSet IndexSet;
13 static const int dim = GridView::dimension;
14
15 public:
16 typedef typename GridView::Intersection Intersection;

20 BoundaryPatchEnclosingVerticesPropertyTrimmed(const GridView& gridView, const Dune::BitSetVector<ncomp>& vertices)
21 : indexSet_(gridView.indexSet()), vertices_(vertices) {}
22
25 bool operator()(const Intersection& i) const {
26 const auto inside = i.inside();
27 int localFaceIndex = i.indexInInside();
28
29 if (inside.impl().isTrimmed()) return false;
30
31 auto refElement = Dune::ReferenceElements<double, dim>::general(inside.type());
32
33 // Get global node ids
34 int n = refElement.size(localFaceIndex, 1, dim);
35
36 // Using ReferenceElement::subEntity is OK here, because we loop
37 // over all subEntities (i.e. sub-vertices) and just return false
38 // if _any_ of them is not marked.
39 for (int ii = 0; ii < n; ii++) {
40 int localVertexIndex = refElement.subEntity(localFaceIndex, 1, ii, dim);
41 if (not(vertices_[indexSet_.subIndex(inside, localVertexIndex, dim)].any())) return false;
42 }
43 return true;
44 }
45
46 private:
47 const IndexSet& indexSet_;
48 const Dune::BitSetVector<ncomp>& vertices_;
49};

54 void forEachUntrimmedBoundaryDOF(const Basis& basis, F&& f) {
55 auto localView = basis.localView();
56 auto seDOFs = subEntityDOFs(basis);
57 const auto& gridView = basis.gridView();
58 for (auto&& element : elements(gridView))
59 if (element.hasBoundaryIntersections() and not element.impl().isTrimmed()) {
60 localView.bind(element);
61 for (const auto& intersection : intersections(gridView, element))
62 if (intersection.boundary())
63 for (auto localIndex : seDOFs.bind(localView, intersection))
64 f(localIndex, localView, intersection);
65 }
66 }
