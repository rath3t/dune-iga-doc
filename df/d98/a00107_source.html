47 [[nodiscard]] double volume(int scaleOrder = 1) const {
48 const auto rule = Dune::QuadratureRules<ctype, mydimension>::rule(
49 this->type(), scaleOrder * mydimension * (*std::ranges::max_element(patchData_->degree)));
50 ctype vol = 0.0;
51 for (auto& gp : rule)

82 auto zeroFirstAndSecondDerivativeOfPosition(const LocalCoordinate& u) const {
83 FieldVector<ctype, dimworld> pos;
84 JacobianTransposed J;
85 FieldMatrix<ctype, dim*(dim + 1) / 2, coorddimension> H;
86 std::array<unsigned int, dim> subDirs;
87 for (int subI = 0, i = 0; i < mydimension; ++i)
88 subDirs[subI++] = i;
89
90 const auto basisFunctionDerivatives = nurbs_.basisFunctionDerivatives(u, 2);
91 auto cpCoordinateNet = netOfSpan(u, patchData_->knotSpans, patchData_->degree,
92 extractControlCoordinates(patchData_->controlPoints));
93 std::array<unsigned int, mydimension> ithVecZero{};
94 pos = Dune::IGA::dot(basisFunctionDerivatives.get(ithVecZero), cpCoordinateNet);
95
96 for (int dir = 0; dir < mydimension; ++dir) {
97 std::array<unsigned int, mydimension> ithVec{};
98 ithVec[subDirs[dir]] = 1;
99 J[dir] = dot(basisFunctionDerivatives.get(ithVec), cpCoordinateNet);
100 }
101 for (int dir = 0; dir < dim; ++dir) {
102 std::array<unsigned int, mydimension> ithVec{};
103 ithVec[subDirs[dir]] = 2; // second derivative in dir direction
104 H[dir] = dot(basisFunctionDerivatives.get(ithVec), cpCoordinateNet);
105 }
106 if constexpr (dim > 1) {
107 std::array<int, dim> mixeDerivs;
108 std::ranges::fill_n(mixeDerivs.begin(), 2, 1); // first mixed derivatives
109 int mixedDireCounter = dim;
110 do {
111 H[mixedDireCounter++] = dot(basisFunctionDerivatives.get(mixeDerivs), cpCoordinateNet);
112 } while (std::ranges::next_permutation(mixeDerivs, std::greater()).found);
113 }
114 return std::make_tuple(pos, J, H);
115 }

124 [[nodiscard]] LocalCoordinate local(const GlobalCoordinate& global) const {
125 if constexpr (dim == 0)
126 return {};
127 else if constexpr (dim != dimworld) {
128 auto [u, Ru, fu, gap] = Dune::IGA::closestPointProjectionByTrustRegion(*this, global);
129 return u;
130 } else {
131 const ctype tolerance = ctype(16) * std::numeric_limits<ctype>::epsilon();
132 LocalCoordinate x = domainMidPoint();
133
134 LocalCoordinate dx{};
135 do { // from multilinearGeometry
136 const GlobalCoordinate dglobal = (*this).global(x) - global;
137 MatrixHelper::template xTRightInvA<mydimension, coorddimension>(jacobianTransposed(x), dglobal, dx);
138 const bool invertible
139 = MatrixHelper::template xTRightInvA<mydimension, coorddimension>(jacobianTransposed(x), dglobal, dx);
140
141 if (!invertible) return LocalCoordinate(std::numeric_limits<ctype>::max());
142 x -= dx;
143 // if local is outside the maximum knot vector span bound, thus we clamp it to it and hope for convergence
144 for (int i = 0; i < mydimension; ++i) {
145 if (Dune::FloatCmp::gt(x[i], patchData_->knotSpans[i].back())) x[i] = patchData_->knotSpans[i].back();
146 if (Dune::FloatCmp::lt(x[i], patchData_->knotSpans[i].front())) x[i] = patchData_->knotSpans[i].front();
147 }
148
149 } while (dx.two_norm2() > tolerance);
150 return x;
151 }
152 }

158 [[nodiscard]] JacobianTransposed jacobianTransposed(const LocalCoordinate& u) const {
159 auto cpCoordinateNet = netOfSpan(u, patchData_->knotSpans, patchData_->degree,
160 extractControlCoordinates(patchData_->controlPoints));
161 return jacobianTransposedImpl(u, cpCoordinateNet);

164 [[nodiscard]] JacobianTransposed jacobianTransposedImpl(const LocalCoordinate& u,
165 const auto& cpCoordinateNet) const {
166 JacobianTransposed result;
167
168 const auto basisFunctionDerivatives = nurbs_.basisFunctionDerivatives(u, 1);
169
170 for (int dir = 0; dir < mydimension; ++dir) {
171 std::array<unsigned int, mydimension> ithVec{};
172 ithVec[dir] = 1;
173 result[dir] = dot(basisFunctionDerivatives.get(ithVec), cpCoordinateNet);
174 }
175 return result;
176 }

178 [[nodiscard]] ctype integrationElement(const LocalCoordinate& local) const {
179 auto j = jacobianTransposed(local);
180 return MatrixHelper::template sqrtDetAAT<mydimension, coorddimension>(j);
181 }
