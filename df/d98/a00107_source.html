78 auto zeroFirstAndSecondDerivativeOfPosition(const LocalCoordinate& u) const {
79 FieldVector<ctype, dimworld> pos;
80 JacobianTransposed J;
81 FieldMatrix<ctype, dim*(dim + 1) / 2, coorddimension> H;
82 std::array<unsigned int, dim> subDirs;
83 for (int subI = 0, i = 0; i < patchDim; ++i)
84 subDirs[subI++] = i;
85
86 const auto basisFunctionDerivatives = nurbs_.basisFunctionDerivatives(u, 2);
87 auto cpCoordinateNet = netOfSpan(u, patchData_->knotSpans, patchData_->degree,
88 extractControlCoordinates(patchData_->controlPoints));
89 std::array<unsigned int, patchDim> ithVecZero{};
90 pos = Dune::IGA::dot(basisFunctionDerivatives.get(ithVecZero), cpCoordinateNet);
91
92 for (int dir = 0; dir < patchDim; ++dir) {
93 std::array<unsigned int, patchDim> ithVec{};
94 ithVec[subDirs[dir]] = 1;
95 J[dir] = dot(basisFunctionDerivatives.get(ithVec), cpCoordinateNet);
96 }
97 for (int dir = 0; dir < dim; ++dir) {
98 std::array<unsigned int, patchDim> ithVec{};
99 ithVec[subDirs[dir]] = 2; // second derivative in dir direction
100 H[dir] = dot(basisFunctionDerivatives.get(ithVec), cpCoordinateNet);
101 }
102 if constexpr (dim > 1) {
103 std::array<int, dim> mixeDerivs;
104 std::ranges::fill_n(mixeDerivs.begin(), 2, 1); // first mixed derivatives
105 int mixedDireCounter = dim;
106 do {
107 H[mixedDireCounter++] = dot(basisFunctionDerivatives.get(mixeDerivs), cpCoordinateNet);
108 } while (std::ranges::next_permutation(mixeDerivs, std::greater()).found);
109 }
110 return std::make_tuple(pos, J, H);
111 }

120 [[nodiscard]] LocalCoordinate local(const GlobalCoordinate& global) const {
121 if constexpr (dim == 0)
122 return {};
123 else if constexpr (dim != dimworld) {
124 auto [u, Ru, fu, gap] = Dune::IGA::closestPointProjectionByTrustRegion(*this, global);
125 return u;
126 } else {
127 const ctype tolerance = ctype(16) * std::numeric_limits<ctype>::epsilon();
128 LocalCoordinate x = domainMidPoint();
129
130 LocalCoordinate dx{};
131 do { // from multilinearGeometry
132 const GlobalCoordinate dglobal = (*this).global(x) - global;
133 MatrixHelper::template xTRightInvA<patchDim, coorddimension>(jacobianTransposed(x), dglobal, dx);
134 const bool invertible
135 = MatrixHelper::template xTRightInvA<patchDim, coorddimension>(jacobianTransposed(x), dglobal, dx);
136
137 if (!invertible) return LocalCoordinate(std::numeric_limits<ctype>::max());
138 x -= dx;
139 // if local is outside the maximum knot vector span bound, thus we clamp it to it and hope for convergence
140 for (int i = 0; i < patchDim; ++i) {
141 if (Dune::FloatCmp::gt(x[i], patchData_->knotSpans[i].back())) x[i] = patchData_->knotSpans[i].back();
142 if (Dune::FloatCmp::lt(x[i], patchData_->knotSpans[i].front())) x[i] = patchData_->knotSpans[i].front();
143 }
144
145 } while (dx.two_norm2() > tolerance);
146 return x;
147 }
148 }

154 [[nodiscard]] JacobianTransposed jacobianTransposed(const LocalCoordinate& u) const {
155 auto cpCoordinateNet = netOfSpan(u, patchData_->knotSpans, patchData_->degree,
156 extractControlCoordinates(patchData_->controlPoints));

160 [[nodiscard]] JacobianTransposed jacobianTransposedImpl(const LocalCoordinate& u,
161 const auto& cpCoordinateNet) const {
162 JacobianTransposed result;
163
164 const auto basisFunctionDerivatives = nurbs_.basisFunctionDerivatives(u, 1);
165
166 for (int dir = 0; dir < patchDim; ++dir) {
167 std::array<unsigned int, patchDim> ithVec{};
168 ithVec[dir] = 1;
169 result[dir] = dot(basisFunctionDerivatives.get(ithVec), cpCoordinateNet);
170 }
171 return result;
172 }

174 [[nodiscard]] ctype integrationElement(const LocalCoordinate& local) const {
175 auto j = jacobianTransposed(local);
176 return MatrixHelper::template sqrtDetAAT<patchDim, coorddimension>(j);
177 }
