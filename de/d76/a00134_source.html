67 template <typename HostIdType>
68 bool operator<(const IdType<HostIdType>& lhs, const IdType<HostIdType>& rhs) {
69 if (lhs.host_or_trimmed == rhs.host_or_trimmed)
70 return lhs.id < rhs.id;
71 else if (lhs.host_or_trimmed < rhs.host_or_trimmed)
72 return true;
