39 void updateImpl() {
40 pointSets_.clear();
41
42 auto const& indexSet = this->gridView().indexSet();
43 std::uint64_t vertexCounter = 0;
44 numCells_ = 0;
45 numPoints_ = 0;
46 for (auto element : elements(this->gridView())) {
47 const size_t elementId = indexSet.index(element);
48
49 auto verticesInSubGrid = geometries_.nVertices(elementId);
50
51 numCells_ += geometries_.nElements(elementId);
52
53 pointSets_.try_emplace(geometries_.geometryType(elementId), 1);
54
55 for (auto& [type, pointSet] : pointSets_)
56 if (pointSet.size() == 0) pointSet.build(type);
57
58 for (std::size_t vIdx = 0; auto& subGridVertex : geometries_.getVertices(elementId))
59 vertexIndex_.emplace(std::array<std::size_t, 2>({elementId, vIdx++}), vertexCounter++);
60
61 numPoints_ += verticesInSubGrid;
62 }
63 }

73 template <class T>
74 [[nodiscard]] std::vector<T> pointsImpl() const {
75 std::vector<T> data(this->numPoints() * 3);
76 auto const& indexSet = this->gridView().indexSet();
77 for (auto element : elements(this->gridView(), partition)) {
78 auto geometry = element.geometry();
79 const size_t elementId = indexSet.index(element);
80
81 for (std::uint64_t vIdx = 0; auto& subGridVertex : geometries_.getVertices(elementId)) {
82 auto v = geometry.global(subGridVertex);
83
84 std::int64_t idx = 3 * vertexIndex_.at({elementId, vIdx++});
85
86 for (std::size_t j = 0; j < v.size(); ++j)
87 data[idx + j] = T(v[j]);
88 for (std::size_t j = v.size(); j < 3u; ++j)
89 data[idx + j] = T(0);
90 }
91 }
92
93 return data;
94 }

103 [[nodiscard]] Cells cellsImpl() const {
104 Cells cells;
105 cells.connectivity.reserve(this->numPoints());
106 cells.offsets.reserve(this->numCells());
107 cells.types.reserve(this->numCells());
108
109 auto const& indexSet = this->gridView().indexSet();
110
111 for (std::int64_t old_o = 0; auto const& ele : elements(this->gridView(), partition)) {
112 const std::size_t elementId = indexSet.index(ele);
113
114 for (std::size_t eIdx = 0; auto& subGridElement : geometries_.getElements(elementId)) {
115 Vtk::CellType cellType(subGridElement.type(), Vtk::CellType::LAGRANGE);
116
117 auto const& pointSet = pointSets_.at(subGridElement.type());
118
119 for (std::size_t i = 0; i < pointSet.size(); ++i) {
120 auto const& p = pointSet[i];
121 auto const& localKey = p.localKey();
122 std::int64_t idx
123 = vertexIndex_.at({elementId, geometries_.vertexSubIndex(elementId, eIdx, localKey.subEntity())});
124
125 cells.connectivity.push_back(idx);
126 }
127 cells.types.push_back(cellType.type());
128 cells.offsets.push_back(old_o += pointSet.size());
129 ++eIdx;
130 }
131 }
132 return cells;
133 }

136 template <class T, class GlobalFunction>
137 [[nodiscard]] std::vector<T> pointDataImpl(GlobalFunction const& fct) const {
138 int nComps = fct.numComponents();
139 std::vector<T> data(this->numPoints() * nComps);
140
141 auto localFct = localFunction(fct);
142 auto const& indexSet = this->gridView().indexSet();
143 for (auto element : elements(this->gridView(), partition)) {
144 localFct.bind(element);
145 auto geometry = element.geometry();
146 const size_t elementId = indexSet.index(element);
147
148 for (std::uint64_t vIdx = 0; auto& subGridVertex : geometries_.getVertices(elementId)) {
149 std::int64_t idx = nComps * vertexIndex_.at({elementId, vIdx++});
150
151 for (std::size_t comp = 0; comp < nComps; ++comp)
152 data[idx + comp] = T(localFct.evaluate(comp, subGridVertex));
153 }
154 }
155
156 return data;
157 }

159 template <class T, class VtkFunction>
160 [[nodiscard]] std::vector<T> cellDataImpl(VtkFunction const& fct) const {
161 int nComps = fct.numComponents();
162 std::vector<T> data;
163 data.reserve(this->numCells_ * nComps);
164
165 auto localFct = localFunction(fct);
166 auto const& indexSet = this->gridView().indexSet();
167
168 for (auto element : elements(this->gridView(), partition)) {
169 localFct.bind(element);
170 auto geometry = element.geometry();
171 const size_t elementId = indexSet.index(element);
172
173 for (auto& subGridElement : geometries_.getElements(elementId)) {
174 auto vecInLocal = subGridElement.center();
175
176 for (std::size_t comp = 0; comp < nComps; ++comp)
177 data.push_back(localFct.evaluate(comp, vecInLocal));
178 }
179 }
180 return data;
181 }
