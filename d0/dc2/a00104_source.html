52 NURBSGeometry(std::shared_ptr<NURBSPatchData<griddim, dimworld, ctype>> patchData,
53 const std::array<Impl::FixedOrFree, griddim>& fixedOrVaryingDirections,
54 const std::array<int, griddim>& thisSpanIndices, const std::shared_ptr<SubGridType> subGrid = nullptr)
55 : patchData_(patchData), fixedOrVaryingDirections_{fixedOrVaryingDirections}, subgrid_(subGrid) {
56 for (int i = 0; i < griddim; ++i) {
57 if (thisSpanIndices[i] + 1 < patchData_->knotSpans[i].size())
58 scaling_[i] = patchData_->knotSpans[i][thisSpanIndices[i] + 1] - patchData_->knotSpans[i][thisSpanIndices[i]];
59 offset_[i] = patchData_->knotSpans[i][thisSpanIndices[i]];
60 }
61 for (int i = 0; i < griddim; ++i)
62 thisSpanIndices_[i] = (thisSpanIndices[i] == patchData->knotSpans[i].size() - 1)
63 ? thisSpanIndices[i] - patchData->degree[i] - 1
64 : thisSpanIndices[i];
65
66 // If we are a vertex and on the rightmost end of the knotspan, we receive here the last index,
67 // For the proper construction of the nurbs and controlpoint net we need the indices end -degree -2
68 // To properly extract for the last span and not for the span after the last one
69 if constexpr (mydim == 0)
70 for (int i = 0; i < griddim; ++i)
71 if (thisSpanIndices[i] == patchData->knotSpans[i].size() - 1)
72 thisSpanIndices_[i] = patchData->knotSpans[i].size() - patchData->degree[i] - 2;
73
74 nurbs_ = Dune::IGA::Nurbs<griddim, ctype>(*patchData, thisSpanIndices_);
75 cpCoordinateNet_
76 = netOfSpan(thisSpanIndices_, patchData_->degree, extractControlCoordinates(patchData_->controlPoints));
77 }

137 [[nodiscard]] LocalCoordinate local(const GlobalCoordinate& global) const {
138 const ctype tolerance = ctype(16) * std::numeric_limits<ctype>::epsilon();
139 if constexpr (mydimension == 0)
140 return {};
141 else if constexpr (mydimension != coorddimension) {
142 auto [u, Ru, fu, gap] = Dune::IGA::closestPointProjectionByTrustRegion(*this, global);
143 return u;
144 } else {
145 LocalCoordinate x = LocalCoordinate(0.5);
146 LocalCoordinate dx{};
147
148 GlobalCoordinate dglobal;
149 do { // from multilinearGeometry
150 dglobal = (*this).global(x) - global;
151 MatrixHelper::template xTRightInvA<mydimension, coorddimension>(jacobianTransposed(x), dglobal, dx);
152 const bool invertible
153 = MatrixHelper::template xTRightInvA<mydimension, coorddimension>(jacobianTransposed(x), dglobal, dx);
154
155 if (!invertible) return LocalCoordinate(std::numeric_limits<ctype>::max());
156 x -= dx;
157 // if local is outside of maximum knot vector span bound, thus we clamp it to it and return
158 // clamp result into boundaries
159 if (Utilities::clampToBoundaryAndCheckIfIsAtAllBoundaries(x, domain())) {
160 break;
161 }
162
163 } while (dx.two_norm2() > tolerance);
164
165 return x;
166 }

173 [[nodiscard]] JacobianTransposed jacobianTransposed(const LocalCoordinate& local) const {
174 JacobianTransposed result;
175 std::array<unsigned int, mydimension> subDirs = getDirectionsOfSubEntityInParameterSpace();
176
177 const auto localInSpan = localToSpan(local);
178 const auto basisFunctionDerivatives = nurbs_.basisFunctionDerivatives(localInSpan, 1);
179
180 for (int dir = 0; dir < mydimension; ++dir) {
181 std::array<unsigned int, griddim> ithVec{};
182 ithVec[subDirs[dir]] = 1;
183 result[dir] = dot(basisFunctionDerivatives.get(ithVec), cpCoordinateNet_);
184 result[dir] *= scaling_[subDirs[dir]]; // transform back to 0..1 domain
185 }
186 return result;
187 }

232 auto secondFundamentalForm(const LocalCoordinate& local) const requires(mydimension == 2) && (coorddimension == 3) {
233 const auto secDerivatives = secondDerivativeOfPosition(local);
234 const auto unitnormal = unitNormal(local);
235 FieldMatrix<ctype, mydimension, mydimension> b;
236 b[0][0] = secDerivatives[0] * unitnormal;
237 b[1][1] = secDerivatives[1] * unitnormal;
238 b[0][1] = b[1][0] = secDerivatives[2] * unitnormal;
239 return b;
240 }

242 auto secondDerivativeOfPosition(const LocalCoordinate& local) const {
243 FieldMatrix<ctype, mydimension*(mydimension + 1) / 2, coorddimension> result;
244 std::array<unsigned int, mydimension> subDirs = getDirectionsOfSubEntityInParameterSpace();
245 const auto localInSpan = localToSpan(local);
246 const auto basisFunctionDerivatives = nurbs_.basisFunctionDerivatives(localInSpan, 2);
247 for (int dir = 0; dir < mydimension; ++dir) {
248 std::array<unsigned int, griddim> ithVec{};
249 ithVec[subDirs[dir]] = 2; // second derivative in dir direction
250 result[dir] = dot(basisFunctionDerivatives.get(ithVec), cpCoordinateNet_);
251 result[dir] *= Dune::power(scaling_[subDirs[dir]], 2); // transform back to 0..1
252 }
253 if constexpr (mydimension > 1 and griddim > 1) {
254 std::array<int, griddim> mixeDerivs;
255 std::ranges::fill(mixeDerivs, 0); // first mixed derivatives
256 if constexpr (mydimension == 2)
257 for (int dir = 0; dir < mydimension; ++dir) {
258 mixeDerivs[subDirs[dir]] = 1;
259 }
260 else
261 std::ranges::fill_n(mixeDerivs.begin(), 2, 1); // first mixed derivatives
262 int mixedDireCounter = mydimension;
263 do {
264 result[mixedDireCounter++] = dot(basisFunctionDerivatives.get(mixeDerivs), cpCoordinateNet_);
265 for (int dir = 0; dir < mydimension; ++dir) {
266 if (mixeDerivs[dir] == 0) continue;
267 result[mixedDireCounter - 1] *= scaling_.at(subDirs[dir]);
268 }
269 if constexpr (mydimension == 2) break;
270
271 } while (std::ranges::next_permutation(mixeDerivs, std::greater()).found);
272 }
273
274 return result;
275 }

277 auto getDirectionsOfSubEntityInParameterSpace() const {
278 std::array<unsigned int, mydimension> subDirs;
279 for (int subI = 0, i = 0; i < griddim; ++i) {
280 if constexpr (mydimension != griddim)
281 if (fixedOrVaryingDirections_[i] == Dune::IGA::Impl::FixedOrFree::fixed) continue;
282 subDirs[subI++] = i;
283 }
284 return subDirs;
285 }
286 auto zeroFirstAndSecondDerivativeOfPosition(const LocalCoordinate& uL) const {
287 const auto u = localToSpan(uL);
288 GlobalCoordinate pos;
289
290 Hessian H;
291 std::array<unsigned int, mydimension> subDirs = getDirectionsOfSubEntityInParameterSpace();
292
293 const auto basisFunctionDerivatives = nurbs_.basisFunctionDerivatives(u, 2);
294
295 std::array<unsigned int, griddim> ithVecZero{};
296 pos = Dune::IGA::dot(basisFunctionDerivatives.get(ithVecZero), cpCoordinateNet_);
297 JacobianTransposed J;
298 for (int dir = 0; dir < mydimension; ++dir) {
299 std::array<unsigned int, griddim> ithVec{};
300 ithVec[subDirs[dir]] = 1;
301 J[dir] = dot(basisFunctionDerivatives.get(ithVec), cpCoordinateNet_);
302 J[dir] *= scaling_.at(subDirs[dir]); // transform back to 0..1 domain
303 }
304
305 for (int dir = 0; dir < mydimension; ++dir) {
306 std::array<unsigned int, griddim> ithVec{};
307 ithVec[subDirs[dir]] = 2; // second derivative in subDirs[dir] direction
308 H[dir] = dot(basisFunctionDerivatives.get(ithVec), cpCoordinateNet_);
309
310 H[dir] *= Dune::power(scaling_.at(subDirs[dir]), 2); // transform back to 0..1
311 }
312 if constexpr (mydimension > 1 and griddim > 1) {
313 std::array<int, griddim> mixeDerivs;
314 std::ranges::fill(mixeDerivs, 0); // first mixed derivatives
315 if constexpr (mydimension == 2)
316 for (int dir = 0; dir < mydimension; ++dir) {
317 mixeDerivs[subDirs[dir]] = 1;
318 }
319 else
320 std::ranges::fill_n(mixeDerivs.begin(), 2, 1); // first mixed derivatives
321 int mixedDireCounter = mydimension;
322 do {
323 H[mixedDireCounter++] = dot(basisFunctionDerivatives.get(mixeDerivs), cpCoordinateNet_);
324 for (int dir = 0; dir < mydimension; ++dir) {
325 if (mixeDerivs[dir] == 0) continue;
326 H[mixedDireCounter - 1] *= scaling_.at(subDirs[dir]);
327 }
328 if constexpr (mydimension == 2) break;
329
330 } while (std::ranges::next_permutation(mixeDerivs, std::greater()).found);
331 }
332
333 return std::make_tuple(pos, J, H);
334 }

345 template <typename ReturnType = Dune::FieldVector<typename LocalCoordinate::value_type, griddim>>
346 auto localToSpan(const LocalCoordinate& local) const {
347 ReturnType localInSpan;
348 if constexpr (LocalCoordinate::dimension != 0) {
349 for (int loci = 0, i = 0; i < griddim; ++i) {
350 localInSpan[i] = (fixedOrVaryingDirections_[i] == Impl::FixedOrFree::free)
351 ? local[loci++] * scaling_[i] + offset_[i]
352 : offset_[i];
353 }
354 } else
355 for (int i = 0; i < griddim; ++i)
356 localInSpan[i] = offset_[i];
357 return localInSpan;
358 }
359
361 template <typename ReturnType = LocalCoordinate>
362 auto spanToLocal(const LocalCoordinate& inSpan) const {
363 ReturnType localL;
364 for (int i = 0; i < griddim; ++i) {
365 localL[i] = (inSpan[i] - offset_[i]) / scaling_[i];
366 localL[i] = clampToDomain(localL[i], domain());
367 }
368
369 return localL;
370 }
