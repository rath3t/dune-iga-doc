51 NURBSGeometry(std::shared_ptr<NURBSPatchData<griddim, dimworld, ctype>> patchData,
52 const std::array<Impl::FixedOrFree, griddim>& fixedOrVaryingDirections,
53 const std::array<int, griddim>& thisSpanIndices, const std::shared_ptr<SubGridType> subGrid = nullptr)
54 : patchData_(patchData), fixedOrVaryingDirections_{fixedOrVaryingDirections}, subgrid_(subGrid) {
55 for (int i = 0; i < griddim; ++i) {
56 if (thisSpanIndices[i] + 1 < patchData_->knotSpans[i].size())
57 scaling_[i] = patchData_->knotSpans[i][thisSpanIndices[i] + 1] - patchData_->knotSpans[i][thisSpanIndices[i]];
58 offset_[i] = patchData_->knotSpans[i][thisSpanIndices[i]];
59 }
60 for (int i = 0; i < griddim; ++i)
61 thisSpanIndices_[i] = (thisSpanIndices[i] == patchData->knotSpans[i].size() - 1)
62 ? thisSpanIndices[i] - patchData->degree[i] - 1
63 : thisSpanIndices[i];
64
65 // If we are a vertex and on the rightmost end of the knotspan, we receive here the last index,
66 // For the proper construction of the nurbs and controlpoint net we need the indices end -degree -2
67 // To properly extract for the last span and not for the span after the last one
68 if constexpr (mydim == 0)
69 for (int i = 0; i < griddim; ++i)
70 if (thisSpanIndices[i] == patchData->knotSpans[i].size() - 1)
71 thisSpanIndices_[i] = patchData->knotSpans[i].size() - patchData->degree[i] - 2;
72
73 nurbs_ = Dune::IGA::Nurbs<griddim, ctype>(*patchData, thisSpanIndices_);
74 cpCoordinateNet_
75 = netOfSpan(thisSpanIndices_, patchData_->degree, extractControlCoordinates(patchData_->controlPoints));
76 }

136 [[nodiscard]] LocalCoordinate local(const GlobalCoordinate& global) const {
137 const ctype tolerance = ctype(16) * std::numeric_limits<ctype>::epsilon();
138 if constexpr (mydimension == 0)
139 return {};
140 else if constexpr (mydimension != coorddimension) {
141 auto [u, Ru, fu, gap] = Dune::IGA::closestPointProjectionByTrustRegion(*this, global);
142 return u;
143 } else {
144 LocalCoordinate x = LocalCoordinate(0.5);
145 LocalCoordinate dx{};
146
147 GlobalCoordinate dglobal;
148 do { // from multilinearGeometry
149 dglobal = (*this).global(x) - global;
150 MatrixHelper::template xTRightInvA<mydimension, coorddimension>(jacobianTransposed(x), dglobal, dx);
151 const bool invertible
152 = MatrixHelper::template xTRightInvA<mydimension, coorddimension>(jacobianTransposed(x), dglobal, dx);
153
154 if (!invertible) return LocalCoordinate(std::numeric_limits<ctype>::max());
155 x -= dx;
156 // if local is outside of maximum knot vector span bound, thus we clamp it to it and return
157 // clamp result into boundaries
158 if (Utilities::clampToBoundaryAndCheckIfIsAtAllBoundaries(x, domain())) {
159 break;
160 }
161
162 } while (dx.two_norm2() > tolerance);
163
164 return x;
165 }

172 [[nodiscard]] JacobianTransposed jacobianTransposed(const LocalCoordinate& local) const {
173 JacobianTransposed result;
174 std::array<unsigned int, mydimension> subDirs = getDirectionsOfSubEntityInParameterSpace();
175
176 const auto localInSpan = localToSpan(local);
177 const auto basisFunctionDerivatives = nurbs_.basisFunctionDerivatives(localInSpan, 1);
178
179 for (int dir = 0; dir < mydimension; ++dir) {
180 std::array<unsigned int, griddim> ithVec{};
181 ithVec[subDirs[dir]] = 1;
182 result[dir] = dot(basisFunctionDerivatives.get(ithVec), cpCoordinateNet_);
183 result[dir] *= scaling_[subDirs[dir]]; // transform back to 0..1 domain
184 }
185 return result;
186 }

231 auto secondFundamentalForm(const LocalCoordinate& local) const requires(mydimension == 2) && (coorddimension == 3) {
232 const auto secDerivatives = secondDerivativeOfPosition(local);
233 const auto unitnormal = unitNormal(local);
234 FieldMatrix<ctype, mydimension, mydimension> b;
235 b[0][0] = secDerivatives[0] * unitnormal;
236 b[1][1] = secDerivatives[1] * unitnormal;
237 b[0][1] = b[1][0] = secDerivatives[2] * unitnormal;
238 return b;
239 }

241 auto secondDerivativeOfPosition(const LocalCoordinate& local) const {
242 FieldMatrix<ctype, mydimension*(mydimension + 1) / 2, coorddimension> result;
243 std::array<unsigned int, mydimension> subDirs = getDirectionsOfSubEntityInParameterSpace();
244 const auto localInSpan = localToSpan(local);
245 const auto basisFunctionDerivatives = nurbs_.basisFunctionDerivatives(localInSpan, 2);
246 for (int dir = 0; dir < mydimension; ++dir) {
247 std::array<unsigned int, griddim> ithVec{};
248 ithVec[subDirs[dir]] = 2; // second derivative in dir direction
249 result[dir] = dot(basisFunctionDerivatives.get(ithVec), cpCoordinateNet_);
250 result[dir] *= Dune::power(scaling_[subDirs[dir]], 2); // transform back to 0..1
251 }
252 if constexpr (mydimension > 1 and griddim > 1) {
253 std::array<int, griddim> mixeDerivs;
254 std::ranges::fill(mixeDerivs, 0); // first mixed derivatives
255 if constexpr (mydimension == 2)
256 for (int dir = 0; dir < mydimension; ++dir) {
257 mixeDerivs[subDirs[dir]] = 1;
258 }
259 else
260 std::ranges::fill_n(mixeDerivs.begin(), 2, 1); // first mixed derivatives
261 int mixedDireCounter = mydimension;
262 do {
263 result[mixedDireCounter++] = dot(basisFunctionDerivatives.get(mixeDerivs), cpCoordinateNet_);
264 for (int dir = 0; dir < mydimension; ++dir) {
265 if (mixeDerivs[dir] == 0) continue;
266 result[mixedDireCounter - 1] *= scaling_.at(subDirs[dir]);
267 }
268 if constexpr (mydimension == 2) break;
269
270 } while (std::ranges::next_permutation(mixeDerivs, std::greater()).found);
271 }
272
273 return result;
274 }

276 auto getDirectionsOfSubEntityInParameterSpace() const {
277 std::array<unsigned int, mydimension> subDirs;
278 for (int subI = 0, i = 0; i < griddim; ++i) {
279 if constexpr (mydimension != griddim)
280 if (fixedOrVaryingDirections_[i] == Dune::IGA::Impl::FixedOrFree::fixed) continue;
281 subDirs[subI++] = i;
282 }
283 return subDirs;
284 }
285 auto zeroFirstAndSecondDerivativeOfPosition(const LocalCoordinate& uL) const {
286 const auto u = localToSpan(uL);
287 FieldVector<ctype, dimworld> pos;
288
289 FieldMatrix<ctype, mydimension*(mydimension + 1) / 2, coorddimension> H;
290 std::array<unsigned int, mydimension> subDirs = getDirectionsOfSubEntityInParameterSpace();
291
292 const auto basisFunctionDerivatives = nurbs_.basisFunctionDerivatives(u, 2);
293
294 std::array<unsigned int, griddim> ithVecZero{};
295 pos = Dune::IGA::dot(basisFunctionDerivatives.get(ithVecZero), cpCoordinateNet_);
296 JacobianTransposed J;
297 for (int dir = 0; dir < mydimension; ++dir) {
298 std::array<unsigned int, griddim> ithVec{};
299 ithVec[subDirs[dir]] = 1;
300 J[dir] = dot(basisFunctionDerivatives.get(ithVec), cpCoordinateNet_);
301 J[dir] *= scaling_.at(subDirs[dir]); // transform back to 0..1 domain
302 }
303
304 for (int dir = 0; dir < mydimension; ++dir) {
305 std::array<unsigned int, griddim> ithVec{};
306 ithVec[subDirs[dir]] = 2; // second derivative in subDirs[dir] direction
307 H[dir] = dot(basisFunctionDerivatives.get(ithVec), cpCoordinateNet_);
308
309 H[dir] *= Dune::power(scaling_.at(subDirs[dir]), 2); // transform back to 0..1
310 }
311 if constexpr (mydimension > 1 and griddim > 1) {
312 std::array<int, griddim> mixeDerivs;
313 std::ranges::fill(mixeDerivs, 0); // first mixed derivatives
314 if constexpr (mydimension == 2)
315 for (int dir = 0; dir < mydimension; ++dir) {
316 mixeDerivs[subDirs[dir]] = 1;
317 }
318 else
319 std::ranges::fill_n(mixeDerivs.begin(), 2, 1); // first mixed derivatives
320 int mixedDireCounter = mydimension;
321 do {
322 H[mixedDireCounter++] = dot(basisFunctionDerivatives.get(mixeDerivs), cpCoordinateNet_);
323 for (int dir = 0; dir < mydimension; ++dir) {
324 if (mixeDerivs[dir] == 0) continue;
325 H[mixedDireCounter - 1] *= scaling_.at(subDirs[dir]);
326 }
327 if constexpr (mydimension == 2) break;
328
329 } while (std::ranges::next_permutation(mixeDerivs, std::greater()).found);
330 }
331
332 return std::make_tuple(pos, J, H);
333 }

344 template <typename ReturnType = Dune::FieldVector<typename LocalCoordinate::value_type, griddim>>
345 auto localToSpan(const LocalCoordinate& local) const {
346 ReturnType localInSpan;
347 if constexpr (LocalCoordinate::dimension != 0) {
348 for (int loci = 0, i = 0; i < griddim; ++i) {
349 localInSpan[i] = (fixedOrVaryingDirections_[i] == Impl::FixedOrFree::free)
350 ? local[loci++] * scaling_[i] + offset_[i]
351 : offset_[i];
352 }
353 } else
354 for (int i = 0; i < griddim; ++i)
355 localInSpan[i] = offset_[i];
356 return localInSpan;
357 }
358
360 template <typename ReturnType = LocalCoordinate>
361 auto spanToLocal(const LocalCoordinate& inSpan) const {
362 ReturnType localL;
363 for (int i = 0; i < griddim; ++i) {
364 localL[i] = (inSpan[i] - offset_[i]) / scaling_[i];
365 localL[i] = clampToDomain(localL[i], domain());
366 }
367
368 return localL;
369 }
