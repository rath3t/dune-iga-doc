20 std::array<int, dim> order;
21 std::ranges::transform(degree, order.begin(), [](const auto& p) { return p + 1; });

25 template <std::integral auto dim>
26 void createPartialSubDerivativPermutations(const FieldVector<int, dim>& v,
27 std::vector<FieldVector<int, dim>>& perm) {
28 perm.resize(0);
29 for (int i = 1; i < std::pow(2, v.size()); ++i) {
30 FieldVector<int, dim> x{};
31 for (int j = 0; j < v.size(); ++j)
32 if ((i & (1 << j)) != 0) {
33 x[j] = v[j];
34 if (v[j] == 0) goto outer;
35 }
36 perm.push_back(x);
37 outer:;
38 }
39 }

41 template <Concept::Vector VectorType>
42 int binom(const VectorType& n, const VectorType& k) {
43 return std::inner_product(n.begin(), n.end(), k.begin(), 1, std::multiplies{},
44 [](auto& ni, auto& ki) { return Dune::binomial(ni, ki); });

63 template <std::floating_point ScalarType, std::integral auto dim>
64 auto generateRefinedKnots(const std::array<std::vector<ScalarType>, dim>& knotSpans, const int dir,
65 const int refinementLevel) {
66 const int newKnotsSizeForEachSpan = Dune::power(2, refinementLevel);
67 auto unique_Knots = knotSpans;
68 std::vector<double> additionalKnots;
69 auto& unique_KnotPerDim = unique_Knots[dir];
70 unique_KnotPerDim.erase(std::ranges::begin(std::ranges::unique(unique_KnotPerDim)), std::end(unique_KnotPerDim));
71
72 for (int i = 0; i < unique_KnotPerDim.size() - 1; ++i) {
73 const double spanLength = unique_KnotPerDim[i + 1] - unique_KnotPerDim[i];
74 const double increment = spanLength / newKnotsSizeForEachSpan;
75 for (int j = 1; j < newKnotsSizeForEachSpan; ++j) {
76 additionalKnots.emplace_back(unique_KnotPerDim[i] + increment * j);
77 }
78 }
79 return additionalKnots;
80 }

94 template <std::integral auto dim, std::integral auto dim2, typename NetValueType>
95 requires(
96 (std::floating_point<
97 NetValueType> || Concept::Vector<NetValueType> || is_instantiation_of<ControlPoint, NetValueType>::value)
98 and (dim == dim2)) auto netOfSpan(std::array<int, dim> subNetStart, const std::array<int, dim>& degree,
99 const MultiDimensionalNet<dim2, NetValueType>& net) {
100 std::array<int, dim> order = Impl::ordersFromDegrees(degree);
101 for (std::size_t i = 0; i < dim; ++i)
102 subNetStart[i] -= degree[i];
103 return net.subNet(subNetStart, order);
104 }

298 static auto basisFunctions(const FieldVector<ScalarType, dim>& u,
299 const std::array<std::vector<ScalarType>, dim>& knots,
300 const std::array<int, dim>& degree, const MultiDimensionalNet<dim, ScalarType>& weights,
301 std::array<int, dim> spIndex) {
302 std::array<DynamicVectorType, (size_t)dim> bSplines;
303
304 for (std::size_t i = 0; i < dim; ++i)
305 bSplines[i] = BsplineBasis<ScalarType>::basisFunctions(u[i], knots[i], degree[i], spIndex[i]);
306
307 auto Nnet = MultiDimensionalNet<dim, ScalarType>(bSplines);
308 // @todo Alex this should also be cached when called with a localView
309 const auto subNetWeights = netOfSpan(spIndex, degree, weights);
310
311 const ScalarType invSumWeight = dot(Nnet, subNetWeights);
312 Nnet *= subNetWeights;
313 Nnet /= invSumWeight;
314 return Nnet;
315 }

327 static auto basisFunctionDerivatives(const FieldVector<ScalarType, dim>& u,
328 const std::array<std::vector<ScalarType>, dim>& knots,
329 const std::array<int, dim>& degree,
330 const MultiDimensionalNet<dim, double>& weights, const int derivativeOrder,
331 std::optional<std::array<int, dim>> spIndex = std::nullopt) {
332 std::array<DynamicMatrixType, dim> bSplineDerivatives;
333 std::array<int, dim> spanIndex = spIndex ? spIndex.value() : findSpan(degree, u, knots);
334
335 for (int i = 0; i < dim; ++i)
336 bSplineDerivatives[i] = BsplineBasis<ScalarType>::basisFunctionDerivatives(u[i], knots[i], degree[i],
337 derivativeOrder, spanIndex[i]);
338
339 std::array<std::vector<ScalarType>, dim> dimArrayOfVectors;
340 FieldVector<int, dim> dimSize(derivativeOrder + 1);
341 MultiDimensionalNet<dim, MultiDimensionalNet<dim, ScalarType>> netOfDerivativeNets(dimSize);
342
343 for (int j = 0; auto& derivNet : netOfDerivativeNets.directGetAll()) {
344 auto multiIndex = netOfDerivativeNets.directToMultiIndex(j++);
345 for (int i = 0; i < multiIndex.size(); ++i)
346 dimArrayOfVectors[i] = bSplineDerivatives[i][multiIndex[i]].container();
347 derivNet = MultiDimensionalNet<dim, ScalarType>(dimArrayOfVectors);
348 }
349
350 MultiDimensionalNet<dim, MultiDimensionalNet<dim, ScalarType>> R = netOfDerivativeNets;
351 MultiDimensionalNet<dim, ScalarType> netsOfWeightfunctions(dimSize);
352 const auto subNetWeights = netOfSpan(spanIndex, degree, weights);
353
354 for (int j = 0; j < R.size(); ++j) {
355 R.directGet(j) *= subNetWeights;
356 netsOfWeightfunctions.directGet(j) = dot(netOfDerivativeNets.directGet(j), subNetWeights);
357 }
358
359 std::vector<FieldVector<int, dim>> perms;
360 // The following is inefficient, it computes too much derivative directions and recycles nothing
361 for (int j = 0; j < R.size(); ++j) {
362 const auto derivOrders = R.template directToMultiIndex<FieldVector<int, dim>>(j);
363 Impl::createPartialSubDerivativPermutations(derivOrders, perms);
364
365 for (const auto& perm : perms) {
366 const MultiDimensionalNetIndex<dim> kNet(perm + FieldVector<int, dim>(1));
367 auto startMultiIndex = perm;
368 std::ranges::transform(startMultiIndex, startMultiIndex.begin(), [](const auto& v) { return (v != 0); });
369 for (int kk = kNet.index(startMultiIndex); kk < kNet.size(); ++kk) {
370 const auto multik = kNet.template directToMultiIndex<FieldVector<int, dim>>(kk);
371 const ScalarType fac = (Impl::binom(perm, multik) * netsOfWeightfunctions.get(multik));
372 const auto& Rdirect_d = R.get(derivOrders - multik);
373 auto& Rdirect = R.directGet(j);
374 for (int i = 0; i < R.directGet(j).size(); ++i)
375 Rdirect.directGet(i) -= fac * Rdirect_d.directGet(i); // generalized Piegl Tiller (4.20)
376 }
377 }
378 R.directGet(j) /= netsOfWeightfunctions.directGet(0);
379 }
380 return R;
381 }

433 template <std::integral auto dim, std::integral auto dimworld, typename ScalarType>
434 auto degreeElevate(const NURBSPatchData<dim, dimworld, ScalarType>& oldData, const int elevationDirection,
435 const int elevationFactor) {
436 using DynamicMatrixType = DynamicMatrix<ScalarType>;
437 using NURBSPatchData = NURBSPatchData<dim, dimworld, ScalarType>;
438 using ControlPointNetType = typename NURBSPatchData::ControlPointNetType;
439 using ControlPointType = typename NURBSPatchData::ControlPointType;
440 const int t = elevationFactor;
441 const int p = oldData.degree[elevationDirection];
442 const int m = oldData.controlPoints.strideSizes()[elevationDirection] + p;
443 const int ph = p + t;
444 const int ph2 = ph / 2;
445 /* Compute Bezier degree elevation coefficients */
446 DynamicMatrixType bezalfs(p + t + 1, p + 1);
447 bezalfs[0][0] = bezalfs[ph][p] = 1.0;
448 for (int i = 1; i <= ph2; ++i) {
449 const ScalarType inv = 1.0 / binomial(ph, i);
450 for (int j = std::max(0, i - t); j <= std::min(p, i); ++j)
451 bezalfs[i][j] = inv * binomial(p, j) * binomial(t, i - j);
452 }
453 for (int i = ph2 + 1; i <= ph - 1; ++i)
454 for (int j = std::max(0, i - t); j <= std::min(p, i); ++j)
455 bezalfs[i][j] = bezalfs[ph - i][p - j];
456
457 ControlPointNetType newCPv(oldData.controlPoints.strideSizes());
458
459 auto oldCPs = oldData.controlPoints;
460
461 auto scaleCPWithW = [](const auto& cp) -> ControlPointType { return {.p = cp.w * cp.p, .w = cp.w}; };
462 std::ranges::transform(oldCPs.directGetAll(), oldCPs.directGetAll().begin(), scaleCPWithW);
463
464 std::vector<ControlPointType> bpts(p + 1), ebpts(ph + 1), nextbpts(p - 1);
465
466 NURBSPatchData newData;
467 newData.degree = oldData.degree;
468 newData.degree[elevationDirection] += t;
469 const auto& U = oldData.knotSpans[elevationDirection];
470 double ua = U[0];
471 std::vector<ScalarType> Uh;
472 for (int j = 0, i = 0; i < U.size(); ++i) // insert knot t times for each unique knot
473 if (j < U.size()) {
474 do {
475 Uh.push_back(U[j]);
476 ++j;
477 } while (j != U.size() && FloatCmp::eq(U[j - 1], U[j]));
478 Uh.insert(Uh.end(), t, U[j - 1]);
479 if (j == U.size()) break;
480 }
481
482 for (int i = 0; i < dim; ++i)
483 if (i == elevationDirection)
484 newData.knotSpans[i] = Uh;
485 else
486 newData.knotSpans[i] = oldData.knotSpans[i];
487
488 auto dimSize = oldData.controlPoints.strideSizes();
489 dimSize[elevationDirection] = Uh.size() - ph - 1;
490 newData.controlPoints = MultiDimensionalNet<dim, ControlPointType>(dimSize);
491 auto& newCPs = newData.controlPoints;
492
493 int totalCurvesInPatch = 1;
494 std::array<int, dim - 1> indicesOther;
495 for (int counter = 0, i = 0; i < dim; ++i)
496 if (i != elevationDirection) {
497 totalCurvesInPatch *= dimSize[i];
498 indicesOther[counter++] = dimSize[i];
499 }
500
501 auto newAndOldCurve = [&elevationDirection, indicesOther, &oldCPs, &newCPs](auto i) {
502 const auto mI
503 = MultiDimensionalNet<dim - 1, int>::template directToMultiIndex<std::array<int, dim - 1>>(indicesOther, i);
504 std::array<int, dim> multiIndex{};
505 for (int c = 0, j = 0; j < dim; ++j)
506 if (j != elevationDirection) multiIndex[j] = mI[c++];
507
508 auto oldCurve = [&oldCPs, &elevationDirection, multiIndex](int i) mutable {
509 multiIndex[elevationDirection] = i;
510 return oldCPs.get(multiIndex);
511 };
512
513 auto newCurve = [&newCPs, &elevationDirection, multiIndex ](int i) mutable -> auto& {
514 multiIndex[elevationDirection] = i;
515 return newCPs.get(multiIndex);
516 };
517 return std::make_pair(oldCurve, newCurve);
518 };
519
520 for (auto [oldCurve, newCurve] :
521 std::ranges::iota_view(0, totalCurvesInPatch) | std::views::transform(newAndOldCurve)) {
522 ua = U[0];
523 int kind = ph + 1;
524 int r = -1;
525 int a = p;
526 int b = p + 1;
527 int cind = 1;
528
529 newCurve(0) = oldCurve(0);
530 for (int i = 0; i <= p; ++i)
531 bpts[i] = oldCurve(i);
532
533 std::vector<ScalarType> alphas(p - 1);
534 while (b < m) {
535 const int bold = b;
536 while (b < m && FloatCmp::eq(U[b], U[b + 1]))
537 ++b;
538 const int mul = b - bold + 1;
539 const auto ub = U[b];
540 const int oldr = r;
541 r = p - mul;
542 const int lbz = (oldr > 0) ? ((oldr + 2) / 2) : 1; // Insert knot u(b) r times
543 const int rbz = (r > 0) ? (ph - (r + 1) / 2) : ph;
544 if (r > 0) // Insert knot to get Bezier segment
545 {
546 const auto numer = ub - ua;
547 for (int k = p; k > mul; --k)
548 alphas[k - mul - 1] = numer / (U[a + k] - ua);
549 for (int j = 1; j <= r; ++j) {
550 const int s = mul + j;
551 for (int k = p; k >= s; --k)
552 bpts[k] = alphas[k - s] * bpts[k] + (1.0 - alphas[k - s]) * bpts[k - 1];
553 nextbpts[r - j] = bpts[p];
554 }
555 } // End of insert knot
556
557 for (int i = lbz; i <= ph; ++i) { // Degree elevate Bezier
558 ebpts[i].setZero();
559 for (int j = std::max(0, i - t); j <= std::min(p, i); ++j)
560 ebpts[i] += bezalfs[i][j] * bpts[j];
561 } // End of degree elevating Bezier
562 if (oldr > 1) { // Must remove knot u = U[a] oldr times
563 int first = kind - 2;
564 int last = kind;
565 const ScalarType den = ub - ua;
566 const ScalarType bet = (ub - Uh[kind - 1]) / den;
567 for (int tr = 1; tr < oldr; ++tr) { // Knot removal loop
568 int i = first;
569 int j = last;
570 int kj = j - kind + 1;
571 while (j - i > tr) {
572 if (i < cind) {
573 const ScalarType alf = (ub - Uh[i]) / (ua - Uh[i]);
574 newCurve(i) = alf * newCurve(i) + (1.0 - alf) * newCurve(i - 1);
575 }
576 if (j >= lbz) {
577 const auto factor = (j - tr <= kind - ph + oldr) ? (ub - Uh[j - tr]) / den : bet;
578 ebpts[kj] = factor * ebpts[kj] + (1.0 - factor) * ebpts[kj + 1];
579 }
580 ++i;
581 --j;
582 --kj;
583 }
584 --first;
585 ++last;
586 }
587 } // End of removing knot, u= U[a]
588 if (a != p) kind += ph - oldr;
589 for (int j = lbz; j <= rbz; ++j, ++cind) // Load ctrl pts into Qw
590 newCurve(cind) = ebpts[j];
591 if (b < m) { // Set up for next pass through loop
592 std::ranges::copy(nextbpts | std::views::take(r), bpts.begin());
593 for (int j = r; j <= p; ++j)
594 bpts[j] = oldCurve(b - p + j);
595 a = b;
596 ++b;
597 ua = ub;
598 }
599 } // End of while-loop (b<m)
600 }
601 std::ranges::transform(newData.controlPoints.directGetAll(), newData.controlPoints.directGetAll().begin(),
602 [](auto& cp) -> ControlPointType { return {.p = cp.p / cp.w, .w = cp.w}; });
603 return newData;
604 }

617 template <std::integral auto dim, std::integral auto dimworld, typename ScalarType>
618 auto knotRefinement(const NURBSPatchData<dim, dimworld, ScalarType>& oldData, const std::vector<double>& newKnots,
619 const int refinementDirection) {
620 assert(std::ranges::is_sorted(newKnots)
621 && "You passed a non-sorted vector of new knots. Sort it first or check if you passed the correct vector.");
622 using NurbsPatchData = NURBSPatchData<dim, dimworld, ScalarType>;
623 using ControlPoint = typename NurbsPatchData::ControlPointType;
624 using namespace std::ranges;
625 std::array<std::vector<double>, dim> newKnotsArray;
626 std::array<int, dim - 1> otherDirections;
627 for (int counter = 0, i = 0; i < dim; ++i) {
628 if (i == refinementDirection) continue;
629 newKnotsArray[i] = oldData.knotSpans[i];
630 otherDirections[counter++] = i;
631 }
632
633 typename NurbsPatchData::ControlPointNetType newCPv(oldData.controlPoints.strideSizes());
634
635 auto oldCPv = oldData.controlPoints;
636 auto oldKnotVec = oldData.knotSpans[refinementDirection];
637
638 auto& newKnotVec = newKnotsArray[refinementDirection];
639
640 // Combine old and new knot vectors
641 newKnotVec.resize(oldKnotVec.size() + newKnots.size());
642 merge(oldKnotVec, newKnots, std::begin(newKnotVec));
643
644 const auto newKSize = newKnots.size();
645 std::array<int, dim> numberOfCPperDir = oldCPv.strideSizes();
646 numberOfCPperDir[refinementDirection] += newKSize;
647
648 newCPv.resize(numberOfCPperDir);
649
650 auto scaleCPWithW = [](const auto& cp) -> ControlPoint { return {.p = cp.w * cp.p, .w = cp.w}; };
651
652 const int p = oldData.degree[refinementDirection];
653 const int a = findSpan(p, newKnots.front(), oldKnotVec);
654 const int b = findSpan(p, newKnots.back(), oldKnotVec);
655
656 auto newSurfI = newCPv.hyperSurfBegin(otherDirections);
657 auto oldSurfI = oldCPv.hyperSurfBegin(otherDirections);
658
659 for (auto hSOld = oldSurfI, hSNew = newSurfI; hSOld != oldCPv.hyperSurfBegin(otherDirections) + (a - p + 1);
660 ++hSOld, ++hSNew)
661 std::ranges::transform(*hSOld, (*hSNew).begin(), scaleCPWithW);
662 for (auto hSOld = oldSurfI + b, hSNew = newSurfI + b; hSOld != oldCPv.hyperSurfEnd(otherDirections);
663 ++hSOld, ++hSNew)
664 std::ranges::transform(*hSOld, (*hSNew).begin(), scaleCPWithW);
665
666 newSurfI = newCPv.hyperSurfEnd(otherDirections) - 1;
667 oldSurfI = oldCPv.hyperSurfEnd(otherDirections) - 1;
668
669 auto currentNewKnot = newKnotVec.end();
670 auto currentOldKnot = oldKnotVec.end();
671
672 for (auto const& currentAdditionalKnot : reverse_view(newKnots)) {
673 while (currentAdditionalKnot <= *(currentOldKnot - 1)
674 && std::distance(oldKnotVec.begin(), currentOldKnot) > a + 1) {
675 std::ranges::transform(*oldSurfI, (*newSurfI).begin(), scaleCPWithW);
676 --currentNewKnot, --currentOldKnot;
677 --newSurfI, --oldSurfI;
678 }
679
680 std::ranges::copy(*(newSurfI + 1), (*newSurfI).begin());
681 ++newSurfI;
682 currentOldKnot -= p;
683 for ([[maybe_unused]] int _ : iota_view{0, p}) {
684 ++newSurfI;
685
686 auto alpha = *currentNewKnot - currentAdditionalKnot;
687 using std::abs;
688 if (abs(alpha) < 1e-7)
689 std::ranges::copy(*newSurfI, (*(newSurfI - 1)).begin());
690 else {
691 alpha /= (*currentNewKnot - *(currentOldKnot));
692 std::ranges::transform(*newSurfI, *(newSurfI - 1), (*(newSurfI - 1)).begin(),
693 [&alpha](auto& cp, auto& cpL) { return alpha * cpL + cp * (1.0 - alpha); });
694 }
695 ++currentOldKnot, ++currentNewKnot;
696 }
697 currentNewKnot -= p + 1;
698 newSurfI -= p + 2;
699 }
700
701 std::ranges::transform(newCPv.directGetAll(), newCPv.directGetAll().begin(),
702 [](auto& cp) -> ControlPoint { return {.p = cp.p / cp.w, .w = cp.w}; });
703
704 return NURBSPatchData<dim, dimworld, ScalarType>(newKnotsArray, newCPv, oldData.degree);
705 }
