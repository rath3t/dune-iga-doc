114 template <std::ranges::sized_range V>
115 explicit MultiDimensionalNet(const std::array<V, netdim>& values) {
116 for (int i = 0; i < netdim; ++i)
117 dimSize_[i] = values[i].size();
118
119 const int localSize = std::accumulate(dimSize_.begin(), dimSize_.end(), 1, std::multiplies{});
120 values_.resize(localSize);
121 for (int i = 0; i < localSize; i += dimSize_[0])
122 std::copy(values[0].begin(), values[0].end(), values_.begin() + i);
123 // https://godbolt.org/z/TbTPszGqT
124 for (int curSize = dimSize_[0], k = 1; k < netdim; curSize *= (dimSize_[k]), ++k)
125 for (int ij = 0; ij < localSize; ij += curSize * (dimSize_[k]))
126 for (int j = 0; auto& v2 : values[k]) {
127 auto countedView = std::views::counted(values_.begin() + ij + (j++) * curSize, curSize);
128 std::ranges::transform(countedView, countedView.begin(), [&v2](auto& v) { return v * v2; });
129 }
130 }

221 auto subNet(const std::array<int, netdim>& start, const std::array<int, netdim>& size) const {
222 std::vector<value_type> subValues;
223 subValues.reserve(std::accumulate(size.begin(), size.end(), 1, std::multiplies{}));
224 if constexpr (netdim == 1)
225 for (int i = 0; i < size[0]; ++i)
226 subValues.push_back(get({start[0] + i}));
227 else if constexpr (netdim == 2)
228 for (int j = 0; j < size[1]; ++j)
229 for (int i = 0; i < size[0]; ++i)
230 subValues.push_back(get({start[0] + i, start[1] + j}));
231 else if constexpr (netdim == 3)
232 for (int k = 0; k < size[2]; ++k) // @todo generalize
233 for (int j = 0; j < size[1]; ++j)
234 for (int i = 0; i < size[0]; ++i)
235 subValues.push_back(get({start[0] + i, start[1] + j, start[2] + k}));
236
237 return MultiDimensionalNet<netdim, value_type>(size, subValues);
238 }

258 static ReturnType directToMultiIndex(const std::array<int, netdim>& dimSize, const int index) {
259 ReturnType multiIndex;
260
261 int help = index;
262 for (int i = 0; i < netdim; ++i) {
263 int temp = help % (dimSize[i]);
264 multiIndex[i] = temp;
265 help -= temp;
266 help = help / dimSize[i];

341 template <typename ArrayType = std::array<int, netdim>>
342 size_t index(const ArrayType& multiIndex) const {
343 assert(!std::ranges::any_of(multiIndex, [](int i) { return i < 0; })
344 && "The passed multiIndex has negative values");
345 assert(!std::ranges::any_of(multiIndex, [id = 0, this](int i) mutable { return i > dimSize_[id++] - 1; })
346 && "The passed multiIndex has too large values");
347
348 size_t index = 0;
349 for (int i = 0; i < netdim; ++i) {
350 int help = 1;
351 for (int j = i - 1; j > -1; --j)
352 help *= dimSize_[j];
353
354 index += help * static_cast<size_t>(multiIndex[i]);
355 }
356 return index;
357 }

399 HyperSurfaceIterator(MultiDimensionalNet<netdim, ValueType>& net, const std::array<int, netdim - 1>& direction,
400 int at)
401 : net_{&net}, direction_{direction}, at_{at} {
402 std::array<int, netdim - 1> indicesSurface;
403 for (int i = 0; i < indicesSurface.size(); ++i)
404 indicesSurface[i] = net.strideSizes().at(direction[i]);
405
406 const int cpSize = std::accumulate(indicesSurface.begin(), indicesSurface.end(), 1, std::multiplies{});
407 std::function<std::array<int, netdim - 1>(int)> func = [indicesSurface](auto i) {
408 return MultiDimensionalNet<netdim - 1, int>::template directToMultiIndex<std::array<int, netdim - 1>>(
409 indicesSurface, i);
410 };
411 viewOverIndices_ = std::ranges::iota_view(0, cpSize) | std::views::transform(func);
412 }

447 std::array<int, netdim> multiIndex;
448 if constexpr (netdim != 1)
449 for (int dirI = 0, i = 0; i < netdim; ++i) {
450 if (dirI < direction_.size() && i == direction_.at(dirI++)) continue;

465 }
