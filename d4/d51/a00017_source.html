63 template <std::size_t dim, typename ScalarType>
64 auto createUniqueKnotSpans(const std::array<std::vector<ScalarType>, dim>& knotSpans) {
65 std::array<std::vector<double>, dim> uniqueKnotVector;
66 for (int i = 0; i < dim; ++i) // create unique knotspan vectors
67 std::ranges::unique_copy(knotSpans[i], std::back_inserter(uniqueKnotVector[i]),
68 [](auto& l, auto& r) { return Dune::FloatCmp::eq(l, r); });
69
70 return uniqueKnotVector;
71 }

255 void globalRefine(int refCount) {
256 for (int i = 0; i < refCount; ++i) {
257 const auto& finestPatchData = patchGeometries.back().patchData_;
258 auto newfinestPatchData = finestPatchData;
259 for (int refDirection = 0; refDirection < dim; ++refDirection) {
260 auto additionalKnots = generateRefinedKnots(finestPatchData.knotSpans, refDirection, 1);
261 newfinestPatchData = knotRefinement<dim>(newfinestPatchData, additionalKnots, refDirection);
262 }
263 patchGeometries.emplace_back(std::move(newfinestPatchData));
264 }
265
266 hostgrid_->globalRefine(refCount);
267 setIndices();
268 }
