336 localIdSet_->update();
337
338 globalIdSet_->update();
339
340 // //////////////////////////////////////////
341 // Create the index sets
342 // //////////////////////////////////////////
343 for (int i = levelIndexSets_.size(); i <= maxLevel(); i++) {
344 auto p = std::make_unique<LevelIndexSet>();
345 levelIndexSets_.emplace_back(std::move(p));
346 }
347
348 for (int i = 0; i <= maxLevel(); i++)
349 if (levelIndexSets_[i]) levelIndexSets_[i]->update(*grid_, i);

351 leafIndexSet_->update(*grid_);
352 }
