89 GlobalCoordinate global(const LocalCoordinate& local) const {
90 const double offset = 0;
91 if constexpr (mydimension == 0)
92 return (localIndexInElement_ == 0) ? GlobalCoordinate(0) : GlobalCoordinate(1 - offset);
93 else if constexpr (mydimension == 1)
94 switch (localIndexInElement_) {
95 case 0:
96 return GlobalCoordinate({0, local[0]});
97 case 1:
98 return GlobalCoordinate({1 - offset, local[0]});
99 case 2:
100 return GlobalCoordinate({local[0], 0});
101 case 3:
102 return GlobalCoordinate({local[0], 1 - offset});
103 default:
104 __builtin_unreachable();
105 }
106 else if constexpr (mydimension == 3)
107 switch (localIndexInElement_) {
108 case 0:
109 return GlobalCoordinate({0, local[0], local[1]});
110 case 1:
111 return GlobalCoordinate({1 - offset, local[0], local[1]});
112 case 2:
113 return GlobalCoordinate({local[0], 0, local[1]});
114 case 3:
115 return GlobalCoordinate({local[0], 1 - offset, local[1]});
116 case 4:
117 return GlobalCoordinate({local[0], local[1], 0});
118 case 5:
119 return GlobalCoordinate({local[0], local[1], 1 - offset});
120 default:
121 __builtin_unreachable();
122 }
123 __builtin_unreachable();

192 case 4:
193 case 5:

197 __builtin_unreachable();
