23 template <std::ranges::random_access_range Range>
24 auto findSpan(const int p, typename std::remove_cvref_t<Range>::value_type u, Range&& U,
25 typename std::remove_cvref_t<Range>::difference_type offset = 0) {
26 if (u <= U[0]) return static_cast<long int>(p);
27 if (u >= U.back())
28 return static_cast<long int>(U.size() - p - 2); // if the coordinate is to big we return to the last non-end span
29 auto it = std::upper_bound(U.begin() + p - 1 + offset, U.end(), u);
30 return static_cast<long int>(std::distance(U.begin(), it) - 1);
31 }

37 std::array<int, dim> res;

80 template <std::ranges::random_access_range Range>
81 auto multiplicity(Range&& U,typename std::remove_cvref_t<Range>::value_type u) {
82
83 auto [lit,rit] =std::ranges::equal_range(U, u);
84 auto rightPos = std::distance(U.begin(), rit) ;
85 const int multiplicity = static_cast<long int>(std::distance(lit, rit));
86 if (u >= U.back())
87 return std::make_pair(multiplicity,static_cast<long int>(rightPos-multiplicity-1)); // if the coordinate is to big we return to the last non-end span
88 if (u <= U[0])
89 return std::make_pair(multiplicity,static_cast<long int>(multiplicity-1));
90 return std::make_pair(multiplicity,static_cast<long int>(std::distance(U.begin(), rit) - 1));
91 }

118 template <std::ranges::random_access_range Range>
119 static auto basisFunctions(typename std::remove_cvref_t<Range>::value_type u, Range&& knots, const int p,
120 std::optional<int> spIndex = std::nullopt) {
121 assert(std::ranges::count(knots.begin(), knots.begin() + p + 1, knots.front()) == p + 1);
122 assert(std::ranges::count(knots.end() - p - 1, knots.end(), knots.back()) == p + 1);
123 assert(spIndex < knots.size() - p - 1);
124 DynamicVectorType N;
125 N.resize(p + 1, 0.0);
126 u = std::clamp(u, knots.front(), knots.back()); // we clamp the value since sometimes we overshoot slightly
127 // due to machine precision
128 if (FloatCmp::eq(u, knots.back())) // early exit
129 {
130 N.back() = 1;
131 return N;
132 }
133 if (FloatCmp::eq(u, knots.front())) {
134 N.front() = 1;
135 return N;
136 }
137
138 // if the index in the knot span is not given we search for it
139 const int sp = spIndex ? spIndex.value() : findSpan(p, u, knots);
140 using namespace std::ranges;
141 auto lDiff = transform_view(reverse_view(std::views::counted(knots.begin() + sp + 1 - p, p)),
142 [&u](auto& kn) { return u - kn; });
143 auto rDiff = transform_view(std::views::counted(knots.begin() + sp + 1, p), [&u](auto& kn) { return kn - u; });
144
145 N[0] = ScalarType(1);
146
147 int j = 0;
148 for (auto lDp = lDiff.begin(); lDp != lDiff.end(); lDp += j + 2, ++j) {
149 ScalarType saved{};
150 for (int r = 0; auto&& rD : rDiff | std::views::take(j + 1)) {
151 const auto& lD = (*lDp);
152 const ScalarType tmp = N[r] / (rD + lD);
153 N[r++] = saved + rD * tmp;
154 saved = lD * tmp;
155 --lDp;
156 }
157 N[j + 1] = saved;
158 }
159 // The basis functions are always >=0!
160 assert(std::ranges::all_of(N, [](const auto& Ni) { return not Dune::FloatCmp::lt(Ni, -1e-8); }));
161
162 return N;
163 }

174 static auto basisFunctionDerivatives(ScalarType u, const std::vector<ScalarType>& knots, const int p,
175 const int derivativeOrder, std::optional<int> spIndex = std::nullopt) {
176 assert(spIndex < knots.size() - p - 1);
177 const int order = p + 1;
178 const int sp = spIndex ? spIndex.value() : findSpan(p, u, knots);
179 using namespace std::ranges;
180
181 DynamicMatrixType dN(derivativeOrder + 1, order);
182
183 std::vector<ScalarType> left(order);
184 std::vector<ScalarType> right(order);
185 DynamicMatrixType ndu(order, order);
186
187 ndu[0][0] = 1.0;
188 for (int j = 1; j <= p; j++) {
189 left[j] = u - knots[sp + 1 - j];
190 right[j] = knots[sp + j] - u;
191 double saved = 0.0;
192 for (int r = 0; r < j; r++) { /* Lower triangle */
193 ndu[j][r] = right[r + 1] + left[j - r];
194 double temp = ndu[r][j - 1] / ndu[j][r];
195 ndu[r][j] = saved + right[r + 1] * temp;
196 saved = left[j - r] * temp;
197 }
198 ndu[j][j] = saved;
199 }
200 for (int j = 0; j <= p; ++j)
201 dN[0][j] = ndu[j][p];
202
203 // Compute the derivatives
204 RowFixedMatrix a;
205 a[0].resize(p + 1);
206 a[1].resize(p + 1);
207 auto& a1Row = a[0];
208 auto& a2Row = a[1];
209 for (int r = 0; r <= p; ++r) {
210 a[0][0] = 1.0;
211
212 // Compute the k-th derivative
213 for (int k = 1; k <= derivativeOrder; ++k) {
214 const int rk = r - k;
215 const int pk = p - k;
216
217 auto& nduRowpk1 = ndu[pk + 1];
218 auto& dNcur = dN[k][r];
219 dNcur = 0.0;
220 if (r >= k) {
221 a2Row[0] = a1Row[0] / nduRowpk1[rk];
222 dNcur = a2Row[0] * ndu[rk][pk];
223 }
224
225 const int j1 = (rk >= -1) ? 1 : -rk;
226 const int j2 = (r - 1 <= pk) ? k - 1 : p - r;
227
228 for (int j = j1; j <= j2; ++j) {
229 a2Row[j] = (a1Row[j] - a1Row[j - 1]) / nduRowpk1[rk + j];
230 dNcur += a2Row[j] * ndu[rk + j][pk];
231 }
232
233 if (r <= pk) {
234 a2Row[k] = -a1Row[k - 1] / nduRowpk1[r];
235 dNcur += a2Row[k] * ndu[r][pk];
236 }
237 std::swap(a2Row, a1Row); // Switch rows
238 }
239 }
240 /* Multiply through by the correct factors */
241 /* (Eq. [2.9]) */
242 for (int r = p, k = 1; k <= derivativeOrder; ++k) {
243 for (int j = 0; j <= p; ++j)
244 dN[k][j] *= r;
245 r *= (p - k);
246 }
247 return dN;
248 }

250 private:
251 std::vector<ScalarType> knots_;
252 int degree_;
253 };
