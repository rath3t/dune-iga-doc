33 // Coordinates in a single knot span differ from coordinates on the B-spline patch
34 // by an affine transformation. This transformation is stored in the diagonal entries.
35 // If trimming is disabled the Jacobian in the parameter space of subentities (edges or surfaces) is a DiagonalMatrixBlock but is treated as a FieldMatrix, since
36 // is also just cubes and an axis-aligned geometry,
37 // if trimming is enabled the Jacobian of the parameterspace for subentities is a potentially fully populated FieldMatrix,
38 // Think about an arbitrary curve in the axis-aligned cube parameter grid
39 // ------->
40 // `____________________C_______D_______`
41 // ::````````````````::``````````'|````::
42 // :: :: `| ::
43 // :: ^F:: `| ::
44 // :: | :: `| ::
45 // :: | :: `| ::
46 // :: |E:: `| ::
47 // :: :: .| x ::
48 // :: :: _: ::
49 // ::................::..........|:....::
50 // ::''''''''''''''''::''''''''''/'''''::
51 // :: :: _:B ::
52 // :: :: .| ::
53 // :: :: `::A ::
54 // :: '::::||::::::' ::
55 // :: ::` :: x ::
56 // :: |' :: ::
57 // :: | x :: ::
58 // ::````````|```````::````````````````::
59 // `____________________________________`
60 // at point A the tangent of the curve (the JacobianInParameterSpace) points into the oblique direction (to B)
61 // for the axis-aligned geometry these tangents are from E to F or from C to D, depending on the orientation of the subentity, which are aligned with the axis
62 using JacobianTransposedInParameterSpace = typename ParameterSpaceGeometry::JacobianTransposed;
63 using GlobalInParameterSpace = typename ParameterSpaceGeometry::GlobalCoordinate;

104 [[nodiscard]] JacobianTransposed jacobianTransposed(const LocalCoordinate& u) const {
105 assert(parameterSpaceGeometry && "Bind the local view first!");
106 // using JacobianTransposed = FieldMatrix<ctype, mydimension, coorddimension>;
107 const auto ouInPatch = globalInParameterSpace(u);
108 const JacobianTransposedInPatch JTinPatch = patchGeometry_->jacobianTransposedImpl(ouInPatch, nurbsLocalView_,localControlPointNet);
109 const JacobianTransposedInParameterSpace jInparaMeterSpace = jacobianTransposedInParameterSpace(u);
110 JacobianTransposed jacobianOfEntity= jInparaMeterSpace*JTinPatch;
111 return jacobianOfEntity;
112 }

126 [[nodiscard]] double volume(int scaleOrder = 1) const {
127 assert(parameterSpaceGeometry && "Bind the local view first!");
128
129 if constexpr(trim) {
130 // if constexpr (mydimension == 2)
131 // if (subgrid_) {
132 // Dune::QuadratureRule<double, mydimension> rule;
133 // fillQuadratureRuleImpl(rule, *subgrid_.get(), (*std::ranges::max_element(patchData().degree)));
134 // Volume vol = 0.0;
135 // for (auto& gp : rule)
136 // vol += integrationElement(gp.position()) * gp.weight();
137 // return vol;
138 // }
139 }
140
141 const auto& rule = Dune::QuadratureRules<ctype, mydimension>::rule(
142 GeometryTypes::cube(mydimension), (*std::ranges::max_element(patchData().degree)));
143 ctype vol = 0.0;
144 for (auto& gp : rule)
145 vol += integrationElement(gp.position()) * gp.weight();
146 return vol;
147 }
