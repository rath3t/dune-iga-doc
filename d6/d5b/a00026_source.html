26 template <std::integral auto dim>
27 void createPartialSubDerivativPermutations(const FieldVector<int, dim>& v,
28 std::vector<FieldVector<int, dim>>& perm) {
29 perm.resize(0);
30 for (int i = 1; i < std::pow(2, v.size()); ++i) {
31 FieldVector<int, dim> x{};
32 for (int j = 0; j < v.size(); ++j)
33 if ((i & (1 << j)) != 0) {
34 x[j] = v[j];
35 if (v[j] == 0) goto outer;
36 }
37 perm.push_back(x);
38 outer:;
39 }
40 }

50 template <std::floating_point ScalarType, std::integral auto dim>
51 auto generateRefinedKnots(const std::array<std::vector<ScalarType>, dim>& knotSpans, const int dir,
52 const int refinementLevel) {
53 const int newKnotsSizeForEachSpan = Dune::power(2, refinementLevel);
54 auto unique_Knots = knotSpans;
55 std::vector<double> additionalKnots;
56 auto& unique_KnotPerDim = unique_Knots[dir];
57 unique_KnotPerDim.erase(std::ranges::begin(std::ranges::unique(unique_KnotPerDim)), std::end(unique_KnotPerDim));
58
59 for (int i = 0; i < unique_KnotPerDim.size() - 1; ++i) {
60 const double spanLength = unique_KnotPerDim[i + 1] - unique_KnotPerDim[i];
61 const double increment = spanLength / newKnotsSizeForEachSpan;
62 for (int j = 1; j < newKnotsSizeForEachSpan; ++j) {
63 additionalKnots.emplace_back(unique_KnotPerDim[i] + increment * j);
64 }
65 }
66 return additionalKnots;
67 }

83 std::array<int, dim> subNetStart, const std::array<int, dim>& degree,

179 static auto basisFunctions(const Dune::FieldVector<ScalarType, dim>& u,
180 const std::array<std::vector<ScalarType>, dim>& knots,
181 const std::array<int, dim>& degree, const MultiDimensionNet<dim, ScalarType>& weights,
182 std::array<int, dim> spIndex) {
183 std::array<DynamicVectorType, (size_t)dim> bSplines;
184
185 for (std::size_t i = 0; i < dim; ++i)
186 bSplines[i] = BsplineBasis<ScalarType>::basisFunctions(u[i], knots[i], degree[i], spIndex[i]);
187
188 auto Nnet = MultiDimensionNet<dim, ScalarType>(bSplines);
189 // TODO Alex this should also be cached when called with a localView
190 const auto subNetWeights = netOfSpan(spIndex, degree, weights);
191
192 const ScalarType invSumWeight = dot(Nnet, subNetWeights);
193 Nnet *= subNetWeights;
194 Nnet /= invSumWeight;
195 return Nnet;
196 }

209 static auto basisFunctionDerivatives(const Dune::FieldVector<ScalarType, dim>& u,
210 const std::array<std::vector<ScalarType>, dim>& knots,
211 const std::array<int, dim>& degree,
212 const MultiDimensionNet<dim, double>& weights, const int derivativeOrder,
213 const bool triangleDerivatives = false,
214 std::optional<std::array<int, dim>> spIndex = std::nullopt) {
215 std::array<DynamicMatrixType, dim> bSplineDerivatives;
216 for (int i = 0; i < dim; ++i)
217 bSplineDerivatives[i] = BsplineBasis<ScalarType>::basisFunctionDerivatives(
218 u[i], knots[i], degree[i], derivativeOrder,
219 (spIndex ? std::optional<int>(spIndex.value()[i]) : std::nullopt));
220
221 std::array<std::vector<ScalarType>, dim> dimArrayOfVectors;
222 FieldVector<int, dim> dimSize(derivativeOrder + 1);
223 MultiDimensionNet<dim, MultiDimensionNet<dim, ScalarType>> netOfDerivativeNets(dimSize);
224
225 for (int j = 0; auto& derivNet : netOfDerivativeNets.directGetAll()) {
226 auto multiIndex = netOfDerivativeNets.directToMultiIndex(j++);
227 for (int i = 0; i < multiIndex.size(); ++i)
228 dimArrayOfVectors[i] = bSplineDerivatives[i][multiIndex[i]].container();
229 derivNet = MultiDimensionNet<dim, ScalarType>(dimArrayOfVectors);
230 }
231
232 MultiDimensionNet<dim, MultiDimensionNet<dim, ScalarType>> R = netOfDerivativeNets;
233 MultiDimensionNet<dim, ScalarType> netsOfWeightfunctions(dimSize);
234 auto subNetStart = spIndex ? spIndex.value() : findSpan(degree, u, knots);
235 const auto subNetWeights = netOfSpan(subNetStart, degree, weights);
236
237 for (int j = 0; j < R.size(); ++j) {
238 R.directGet(j) *= subNetWeights;
239 netsOfWeightfunctions.directGet(j) = dot(netOfDerivativeNets.directGet(j), subNetWeights);
240 }
241
242 std::vector<FieldVector<int, dim>> perms;
243 for (int j = 0; j < R.size(); ++j) {
244 const auto derivOrders = R.template directToMultiIndex<FieldVector<int, dim>>(j);
245 if (triangleDerivatives)
246 if (std::accumulate(derivOrders.begin(), derivOrders.end(), derivativeOrder)) continue;
247 Impl::createPartialSubDerivativPermutations(derivOrders, perms);
248
249 for (const auto& perm : perms) {
250 const MultiDimensionNetIndex<dim> kNet(perm + FieldVector<int, dim>(1));
251 auto startMultiIndex = perm;
252 std::ranges::transform(startMultiIndex, startMultiIndex.begin(), [](const auto& v) { return (v != 0); });
253 for (int kk = kNet.index(startMultiIndex); kk < kNet.size(); ++kk) {
254 const auto multik = kNet.template directToMultiIndex<FieldVector<int, dim>>(kk);
255 const ScalarType fac = (Impl::binom(perm, multik) * netsOfWeightfunctions.get(multik));
256 const auto& Rdirect_d = R.get(derivOrders - multik);
257 auto& Rdirect = R.directGet(j);
258 for (int i = 0; i < R.directGet(j).size(); ++i)
259 Rdirect.directGet(i) -= fac * Rdirect_d.directGet(i); // generalized Piegl Tiller (4.20)
260 }
261 }
262 R.directGet(j) /= netsOfWeightfunctions.directGet(0);
263 }
264 return R;
265 }

282 template <std::integral auto dim, std::integral auto dimworld, typename ScalarType_>
283 auto degreeElevate(const NURBSPatchData<dim, dimworld, ScalarType_>& oldData, const int refinementDirection,
284 const int elevationFactor) {
285 using ScalarType = ScalarType_;
286 using DynamicMatrixType = Dune::DynamicMatrix<ScalarType>;
287 using NURBSPatchData = NURBSPatchData<dim, dimworld, ScalarType_>;
288 using ControlPointNetType = typename NURBSPatchData::ControlPointNetType;
289 using ControlPointType = typename NURBSPatchData::ControlPointType;
290 const int t = elevationFactor;
291 const int p = oldData.degree[refinementDirection];
292 const int m = oldData.controlPoints.strideSizes()[refinementDirection] + p;
293 const int ph = p + t;
294 const int ph2 = ph / 2;
295 /* Compute Bezier degree elevation coefficients */
296 DynamicMatrixType bezalfs(p + t + 1, p + 1);
297 bezalfs[0][0] = bezalfs[ph][p] = 1.0;
298 for (int i = 1; i <= ph2; ++i) {
299 const ScalarType inv = 1.0 / Dune::binomial(ph, i);
300 for (int j = std::max(0, i - t); j <= std::min(p, i); ++j)
301 bezalfs[i][j] = inv * Dune::binomial(p, j) * Dune::binomial(t, i - j);
302 }
303 for (int i = ph2 + 1; i <= ph - 1; ++i)
304 for (int j = std::max(0, i - t); j <= std::min(p, i); ++j)
305 bezalfs[i][j] = bezalfs[ph - i][p - j];
306
307 ControlPointNetType newCPv(oldData.controlPoints.strideSizes());
308
309 auto oldCPs = oldData.controlPoints;
310
311 auto scaleCPWithW = [](const auto& cp) -> ControlPointType { return {.p = cp.w * cp.p, .w = cp.w}; };
312 std::ranges::transform(oldCPs.directGetAll(), oldCPs.directGetAll().begin(), scaleCPWithW);
313
314 std::vector<ControlPointType> bpts(p + 1), ebpts(ph + 1), nextbpts(p - 1);
315
316 NURBSPatchData newData;
317 newData.degree = oldData.degree;
318 newData.degree[refinementDirection] += t;
319 const auto& U = oldData.knotSpans[refinementDirection];
320 double ua = U[0];
321 std::vector<ScalarType> Uh;
322 for (int j = 0, i = 0; i < U.size(); ++i) // insert knot t times for each unique knot
323 if (j < U.size()) {
324 do {
325 Uh.push_back(U[j]);
326 ++j;
327 } while (j != U.size() && Dune::FloatCmp::eq(U[j - 1], U[j]));
328 Uh.insert(Uh.end(), t, U[j - 1]);
329 if (j == U.size()) break;
330 }
331
332 for (int i = 0; i < dim; ++i)
333 if (i == refinementDirection)
334 newData.knotSpans[i] = Uh;
335 else
336 newData.knotSpans[i] = oldData.knotSpans[i];
337
338 auto dimSize = oldData.controlPoints.strideSizes();
339 dimSize[refinementDirection] = Uh.size() - ph - 1;
340 newData.controlPoints = MultiDimensionNet<dim, ControlPointType>(dimSize);
341 auto& newCPs = newData.controlPoints;
342
343 int totalCurvesInPatch = 1;
344 std::array<int, dim - 1> indicesOther;
345 for (int counter = 0, i = 0; i < dim; ++i)
346 if (i != refinementDirection) {
347 totalCurvesInPatch *= dimSize[i];
348 indicesOther[counter++] = dimSize[i];
349 }
350
351 auto newAndOldCurve = [&refinementDirection, indicesOther, &oldCPs, &newCPs](auto i) {
352 const auto mI
353 = MultiDimensionNet<dim - 1, int>::template directToMultiIndex<std::array<int, dim - 1>>(indicesOther, i);
354 std::array<int, dim> multiIndex{};
355 for (int c = 0, j = 0; j < dim; ++j)
356 if (j != refinementDirection) multiIndex[j] = mI[c++];
357
358 auto oldCurve = [&oldCPs, &refinementDirection, multiIndex](int i) mutable {
359 multiIndex[refinementDirection] = i;
360 return oldCPs.get(multiIndex);
361 };
362
363 auto newCurve = [&newCPs, &refinementDirection, multiIndex ](int i) mutable -> auto& {
364 multiIndex[refinementDirection] = i;
365 return newCPs.get(multiIndex);
366 };
367 return std::make_pair(oldCurve, newCurve);
368 };
369
370 for (auto [oldCurve, newCurve] :
371 std::ranges::iota_view(0, totalCurvesInPatch) | std::views::transform(newAndOldCurve)) {
372 ua = U[0];
373 int kind = ph + 1;
374 int r = -1;
375 int a = p;
376 int b = p + 1;
377 int cind = 1;
378
379 newCurve(0) = oldCurve(0);
380 for (int i = 0; i <= p; ++i)
381 bpts[i] = oldCurve(i);
382
383 std::vector<ScalarType> alphas(p - 1);
384 while (b < m) {
385 const int bold = b;
386 while (b < m && Dune::FloatCmp::eq(U[b], U[b + 1]))
387 ++b;
388 const int mul = b - bold + 1;
389 const auto ub = U[b];
390 const int oldr = r;
391 r = p - mul;
392 const int lbz = (oldr > 0) ? ((oldr + 2) / 2) : 1; // Insert knot u(b) r times
393 const int rbz = (r > 0) ? (ph - (r + 1) / 2) : ph;
394 if (r > 0) // Insert knot to get Bezier segment
395 {
396 const auto numer = ub - ua;
397 for (int k = p; k > mul; --k)
398 alphas[k - mul - 1] = numer / (U[a + k] - ua);
399 for (int j = 1; j <= r; ++j) {
400 const int s = mul + j;
401 for (int k = p; k >= s; --k)
402 bpts[k] = alphas[k - s] * bpts[k] + (1.0 - alphas[k - s]) * bpts[k - 1];
403 nextbpts[r - j] = bpts[p];
404 }
405 } // End of insert knot
406
407 for (int i = lbz; i <= ph; ++i) { // Degree elevate Bezier
408 ebpts[i].setZero();
409 for (int j = std::max(0, i - t); j <= std::min(p, i); ++j)
410 ebpts[i] += bezalfs[i][j] * bpts[j];
411 } // End of degree elevating Bezier
412 if (oldr > 1) { // Must remove knot u = U[a] oldr times
413 int first = kind - 2;
414 int last = kind;
415 const ScalarType den = ub - ua;
416 const ScalarType bet = (ub - Uh[kind - 1]) / den;
417 for (int tr = 1; tr < oldr; ++tr) { // Knot removal loop
418 int i = first;
419 int j = last;
420 int kj = j - kind + 1;
421 while (j - i > tr) {
422 if (i < cind) {
423 const ScalarType alf = (ub - Uh[i]) / (ua - Uh[i]);
424 newCurve(i) = alf * newCurve(i) + (1.0 - alf) * newCurve(i - 1);
425 }
426 if (j >= lbz) {
427 const auto factor = (j - tr <= kind - ph + oldr) ? (ub - Uh[j - tr]) / den : bet;
428 ebpts[kj] = factor * ebpts[kj] + (1.0 - factor) * ebpts[kj + 1];
429 }
430 ++i;
431 --j;
432 --kj;
433 }
434 --first;
435 ++last;
436 }
437 } // End of removing knot, u= U[a]
438 if (a != p) kind += ph - oldr;
439 for (int j = lbz; j <= rbz; ++j, ++cind) // Load ctrl pts into Qw
440 newCurve(cind) = ebpts[j];
441 if (b < m) { // Set up for next pass through loop
442 std::ranges::copy(nextbpts | std::views::take(r), bpts.begin());
443 for (int j = r; j <= p; ++j)
444 bpts[j] = oldCurve(b - p + j);
445 a = b;
446 ++b;
447 ua = ub;
448 }
449 } // End of while-loop (b<m)
450 }
451 std::ranges::transform(newData.controlPoints.directGetAll(), newData.controlPoints.directGetAll().begin(),
452 [](auto& cp) -> ControlPointType { return {.p = cp.p / cp.w, .w = cp.w}; });
453 return newData;
454 }

456 template <std::integral auto dim, std::integral auto dimworld, typename ScalarType>
457 auto knotRefinement(const NURBSPatchData<dim, dimworld, ScalarType>& oldData, const std::vector<double>& newKnots,
458 const int refinementDirection) {
459 assert(std::ranges::is_sorted(newKnots)
460 && "You passed a non-sorted vector of new knots. Sort it first or check if you passed the correct vector.");
461 using NurbsPatchData = NURBSPatchData<dim, dimworld, ScalarType>;
462 using ControlPoint = typename NurbsPatchData::ControlPointType;
463 using namespace std::ranges;
464 std::array<std::vector<double>, dim> newKnotsArray;
465 std::array<int, dim - 1> otherDirections;
466 for (int counter = 0, i = 0; i < dim; ++i) {
467 if (i == refinementDirection) continue;
468 newKnotsArray[i] = oldData.knotSpans[i];
469 otherDirections[counter++] = i;
470 }
471
472 typename NurbsPatchData::ControlPointNetType newCPv(oldData.controlPoints.strideSizes());
473
474 auto oldCPv = oldData.controlPoints;
475 auto oldKnotVec = oldData.knotSpans[refinementDirection];
476
477 auto& newKnotVec = newKnotsArray[refinementDirection];
478
479 newKnotVec.reserve(oldKnotVec.size() + newKnots.size());
480 merge(oldKnotVec, newKnots, std::back_inserter(newKnotVec));
481
482 const auto newKSize = newKnots.size();
483 std::array<int, dim> numberOfCPperDir = oldCPv.strideSizes();
484 numberOfCPperDir[refinementDirection] += newKSize;
485
486 newCPv.resize(numberOfCPperDir);
487
488 auto scaleCPWithW = [](const auto& cp) -> ControlPoint { return {.p = cp.w * cp.p, .w = cp.w}; };
489
490 const int p = oldData.degree[refinementDirection];
491 const int a = findSpan(p, newKnots.front(), oldKnotVec);
492 const int b = findSpan(p, newKnots.back(), oldKnotVec);
493
494 auto newSurfI = newCPv.hyperSurfBegin(otherDirections);
495 auto oldSurfI = oldCPv.hyperSurfBegin(otherDirections);
496
497 for (auto hSOld = oldSurfI, hSNew = newSurfI; hSOld != oldCPv.hyperSurfBegin(otherDirections) + (a - p + 1);
498 ++hSOld, ++hSNew)
499 std::ranges::transform(*hSOld, (*hSNew).begin(), scaleCPWithW);
500 for (auto hSOld = oldSurfI + b, hSNew = newSurfI + b; hSOld != oldCPv.hyperSurfEnd(otherDirections);
501 ++hSOld, ++hSNew)
502 std::ranges::transform(*hSOld, (*hSNew).begin(), scaleCPWithW);
503
504 newSurfI = newCPv.hyperSurfEnd(otherDirections) - 1;
505 oldSurfI = oldCPv.hyperSurfEnd(otherDirections) - 1;
506
507 auto currentNewKnot = newKnotVec.end();
508 auto currentOldKnot = oldKnotVec.end();
509
510 for (auto const& currentAdditionalKnot : reverse_view(newKnots)) {
511 while (currentAdditionalKnot <= *(currentOldKnot - 1)
512 && std::distance(oldKnotVec.begin(), currentOldKnot) > a + 1) {
513 std::ranges::transform(*oldSurfI, (*newSurfI).begin(), scaleCPWithW);
514 --currentNewKnot, --currentOldKnot;
515 --newSurfI, --oldSurfI;
516 }
517
518 std::ranges::copy(*(newSurfI + 1), (*newSurfI).begin());
519 ++newSurfI;
520 currentOldKnot -= p;
521 for ([[maybe_unused]] int _ : iota_view{0, p}) {
522 ++newSurfI;
523
524 auto alpha = *currentNewKnot - currentAdditionalKnot;
525 using std::abs;
526 if (abs(alpha) < 1e-7)
527 std::ranges::copy(*newSurfI, (*(newSurfI - 1)).begin());
528 else {
529 alpha /= (*currentNewKnot - *(currentOldKnot));
530 std::ranges::transform(*newSurfI, *(newSurfI - 1), (*(newSurfI - 1)).begin(),
531 [&alpha](auto& cp, auto& cpL) { return alpha * cpL + cp * (1.0 - alpha); });
532 }
533 ++currentOldKnot, ++currentNewKnot;
534 }
535 currentNewKnot -= p + 1;
536 newSurfI -= p + 2;
537 }
538
539 std::ranges::transform(newCPv.directGetAll(), newCPv.directGetAll().begin(),
540 [](auto& cp) -> ControlPoint { return {.p = cp.p / cp.w, .w = cp.w}; });
541
542 return NURBSPatchData<dim, dimworld, ScalarType>(newKnotsArray, newCPv, oldData.degree);
543 }
