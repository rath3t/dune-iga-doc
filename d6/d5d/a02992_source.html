320 void globalRefine(int refCount) {
321 for (int i = 0; i < refCount; ++i) {
322 const auto& finestPatchData = patchGeometries_.back().patchData();
323 auto newfinestPatchData = finestPatchData;
324
325 auto& olduniqueKnotVector = patchGeometries_.back().uniqueKnotVector();
326 std::array<std::vector<ctype>, dim> newUniqueKnotVecs;
327 for (int refDirection = 0; refDirection < dim; ++refDirection) {
328 auto additionalKnots = Splines::generateRefinedKnots(finestPatchData.knotSpans, refDirection, 1);
329 newfinestPatchData = Splines::knotRefinement<dim>(newfinestPatchData, additionalKnots, refDirection);
330 auto& newKnotVec = newUniqueKnotVecs[refDirection];
331
332 // compute new unique Knot vectors from refinement
333 newKnotVec.resize(olduniqueKnotVector[refDirection].size() + additionalKnots.size());
334 std::ranges::merge(olduniqueKnotVector[refDirection], additionalKnots, std::begin(newKnotVec));
335 }
336
337 patchGeometries_.emplace_back(newfinestPatchData, newUniqueKnotVecs);
338 patchGeometriesUnElevated.emplace_back(patchGeometries_.back());
339 }
340
341 trimmer_.parameterSpaceGrid().globalRefine(refCount);
342 setIndices();
343 }

360 void globalRefineInDirection(const std::array<int, dim>& refines) {
361 const auto& finestPatchData = patchGeometries_.back().patchData();
362 auto newfinestPatchData = finestPatchData;
363 for (int dir = 0; auto refinesInDirection : refines) {
364 if (refinesInDirection == 0) {
365 ++dir;
366 continue;
367 }
368 auto additionalKnots = Splines::generateRefinedKnots(finestPatchData.knotSpans, dir, refinesInDirection);
369 newfinestPatchData = Splines::knotRefinement<dim>(newfinestPatchData, additionalKnots, dir);
370 ++dir;
371 }
372 auto newGrid = PatchGrid(newfinestPatchData);
373 *this = std::move(newGrid);
374 }

381 void degreeElevate(const std::array<int, dim>& elevationFactors, int lvl) {
382 if (lvl > maxLevel() and lvl >= 0) DUNE_THROW(Dune : RangeError, "This level does not exist");
383 auto& patchData = patchGeometries_[lvl].patchData();
384 patchGeometriesUnElevated[lvl] = GeometryKernel::NURBSPatch<dim, dimworld, ctype>(patchData);
385 for (int dir = 0; auto elevatesInDirection : elevationFactors) {
386 if (elevatesInDirection == 0) {
387 ++dir;
388 continue;
389 }
390 patchData = Splines::degreeElevate(patchData, dir, elevatesInDirection);
391 ++dir;
392 }
393 patchGeometries_[lvl] = GeometryKernel::NURBSPatch<dim, dimworld, ctype>(patchData);
394 }
