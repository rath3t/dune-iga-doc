44 enum class HostOrTrimmed{host,trimmed};
45 HostOrTrimmed host_or_trimmed;
46 HostIdType id;
47
48 friend std::ostream& operator<<(std::ostream& stream, const IdType& id) {
49 stream << "Type: " << std::string(id.host_or_trimmed==HostOrTrimmed::host? "Host": "Trimmed") << ", Key: " << id.id << "\n";
50 return stream;
51 }

54 template<typename HostIdType>
55 bool operator<(const IdType<HostIdType>& lhs, const IdType<HostIdType>& rhs) {
56
57 if(lhs.host_or_trimmed==rhs.host_or_trimmed)
58 return lhs.id < rhs.id;
59 else if(lhs.host_or_trimmed<rhs.host_or_trimmed)
60 return true;
61 else
62 return false;
