59 void finish() {
60 // Sort the points in counter clockwise manner such that the first point is in the lower left
61 auto getPoint = [](auto&& vertexVariant) { return vertexVariant.pt; };
62
63 const auto minVertex
64 = *std::ranges::min_element(vertices_, [&](const VertexVariant& aV, const VertexVariant& bV) {
65 const auto a = std::visit(getPoint, aV);
66 const auto b = std::visit(getPoint, bV);
67 return (a.y < b.y) || (a.y == b.y && a.x < b.x);
68 });
69
70 auto polarAngle
71 = [](const auto& p, const auto& reference) -> double { return atan2(p.y - reference.y, p.x - reference.x); };
72
73 // Sort the points based on polar angle with respect to the reference point
74 auto comparePoints
75 = [&](const VertexVariant& aV, const VertexVariant& bV, const VertexVariant& referenceV) -> bool {
76 const auto a = std::visit(getPoint, aV);
77 const auto b = std::visit(getPoint, bV);
78 const auto reference = std::visit(getPoint, referenceV);
79 const double angleA = polarAngle(a, reference);
80 const double angleB = polarAngle(b, reference);
81
82 if (angleA != angleB) return angleA < angleB;
83
84 // If two points have the same angle, sort based on distance to the reference point
85 return (a.x - reference.x) * (a.x - reference.x) + (a.y - reference.y) * (a.y - reference.y)
86 < (b.x - reference.x) * (b.x - reference.x) + (b.y - reference.y) * (b.y - reference.y);
87 };
88
89 // Sort counter-clockwise
90 std::ranges::sort(vertices_, [&](const VertexVariant& aV, const VertexVariant& bV) {
91 return comparePoints(aV, bV, minVertex);
92 });
93
94 // Remove duplicate entries for NewVertices, begin at the first host index
95 const auto it = std::ranges::find_if(
96 vertices_, [](const VertexVariant& vV) { return std::holds_alternative<HostVertex>(vV); });
97 if (it == vertices_.end()) DUNE_THROW(Dune::NotImplemented, "Algorithm needs at least one HostVertex to work");
98
99 std::ranges::rotate(vertices_, it);
100 for (const auto& vV : vertices_) {
101 }
102 }

110 inline int giveEdgeIdx(const std::size_t e1, const std::size_t e2) {
111 assert(e1 < 4 and e2 < 4);
112 if ((e1 == 0 and e2 == 1) or (e1 == 1 and e2 == 0)) return 0;
113 if ((e1 == 1 and e2 == 2) or (e1 == 2 and e2 == 1)) return 1;
114 if ((e1 == 2 and e2 == 3) or (e1 == 3 and e2 == 2)) return 2;
115 if ((e1 == 3 and e2 == 0) or (e1 == 0 and e2 == 3)) return 3;
116 assert(false);

121 auto clipElementRectangle(const auto& element, const auto& patchTrimData)
122 -> std::tuple<ElementTrimFlag, ClippingResult> {
123 using namespace Clipper2Lib;
124
125 auto eleGeo = element.geometry();
126 constexpr std::array vIdxMapping = {0u, 1u, 3u, 2u};
127
128 PathD eleRect;
129 for (const auto i : std::views::iota(0u, 4u)) {
130 auto corner = eleGeo.corner(vIdxMapping[i]);
131 eleRect.emplace_back(corner[0], corner[1], i);
132 }
133
134 PathsD trimmingCurves;
135 PathD tempPath;
136 constexpr int N = 10;
137 // @todo store param value of sampled points on trimmingCurve
138 for (auto loop : patchTrimData.loops()) {
139 tempPath.clear();
140 for (int i = 100; auto& curve : loop.curves()) {
141 for (auto v : Utilities::linspace(curve.domain()[0], N)) {
142 auto fV = curve.global({v});
143 tempPath.emplace_back(fV[0], fV[1], i++);
144 }
145 // Add an additional point just outside of the element
146 auto localLastPoint = curve.domain()[0][1];
147 auto lastPoint = curve.global(localLastPoint);
148
149 auto scale = element.geometry().volume() / 10;
150
151 auto dx = curve.jacobian({localLastPoint}) * scale;
152 tempPath.emplace_back(lastPoint[0] + dx[0], lastPoint[1] + dx[1], i++);
153
154 i += 99 - N;
155 }
156 trimmingCurves.push_back(tempPath);
157 }
158
159 // todo should this be hardcoded here?
160 std::array dirs{FieldVector<double, 2>{1.0, 0.0}, FieldVector<double, 2>{0, 1}, FieldVector<double, 2>{-1, 0},
161 FieldVector<double, 2>{0, -1}};
162
163 // @todo check z value if same as eleRect -> full
164 auto isFullElement = [&](const auto& clippedEdges) { return FloatCmp::eq(Area(clippedEdges), Area(eleRect)); };
165
166 // First determine if element is trimmed
167 // @todo try this with clipper for non-boundary elements with ClipperD
168 const auto intersectResult = Intersect({eleRect}, trimmingCurves, FillRule::NonZero, 2);
169
170 if (intersectResult.empty()) return std::make_tuple(ElementTrimFlag::empty, ClippingResult{eleRect});
171 if (isFullElement(intersectResult.front())) {
172 matplot::rectangle(eleRect[0].x, eleRect[1].y, eleRect[1].x - eleRect[0].x, eleRect[3].y - eleRect[0].y);
173 return std::make_tuple(ElementTrimFlag::full, ClippingResult{eleRect});
174 }
175
176 // Element is trimmed
177 ClipperD clipper(8);
178 ClippingResult result(eleRect);
179
180 auto checkParallel = [&](const auto& curve, const int edgeIndex) -> bool {
181 return curve.degree().front() == 1
182 and std::get<0>(findIntersectionLinearCurveAndLine(curve, eleGeo.corner(vIdxMapping[edgeIndex]),
183 dirs[edgeIndex], {curve.domainMidPoint()[0], 0.5}))
184 == FindIntersectionCurveAndLineResult::linesParallel;
185 };

187 clipper.SetZCallback([&](const PointD& e1bot, const PointD& e1top, const PointD& e2bot, const PointD& e2top,
188 const PointD& pt) {
189 if (not checkParallel(patchTrimData.loops().front().curves()[toCurveIdx(e2bot.z)], giveEdgeIdx(e1bot.z, e1top.z)))
190 result.addNewVertex(giveEdgeIdx(e1bot.z, e1top.z), pt, e2bot.z);
191 });
192
193 PathsD resultClosedPaths{};
194 PathsD resultOpenPaths{};
195
196 for (const auto i : std::views::iota(0, 4)) {
197 // std::cout << "Edge " << i << std::endl;
198
199 clipper.Clear();
200 clipper.AddClip(trimmingCurves);
201
202 clipper.AddOpenSubject(PathsD{{eleRect[i], eleRect[nextEntityIdx(i, 1)]}});
203 clipper.Execute(ClipType::Intersection, FillRule::NonZero, resultClosedPaths, resultOpenPaths);
204 // We receive here all intersection points but not the sampled curve points
205 // resultClosedPaths should be empty
206 assert(resultClosedPaths.empty());
207 assert(resultOpenPaths.size() <= 1);
208
209 if (not resultOpenPaths.empty())
210 for (const auto& p : resultOpenPaths.front()) {
211 if (p.z < 4)
212 result.addOriginalVertex(p);
213 else {
214 // Check if this is neeeded
215 std::cout << "maybe we need thhis point\n";
216 }
217 }
218 }

220 auto isPointInTrimmingCurves = [&](const auto& pt1) -> bool {
221 return std::ranges::any_of(trimmingCurves, [&](const auto& curve) {
222 auto it = std::ranges::find_if(
223 curve, [&](const auto& pt2) { return FloatCmp::eq(pt1.x, pt2.x) and FloatCmp::eq(pt1.y, pt2.y); });
224 return it != curve.end();
225 });

228 // check if any element vertex is a starting point of one of the trimming curves as the algorihtm cannot find them
229 // @todo only for boundary elements
230 for (const auto& cP : eleRect)

233 result.finish();
