34 template <typename Geometry>
35 typename Geometry::LocalCoordinate findClosestParameterSpaceCoordinate(
36 const Geometry& geo, const typename Geometry::GlobalCoordinate& global) {
37 static constexpr int mydimension = Geometry::mydimension;
38 static constexpr int worlddimension = Geometry::worlddimension;
39 using ctype = typename Geometry::ctype;
40 using LocalCoordinate = typename Geometry::LocalCoordinate;
41 using GlobalCoordinate = typename Geometry::GlobalCoordinate;
42 using MatrixHelper = typename Geometry::MatrixHelper;
43 if constexpr (mydimension == 0)
44 return {};
45 else if constexpr (mydimension != worlddimension) {
46 auto [u, Ru, fu, gap] = closestPointProjectionByTrustRegion(geo, global);
47 return u;
48 } else {
49 const ctype tolerance = ctype(16) * std::numeric_limits<ctype>::epsilon();
50 LocalCoordinate x = geo.domainMidPoint();
51
52 LocalCoordinate dx{};
53 do { // from multilinearGeometry
54 const GlobalCoordinate dglobal = geo.global(x) - global;
55 MatrixHelper::template xTRightInvA<mydimension, worlddimension>(geo.jacobianTransposed(x), dglobal, dx);
56 const bool invertible
57 = MatrixHelper::template xTRightInvA<mydimension, worlddimension>(geo.jacobianTransposed(x), dglobal, dx);
58
59 if (!invertible) return LocalCoordinate(std::numeric_limits<ctype>::max());
60 x -= dx;
61 // if local is outside of maximum knot vector span bound, thus we clamp it to it and return
62 // clamp result into boundaries
63 if (Dune::IGANEW::Utilities::clampToBoundaryAndCheckIfIsAtAllBoundaries(x, geo.domain())) {
64 break;
65 }
66
67 } while (dx.two_norm2() > tolerance);
68 return x;
69 }
70 }

91 template <typename LocalCoordinate, typename NurbsLocalView, typename ControlPointCoordinateNetType>
92 auto hessian(const LocalCoordinate& local, const NurbsLocalView& nurbsLocalView,
93 const ControlPointCoordinateNetType& localControlPointNet) {
94 static constexpr int mydimension = NurbsLocalView::dimension;
95 static constexpr int worlddimension = ControlPointCoordinateNetType::value_type::dimension;
96 using ctype = typename ControlPointCoordinateNetType::value_type::value_type;
97
98 FieldMatrix<ctype, mydimension*(mydimension + 1) / 2, worlddimension> H;
99
100 const auto basisFunctionDerivatives = nurbsLocalView.basisFunctionDerivatives(local, 2);
101
102 for (int dir = 0; dir < mydimension; ++dir) {
103 std::array<unsigned int, mydimension> ithVec{};
104 ithVec[dir] = 2; // second derivative in dir direction
105 H[dir] = dot(basisFunctionDerivatives.get(ithVec), localControlPointNet);
106 }
107 if constexpr (mydimension > 1) {
108 std::array<int, mydimension> mixeDerivs;
109 std::ranges::fill(mixeDerivs, 0); // first mixed derivatives
110 if constexpr (mydimension == 2)
111 for (int dir = 0; dir < mydimension; ++dir) {
112 mixeDerivs[dir] = 1;
113 }
114 else
115 std::ranges::fill_n(mixeDerivs.begin() + 1, 2, 1); // first mixed derivatives
116 int mixedDireCounter = mydimension;
117 do {
118 H[mixedDireCounter++] = dot(basisFunctionDerivatives.get(mixeDerivs), localControlPointNet);
119 if constexpr (mydimension == 2) break;
120 } while (std::ranges::next_permutation(mixeDerivs, std::less()).found);
121 }
122 return H;
123 }

143 static constexpr int mydimension = NurbsLocalView::dimension;
144 static constexpr int worlddimension = ControlPointCoordinateNetType::value_type::dimension;
145 using ctype = typename ControlPointCoordinateNetType::value_type::value_type;
146

148

150

152 std::array<unsigned int, mydimension> ithVec{};
