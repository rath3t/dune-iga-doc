23 // At the moment only gridDim und worldDim == 2 or 3 supported
24 template <int gridDim, int worldDim, typename ScalarType = double>
25 requires(gridDim == 2) && (worldDim == 2 || worldDim == 3) && (gridDim <= worldDim) class IbraReader {
26 public:
27 using Grid = Dune::IGA::NURBSGrid<gridDim, worldDim, ScalarType>;
28 using ControlPoint = Dune::IGA::NURBSPatchData<gridDim, worldDim>::ControlPointType;
29 using PatchData = Dune::IGA::NURBSPatchData<gridDim, worldDim>;
30 using ControlPointNetType = Dune::IGA::MultiDimensionalNet<gridDim, ControlPoint>;
31
32 static std::unique_ptr<Grid> read(const std::string& fileName, const bool trim = true,
33 std::array<int, 2> elevateDegree = {0, 0},
34 std::array<int, 2> preKnotRefine = {0, 0},
35 std::array<int, 2> postKnotRefine = {0, 0}) {
36 std::ifstream ibraInputFile;
37 ibraInputFile.open(fileName);
38 return read(ibraInputFile, trim, elevateDegree, preKnotRefine, postKnotRefine);
39 }

41 template <typename InputStringType>
42 requires(not std::convertible_to<
43 std::string, InputStringType> and not std::convertible_to<InputStringType, const char*>) static std::
44 unique_ptr<Grid> read(InputStringType& ibraInputFile, const bool trim = true,
45 std::array<int, 2> elevateDegree = {0, 0}, std::array<int, 2> preKnotRefine = {0, 0},
46 std::array<int, 2> postKnotRefine = {0, 0}) {
47 using json = nlohmann::json;
48
49 // Result
50 std::vector<Ibra::Surface<worldDim>> surfaces;
51 std::vector<Ibra::Curve2D> curves2D;
52
53 std::vector<Ibra::BrepRepresentation> brepRepresentations;
54 std::vector<Ibra::BrepLoopRepresentation> brepLoopRepresentations;
55 std::vector<Ibra::BrepTrimRepresentation> brepTrimRepresentations;
56
57 json ibraJson;
58 try {
59 ibraJson = json::parse(ibraInputFile);
60
61 for (auto& j : ibraJson) {
62 auto geo = j.get<Ibra::IbraBase>();
63
64 switch (geo.type) {
65 case Ibra::Type::NurbsSurfaceGeometry3D:
66 surfaces.push_back(j.get<Ibra::Surface<worldDim>>());
67 break;
68 case Ibra::Type::NurbsCurveGeometry2D:
69 curves2D.push_back(j.get<Ibra::Curve2D>());
70 break;
71 case Ibra::Type::BrepType:
72 brepRepresentations.push_back(j.get<Ibra::BrepRepresentation>());
73 break;
74 case Ibra::Type::BrepLoopType:
75 brepLoopRepresentations.push_back(j.get<Ibra::BrepLoopRepresentation>());
76 break;
77 case Ibra::Type::BrepTrimType:
78 brepTrimRepresentations.push_back(j.get<Ibra::BrepTrimRepresentation>());
79 break;
80 default:
81 // Do nothing -- keep the compiler happy
82 break;
83 }
84 }
85 } catch (json::parse_error& ex) {
86 DUNE_THROW(Dune::IOError, "Error in reading input stream: "
87 << ", parse error at byte " << ex.byte << " What: " << ex.what());
88 }
89
90 // Make Connections
91 auto brepRepresentation = brepRepresentations[0];
92
93 Ibra::Brep brep{brepRepresentation, curves2D, surfaces, brepLoopRepresentations, brepTrimRepresentations};
94 // Each surface in a brep is one Patch, as for now only one brep is supported in NURBSGrid
95 assert(brep.surfaces.size() == 1);
96
97 std::cout << "IbraReader successfully read in 1 patch with " << brep.trims.size() << " edges." << std::endl;
98
99 // Reader has done its job, now NURBSGrid can be constructed
100
101 Ibra::Surface<worldDim> _surface = brep.surfaces[0];
102 const std::array<std::vector<double>, gridDim> knotSpans = _surface.compileKnotVectors();
103 const std::vector<std::vector<ControlPoint>> controlPoints = _surface.transformControlPoints();
104 std::array<int, gridDim> dimsize = _surface.n_controlPoints;
105
106 auto controlNet = ControlPointNetType(dimsize, controlPoints);
107 PatchData _patchData{knotSpans, controlNet, _surface.degree};
108 // Optional Pre Refinement of knots
109 for (int i = 0; i < gridDim; ++i) {
110 if (preKnotRefine[i] > 0) {
111 auto additionalKnots = generateRefinedKnots(knotSpans, i, preKnotRefine[i]);
112 _patchData = knotRefinement<gridDim>(_patchData, additionalKnots, i);
113 }
114 }
115
116 // Optional Degree Elevate
117 for (int i = 0; i < gridDim; ++i)
118 if (elevateDegree[i] > 0) _patchData = degreeElevate(_patchData, i, elevateDegree[i]);
119
120 // Optional Post Refinement of knots
121 for (int i = 0; i < gridDim; ++i) {
122 if (postKnotRefine[i] > 0) {
123 auto additionalKnots = generateRefinedKnots(knotSpans, i, postKnotRefine[i]);
124 _patchData = knotRefinement<gridDim>(_patchData, additionalKnots, i);
125 }
126 }
127
128 // Make the trimData and pass them into the grid
129 // So the grid can figure out what to do with it
130 auto trimData = constructGlobalBoundaries(brep);
131
132 if (trim)
133 return std::make_unique<Grid>(_patchData, trimData);
134 else
135 return std::make_unique<Grid>(_patchData);
136 }
137
138 private:
139 static auto constructGlobalBoundaries(const Ibra::Brep<worldDim>& brep) -> std::shared_ptr<TrimData> {
140 auto data = std::make_shared<TrimData>();
141 std::vector<Boundary> boundaries;
142 std::vector<Ibra::BrepLoop> loops = brep.loops;
143 for (Ibra::BrepLoop& loop : loops) {
144 boundaries.clear();
145 for (Ibra::BrepTrim& trim : loop.trims) {
146 boundaries.emplace_back(trim);
147 }
148 data->addLoop(boundaries);
149 }
150
151 return data;
152 }
153 };
