34 void addLoop(const BoundaryLoop& loop) {
35 Clipper2Lib::PathD path;
36 for (idx_t i = loops_.empty() ? splitter_ : loops_.back().back().z + 1; const auto& curve : loop.curves()) {
37 if (curve.degree().front() == 1) {
38 auto p1 = curve.corner(0);
39 auto p2 = curve.corner(1);
40 path.emplace_back(p1[0], p1[1], i);
41 path.emplace_back(p2[0], p2[1], i + splitter_ - 1);
42 i += splitter_;
43 continue;
44 }
45 for (const auto v : Utilities::linspace(curve.domain()[0], splitter_)) {
46 auto fV = curve.global({v});
47 path.emplace_back(fV[0], fV[1], ++i);
48 }
49 }
50 loops_.push_back(path);
51 loopIndices_.push_back(loops_.back().back().z);
52 }

54 [[nodiscard]] auto getIndices(const idx_t val) const -> std::pair<size_t, size_t> {

109 // Determine points of the outer boundary that are not on the edges
110 auto isInsidePatch = [&](const FieldVector<ctype, 2>& pt) {
111 const bool isInsideU = FloatCmp::gt(pt[0], domainU.front()) and FloatCmp::lt(pt[0], domainU.back());
112 const bool isInsideV = FloatCmp::gt(pt[1], domainV.front()) and FloatCmp::lt(pt[1], domainV.back());
113
114 return isInsideU and isInsideV;
115 };

117 pointsInPatch_.push_back({});
118 for (size_t i = 0; const auto& curve : loops_.front().curves()) {
119 if (const auto pt = curve.corner(1); isInsidePatch(pt)) {
