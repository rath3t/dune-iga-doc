7 inline int giveEdgeIdx(const std::size_t e1, const std::size_t e2) {

28 auto isPointOnLine(const auto& pt, const auto& p1, const auto& p2) -> bool {
29 // Check if the points are collinear using the slope formula
30 auto x = pt[0];
31 auto y = pt[1];
32 auto x1 = p1[0];
33 auto y1 = p1[1];
34 auto x2 = p2[0];
35 auto y2 = p2[1];
36
37 // Check if the line is horizontal
38 if (std::abs(y1 - y2) < 1e-6) {
39 return std::abs(y - y1) < 1e-6 && ((x >= x1 && x <= x2) || (x >= x2 && x <= x1));
40 }
41 // Check if the line is vertical
42 if (std::abs(x1 - x2) < 1e-6) {
43 return std::abs(x - x1) < 1e-6 && ((y >= y1 && y <= y2) || (y >= y2 && y <= y1));
44 }
45
46 // Check if the given point lies on the line
47 const double slope = (y2 - y1) / (x2 - x1);
48 return std::abs((y - y1) - slope * (x - x1)) < 1e-6;
49 }

108 void finish() {
109 // Sort the points in counter clockwise manner such that the first point is in the lower left
110 auto getPoint = [](auto&& vertexVariant) { return vertexVariant.pt; };
111
112 const auto minVertex
113 = *std::ranges::min_element(vertices_, [&](const VertexVariant& aV, const VertexVariant& bV) {
114 const auto a = std::visit(getPoint, aV);
115 const auto b = std::visit(getPoint, bV);
116 return (a.y < b.y) || (a.y == b.y && a.x < b.x);
117 });
118
119 auto polarAngle
120 = [](const auto& p, const auto& reference) -> double { return atan2(p.y - reference.y, p.x - reference.x); };
121
122 // Sort the points based on polar angle with respect to the reference point
123 auto comparePoints
124 = [&](const VertexVariant& aV, const VertexVariant& bV, const VertexVariant& referenceV) -> bool {
125 const auto a = std::visit(getPoint, aV);
126 const auto b = std::visit(getPoint, bV);
127 const auto reference = std::visit(getPoint, referenceV);
128 const double angleA = polarAngle(a, reference);
129 const double angleB = polarAngle(b, reference);
130
131 if (angleA != angleB) return angleA < angleB;
132
133 // If two points have the same angle, sort based on distance to the reference point
134 return (a.x - reference.x) * (a.x - reference.x) + (a.y - reference.y) * (a.y - reference.y)
135 < (b.x - reference.x) * (b.x - reference.x) + (b.y - reference.y) * (b.y - reference.y);
136 };
137
138 // Sort counter-clockwise
139 std::ranges::sort(vertices_, [&](const VertexVariant& aV, const VertexVariant& bV) {
140 return comparePoints(aV, bV, minVertex);
141 });
142
143 const auto it = std::ranges::find_if(
144 vertices_, [](const VertexVariant& vV) { return std::holds_alternative<HostVertex>(vV); });
145 if (it != vertices_.end())
146 std::ranges::rotate(vertices_, it);
147 else
148 std::cout << "Warning, no HostVertex" << std::endl;
149 }
