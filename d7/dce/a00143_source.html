131 [[nodiscard]] double volume() const {
132 const auto rule = Dune::QuadratureRules<ctype, mydimension>::rule(
133 this->type(), mydimension * (*std::ranges::max_element(patchData_.degree)));
134 ctype vol = 0.0;
135 for (auto& gp : rule) {
136 vol += integrationElement(gp.position()) * gp.weight();
137 }
138 return vol;
139 }

145 [[nodiscard]] int corners() const { return 1 << mydimension; }

152 [[nodiscard]] GlobalCoordinate corner(int k) const {
153 LocalCoordinate localcorner;
154 for (size_t i = 0; i < mydimension; i++) {
155 localcorner[i] = (k & (1 << i)) ? uniqueKnotVector()[i].back() : uniqueKnotVector()[i].front();
156 }
157 return global(localcorner);
158 }

165 [[nodiscard]] FieldVector<ctype, worlddimension> global(const LocalCoordinate& u) const {

281 /* @brief Calculate NURBS and control point net for a given local coordinate. */
282 auto calculateNurbsAndControlPointNet(const LocalCoordinate& u) const {
283 auto subNetStart = Splines::findSpan(patchData_.degree, u, patchData_.knotSpans);
284 auto cpCoordinateNet = Splines::netOfSpan(subNetStart, patchData_.degree,
285 Splines::extractControlCoordinates(patchData_.controlPoints));
286 auto nurbsLocalView = nurbs_.localView();
287 nurbsLocalView.bind(subNetStart);
288 return std::make_tuple(nurbsLocalView, cpCoordinateNet, subNetStart);
289 }
290
291 /* @brief Calculate zero first and second derivatives of the position for a given local coordinate. */
292 static auto zeroFirstAndSecondDerivativeOfPositionImpl(const LocalCoordinate& u,
293 const NurbsLocalView& nurbsLocalView,
294 const ControlPointCoordinateNetType& localControlPointNet) {
295 return std::make_tuple(GeometryKernel::position(u, nurbsLocalView, localControlPointNet),
296 GeometryKernel::jacobianTransposed(u, nurbsLocalView, localControlPointNet),
297 GeometryKernel::hessian(u, nurbsLocalView, localControlPointNet));
298 }
