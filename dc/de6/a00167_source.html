13 template <int dim, int dimworld, typename ScalarType>

16 const PatchTrimData& patchTrimData) {
17 std::cout << "START " << std::endl;
18 auto geo = element.geometry();
19 std::array<FieldVector<double, 2>, 4> corners; // see dune book page 127 Figure 5.12
20 corners[0] = geo.corner(0);
21 corners[1] = geo.corner(1);
22 corners[2] = geo.corner(3);
23 corners[3] = geo.corner(2);
24
25 auto [flag, result] = Impl::clipElementRectangle(element, patchTrimData);
26
27 // Create ElementTrimData with exact intersection Points and a geometry representation of the element edges
28 ElementTrimData elementTrimData(flag);
29
30 if (flag != ElementTrimFlag::trimmed) return elementTrimData;
31
32 auto nextEntity = [&](const int i) { return (i + 1) % result.vertices_.size(); };
33 auto isNewVertex = [](const auto& vV) { return std::holds_alternative<Impl::ClippingResult::NewVertex>(vV); };
34 auto getPt = [](auto&& vV) { return vV.pt; };
35 auto getHostIdx = [](const auto& vV) { return std::get<Impl::ClippingResult::HostVertex>(vV).hostIdx; };
36 auto getEdgeIdx = [](const auto& vV) { return std::get<Impl::ClippingResult::NewVertex>(vV).onEdgeIdx; };
37 auto getTrimmingCurveZ
38 = [](const auto& vV) { return std::get<Impl::ClippingResult::NewVertex>(vV).trimmingCurveZ; };
39
40 // @todo could be done via Jacobian of the edges
41 std::array<FieldVector<ScalarType, dim>, 4> dirs{
42 FieldVector<ScalarType, dim>{1.0, 0.0}, FieldVector<ScalarType, dim>{0, 1}, FieldVector<ScalarType, dim>{-1, 0},
43 FieldVector<ScalarType, dim>{0, -1}};
44
45 auto approxSamePoint
46 = [](const Clipper2Lib::PointD& pt1, const Dune::FieldVector<ScalarType, dim>& pt2, const double prec) -> bool {
47 return FloatCmp::eq(pt1.x, pt2[0], prec) and FloatCmp::eq(pt1.y, pt2[1], prec);
48 };
49 auto getTrimmingCurveIdx
50 = [&](auto& vV) -> size_t { return static_cast<size_t>(std::floor((getTrimmingCurveZ(vV) / 100) - 1)); };

52 auto callFindIntersection
53 = [&](const size_t tcIdx, const int edgeIdx, auto ip) -> std::pair<double, FieldVector<ScalarType, dim>> {
54 // @todo gerneralize for more than one loop
55 auto curvePatchGeo = patchTrimData.loops().front().curves()[tcIdx];
56 auto pos = corners[edgeIdx];
57 auto dir = dirs[edgeIdx];
58 // @todo make better guess by maybe using z Val with lookup-table
59 auto guessTParam = FieldVector<ScalarType, dim>{curvePatchGeo.domainMidPoint()[0], 0.5};
60
61 auto [success, tParam, curvePoint] = findIntersectionCurveAndLine(curvePatchGeo, pos, dir, guessTParam);
62 if (success == FindIntersectionCurveAndLineResult::sucess) return std::make_pair(tParam[0], curvePoint);
63 if (success == FindIntersectionCurveAndLineResult::linesParallel) {
64 DUNE_THROW(Dune::GridError, "Couldn't find intersection Point, lines are parallel");
65 }
66
67 assert(approxSamePoint(ip, curvePoint, 1e-4) && "Intersection Point and Curve Point are not the same");
68
69 auto domain = curvePatchGeo.domain();
70 if (FloatCmp::eq(curvePoint, curvePatchGeo.global({domain[0].front()})))
71 return std::make_pair(domain[0][0], curvePoint);
72 if (FloatCmp::eq(curvePoint, curvePatchGeo.global({domain[0].back()})))
73 return std::make_pair(domain[0].back(), curvePoint);
74
75 DUNE_THROW(Dune::GridError, "Couldn't find intersection Point");
76 };

78 auto createHostGeometry = [&](auto& vertex1, auto& vertex2) -> TrimmingCurve {
79 const std::array<std::vector<double>, 1> knotSpans = {{{0, 0, 1, 1}}};
80
81 const std::vector<typename TrimmingCurve::ControlPointType> controlPoints
82 = {{{.p = vertex1, .w = 1}, {.p = vertex2, .w = 1}}};
83 auto controlNet = NURBSPatchData<1, 2>::ControlPointNetType(controlPoints);
84 typename TrimmingCurve::PatchData patchData(knotSpans, controlNet, {1});
85
86 return GeometryKernel::NURBSPatch(patchData);
87 };

89 auto createTrimmingCurveSlice = [&](auto& curveIdx, double t1, double t2) -> TrimmingCurve {
90 return sliceCurve(patchTrimData.loops().front().curves()[curveIdx], {t1, t2});
91 };

97 // State
98 std::vector<FieldVector<ScalarType, dim>> foundVertices;
99 bool isOnNewEdge = false;
100 size_t currentCurveIdx = std::numeric_limits<size_t>::infinity();
101 double currentT = std::numeric_limits<double>::infinity();
102
103 for (const auto i : std::views::iota(0u, result.vertices_.size())) {
104 auto vV1 = result.vertices_[i];
105 auto vV2 = result.vertices_[nextEntity(i)];
106
107 auto pt1 = std::visit(getPt, vV1);
108 auto pt2 = std::visit(getPt, vV2);
109
110 // First case edge is completly untrimmed
111 if (!isNewVertex(vV1) and !isNewVertex(vV2)) {
112 elementTrimData.addEdge(Impl::giveEdgeIdx(getHostIdx(vV1), getHostIdx(vV2)));
113 foundVertices.push_back({pt2.x, pt2.y});
114 continue;
115 }
116
117 // Second case edge begins on a hostVertes and ends on a newVertex
118 if (!isNewVertex(vV1) and isNewVertex(vV2)) {
119 currentCurveIdx = getTrimmingCurveIdx(vV2);
120 auto [tParam, curvePoint] = callFindIntersection(currentCurveIdx, getEdgeIdx(vV2), pt2);
121 std::cout << "Found: " << curvePoint << " From Clipping: " << pt2.x << " " << pt2.y << " t: " << tParam
122 << std::endl;
123
124 FieldVector<ScalarType, dim> p
125 = foundVertices.empty() ? FieldVector<ScalarType, dim>{pt1.x, pt1.y} : foundVertices.back();
126 auto trimmedEdge = createHostGeometry(p, curvePoint);
127 elementTrimData.addEdgeHostNew(getEdgeIdx(vV2), trimmedEdge, curvePoint);
128 foundVertices.push_back(curvePoint);
129
130 isOnNewEdge = true;
131 currentT = tParam;
132 continue;
133 }
134 // Third case newVertex - newVertex
135 if (isNewVertex(vV1), isNewVertex(vV2)) {
136 assert(getTrimmingCurveIdx(vV2) == currentCurveIdx);
137
138 auto [tParam, curvePoint] = callFindIntersection(currentCurveIdx, getEdgeIdx(vV2), pt2);
139 std::cout << "Found: " << curvePoint << " From Clipping: " << pt2.x << " " << pt2.y << " t: " << tParam
140 << std::endl;
141
142 auto elementTrimmingCurve = createTrimmingCurveSlice(currentCurveIdx, currentT, tParam);
143 elementTrimData.addEdgeNewNew(elementTrimmingCurve, curvePoint);
144 foundVertices.push_back(curvePoint);
145
146 isOnNewEdge = false;
147 currentT = std::numeric_limits<double>::infinity();
148
149 continue;
150 }
151 // Fourth case edge begins on a newVertex and ends in a HostVertex
152 if (isNewVertex(vV1), !isNewVertex(vV2)) {
153 auto v2 = Dune::FieldVector<ScalarType, dim>{pt2.x, pt2.y};
154
155 FieldVector<ScalarType, dim> p;
156 if (foundVertices.empty())
157 std::tie(std::ignore, p) = callFindIntersection(getTrimmingCurveIdx(vV1), getEdgeIdx(vV1), pt1);
158 else
159 p = foundVertices.back();
160
161 auto trimmedEdge = createHostGeometry(p, v2);
162 elementTrimData.addEdgeNewHost(getEdgeIdx(vV1), trimmedEdge, getHostIdx(vV2));
163
164 foundVertices.push_back({pt2.x, pt2.y});
165 continue;
166 }
167 }
168
169 return elementTrimData;
170 }
