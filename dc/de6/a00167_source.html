28 void addLoop(const BoundaryLoop& loop) {
29 Clipper2Lib::PathD path;
30 for (idx_t i = loops_.empty() ? splitter_ : loops_.back().back().z + 1;
31 const auto& curve : loop.curves()) {
32 if (curve.degree().front() == 1) {
33 auto p1 = curve.corner(0);
34 auto p2 = curve.corner(1);
35 path.emplace_back(p1[0], p1[1], i);
36 path.emplace_back(p2[0], p2[1], i + splitter_ - 1);
37 i += splitter_;
38 continue;
39 }
40 for (const auto v : Utilities::linspace(curve.domain()[0], splitter_)) {
41 auto fV = curve.global({v});
42 path.emplace_back(fV[0], fV[1], ++i);
43 }
44 }
45 loops_.push_back(path);
46 loopIndices_.push_back(loops_.back().back().z);
47 }
