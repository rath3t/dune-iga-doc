69 auto figure = matplot::figure(true);

72 constexpr std::array<std::array<int, 2>, 4> edgeLookUp{std::array{0, 1}, {1, 3}, {3, 2}, {2, 0}};

74
75 auto plotEllipse = [](const Vertex& v) {
76 constexpr auto w = 0.025;
77 const auto c = matplot::ellipse(v[0] - (w / 2), v[1] - (w / 2), w, w);
78 c->color("blue");
79 };

81 auto plotLine = [](std::vector<Vertex>& vs, bool thin = false) {
82 std::vector<double> x;
83 std::ranges::transform(vs, std::back_inserter(x), [](auto& v) { return v[0]; });
84
85 std::vector<double> y;
86 std::ranges::transform(vs, std::back_inserter(y), [](auto& v) { return v[1]; });
87 if (thin)
88 matplot::plot(x, y)->line_width(0.5).color("grey");
89 else
90 matplot::plot(x, y)->line_width(2).color("red");
91 };
92
93 for (auto c : edgeLookUp) {
94 std::vector<Vertex> vs{eleGeometry.corner(c[0]), eleGeometry.corner(c[1])};
95 plotLine(vs, true);
96 }
97
98 for (auto& v : vertices_) {
99 if (v.geometry.has_value())
100 plotEllipse(v.geometry.value());
101 else
102 plotEllipse(eleGeometry.corner(idxLookUp[v.idx]));
103 }
104
105 for (auto& curve : edges_) {
106 if (curve.geometry.has_value()) {
107 std::vector<Vertex> vs;
108 for (double u : Utilities::linspace(curve.geometry.value().domain().front(), 40))
109 vs.push_back(curve.geometry.value().global(u));
110 plotLine(vs);
111 } else {
112 auto cornerIdx = edgeLookUp[curve.idx];
113 std::vector<Vertex> vs{eleGeometry.corner(cornerIdx.front()), eleGeometry.corner(cornerIdx.back())};
114 plotLine(vs);
115 }
116 }
117
118 matplot::axis(matplot::equal);
119 matplot::save(filename, "gif");
120 }
