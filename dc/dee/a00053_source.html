57 // Coordinates in a single knot span differ from coordinates on the B-spline patch
58 // by an affine transformation. This transformation is stored in the diagonal entries.
59 // If trimming is disabled the Jacobian in the parameter space of subentities (edges or surfaces) is a
60 // DiagonalMatrixBlock but is treated as a FieldMatrix, since is also just cubes and an axis-aligned geometry, if
61 // trimming is enabled the Jacobian of the parameterspace for subentities is a potentially fully populated
62 // FieldMatrix, Think about an arbitrary curve in the axis-aligned cube parameter grid
63 // ------->
64 // `____________________C_______D_______`
65 // ::````````````````::``````````'|````::
66 // :: :: `| ::
67 // :: ^F:: `| ::
68 // :: | :: `| ::
69 // :: | :: `| ::
70 // :: |E:: `| ::
71 // :: :: .| x ::
72 // :: :: _: ::
73 // ::................::..........|:....::
74 // ::''''''''''''''''::''''''''''/'''''::
75 // :: :: _:B ::
76 // :: :: .| ::
77 // :: :: `::A ::
78 // :: '::::||::::::' ::
79 // :: ::` :: x ::
80 // :: |' :: ::
81 // :: | x :: ::
82 // ::````````|```````::````````````````::
83 // `____________________________________`
84 // at point A the tangent of the curve (the JacobianInParameterSpace) points into the oblique direction (to B)
85 // for the axis-aligned geometry these tangents are from E to F or from C to D, depending on the orientation of the
86 // subentity, which are aligned with the axis
87 using JacobianTransposedInParameterSpace = typename ParameterSpaceGeometry::JacobianTransposed;
88 using GlobalInParameterSpace = typename ParameterSpaceGeometry::GlobalCoordinate;
89

263 Hessian hessian(const LocalCoordinate& u) const {
264 assert(parameterSpaceGeometry && "Bind the local view first!");
265 const auto ouInPatch = globalInParameterSpace(u);
266
267 const auto [p, dfdg, dfdgdg] = patchGeometry_->zeroFirstAndSecondDerivativeOfPositionImpl(
268 ouInPatch, nurbsLocalView_, localControlPointNet);
269
270 static_assert(std::is_same_v<std::remove_const_t<decltype(dfdg)>, PatchJacobianTransposed>);
271 static_assert(std::is_same_v<std::remove_const_t<decltype(dfdgdg)>, PatchHessian>);
272 static_assert(std::is_same_v<std::remove_const_t<decltype(p)>, GlobalCoordinate>);
273
274 const JacobianTransposedInParameterSpace dgdt = jacobianTransposedInParameterSpace(u);
275 static_assert(JacobianTransposedInParameterSpace::rows==mydimension);
276 static_assert(JacobianTransposedInParameterSpace::cols==gridDimension);
277 // dgdt (1x2, curve on surface), (2x2 surface in surface), (2x3 surface in 3D), (3x3 volume in 3D)
278 FieldMatrix<ctype, patchNumberOfSecondDerivatives, numberOfSecondDerivatives> dgdtSquared(0);
279
280 for (int patchIndex = 0; auto [patchRow, patchCol] : Impl::voigtIndices<gridDimension>()) {
281 for (int myIndex = 0; auto [myRow, myCol] : Impl::voigtIndices<mydimension>()) {
282 if constexpr (not std::is_same_v<JacobianTransposedInParameterSpace,
283 DiagonalMatrix<ctype, gridDimension>>)
284 dgdtSquared[patchIndex++][myIndex++]
285 = (dgdt[myRow][patchRow]*dgdt[myCol][patchCol]+ dgdt[myCol][patchRow]*dgdt[myRow][patchCol])
286 * ((patchRow == patchCol) ? 0.5 : 1.0);
287 else {
288 const auto dgdtmyRowpatchRow = (myRow == patchRow) ? dgdt[myRow][patchRow] : 0;
289 const auto dgdtmyColpatchCol = (myCol == patchCol) ? dgdt[myCol][patchCol] : 0;
290 const auto dgdtmyColpatchRow = (myRow == patchCol) ? dgdt[myCol][patchRow] : 0;
291 const auto dgdtmyRowpatchCol = (myCol == patchRow) ? dgdt[myRow][patchCol] : 0;
292 dgdtSquared[patchIndex++][myIndex++]
293 = (dgdtmyRowpatchRow * dgdtmyColpatchCol + dgdtmyColpatchRow * dgdtmyRowpatchCol)
294 * ((patchRow == patchCol) ? 0.5 : 1.0);
295 }
296 }
297 }
298
299 Hessian h;
300 // h = mydimension * (mydimension + 1) / 2 X worlddimension = 1x2 (curve in 2D), 1x3 (curve in 3D), (3x2 surface
301 // in 2D), (3x3 surface in 3D), (6x3 volume in 3D)
302 //
303 // dfdgdg = gridDimension*(gridDimension + 1) / 2 X worlddimension
304 // = 3x2 (curve on surface flat),3x3 (curve on surface in 3D), 6x3 (curve in volume), 6x3 (surface in volume),(3x3
305 // surface in surface in 3D)
306 //
307 // dgdtSquared = gridDimension * (gridDimension + 1) / 2 X mydimension * (mydimension +
308 // 1) / 2 = 3x1 (curve on surface), 6x1 (curve in volume), 6x3 (surface in volume), 3x3 ( surface in surface), 6x6
309 // ( volume in volume)
310 h = transposedView(dgdtSquared) * dfdgdg;
311
312 /* if trimming is enabled the parameter space geometry is potentially non-linear,
313 * the resutling Hessian has another contribution due to chain-rule, namely the second derivative of g */
314 if constexpr (trim == Trimming::Enabled) {
315 const auto dgdtdt = parameterSpaceGeometry.value().hessian(ouInPatch);
316
317 assert(trim == Trimming::Disabled
318 && "This can not be checked yet. Check if this works with trimming and then remove assert");
319 h += transpose(transposedView(dfdg) * dgdtdt);
320 }
321 return h;
322 }
