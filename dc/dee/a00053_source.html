85 explicit NURBSGrid(const NURBSPatchData<dim, dimworld, ScalarType>& nurbsPatchData,

89 coarsestPatchRepresentation_{nurbsPatchData},
90 currentPatchRepresentation_{coarsestPatchRepresentation_},

97 static_assert(dim <= 3, "Higher grid dimensions are unsupported");
98 assert(nurbsPatchData.knotSpans[0].size() - nurbsPatchData.degree[0] - 1
99 == nurbsPatchData.controlPoints.strideSizes()[0]
100 && "The size of the controlpoints and the knotvector size do not match in the first direction");
101 if constexpr (dim > 1)
102 assert(nurbsPatchData.knotSpans[1].size() - nurbsPatchData.degree[1] - 1
103 == nurbsPatchData.controlPoints.strideSizes()[1]
104 && "The size of the controlpoints and the knotvector size do not match in the second direction");
105 if constexpr (dim > 2)
106 assert(nurbsPatchData.knotSpans[2].size() - nurbsPatchData.degree[2] - 1
107 == nurbsPatchData.controlPoints.strideSizes()[2]
108 && "The size of the controlpoints and the knotvector strideSizes do not match in the third direction");
109 // FIXME check sanity of knotvector and degree
110 createEntities();
111 }

115 void globalRefine(int refinementLevel) { globalRefine(refinementLevel, false); }
116 void globalRefine(int refinementLevel, bool omitTrim) {
117 if (refinementLevel == 0) return;
118 for (int refDirection = 0; refDirection < dim; ++refDirection) {
119 auto additionalKnots
120 = generateRefinedKnots(currentPatchRepresentation_.knotSpans, refDirection, refinementLevel);
121 currentPatchRepresentation_ = knotRefinement<dim>(currentPatchRepresentation_, additionalKnots, refDirection);
122 }
123 updateStateAfterRefinement(omitTrim);
124 }
125
126 void globalRefineInDirection(const int dir, const int refinementLevel, bool omitTrim = false) {
127 if (refinementLevel == 0) return;
128 auto additionalKnots = generateRefinedKnots(currentPatchRepresentation_.knotSpans, dir, refinementLevel);
129 currentPatchRepresentation_ = knotRefinement<dim>(currentPatchRepresentation_, additionalKnots, dir);
130 updateStateAfterRefinement(omitTrim);
131 }
132
133 void degreeElevateInDirection(const int dir, const int elevationFactor, bool omitTrim = false) {
134 if (elevationFactor == 0) return;
135 currentPatchRepresentation_ = degreeElevate(currentPatchRepresentation_, dir, elevationFactor);
136 updateStateAfterRefinement(omitTrim);
137 }
138
139 void globalDegreeElevate(const int elevationFactor, bool omitTrim = false) {
140 if (elevationFactor == 0) return;
141 oldPatchRepresentation_.push_back(currentPatchRepresentation_);
142 for (int refDirection = 0; refDirection < dim; ++refDirection)
143 currentPatchRepresentation_ = degreeElevate(currentPatchRepresentation_, refDirection, elevationFactor);
144 updateStateAfterRefinement(omitTrim);
145 }

147 void globalMultiRefine(const int global, const int uDir, const int vDir) {

149 this->globalRefineInDirection(0, uDir, vDir > 0);
150 this->globalRefineInDirection(1, vDir);

194 void updateStateAfterRefinement(bool omitTrim = false) {
195 leafPatches_->clear();
196 leafPatches_->emplace_back(currentPatchRepresentation_, omitTrim ? std::nullopt : trimData_);

200 indexSet_ = std::make_unique<NURBSGridLeafIndexSet<const NURBSGrid>>((this->leafGridView().impl()));

204 void createEntities() {
205 for (int currentPatchId = 0; auto&& patch : *leafPatches_.get()) {
206 Dune::Hybrid::forEach(Dune::Hybrid::integralRange(Dune::index_constant<dimension + 1>()), [&](const auto i) {
207 std::get<i>(*entityVector.get()).reserve(patch.size(i));
208 for (unsigned int j = 0; j < patch.size(i); ++j) {
209 std::get<i>(*entityVector.get())
210 .emplace_back(NURBSGridEntity<i, dimension, const NURBSGrid>(*leafGridView_, j, currentPatchId));
211 }
212 });
213 ++currentPatchId;
214 }
215 }
216 void updateEntities() {
217 entityVector = std::make_unique<decltype(gridEntityTupleGenerator<NURBSGrid, dimension>(
218 std::make_integer_sequence<int, dimension + 1>()))>();
219
220 createEntities();
221 }
