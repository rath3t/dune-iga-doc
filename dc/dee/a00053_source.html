11 /* \brief Create a circular arc with different customization options
12 * \tparam ScalarType the field type (use float, double, complex, etc)
13 * \param radius Radius of the arc
14 * \param startAngle starting angle of the arc in degrees
15 * \param endAngle end angle of the arc in degrees
16 * \param origin center of the circle
17 * \param X first base vector of plane where the arc should reside
18 * \param Y second base vector of plane where the arc should reside
19 * \return NURBSPatchData representing the arc
20 */
21 template <typename ScalarType = double>
22 NURBSPatchData<1, 3, ScalarType> makeCircularArc(const ScalarType radius = 1.0, const ScalarType startAngle = 0.0,
23 ScalarType endAngle = 360.0,
24 const Dune::FieldVector<ScalarType, 3>& origin = {0, 0, 0},
25 const Dune::FieldVector<ScalarType, 3>& X = {1, 0, 0},
26 const Dune::FieldVector<ScalarType, 3>& Y = {0, 1, 0}) {
27 using GlobalCoordinateType = typename NURBSPatchData<1, 3, ScalarType>::GlobalCoordinateType;
28 const auto pi = std::numbers::pi_v<ScalarType>;
29
30 if (endAngle < startAngle) endAngle += 360.0;
31 const ScalarType theta = endAngle - startAngle;
32 const int narcs = std::ceil(theta / 90);
33
34 typename NURBSPatchData<1, 3, ScalarType>::ControlPointNetType circleCPs(2 * narcs + 1);
35 const ScalarType dtheta = theta / narcs * pi / 180;
36 const int n = 2 * narcs;
37 const ScalarType w1 = cos(dtheta / 2.0);
38 GlobalCoordinateType PO = origin + radius * cos(startAngle) * X + radius * sin(startAngle) * Y;
39 GlobalCoordinateType TO = -sin(startAngle) * X + cos(startAngle) * Y;
40 circleCPs.directGet(0).p = PO;
41 ScalarType angle = startAngle;
42 for (int index = 0, i = 0; i < narcs; ++i) {
43 angle += dtheta;
44 const GlobalCoordinateType P2 = origin + radius * cos(angle) * X + radius * sin(angle) * Y;
45 circleCPs.directGet(index + 2).p = P2;
46 const GlobalCoordinateType T2 = -sin(angle) * X + cos(angle) * Y;
47 const GlobalCoordinateType P1 = intersect3DLines(PO, TO, P2, T2);
48 circleCPs.directGet(index + 1) = {.p = P1, .w = w1};
49 index += 2;
50 if (i < narcs - 1) {
51 PO = P2;
52 TO = T2;
53 }
54 }
55
56 auto knotVec = std::array<std::vector<double>, 1>{};
57 auto& U = knotVec[0];
58 U.resize(2 * (narcs + 2));
59 if (narcs == 1) {
60 } else if (narcs == 2) {
61 U[3] = U[4] = 1.0 / 2.0;
62 } else if (narcs == 3) {
63 U[3] = U[4] = 1.0 / 3.0;
64 U[5] = U[6] = 2.0 / 3.0;
65 } else {
66 U[3] = U[4] = 1.0 / 4.0;
67 U[5] = U[6] = 2.0 / 4.0;
68 U[7] = U[8] = 3.0 / 4.0;
69 }
70
71 std::ranges::fill_n(U.begin(), 3, 0.0);
72 std::ranges::fill_n(std::ranges::reverse_view(U).begin(), 3, 1.0);
73 return NURBSPatchData<1, 3, ScalarType>(knotVec, circleCPs, {2});
74 }
