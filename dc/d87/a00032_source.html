29 class CurveLoopIndexEncoder {
30 static constexpr int loopBits = 4;
31 static constexpr int curveBits = 8;
32 static constexpr int sampleBits = 12;
33
34 static constexpr u_int64_t loopMask{(1ULL << loopBits) - 1};
35 static constexpr u_int64_t curveMask{(1ULL << curveBits) - 1};
36 static constexpr u_int64_t sampleMask{(1ULL << sampleBits) - 1};
37 public:
38
39 static int64_t encode(int loopIndex,int curveIndex, int sampleIndex) {
40 // the left hand side shifts the curve index value away from loop values and then inserts the curve index
41 // example curveBits = 4 and loopBits = 3 and sampleBits=12;
42 // curveIndex = 3 and loopIndex = 2 = 00010_b , sampleIndex= 200= 11001000_b
43 // curveIndex = 3 = 00011_b
44 // curveIndex<< (loopBits+ sampleBits) = 110,000,000,000,000,000_b // shift the curve index 16 bits to the left
45 // loopIndex<< sampleBits = 10,000,000,000,000_b // shift the loop index 12 bits to the left
46 // sampleIndex = 11,001,000_b // shift the loop index 12 bits to the left
47 // bitwise or of the three values: 110,010,000,011,001,000_b //
48 // (curveIndex<< loopBits) | loopIndex = 11000_b | 00010_b = 11010_b = 205000_d
49 return (static_cast<u_int64_t>(curveIndex) << (loopBits + sampleBits)) |
50 (static_cast<u_int64_t>(loopIndex) << sampleBits) |
51 sampleIndex;
52 }

65 static IndexResult decode(int64_t singleIndex) {
66 // example curveBits = 4 and loopBits = 3;
67 // loopMask = 00111
68 // example singleIndex = 11010_b = 26
69 // curveIndex = 3 and loopIndex = 2 = 00010_b
70 // curveIndex = 3 = 00011_b
71 // loopMask =
72 // (singleIndex >> loopBits) = 00011_b = curveIndex
73 // since the sample index is stored in the right-most bits, we only have to apply the bitwise and, which returns simply the sample index
74 // 11010_b AND
75 // 00111_b
76 // 00010_b = 2
77 int sampleIndex = singleIndex & sampleMask;
78 const int loopAndCurveIndex = singleIndex >> sampleBits;
79 int loopIndex = loopAndCurveIndex & loopMask;
80 int curveIndex = (loopAndCurveIndex >> loopBits) & curveMask;
81 return {loopIndex,curveIndex, sampleIndex};

83 }

104 Clipper2Lib::PathD path;
105 // since the first
106 for (int curveIndex=0; const auto& curve : loop.curves()) {
107
108 if (curve.affine() == 1 and loops_.empty()) {
109 auto p1 = curve.corner(0);
110 auto p2 = curve.corner(1);
111 path.emplace_back(p1[0], p1[1], getZValue(loops_.size(),curveIndex,0));
112 path.emplace_back(p2[0], p2[1], getZValue(loops_.size(),curveIndex,1) );
113 }else {
114 for (int vi=0;const auto v : Utilities::linspace(curve.domain()[0], splitter_)) {
115 auto fV = curve.global({v});
116 path.emplace_back(fV[0], fV[1], getZValue(loops_.size(),curveIndex,vi++));

118 }

120 }

171 manager_.splitter_ = parameters.splitter;

177 // Determine points of the outer boundary that are not on the edges
178 auto isInsidePatch = [&](const FieldVector<ctype, 2>& pt) {
179 const bool isInsideU = FloatCmp::gt(pt[0], domainU.front()) and FloatCmp::lt(pt[0], domainU.back());
180 const bool isInsideV = FloatCmp::gt(pt[1], domainV.front()) and FloatCmp::lt(pt[1], domainV.back());

182 return isInsideU and isInsideV;
183 };

187 for (size_t i = 0; const auto& curve : loop.curves()) {
188 if (const auto pt = curve.corner(1); isInsidePatch(pt)) {
