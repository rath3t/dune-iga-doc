12 struct TransformToSpan {
13 std::array<double, dim> scaling_{};
14 std::array<double, dim> offset_{};
15
16 TransformToSpan() = default;
17 explicit TransformToSpan(std::pair<std::array<double, dim>, std::array<double, dim>>& input)
18 : scaling_(input.first), offset_(input.second){};

20 template <typename VecType>
21 [[nodiscard]] auto transform(const VecType& cp) const -> VecType {
22 VecType local{};
23 for (int i = 0; i < dim; ++i) {
24 local[i] = (cp[i] - offset_[i]) / scaling_[i];
25 local[i] = std::clamp(local[i], 0.0, 1.0);
26 }
27 return local;
28 }

30 [[nodiscard]] auto transform(const Clipper2Lib::PointD& p) const -> Clipper2Lib::PointD {
31 std::array<double, 2> cp{p.x, p.y};
32 auto cpLoc = transform(cp);
33 return {cpLoc[0], cpLoc[1]};
34 }
35 };

37 [[nodiscard]] double calculateBoundaryLoopLength(const std::vector<Boundary>& loop,
38 const TransformToSpan<2>& transformer, unsigned int div = 200) {
39 Clipper2Lib::PathD polygon;
40 polygon.reserve(div * loop.size());
41
42 for (auto& boundary : loop) {
43 auto path = boundary.path(div);
44 for (Clipper2Lib::PointD point : path)
45 polygon.emplace_back(transformer.transform(point));
46 }
47 return std::fabs(Clipper2Lib::Length(polygon));
48 }

66 auto splitBoundariesImpl(const std::vector<Boundary>& boundaries, int maxSubSample, double tolerance,
67 const TransformToSpan<2>& transformer) {
68 bool checkTarget = false;
69
70 double targetLength = 0.0;
71 if (tolerance not_eq 0.0) {
72 checkTarget = true;
73 targetLength = calculateBoundaryLoopLength(boundaries, transformer);
74 }
75 auto refineMap = determineCurvedBoundaries(boundaries);
76
77 // Get all domains
78 std::vector<DomainInformation> domains;
79 domains.reserve((1 << maxSubSample) * boundaries.size());
80
81 for (auto i : std::views::iota(0u, boundaries.size()))
82 domains.emplace_back(boundaries[i].domain, i);
83
84 std::vector<DomainInformation> tempDomainInfos;
85 tempDomainInfos.reserve((1 << maxSubSample) * boundaries.size());
86
87 for (auto _ : std::views::iota(0, maxSubSample)) {
88 tempDomainInfos.clear();
89 tempDomainInfos.insert(tempDomainInfos.end(), domains.begin(), domains.end());
90 domains.clear();
91 for (auto& domainInfo : tempDomainInfos) {
92 bool refine = refineMap[domainInfo.localIndex];
93
94 if (refine) {
95 std::array<DomainType, 2> newDomains = Utilities::splitDomainInHalf(domainInfo.domain);
96 domains.emplace_back(newDomains[0], domainInfo.localIndex);
97 domains.emplace_back(newDomains[1], domainInfo.localIndex);
98 } else
99 domains.push_back(domainInfo);
100 }
101 // Check convergence
102 if (checkTarget) {
103 auto computeCurrentLength = [&](const std::vector<DomainInformation>& domainInfos) -> double {
104 Clipper2Lib::PathD path;
105 for (const auto& domainInfo : domainInfos) {
106 auto u = transformer.transform(boundaries[domainInfo.localIndex].nurbsGeometry(domainInfo.domain.left()));
107 path.emplace_back(u[0], u[1]);
108 }
109 return std::fabs(Clipper2Lib::Length(path, true));
110 };
111 auto actLength = computeCurrentLength(domains);
112 if (std::fabs(actLength - targetLength) / targetLength < tolerance) break;
113 }
114 }
115
116 // Create split outerBoundaries from DomainInformation
117 std::vector<Boundary> newBoundaries;
118 std::ranges::transform(domains, std::back_inserter(newBoundaries),
119 [&boundaries](const auto& domain_info) -> Boundary {
120 return {boundaries[domain_info.localIndex].nurbsGeometry, domain_info.domain};
121 });
122 return newBoundaries;
123 }

125 template <int dim, typename Transformer = TransformToSpan<dim>, typename Index = std::uint64_t,
126 typename Point = Dune::FieldVector<double, dim>>
127 auto triangulate(auto& boundaries, const Transformer& transformer)
128 -> std::pair<std::vector<Index>, std::vector<Point>> {
129 // Construct mesh with Earcut
130 // C.f. https://github.com/mapbox/earcut.hpp
131
132 auto [splitOuter, splitInner] = boundaries;
133
134 // Create array of points
135 auto vertices = std::vector<Point>();
136
137 vertices.reserve(splitOuter.size());
138 for (auto& boundary : splitOuter)
139 vertices.push_back(transformer.transform(boundary.endPoints.front()));
140
141 std::vector<std::vector<Point>> polygonInput;
142 polygonInput.push_back(vertices);
143
144 if (splitInner)
145 for (auto& innerLoop : splitInner.value()) {
146 assert(innerLoop.size() > 1);
147 std::vector<Point> holeInput;
148 for (auto& boundary : innerLoop) {
149 auto v = transformer.transform(boundary.endPoints.front());
150 vertices.push_back(v);
151 holeInput.push_back(v);
152 }
153 polygonInput.push_back(holeInput);
154 }
155
156 return {mapbox::earcut<Index>(polygonInput), vertices};
157 }

160 struct GridBoundarySegment : Dune::BoundarySegment<dim, dim, double> {

164 Dune::FieldVector<double, dim> operator()(const Dune::FieldVector<double, 1>& localI) const override {
165 // u has to be mapped on the domain of 0 to 1
166 const auto local = std::clamp(localI[0], 0.0, 1.0);
167 double u = Utilities::mapToRange(local, Utilities::Domain<double>{}, boundary.domain);
168 return transformer.transform(boundary.nurbsGeometry(u));
169 }

171 Boundary boundary;
172 };
