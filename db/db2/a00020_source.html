82 template<std::size_t dim,typename ScalarType>
83 auto createUniqueKnotSpans(const std::array<std::vector<ScalarType>, dim>& knotSpans) {
84 std::array<std::vector<double>, dim>uniqueKnotVector;
85 for (int i = 0; i < dim; ++i) // create unique knotspan vectors
86 std::ranges::unique_copy(knotSpans[i], std::back_inserter(uniqueKnotVector[i]),
87 [](auto& l, auto& r) { return Dune::FloatCmp::eq(l, r); });
88
89 return uniqueKnotVector;
90 }

328 void globalRefine (int refCount) {
329 for (int i = 0; i < refCount; ++i) {
330 const auto& finestPatchData = patchGeometries.back().patchData_;
331 auto newfinestPatchData = finestPatchData;
332 for (int refDirection = 0; refDirection < dim; ++refDirection) {
333 auto additionalKnots = IGA::generateRefinedKnots(finestPatchData.knotSpans, refDirection, 1);
334 newfinestPatchData = knotRefinement<dim>(newfinestPatchData, additionalKnots, refDirection);
335 }
336 patchGeometries.emplace_back(std::move(newfinestPatchData));
337
338 }
339
340 hostgrid_->globalRefine(refCount);
341 setIndices();
342 }

488 }

490 for (int i=0; i<=maxLevel(); i++)
