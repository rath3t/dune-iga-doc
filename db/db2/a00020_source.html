79 template<std::size_t dim,typename ScalarType>
80 auto createUniqueKnotSpans(const std::array<std::vector<ScalarType>, dim>& knotSpans) {
81 std::array<std::vector<double>, dim>uniqueKnotVector;
82 for (int i = 0; i < dim; ++i) // create unique knotspan vectors
83 std::ranges::unique_copy(knotSpans[i], std::back_inserter(uniqueKnotVector[i]),
84 [](auto& l, auto& r) { return Dune::FloatCmp::eq(l, r); });
85
86 return uniqueKnotVector;
87 }

93 return grid;
94 }

322 void globalRefine (int refCount) {
323 for (int i = 0; i < refCount; ++i) {
324 const auto& finestPatchData = patchGeometries.back().patchData_;
325 auto newfinestPatchData = finestPatchData;
326 for (int refDirection = 0; refDirection < dim; ++refDirection) {
327 auto additionalKnots = IGA::generateRefinedKnots(finestPatchData.knotSpans, refDirection, 1);
328 newfinestPatchData = knotRefinement<dim>(newfinestPatchData, additionalKnots, refDirection);
329 }
330 patchGeometries.emplace_back(std::move(newfinestPatchData));
331
332 }
333
334 hostgrid_->globalRefine(refCount);
335 setIndices();
336 }

475 // //////////////////////////////////////////

479 auto p

481 levelIndexSets_.emplace_back(std::move(p));
482 }
483
484 for (int i=0; i<=maxLevel(); i++)
485 if (levelIndexSets_[i])
486 levelIndexSets_[i]->update(*this, i);
487
488 leafIndexSet_.update(*this);

490 }
