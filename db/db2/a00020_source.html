63 template <std::size_t dim, typename ScalarType>
64 auto createUniqueKnotSpans(const std::array<std::vector<ScalarType>, dim>& knotSpans) {
65 std::array<std::vector<double>, dim> uniqueKnotVector;
66 for (int i = 0; i < dim; ++i) // create unique knotspan vectors
67 std::ranges::unique_copy(knotSpans[i], std::back_inserter(uniqueKnotVector[i]),
68 [](auto& l, auto& r) { return Dune::FloatCmp::eq(l, r); });
69
70 return uniqueKnotVector;
71 }

260 void globalRefine(int refCount) {
261 for (int i = 0; i < refCount; ++i) {
262 const auto& finestPatchData = patchGeometries.back().patchData_;
263 auto newfinestPatchData = finestPatchData;
264 for (int refDirection = 0; refDirection < dim; ++refDirection) {
265 auto additionalKnots = IGA::generateRefinedKnots(finestPatchData.knotSpans, refDirection, 1);
266 newfinestPatchData = knotRefinement<dim>(newfinestPatchData, additionalKnots, refDirection);
267 }
268 patchGeometries.emplace_back(std::move(newfinestPatchData));
269 }
270
271 hostgrid_->globalRefine(refCount);
272 setIndices();
273 }

373 globalIdSet_.update();

375 // //////////////////////////////////////////
376 // Create the index sets
377 // //////////////////////////////////////////
378 for (int i = levelIndexSets_.size(); i <= maxLevel(); i++) {
379 auto p = std::make_unique<PatchGridLevelIndexSet<const PatchGrid>>();
380 levelIndexSets_.emplace_back(std::move(p));
381 }
382
383 for (int i = 0; i <= maxLevel(); i++)
384 if (levelIndexSets_[i]) levelIndexSets_[i]->update(*this, i);
385
386 leafIndexSet_.update(*this);
387 }
