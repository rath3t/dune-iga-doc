31 constexpr int N = 5;
32
33 for (auto loop : trimData.loops()) {
34 tempPath.clear();
35 for (int i = 100; auto& curve: loop.curves()) {
36 for (auto v : Utilities::linspace(curve.domain()[0], N)) {
37 auto fV = curve.global({v});
38 tempPath.emplace_back(fV[0], fV[1], i++);
39 }
40 // Add an additional point just outside of the element
41 auto localLastPoint = curve.domain()[0][1];
42 auto lastPoint = curve.global(localLastPoint);
43
44 auto scale = element.geometry().volume() / 10;
45
46 auto dx = curve.jacobian({localLastPoint}) * scale;
47 tempPath.emplace_back(lastPoint[0] + dx[0], lastPoint[1] + dx[1], i++);
48
49 i += 99 - N;
50 }
51 clipPaths.push_back(tempPath);
52 }
