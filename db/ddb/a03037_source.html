305 void globalRefine(int refCount) {
306 for (int i = 0; i < refCount; ++i) {
307 const auto& finestPatchData = patchGeometries_.back().patchData();
308 auto newfinestPatchData = finestPatchData;
309
310 auto& olduniqueKnotVector = patchGeometries_.back().uniqueKnotVector();
311 std::array<std::vector<ctype>, dim> newUniqueKnotVecs;
312 for (int refDirection = 0; refDirection < dim; ++refDirection) {
313 auto additionalKnots = Splines::generateRefinedKnots(finestPatchData.knotSpans, refDirection, 1);
314 newfinestPatchData = Splines::knotRefinement<dim>(newfinestPatchData, additionalKnots, refDirection);
315 auto& newKnotVec = newUniqueKnotVecs[refDirection];
316
317 // compute new unique Knot vectors from refinement
318 newKnotVec.resize(olduniqueKnotVector[refDirection].size() + additionalKnots.size());
319 std::ranges::merge(olduniqueKnotVector[refDirection], additionalKnots, std::begin(newKnotVec));
320 }
321
322 patchGeometries_.emplace_back(newfinestPatchData, newUniqueKnotVecs);
323 patchGeometriesUnElevated.emplace_back(patchGeometries_.back());
324 }
325
326 trimmer_.parameterSpaceGrid().globalRefine(refCount);
327 setIndices();
328 }

345 void globalRefineInDirection(const std::array<int, dim>& refines) {
346 const auto& finestPatchData = patchGeometries_.back().patchData();
347 auto newfinestPatchData = finestPatchData;
348 for (int dir = 0; auto refinesInDirection : refines) {
349 if (refinesInDirection == 0) {
350 ++dir;
351 continue;
352 }
353 auto additionalKnots = Splines::generateRefinedKnots(finestPatchData.knotSpans, dir, refinesInDirection);
354 newfinestPatchData = Splines::knotRefinement<dim>(newfinestPatchData, additionalKnots, dir);
355 ++dir;
356 }
357 auto newGrid = PatchGrid(newfinestPatchData);
358 *this = std::move(newGrid);
359 }

366 void degreeElevate(const std::array<int, dim>& elevationFactors, int lvl) {
367 if (lvl > maxLevel() and lvl >= 0) DUNE_THROW(Dune : RangeError, "This level does not exist");
368 auto& patchData = patchGeometries_[lvl].patchData();
369 patchGeometriesUnElevated[lvl] = GeometryKernel::NURBSPatch<dim, dimworld, ctype>(patchData);
370 for (int dir = 0; auto elevatesInDirection : elevationFactors) {
371 if (elevatesInDirection == 0) {
372 ++dir;
373 continue;
374 }
375 patchData = Splines::degreeElevate(patchData, dir, elevatesInDirection);
376 ++dir;
377 }
378 patchGeometries_[lvl] = GeometryKernel::NURBSPatch<dim, dimworld, ctype>(patchData);
379 }

385 void degreeElevateOnAllLevels(const std::array<int, dim>& elevationFactors) {
386 for (int lvl = 0; lvl < maxLevel() + 1; ++lvl)
387 degreeElevate(elevationFactors, lvl);
