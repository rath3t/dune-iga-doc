32 if constexpr (dim == 1) {

82 // Coordinates in a single knot span differ from coordinates on the B-spline patch
83 // by an affine transformation. This transformation is stored in the diagonal entries.
84 // If trimming is disabled the Jacobian in the parameter space of subentities (edges or surfaces) is a
85 // DiagonalMatrixBlock but is treated as a FieldMatrix, since is also just cubes and an axis-aligned geometry, if
86 // trimming is enabled the Jacobian of the parameterspace for subentities is a potentially fully populated
87 // FieldMatrix, Think about an arbitrary curve in the axis-aligned cube parameter grid
88 // ------->
89 // `____________________C_______D_______`
90 // ::````````````````::``````````'|````::
91 // :: :: `| ::
92 // :: ^F:: `| ::
93 // :: | :: `| ::
94 // :: | :: `| ::
95 // :: |E:: `| ::
96 // :: :: .| x ::
97 // :: :: _: ::
98 // ::................::..........|:....::
99 // ::''''''''''''''''::''''''''''/'''''::
100 // :: :: _:B ::
101 // :: :: .| ::
102 // :: :: `::A ::
103 // :: '::::||::::::' ::
104 // :: ::` :: x ::
105 // :: |' :: ::
106 // :: | x :: ::
107 // ::````````|```````::````````````````::
108 // `____________________________________`
109 // at point A the tangent of the curve (the JacobianInParameterSpace) points into the oblique direction (to B)
110 // for the axis-aligned geometry these tangents are from E to F or from C to D, depending on the orientation of
111 // the subentity, which are aligned with the axis
112 using JacobianTransposedInParameterSpace = typename ParameterSpaceGeometry::JacobianTransposed;
113 using GlobalInParameterSpace = typename ParameterSpaceGeometry::GlobalCoordinate;
114

312 Hessian hessian(const LocalCoordinate& local) const {
313 checkState();
314 const auto ouInPatch = globalInParameterSpace(local);
315
316 const auto [p, dfdg, dfdgdg] = patchGeometry_->zeroFirstAndSecondDerivativeOfPositionImpl(
317 ouInPatch, nurbsLocalView_, localControlPointNet);
318
319 const JacobianTransposedInParameterSpace dgdt = jacobianTransposedInParameterSpace(local);
320 static_assert(JacobianTransposedInParameterSpace::rows == mydimension);
321 static_assert(JacobianTransposedInParameterSpace::cols == gridDimension);
322 // dgdt (1x2, curve on surface), (2x2 surface in surface), (2x3 surface in 3D), (3x3 volume in 3D)
323 FieldMatrix<ctype, patchNumberOfSecondDerivatives, numberOfSecondDerivatives> dgdtSquared(0);
324 for (int patchIndex = 0; auto [patchRow, patchCol] : Impl::voigtIndices<gridDimension>()) {
325 for (int myIndex = 0; auto [myRow, myCol] : Impl::voigtIndices<mydimension>()) {
326 if constexpr (not std::is_same_v<JacobianTransposedInParameterSpace, DiagonalMatrix<ctype, gridDimension>>)
327 dgdtSquared[patchIndex][myIndex]
328 = (dgdt[myRow][patchRow] * dgdt[myCol][patchCol] + dgdt[myCol][patchRow] * dgdt[myRow][patchCol])
329 * ((patchRow == patchCol) ? 0.5 : 1.0);
330 else {
331 const auto dgdtmyRowpatchRow = (myRow == patchRow) ? dgdt[myRow][patchRow] : 0;
332 const auto dgdtmyColpatchCol = (myCol == patchCol) ? dgdt[myCol][patchCol] : 0;
333 const auto dgdtmyColpatchRow = (myCol == patchRow) ? dgdt[myCol][patchRow] : 0;
334 const auto dgdtmyRowpatchCol = (myRow == patchCol) ? dgdt[myRow][patchCol] : 0;
335 dgdtSquared[patchIndex][myIndex]
336 = (dgdtmyRowpatchRow * dgdtmyColpatchCol + dgdtmyColpatchRow * dgdtmyRowpatchCol)
337 * ((patchRow == patchCol) ? 0.5 : 1.0);
338 }
339 ++myIndex;
340 }
341 ++patchIndex;
342 }
343
344 // h = mydimension * (mydimension + 1) / 2 X worlddimension = 1x2 (curve in 2D), 1x3 (curve in 3D), (3x2 surface
345 // in 2D), (3x3 surface in 3D), (6x3 volume in 3D)
346 //
347 // dfdgdg = gridDimension*(gridDimension + 1) / 2 X worlddimension
348 // = 3x2 (curve on surface flat),3x3 (curve on surface in 3D), 6x3 (curve in volume), 6x3 (surface in
349 // volume),(3x3 surface in surface in 3D)
350 //
351 // dgdtSquared = gridDimension * (gridDimension + 1) / 2 X mydimension * (mydimension +
352 // 1) / 2 = 3x1 (curve on surface), 6x1 (curve in volume), 6x3 (surface in volume), 3x3 ( surface in surface),
353 // 6x6 ( volume in volume)
354 Hessian h = transposedView(dgdtSquared) * dfdgdg;
355
356 /* if trimming is enabled the parameter space geometry is potentially non-linear,
357 * the resutling Hessian has another contribution due to chain-rule, namely the second derivative of g */
358 if constexpr (not Trimmer::template isLocalGeometryLinear<codim>) {
359 // const auto dgdtdt = parameterSpaceGeometry->hessian(ouInPatch);
360 //
361 // assert(TrimmerType::template isLocalGeometryLinear<
362 // codim> && "This can not be checked yet. Check if this works with trimming and then remove
363 // assert");
364 // h += transpose(transposedView(dfdg) * dgdtdt);
365 }
366 return h;
367 }
