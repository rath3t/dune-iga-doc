14 template <int dim, int dimworld, typename ScalarType>
15 auto TrimmerImpl<dim, dimworld, ScalarType>::trimElement(const auto& element, const PatchTrimData& patchTrimData) {
16 // std::cout << "START " << std::endl;
17 auto geo = element.geometry();
18 std::array<FieldVector<double, 2>, 4> corners; // see dune book page 127 Figure 5.12
19 corners[0] = geo.corner(0);
20 corners[1] = geo.corner(1);
21 corners[2] = geo.corner(3);
22 corners[3] = geo.corner(2);
23
24 auto [flag, result] = Util::clipElementRectangle(element, patchTrimData);
25
26 // Create ElementTrimData with exact intersection Points and a geometry representation of the element edges
27 ElementTrimData elementTrimData(flag, element);
28
29 if (flag != ElementTrimFlag::trimmed) return elementTrimData;
30
31 /*
32 * Visitors to get vertex properties
33 */
34
35 auto nextEntity = [&](const int i) { return (i + 1) % result.vertices_.size(); };
36 auto isNewVertex = [](const auto& vV) { return std::holds_alternative<Util::ClippingResult::NewVertex>(vV); };
37 auto isHostVertex = [](const auto& vV) { return std::holds_alternative<Util::ClippingResult::HostVertex>(vV); };
38 auto isInsideVertex = [](const auto& vV) { return std::holds_alternative<Util::ClippingResult::InsideVertex>(vV); };
39 auto getPt = [](auto&& vV) { return vV.pt; };
40 auto getHostIdx = [](const auto& vV) { return std::get<Util::ClippingResult::HostVertex>(vV).hostIdx; };
41 auto getEdgeIdx = [](const auto& vV) { return std::get<Util::ClippingResult::NewVertex>(vV).onEdgeIdx; };
42 auto getCurveI = [](const auto& vV) {return std::get<Util::ClippingResult::InsideVertex>(vV).curveIdxI; };
43 auto getCurveJ = [](const auto& vV) {return std::get<Util::ClippingResult::InsideVertex>(vV).curveIdxJ; };
44 auto getLoopIdx = [](const auto& vV) {return std::get<Util::ClippingResult::InsideVertex>(vV).loopIdx; };
45 auto getTrimmingCurveIdx = [&](auto& vV) -> std::pair<size_t, size_t> {
46 return patchTrimData.getIndices(std::get<Util::ClippingResult::NewVertex>(vV).trimmingCurveZ);
47 };
48
50 // Here the actual code is starting
52
53 // Major todo: Create fallback to straight line if alog is not able to find correct Trimming CurveIdx
54 // Also todo: Make a second algo that just connects the points as lines (as template?)
55
56 // State
57 std::vector<FieldVector<ScalarType, dim>> foundVertices;
58 bool isOnNewEdge = false;
59 std::pair currentCurveIdx = {std::numeric_limits<size_t>::infinity(), std::numeric_limits<size_t>::infinity()};
60 double currentT = std::numeric_limits<double>::infinity();
61
62 for (const auto i : std::views::iota(0u, result.vertices_.size())) {
63 auto vV1 = result.vertices_[i];
64 auto vV2 = result.vertices_[nextEntity(i)];
65
66 auto pt1 = std::visit(getPt, vV1);
67 auto pt2 = std::visit(getPt, vV2);
68
69 // First case edge is completly untrimmed
70 if (isHostVertex(vV1) and isHostVertex(vV2)) {
71 elementTrimData.addEdge(Util::giveEdgeIdx(getHostIdx(vV1), getHostIdx(vV2)));
72 foundVertices.push_back({pt2.x, pt2.y});
73 continue;
74 }
75
76 // Second case edge begins on a hostVertes and ends on a newVertex
77 if (isHostVertex(vV1) and isNewVertex(vV2)) {
78 currentCurveIdx = getTrimmingCurveIdx(vV2);
79 auto [tParam, curvePoint]
80 = Util::callFindIntersection(patchTrimData.getCurve(currentCurveIdx), getEdgeIdx(vV2), pt2, corners);
81 std::cout << "Found: " << curvePoint << " From Clipping: " << pt2.x << " " << pt2.y << " t: " << tParam
82 << std::endl;
83
84 FieldVector<ScalarType, dim> p
85 = foundVertices.empty() ? FieldVector<ScalarType, dim>{pt1.x, pt1.y} : foundVertices.back();
86 auto trimmedEdge = Util::createHostGeometry<TrimmingCurve>(p, curvePoint);
87 elementTrimData.addEdgeHostNew(getEdgeIdx(vV2), trimmedEdge, curvePoint);
88 foundVertices.push_back(curvePoint);
89
90 isOnNewEdge = true;
91 currentT = tParam;
92 continue;
93 }
94 // Third case newVertex - newVertex
95 if (isNewVertex(vV1) and isNewVertex(vV2)) {
96 if (foundVertices.empty()) {
97 currentCurveIdx = getTrimmingCurveIdx(vV1);
98 FieldVector<ScalarType, dim> curvePoint;
99 std::tie(currentT, curvePoint)
100 = Util::callFindIntersection(patchTrimData.getCurve(currentCurveIdx), getEdgeIdx(vV1), pt1, corners);
101 std::cout << "Found: " << curvePoint << " From Clipping: " << pt1.x << " " << pt1.y << " t: " << currentT
102 << std::endl;
103 foundVertices.push_back(curvePoint);
104 }
105 assert(getTrimmingCurveIdx(vV2) == currentCurveIdx);
106
107 auto [tParam, curvePoint]
108 = Util::callFindIntersection(patchTrimData.getCurve(currentCurveIdx), getEdgeIdx(vV2), pt2, corners);
109 std::cout << "Found: " << curvePoint << " From Clipping: " << pt2.x << " " << pt2.y << " t: " << tParam
110 << std::endl;
111
112 if (currentT > tParam) {
113 assert(getEdgeIdx(vV1) == getEdgeIdx(vV2));
114 auto trimmedEdge = Util::createHostGeometry<TrimmingCurve>(foundVertices.back(), curvePoint);
115 elementTrimData.addEdgeNewNewOnHost(getEdgeIdx(vV2), trimmedEdge, curvePoint);
116 isOnNewEdge = true;
117 currentT = tParam;
118 } else {
119 auto elementTrimmingCurve
120 = Util::createTrimmingCurveSlice(patchTrimData.getCurve(currentCurveIdx), currentT, tParam);
121 elementTrimData.addEdgeNewNew(elementTrimmingCurve, curvePoint);
122 isOnNewEdge = false;
123 currentT = std::numeric_limits<double>::infinity();
124 }
125 foundVertices.push_back(curvePoint);
126
127 continue;
128 }
129 // Fourth case edge begins on a newVertex and ends in a HostVertex
130 if (isNewVertex(vV1), isHostVertex(vV2)) {
131 auto v2 = Dune::FieldVector<ScalarType, dim>{pt2.x, pt2.y};
132
133 FieldVector<ScalarType, dim> p;
134 if (foundVertices.empty())
135 std::tie(std::ignore, p) = Util::callFindIntersection(patchTrimData.getCurve(getTrimmingCurveIdx(vV1)),
136 getEdgeIdx(vV1), pt1, corners);
137 else
138 p = foundVertices.back();
139
140 auto trimmedEdge = Util::createHostGeometry<TrimmingCurve>(p, v2);
141 elementTrimData.addEdgeNewHost(getEdgeIdx(vV1), trimmedEdge, getHostIdx(vV2));
142
143 foundVertices.push_back({pt2.x, pt2.y});
144 continue;
145 }
146 // Additional cases to cover inside vertices
147 if (isNewVertex(vV1), isInsideVertex(vV2)) {
148 if (foundVertices.empty()) {
149 currentCurveIdx = getTrimmingCurveIdx(vV1);
150 FieldVector<ScalarType, dim> curvePoint;
151 std::tie(currentT, curvePoint)
152 = Util::callFindIntersection(patchTrimData.getCurve(currentCurveIdx), getEdgeIdx(vV1), pt1, corners);
153 std::cout << "Found: " << curvePoint << " From Clipping: " << pt1.x << " " << pt1.y << " t: " << currentT
154 << std::endl;
155 foundVertices.push_back(curvePoint);
156 }
157 const auto& curve = patchTrimData.loops()[getLoopIdx(vV2)].curves()[getCurveI(vV2)];
158 double tParam = curve.domain().front().back();
159 auto elementTrimmingCurve = Util::createTrimmingCurveSlice(curve, currentT, tParam);
160 auto curvePoint = curve.corner(1);
161 elementTrimData.addEdgeNewNew(elementTrimmingCurve, curvePoint);
162 foundVertices.push_back(curvePoint);
163 continue;
164 }
165 if (isInsideVertex(vV1), isNewVertex(vV2)) {
166 const auto& curve = patchTrimData.loops()[getLoopIdx(vV1)].curves()[getCurveJ(vV1)];
167 currentT = curve.domain().front().front();
168 auto [tParam, curvePoint]
169 = Util::callFindIntersection(curve, getEdgeIdx(vV2), pt2, corners);
170 std::cout << "Found: " << curvePoint << " From Clipping: " << pt2.x << " " << pt2.y << " t: " << tParam
171 << std::endl;
172 auto elementTrimmingCurve
173 = Util::createTrimmingCurveSlice(curve, currentT, tParam);
174 elementTrimData.addEdgeNewNew(elementTrimmingCurve, curvePoint);
175 foundVertices.push_back(curvePoint);
176 continue;
177 }
178 }
179 elementTrimData.finalize();
180 return elementTrimData;
181 }
