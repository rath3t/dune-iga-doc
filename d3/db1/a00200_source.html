85 // Coordinates in a single knot span differ from coordinates on the B-spline patch
86 // by an affine transformation. This transformation is stored in the diagonal entries.
87 // If trimming is disabled the Jacobian in the parameter space of subentities (edges or surfaces) is a
88 // DiagonalMatrixBlock but is treated as a FieldMatrix, since is also just cubes and an axis-aligned geometry, if
89 // trimming is enabled the Jacobian of the parameterspace for subentities is a potentially fully populated
90 // FieldMatrix, Think about an arbitrary curve in the axis-aligned cube parameter grid
91 // ------->
92 // `____________________C_______D_______`
93 // ::````````````````::``````````'|````::
94 // :: :: `| ::
95 // :: ^F:: `| ::
96 // :: | :: `| ::
97 // :: | :: `| ::
98 // :: |E:: `| ::
99 // :: :: .| x ::
100 // :: :: _: ::
101 // ::................::..........|:....::
102 // ::''''''''''''''''::''''''''''/'''''::
103 // :: :: _:B ::
104 // :: :: .| ::
105 // :: :: `::A ::
106 // :: '::::||::::::' ::
107 // :: ::` :: x ::
108 // :: |' :: ::
109 // :: | x :: ::
110 // ::````````|```````::````````````````::
111 // `____________________________________`
112 // at point A the tangent of the curve (the JacobianInParameterSpace) points into the oblique direction (to B)
113 // for the axis-aligned geometry these tangents are from E to F or from C to D, depending on the orientation of
114 // the subentity, which are aligned with the axis
115 using JacobianTransposedInParameterSpace = typename ParameterSpaceGeometry::JacobianTransposed;
116 using GlobalInParameterSpace = typename ParameterSpaceGeometry::GlobalCoordinate;
117

316 Hessian hessian(const LocalCoordinate& local) const {
317 checkState();
318 const auto ouInPatch = globalInParameterSpace(local);
319
320 const auto [p, dfdg, dfdgdg] = patchGeometry_->zeroFirstAndSecondDerivativeOfPositionImpl(
321 ouInPatch, nurbsLocalView_, localControlPointNet);
322
323 const JacobianTransposedInParameterSpace dgdt = jacobianTransposedInParameterSpace(local);
324 static_assert(JacobianTransposedInParameterSpace::rows == mydimension);
325 static_assert(JacobianTransposedInParameterSpace::cols == gridDimension);
326 // dgdt (1x2, curve on surface), (2x2 surface in surface), (2x3 surface in 3D), (3x3 volume in 3D)
327 FieldMatrix<ctype, patchNumberOfSecondDerivatives, numberOfSecondDerivatives> dgdtSquared(0);
328 for (int patchIndex = 0; auto [patchRow, patchCol] : Impl::voigtIndices<gridDimension>()) {
329 for (int myIndex = 0; auto [myRow, myCol] : Impl::voigtIndices<mydimension>()) {
330 if constexpr (not std::is_same_v<JacobianTransposedInParameterSpace, DiagonalMatrix<ctype, gridDimension>>)
331 dgdtSquared[patchIndex][myIndex]
332 = (dgdt[myRow][patchRow] * dgdt[myCol][patchCol] + dgdt[myCol][patchRow] * dgdt[myRow][patchCol])
333 * ((patchRow == patchCol) ? 0.5 : 1.0);
334 else {
335 const auto dgdtmyRowpatchRow = (myRow == patchRow) ? dgdt[myRow][patchRow] : 0;
336 const auto dgdtmyColpatchCol = (myCol == patchCol) ? dgdt[myCol][patchCol] : 0;
337 const auto dgdtmyColpatchRow = (myCol == patchRow) ? dgdt[myCol][patchRow] : 0;
338 const auto dgdtmyRowpatchCol = (myRow == patchCol) ? dgdt[myRow][patchCol] : 0;
339 dgdtSquared[patchIndex][myIndex]
340 = (dgdtmyRowpatchRow * dgdtmyColpatchCol + dgdtmyColpatchRow * dgdtmyRowpatchCol)
341 * ((patchRow == patchCol) ? 0.5 : 1.0);
342 }
343 ++myIndex;
344 }
345 ++patchIndex;
346 }
347
348 // h = mydimension * (mydimension + 1) / 2 X worlddimension = 1x2 (curve in 2D), 1x3 (curve in 3D), (3x2 surface
349 // in 2D), (3x3 surface in 3D), (6x3 volume in 3D)
350 //
351 // dfdgdg = gridDimension*(gridDimension + 1) / 2 X worlddimension
352 // = 3x2 (curve on surface flat),3x3 (curve on surface in 3D), 6x3 (curve in volume), 6x3 (surface in
353 // volume),(3x3 surface in surface in 3D)
354 //
355 // dgdtSquared = gridDimension * (gridDimension + 1) / 2 X mydimension * (mydimension +
356 // 1) / 2 = 3x1 (curve on surface), 6x1 (curve in volume), 6x3 (surface in volume), 3x3 ( surface in surface),
357 // 6x6 ( volume in volume)
358 Hessian h = transposedView(dgdtSquared) * dfdgdg;
359
360 /* if trimming is enabled the parameter space geometry is potentially non-linear,
361 * the resutling Hessian has another contribution due to chain-rule, namely the second derivative of g */
362 if constexpr (not Trimmer::template isLocalGeometryLinear<codim>) {
363 // const auto dgdtdt = parameterSpaceGeometry->hessian(ouInPatch);
364 //
365 // assert(TrimmerType::template isLocalGeometryLinear<
366 // codim> && "This can not be checked yet. Check if this works with trimming and then remove
367 // assert");
368 // h += transpose(transposedView(dfdg) * dgdtdt);
369 }
370 return h;
371 }
