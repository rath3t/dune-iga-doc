196 auto subEntities(int i, int c, int cc) const {
197 // auto subEntityName = [](std::size_t codim)-> std::string {

203 if (trimData_.flag() == ElementTrimFlag::full) {
204 auto range = cubeGeometry.subEntities(i, c, cc);
205 return SubEntityRangeImpl(range.begin(), range.end(), true, i, c, cc);
206 }
207 if (trimData_.flag() == ElementTrimFlag::trimmed) {
208 if (cc < c)
209 return SubEntityRangeImpl(dummyArray.begin(), dummyArray.end());
210 else if (i == 0 and c == 0
211 and cc == 0) // iterate over the entity itself (only one element in storage with value zero)
212 {
213 SubEntityRangeImpl range(elementIndexDummy.begin(), elementIndexDummy.end());
214 return range;
215 } else
216
217 if (i == 0 and c == 0 and cc == 1) // iterate over the edges of the element
218 {
219 SubEntityRangeImpl range(edgeIndices.begin(), edgeIndices.end());
220 return range;
221 } else if (c == 0 and cc == 2) // iterate over the vertices of the element
222 {
223 SubEntityRangeImpl range(verticesIndices.begin(), verticesIndices.end() - 1);
224 return range;
225 }
226
227 else if (c == 1 and cc == 2) // iterate over the vertices of edge
228 {
229 SubEntityRangeImpl range(verticesIndices.begin() + i, verticesIndices.begin() + i + 2);
230 return range;
231 }
232
233 else if (c == cc) // iterate over the vertices of edge
234 {
235 if (c == 2) {
236 SubEntityRangeImpl range(verticesIndices.begin() + i, verticesIndices.begin() + i + 1);
237 return range;
238 } else if (cc == 1) {
239 SubEntityRangeImpl range(edgeIndices.begin() + i, edgeIndices.begin() + i + 1);
240 return range;
241 }
242 }
243 }
244
245 assert(false);

271 bool contains(std::size_t i) const {
272 if (host_)
273 return cubeGeometry.subEntities(i_, c_, cc_).contains(i);
274 else {
275 if (size_ == 0)
276 return false;
277 else if (size_ == 2) // two vertices at one edge case
278 {
279 const int back = *(end() - 1);
280 const int front = *begin();
281 return i == front or i == back;
282 }

414 explicit TrimmedReferenceElement(const TrimDataType& trimData) : trimData_{trimData} {
415 if (trimData_.flag() == ElementTrimFlag::trimmed) {
416 edgeIndices.resize(trimData_.edges().size());
417 for (auto eI : Dune::range(trimData_.edges().size()))
418 edgeIndices[eI] = eI;
419
420 verticesIndices.resize(trimData_.vertices().size() + 1);
421 for (auto vI : Dune::range(trimData_.vertices().size())) {
422 verticesIndices[vI] = vI;
423 }
424 // to mimic a circular vector we add the first vertex at the end
425 verticesIndices.back() = 0;
