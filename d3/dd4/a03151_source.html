44 template <typename HostIdType>
45 struct IdType {
46 enum class HostOrTrimmed { host, trimmed };
47 HostOrTrimmed elementState{HostOrTrimmed::host};
48 HostIdType id{};
49
50 bool operator==(const IdType& other) const { return (elementState == other.elementState) and (id == other.id); }
51
52 friend std::ostream& operator<<(std::ostream& stream, const IdType& id) {
53 stream << "Type: " << std::string(id.elementState == HostOrTrimmed::host ? "Host" : "Trimmed")
54 << ", Key: " << id.id << "\n";
55 return stream;
56 }
57 };

59 template <typename HostIdType>
60 bool operator<(const IdType<HostIdType>& lhs, const IdType<HostIdType>& rhs) {
61 if (lhs.elementState == rhs.elementState)
62 return lhs.id < rhs.id;
63 else if (lhs.elementState < rhs.elementState)
64 return true;
65 else
66 return false;
67 }
