template<int dim, int dimworld, typename ScalarType>
class Dune::IGANEW::DefaultTrim::TrimmerImpl< dim, dimworld, ScalarType >

PatchTrimData struct representing trim data for a patch.

Trimmer struct representing a trimmer with identity trimming (no trimming at all).
