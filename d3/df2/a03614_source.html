19 inline Type typeForTypeString(const std::string& typeString) {
20 if (typeString == "NurbsCurveGeometry2D")
21 return Type::NurbsCurveGeometry2D;
22 else if (typeString == "NurbsSurfaceGeometry3D")
23 return Type::NurbsSurfaceGeometry3D;
24 else if (typeString == "BrepLoop")
25 return Type::BrepLoopType;
26 else if (typeString == "BrepTrim")
27 return Type::BrepTrimType;
28 else if (typeString == "Brep")
29 return Type::BrepType;
30 else

45 template <int dim, int worldDim>
46 requires(dim < 3) && (worldDim >= dim) struct IbraNURBSData : public IbraBase {
47 std::array<int, dim> degree{};
48 std::array<int, dim> n_controlPoints{};
49 std::array<std::vector<double>, dim> knots;
50 Dune::DynamicMatrix<double> controlPoints;
51 std::vector<double> weights;
52
53 template <typename ControlPointType>
54 [[nodiscard]] std::vector<std::vector<ControlPointType>> transformControlPoints() const {
55 if constexpr (dim == 1) {
56 std::vector<ControlPointType> cps;
57 for (int i = 0; i < n_controlPoints[0]; ++i) {
58 cps.push_back(controlPointAt<ControlPointType>({i}));
59 }
60 return {cps};
61 } else if constexpr (dim == 2) {
62 std::vector<std::vector<ControlPointType>> vec;
63 std::vector<ControlPointType> cpTemp(n_controlPoints[1]);
64 for (int i = 0; i < n_controlPoints[0]; ++i) {
65 cpTemp.clear();
66 for (int j = 0; j < n_controlPoints[1]; ++j) {
67 cpTemp.push_back(controlPointAt<ControlPointType>({i, j}));
68 }
69 vec.push_back(cpTemp);
70 }
71 return vec;
72 }
73 __builtin_unreachable();
74 }
75
76 [[nodiscard]] std::array<std::vector<double>, dim> compileKnotVectors() const {
77 std::array<std::vector<double>, dim> knotVec;
78
79 // Insert first and last knot repeatedly
80 for (int i = 0; i < dim; ++i) {
81 knotVec[i].insert(knotVec[i].end(), knots[i][0]);
82 knotVec[i].insert(knotVec[i].end(), knots[i].begin(), knots[i].end());
83 knotVec[i].insert(knotVec[i].end(), knots[i].back());
84 }
85 return knotVec;
86 }
87
88 private:
89 template <typename ControlPointType>
90 [[nodiscard]] ControlPointType controlPointAt(std::array<int, dim> idx) const {
91 int row = (dim == 1) ? idx[0] : idx[0] * n_controlPoints[1] + idx[1];
92
93 Dune::FieldVector<double, worldDim> p;
94 for (int k = 0; k < worldDim; ++k)
95 p[k] = controlPoints[row][k];
96
97 double w = weights[row];
98
99 return {p, w};
100 }
101 };

118 BrepTrim(BrepTrimRepresentation& trimRepresentation, std::vector<Curve2D>& allCurves)
119 : domain{trimRepresentation.domain}, IbraBase{static_cast<IbraBase&>(trimRepresentation)} {
120 auto it = std::find_if(allCurves.begin(), allCurves.end(),
121 [trimRepresentation](auto x) { return (x.key == trimRepresentation.geometry); });
122 if (it != allCurves.end())
123 geometry = *it;
124 else
125 DUNE_THROW(Dune::InvalidStateException, "Couldn't find geometry in BrepTrim: " << key);
126 }

128 template <typename TrimmingCurve>
129 auto asCurve() const {
130 using ControlPointType = typename TrimmingCurve::ControlPointType;
131 using ControlPointNetType = typename TrimmingCurve::ControlPointNetType;
132 using PatchData = typename TrimmingCurve::PatchData;
133
134 const auto cp = geometry.transformControlPoints<ControlPointType>()[0];
135 std::array dimSize{static_cast<int>(cp.size())};
136 std::array knotSpans{geometry.compileKnotVectors()};
137 ControlPointNetType controlNet{dimSize, cp};
138
139 return TrimmingCurve(PatchData(knotSpans, controlNet, std::array(geometry.degree)));
140 }
141 };

174 template <int worldDim>
175 struct Brep : IbraBase {
176 std::vector<Curve2D> curves2D;
177 using SurfaceType = Surface<worldDim>;
178 std::vector<SurfaceType> surfaces;
179 std::vector<BrepLoop> loops;
180 std::vector<BrepTrim> trims;
181
182 Brep(const BrepRepresentation& representation, std::vector<Curve2D>& allCurves,
183 std::vector<SurfaceType>& allSurfaces, std::vector<Ibra::BrepLoopRepresentation>& allBrepLoopRepresentations,
184 std::vector<Ibra::BrepTrimRepresentation>& allBrepTrimRepresentations)
185 : IbraBase{static_cast<const BrepRepresentation&>(representation)} {
186 // Lambda to check if a given key string is in a list of strings
187 auto objectIsInList = [](const std::vector<std::string>& strRepr) {
188 return [&](const auto& keyString) -> bool { return std::ranges::find(strRepr, keyString) != strRepr.end(); };
189 };
190
191 // Find in all Curves
192 std::ranges::copy_if(allCurves, std::back_inserter(curves2D), objectIsInList(representation.curve_geometries_2d),
193 &Curve2D::key);
194 // and all Surfaces
195 std::ranges::copy_if(allSurfaces, std::back_inserter(surfaces),
196 objectIsInList(representation.surface_geometries_3d), &SurfaceType::key);
197
198 // Find all Trims
199 for (auto& trimRepr : allBrepTrimRepresentations) {
200 if (objectIsInList(representation.trims)(trimRepr.key)) trims.emplace_back(trimRepr, curves2D);
201 }
202 // All Loops
203 for (auto& loopRepr : allBrepLoopRepresentations) {
204 if (objectIsInList(representation.loops)(loopRepr.key)) loops.emplace_back(loopRepr, trims);
205 }
206
207 // Asserts if all geometries were found
208 assert(curves2D.size() == representation.curve_geometries_2d.size());
209 assert(surfaces.size() == representation.surface_geometries_3d.size());
210 assert(loops.size() == representation.loops.size());
211 assert(trims.size() == representation.trims.size());
212 }
213 };

229 template <int worldDim>
230 void from_json(const json& j, Curve<worldDim>& curve) {
231 getGenerics(j, curve);
232
233 // Degree
234 j.at("degree").get_to(curve.degree[0]);
235
236 // Knots (u)
237 std::ranges::copy(j.at("knots"), std::back_inserter(curve.knots[0]));
238
239 // Poles aka Control Points
240 j.at("nb_poles").get_to(curve.n_controlPoints[0]);
241
242 curve.controlPoints.resize(curve.n_controlPoints[0], worldDim);
243
244 for (int i = 0; i < curve.n_controlPoints[0]; ++i)
245 for (int k = 0; k < worldDim; ++k)
246 curve.controlPoints[i][k] = j.at("poles")[i][k];
247
248 if (j.contains("weights"))
249 std::ranges::copy(j.at("weights"), std::back_inserter(curve.weights));
250 else {
251 curve.weights.resize(curve.n_controlPoints[0]);
252 std::ranges::fill(curve.weights, 1.0);
253 }
254 }

256 template <int worldDim>
257 void from_json(const json& j, Surface<worldDim>& surface) {
258 getGenerics(j, surface);
259
260 // Degree
261 j.at("degree_u").get_to(surface.degree[0]);
262 j.at("degree_v").get_to(surface.degree[1]);
263
264 // Knots (u,v)
265 std::ranges::copy(j.at("knots_u"), std::back_inserter(surface.knots[0]));
266 std::ranges::copy(j.at("knots_v"), std::back_inserter(surface.knots[1]));
267
268 // Poles aka Control Points
269 j.at("nb_poles_u").get_to(surface.n_controlPoints[0]);
270 j.at("nb_poles_v").get_to(surface.n_controlPoints[1]);
271
272 int total_controlPoints = surface.n_controlPoints[0] * surface.n_controlPoints[1];
273 surface.controlPoints.resize(total_controlPoints, worldDim);
274
275 for (int i = 0; i < total_controlPoints; ++i)
276 for (int k = 0; k < worldDim; ++k)
277 surface.controlPoints[i][k] = j.at("poles")[i][k];
278
279 if (j.contains("weights"))
280 std::ranges::copy(j.at("weights"), std::back_inserter(surface.weights));
281 else {
282 surface.weights.resize(total_controlPoints);
283 std::ranges::fill(surface.weights, 1.0);
284 }
285 }
