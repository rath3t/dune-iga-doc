88 NURBSGridEntity(const GridView& gridView, unsigned int directIndex, unsigned int patchID)
89 : NURBSGridView_(&gridView),
90 directIndex_(directIndex),
91 patchID_{patchID},
92 parType_{PartitionType::InteriorEntity},
93 trimFlag{NURBSGridView_->impl().getPatch(patchID_).getTrimFlag(directIndex_)},
94 intersections_{std::make_shared<std::vector<Intersection>>()} // FIXME why shared?
95 {
96 intersections_->reserve(this->subEntities(1));
97
98 for (int innerLocalIndex = 0, outerLocalIndex = 1; innerLocalIndex < this->subEntities(1); ++innerLocalIndex) {
99 const auto& eleNet = NURBSGridView_->impl().getPatch(patchID_).elementNet_;
100 auto nurbsDirectIndex = NURBSGridView_->impl().getPatch(patchID_).template getDirectIndex<0>(directIndex_);
101 auto multiIndex = eleNet->directToMultiIndex(nurbsDirectIndex);
102 multiIndex[static_cast<int>(std::floor(innerLocalIndex / 2))]
103 += ((innerLocalIndex % 2)
104 ? 1
105 : Impl::noNeighbor); // increase the multiIndex depending on where the outer element should lie
106 auto directOuterIndex = (eleNet->isValid(multiIndex)) ? eleNet->index(multiIndex) : Impl::noNeighbor;
107 directOuterIndex = getRealIndexForOuterIndex(directOuterIndex);
108 intersections_->emplace_back(NURBSintersection<GridImpl>(innerLocalIndex, outerLocalIndex, directIndex_,
109 directOuterIndex, *NURBSGridView_));
110 outerLocalIndex += ((innerLocalIndex - 1) % 2) ? -1 : 3;
111 }
112 }

116 void fillQuadratureRule(Dune::QuadratureRule<double, dim>& vector, const std::optional<int>& p_order = std::nullopt,
117 const QuadratureType::Enum qt = QuadratureType::GaussLegendre) const {
118 vector.clear();
119 int order = p_order.value_or(mydimension
120 * (*std::ranges::max_element(NURBSGridView_->impl().getPatchData(patchID_).degree)));
121 if constexpr (dim == 2)
122 if (isTrimmed()) {
123 auto elementRepr = NURBSGridView_->impl().getPatch(patchID_).getElementSubGrid(directIndex_);
124 fillQuadratureRuleImpl(vector, *elementRepr, order, qt);
125 return;
126 }
127 const auto& rule = Dune::QuadratureRules<double, mydimension>::rule(this->type(), order, qt);
128 vector.insert(vector.end(), rule.begin(), rule.end());
129 }

170 template <int codimSub>
171 typename GridImpl::Traits::template Codim<codimSub>::Entity subEntity(int i) const {
172 if constexpr (codimSub == 0) {
173 assert(i == 0);
174 return *this;
175 } else if constexpr (codimSub == mydimension) // vertices from elements
176 {
177 auto globalIndex
178 = NURBSGridView_->impl().getPatch(patchID_).getGlobalVertexIndexFromElementIndex(directIndex_, i);
179 return NURBSGridView_->impl().template getEntity<codimSub>(globalIndex);
180 } else if constexpr (mydimension - codimSub == 1) // edges from elements
181 {
182 auto globalIndex
183 = NURBSGridView_->impl().getPatch(patchID_).getGlobalEdgeIndexFromElementIndex(directIndex_, i);
184 return NURBSGridView_->impl().template getEntity<codimSub>(globalIndex);
185 } else if constexpr (mydimension - codimSub == 2) // surfaces from elements
186 {
187 auto globalIndex
188 = NURBSGridView_->impl().getPatch(patchID_).getGlobalSurfaceIndexFromElementIndex(directIndex_, i);
189 return NURBSGridView_->impl().template getEntity<codimSub>(globalIndex);
190 }
191 throw std::logic_error("The requested subentity codim combination is not supported ");
192 }
