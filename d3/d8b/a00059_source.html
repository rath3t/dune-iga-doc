133 void findInnerLoops(std::optional<InnerLoops>& innerLoops, const ClipperPaths& clippedEdges) const {
134 assert(clip_.size() > 1);
135 assert(clippedEdges.size() > 1);
136 std::vector<int> foundLoops;
137
138 auto checkEquality = [](const ClipperPath& p1, const ClipperPath& p2) -> bool {
139 auto accumulator = [](intType rhs, const ClipperPoint& point) -> intType { return rhs + point.x + point.y; };
140 auto sumP1 = std::accumulate(p1.begin(), p1.end(), intType(0), accumulator);
141 auto sumP2 = std::accumulate(p2.begin(), p2.end(), intType(0), accumulator);
142
143 return intCmpEq(sumP1, sumP2, 100l);
144 };
145
146 for (auto it1 = clippedEdges.cbegin() + 1; it1 < clippedEdges.cend(); ++it1)
147 for (auto it2 = clip_.cbegin() + 1; it2 < clip_.cend(); ++it2)
148 if (checkEquality(*it1, *it2)) foundLoops.push_back(std::distance(clip_.begin(), it2));
149
150 if (not foundLoops.empty()) innerLoops.emplace(foundLoops);
151 }

153 void clipElement(const auto& element) {
154 auto index = indexSet_.index(element);
155 auto corners = getElementCorners(index);
156 auto elementEdges = getElementEdgesFromElementCorners(corners);
157
158 ClipperPaths clippedEdges;
159 // Rule NonZero to ensure that inner boundaries are defined clockwise and outer boundaries counter-clockwise
160 /*
161 if (clip_.size() == 1) {
162 Clipper2Lib::Rect<intType> elementRect{corners[0][0], corners[1][1], corners[1][0], corners[3][1]};
163 clippedEdges = Clipper2Lib::RectClip(elementRect, clip_);
164 } else
165 */
166 clippedEdges = Clipper2Lib::Intersect(ClipperPaths{elementEdges}, clip_, Clipper2Lib::FillRule::NonZero);
167
168 // At the moment there is no hole, if there are more than 2 Paths, then there is a hole
169 std::optional<InnerLoops> innerLoops = std::nullopt;
170 if (clippedEdges.size() > 1) {
171 findInnerLoops(innerLoops, clippedEdges);
172 }
173
174 // If the clippedEdges are empty, this means the element is outside the clip -> empty
175 // And if the clip has only 4 corners and these corners are the same as the element -> full
176 if (clippedEdges.empty())
177 trimFlags_.push_back(ElementTrimFlag::empty);
178 else {
179 if (isFullElement(clippedEdges.front(), corners) and not innerLoops.has_value())
180 trimFlags_.push_back(ElementTrimFlag::full);
181 else
182 trimFlags_.push_back(ElementTrimFlag::trimmed);
183 }
184
185 if (trimFlags_.back() != ElementTrimFlag::trimmed) {
186 clipResults_.push_back(std::nullopt);
187 return;
188 }
189
190 // Fill pointMap to store the intersection points in regard to their corresponding edges
191 auto pointMap = std::make_unique<IntersectionPointMap>();
192
193 // Look if there are two distinct regions // Not yet implemented
194 int regions = clippedEdges.size() - (innerLoops.has_value() ? innerLoops.value().loopIndices.size() : 0);
195 if (regions > 1) std::cerr << "More than one region, undefined behaviour, inspect output " << std::endl;
196
197 // This keeps track of how many intersection Points there are at a given edge
198 std::array<int, 4> edgeCounter{0, 0, 0, 0};
199 std::array<int, 4> nodeCounter{0, 0, 0, 0};
200
201 for (auto& point : clippedEdges.front()) {
202 auto [isOnAnyEdge, edgeThePointIsOn] = pointOnAnyEdge(point, elementEdges);
203 auto [isOnAnyNode, nodeThePointIsOn] = pointOnAnyVertex(point, corners);
204
205 // If is onAnyNode is true, then isOnAnyEdge automatically is also true, so test onAnyNode first
206 if (isOnAnyNode)
207 nodeCounter[nodeThePointIsOn] = 1;
208 else if (isOnAnyEdge) {
209 (*pointMap)[edgeThePointIsOn].emplace_back(point);
210 ++edgeCounter[edgeThePointIsOn];
211 }
212 }
213
214 // Sort IntersectionPoints counter-clockwise
215 sortIntersectionPoints(*pointMap);
216
217 clipResults_[index].emplace(std::move(pointMap), edgeCounter, nodeCounter, std::move(innerLoops));
218 }

220 std::optional<ElementBoundaries> constructElementBoundaries(int directIndex) {
221 std::optional<ElementBoundaries> result{};
222 try {
223 std::vector<Boundary> elementBoundaries;
224 auto& clipResult = clipResults_[directIndex].value();
225 auto corners = getElementCorners(directIndex);
226
227 // Determine the node where we begin to trace
228 const auto nodeCounter = clipResult.nodeCounter;
229 auto it_nodes = std::ranges::find_if(nodeCounter, [](auto x) { return x > 0; });
230
231 if (it_nodes == nodeCounter.end())
232 throw std::runtime_error(
233 "No Curve tracing scheme implemented for elements that are only trimmed on single element edges.");
234
235 auto nodeToBegin = std::distance(nodeCounter.begin(), it_nodes);
236 auto state = FindNextBoundaryLoopState(nodeToBegin, corners, clipResult);
237
238 // Iteratively find all the elementBoundaries
239 bool found;
240 do {
241 found = findNextBoundary(&state, elementBoundaries);
242 } while (!found);
243
244 // Success
245 result.emplace(elementBoundaries, extractInnerBoundaries(clipResult));
246 } catch (const std::runtime_error& e) {
247 std::cout << "Trimming encountered problem: " << e.what() << "\n Element is ignored" << std::endl;
248 }
249 return result;
250 }

252 std::optional<std::vector<std::vector<Boundary>>> extractInnerBoundaries(const ClippingResult& clipResult) const {
253 if (not clipResult.innerLoops.has_value()) return std::nullopt;
254
255 std::vector<std::vector<Boundary>> innerBoundaries;
256 innerBoundaries.resize(clipResult.innerLoops.value().loopIndices.size());
257
258 for (auto c = 0u; auto& loopIndex : clipResult.innerLoops.value().loopIndices) {
259 assert(loopIndex < trimData_->boundaryLoops.size());
260 innerBoundaries[c].reserve(trimData_->boundaryLoops[loopIndex].size());
261 std::ranges::copy(trimData_->boundaryLoops[loopIndex], std::back_inserter(innerBoundaries[c]));
262 ++c;
263 }
264
265 return std::make_optional(innerBoundaries);
266 }

268 // @todo Get rid of weird and redundant naming of outer-loop variables
269 bool findNextBoundary(FindNextBoundaryLoopState* state, std::vector<Boundary>& elementBoundaries) {
270 int node = state->currentNode;
271 int nextEntity = node;
272
273 // Check if nextEntity has no intersections and next node has one
274 if (state->isCurrentlyOnNode && !state->hasIntersectionPointOnEdgeNr(nextEntity)
275 && state->hasIntersectionPointOnNodeNr(nextEntityIdx(node, 1))) {
276 elementBoundaries.push_back(boundaryForEdge(state->corners, nextEntity));
277
278 // Update Node and Point
279 node = nextEntityIdx(node, 1);
280 state->currentNode = node;
281 state->currentNodePoint = toFloatDomain(state->corners[node]);
282
283 return (node == state->nodeToBegin);
284 }
285 if (state->isCurrentlyOnNode) {
286 // Determine the next nextEntity with an IP
287 for (int i = 0; i < 4; ++i) {
288 auto nextEdge = nextEntityIdx(node, i);
289 auto nextNode = nextEntityIdx(node, i + 1);
290
291 if (state->hasIntersectionPointOnEdgeNr(nextEdge)) {
292 nextEntity = nextEdge;
293 break;
294 }
295 if (state->hasIntersectionPointOnNodeNr(nextNode)) {
296 nextEntity = nextNode;
297 break;
298 }
299 }
300 // @todo Assert if nextEntity was found or not
301
302 // Set node for the next step
303 state->currentNode = node;
304 state->currentNodePoint = toFloatDomain(state->corners[node]);
305
306 int boundaryIndexThatHasIntersectionPoint;
307 IntersectionResult edgeIntersectResult;
308
309 if (node == nextEntity) {
311 auto findNextBoundaryResult
312 = findBoundaryThatHasIntersectionPoint(state->pointMapPtr.get(), nextEntity, state->corners);
313
314 if (not findNextBoundaryResult.has_value())
315 throw std::runtime_error("findBoundaryThatHasIntersectionPoint not successful");
316
317 boundaryIndexThatHasIntersectionPoint = findNextBoundaryResult.value().edge;
318 edgeIntersectResult = findNextBoundaryResult.value().result;
319
320 elementBoundaries.emplace_back(state->currentNodePoint, edgeIntersectResult.globalResult);
321 } else {
322 auto findNextBoundaryResult = findNextBoundaryThatHasIntersectionPointOnNode(state, node);
323
324 if (not findNextBoundaryResult.has_value())
325 throw std::runtime_error("findNextBoundaryThatHasIntersectionPointOnNode not successful");
326
327 boundaryIndexThatHasIntersectionPoint = findNextBoundaryResult.value().edge;
328 edgeIntersectResult = findNextBoundaryResult.value().result;
329 }
330
332 auto traceResult
333 = traceCurve(state, {boundaryIndexThatHasIntersectionPoint, edgeIntersectResult.localResult, nextEntity});
334
335 if (not traceResult.has_value()) throw std::runtime_error("traceCurve not successful");
336
337 for (const auto& result : traceResult.value()) {
338 auto curveToTrace = globalBoundaries_[result.boundaryIdx].nurbsGeometry;
339 Boundary newBoundary = Boundary(curveToTrace, Utilities::Domain<double>{result.beginU, result.intersectU});
340 elementBoundaries.push_back(newBoundary);
341 }
342
343 auto lastResult = traceResult.value().back();
344 state->isCurrentlyOnNode = lastResult.onNode;
345 state->currentNodePoint = lastResult.intersectionPoint;
346
347 if (lastResult.onNode)
348 state->currentNode = lastResult.edge;
349 else {
350 state->edgeTheLoopIsOn = lastResult.edge;
351 state->moreIntersectionPointsOnEdge
352 = getNextIntersectionPointOnEdge(state->pointMapPtr.get(), state->edgeTheLoopIsOn).second;
353 }
354 return lastResult.onNode && state->currentNode == state->nodeToBegin;
355 }
356
357 if (!(state->isCurrentlyOnNode) && state->moreIntersectionPointsOnEdge) {
358 nextEntity = state->edgeTheLoopIsOn;
359
361
362 auto findNextBoundaryResult
363 = findBoundaryThatHasIntersectionPoint(state->pointMapPtr.get(), nextEntity, state->corners);
364
365 if (not findNextBoundaryResult.has_value())
366 throw std::runtime_error("findBoundaryThatHasIntersectionPoint not successful");
367
368 auto [boundaryIndexThatHasIntersectionPoint, edgeIntersectResult] = findNextBoundaryResult.value();
369 elementBoundaries.emplace_back(state->currentNodePoint, edgeIntersectResult.globalResult);
370
372
373 auto traceResult
374 = traceCurve(state, {boundaryIndexThatHasIntersectionPoint, edgeIntersectResult.localResult, nextEntity});
375
376 if (not traceResult.has_value()) throw std::runtime_error("tracCurve not successful");
377
378 for (const auto& result : traceResult.value()) {
379 auto curveToTrace = globalBoundaries_[result.boundaryIdx].nurbsGeometry;
380 Boundary newBoundary = Boundary(curveToTrace, Utilities::Domain<double>{result.beginU, result.intersectU});
381 elementBoundaries.push_back(newBoundary);
382 }
383
384 auto lastResult = traceResult.value().back();
385 state->isCurrentlyOnNode = lastResult.onNode;
386 state->currentNodePoint = lastResult.intersectionPoint;
387
388 if (lastResult.onNode)
389 state->currentNode = lastResult.edge;
390 else {
391 state->edgeTheLoopIsOn = lastResult.edge;
392 state->moreIntersectionPointsOnEdge
393 = getNextIntersectionPointOnEdge(state->pointMapPtr.get(), state->edgeTheLoopIsOn).second;
394 }
395
396 return lastResult.onNode && state->currentNode == state->nodeToBegin;
397 }
398
399 if (!(state->isCurrentlyOnNode) && !(state->moreIntersectionPointsOnEdge)) {
400 elementBoundaries.emplace_back(state->currentNodePoint,
401 toFloatDomain(state->corners[nextEntityIdx(state->edgeTheLoopIsOn, 1)]));
402
403 state->isCurrentlyOnNode = true;
404 node = nextEntityIdx(state->edgeTheLoopIsOn, 1);
405
406 // Update Node and Point
407 state->currentNode = node;
408 state->currentNodePoint = toFloatDomain(state->corners[node]);
409 state->edgeTheLoopIsOn = -1;
410
411 return node == state->nodeToBegin;
412 }
413 __builtin_unreachable();
414 }

440 std::optional<FindNextBoundaryResult> findBoundaryThatHasIntersectionPoint(IntersectionPointMap* pointMapPtr,
441 int edge,
442 std::vector<IntPoint>& corners) {
443 auto edgeCurve = boundaryForEdge(corners, edge);
444 Point intersectionPointGuess = getNextIntersectionPointOnEdge(pointMapPtr, edge).first;
445
446 for (int i = 0; auto& boundary : globalBoundaries_) {
447 if (areLinesParallel(boundary.nurbsGeometry, edgeCurve.nurbsGeometry)) {
448 ++i;
449 continue;
450 }
451
452 auto edgeIntersectResult = findclosestPointOnCurve(boundary.nurbsGeometry, intersectionPointGuess);
453
454 if (edgeIntersectResult.found) {
455 markIntersectionPointAsVisited(pointMapPtr, edge, intersectionPointGuess);
456 assert(Dune::FloatCmp::eq(edgeIntersectResult.globalResult, intersectionPointGuess, 1e-3));
457 return std::make_optional(FindNextBoundaryResult(i, edgeIntersectResult));
458 }
459 ++i;
460 }
461
462 return std::nullopt;
463 }
464 std::optional<FindNextBoundaryResult> findNextBoundaryThatHasIntersectionPointOnNode(
465 FindNextBoundaryLoopState* state, int node) {
466 auto nodeCounter = state->nodeCounter;
467 if (nodeCounter[node] != 1) return std::nullopt;
468
469 Point intersectionPointGuess = toFloatDomain(state->corners[node]);
470 for (int i = 0; auto& boundary : globalBoundaries_) {
471 auto result = findclosestPointOnCurve(boundary.nurbsGeometry, intersectionPointGuess);
472
473 if (result.found) return FindNextBoundaryResult(i, result);
474 ++i;
475 }
476 return std::nullopt;
477 }
478 // @todo: Get rid of initializer lists
479 std::optional<std::vector<TraceCurveOutput>> traceCurve(FindNextBoundaryLoopState* state,
480 TraceCurveInput traceCurveInput) {
481 CurveGeometry curveToTrace = globalBoundaries_[traceCurveInput.boundaryIdx].nurbsGeometry;
482 auto [startPoint, endPoint] = curveStartEndPoint(curveToTrace);
483
484 // There is an edge case where we have a closed curve (e.g. a circle) that has endPoint and startPoint at the same
485 // point, and that point is in the element
486 if (Dune::FloatCmp::eq(startPoint, endPoint) && pointInElement(startPoint, state->corners)) {
487 auto traceCurveResult = traceCurveImpl(state, curveToTrace, traceCurveInput.startEdge);
488 if (traceCurveResult.has_value())
489 return std::make_optional<std::vector<TraceCurveOutput>>(
490 {{traceCurveInput.boundaryIdx, traceCurveInput.startU, curveToTrace.domain()[0].right(),
491 traceCurveResult->intersectionPoint, -1, false},
492 {traceCurveInput.boundaryIdx, curveToTrace.domain()[0].left(), traceCurveResult->intersectU,
493 traceCurveResult->intersectionPoint, traceCurveResult->edge, false}});
494 else
495 return std::nullopt;
496 }
497 // ... or the point is on an edge, even nastier
498 if (Dune::FloatCmp::eq(startPoint, endPoint) && pointInElementOrOnEdge(startPoint, state->corners)) {
499 if (Dune::FloatCmp::ne(curveToTrace(traceCurveInput.startU), startPoint)) {
500 auto traceCurveResult = traceCurveImpl(state, curveToTrace, traceCurveInput.startEdge);
501 if (traceCurveResult.has_value())
502 return std::make_optional<std::vector<TraceCurveOutput>>(
503 {{traceCurveInput.boundaryIdx, traceCurveInput.startU, curveToTrace.domain()[0].right(),
504 traceCurveResult->intersectionPoint, traceCurveResult->edge, false}});
505 }
506 }
507
508 // Check the boundary that was given in traceInput
509 auto traceCurveResult = traceCurveImpl(state, curveToTrace, traceCurveInput.startEdge);
510
511 if (traceCurveResult.has_value())
512 return std::make_optional<std::vector<TraceCurveOutput>>(
513 {{traceCurveInput.boundaryIdx, traceCurveInput.startU, traceCurveResult->intersectU,
514 traceCurveResult->intersectionPoint, traceCurveResult->edge, traceCurveResult.value().onNode}});
515
516 // If this is unsuccessful the curve is split into two parametrisation
517
518 int status = -1;
519 // Now check if the startPoint or endPoint of the curve is in the element
520 if (pointInElement(endPoint, state->corners))
521 status = 1;
522 else if (pointInElement(startPoint, state->corners))
523 status = 0;
524
525 // If neither start nor endpoint is in the element then traceCurveImpl most likely made a mistake
526 if (status == -1) return std::nullopt;
527
528 // Now we check if there is another boundary that begins at this point and trace this potential curve
529 Point checkPoint = (status == 1) ? endPoint : startPoint;
530 double endUOfCurve1 = (status == 1) ? curveToTrace.domain()[0].right() : curveToTrace.domain()[0].left();
531 double startUOfCurve2 = std::numeric_limits<double>::quiet_NaN();
532
533 // Find the next BoundarySegment
534 auto it = std::ranges::find_if(globalBoundaries_, [&](auto& _boundary) {
535 auto [startPointOfBoundaryToCheck, endPointOfBoundaryToCheck] = curveStartEndPoint(_boundary.nurbsGeometry);
536 // Check if it's the same boundary Geometry
537 if (startPointOfBoundaryToCheck == startPoint && endPointOfBoundaryToCheck == endPoint) return false;
538
539 if (Dune::FloatCmp::eq(checkPoint, startPointOfBoundaryToCheck, 1e-8)) {
540 startUOfCurve2 = _boundary.domain.left();
541 return true;
542 } else if (Dune::FloatCmp::eq(checkPoint, endPointOfBoundaryToCheck, 1e-8)) {
543 startUOfCurve2 = _boundary.domain.right();
544 return true;
545 }
546 return false;
547 });
548
549 if (it != globalBoundaries_.end()) {
550 int newBoundaryIdx = static_cast<int>(std::ranges::distance(globalBoundaries_.begin(), it));
551 auto curveToTrace2 = globalBoundaries_[newBoundaryIdx].nurbsGeometry;
552 auto traceCurveResult2 = traceCurveImpl(state, curveToTrace2, traceCurveInput.startEdge);
553
554 assert(Dune::FloatCmp::eq(curveToTrace(endUOfCurve1), curveToTrace2(startUOfCurve2), 1e-8));
555
556 if (traceCurveResult2.has_value())
557 return std::make_optional<std::vector<TraceCurveOutput>>(
558 {{traceCurveInput.boundaryIdx, traceCurveInput.startU, endUOfCurve1, checkPoint, -1, false},
559 {newBoundaryIdx, startUOfCurve2, traceCurveResult2->intersectU, traceCurveResult2->intersectionPoint,
560 traceCurveResult2->edge, false}});
561 }
562
563 return std::nullopt;
564 }

570 static Clipper2Lib::Paths<intType> getClip(TrimData* trimData) {
571 Clipper2Lib::Paths<intType> clipPaths;
572 Clipper2Lib::Path<intType> tempPath;
573 for (auto& loop : trimData->boundaryLoops) {
574 tempPath.clear();
575 for (auto& boundary : loop)
576 for (auto& point : boundary.path<intType>(pathSamples, scale))
577 tempPath.push_back(point);
578
579 // Sanitize Path, so we don't get duplicate intersection Points on corners
580 tempPath = Clipper2Lib::TrimCollinear(tempPath);
581 clipPaths.push_back(tempPath);
582 }
583
584 return clipPaths;
585 }

621 std::array<int, 2> indices{0, static_cast<int>(curve.patchData_.controlPoints.strideSizes()[0]) - 1};

646 static double findGoodStartingPoint(CurveGeometry& curve, Point& intersectionPoint, int N) {
647 auto linSpace = Utilities::linspace(curve.domain()[0], N);
648 std::vector<double> distances;
649 std::ranges::transform(linSpace, std::back_inserter(distances),
650 [&](const auto& u) { return distance(curve(u), intersectionPoint); });
651 auto min_it = std::ranges::min_element(distances);

653 auto min_idx = std::ranges::distance(distances.begin(), min_it);
654 return linSpace[min_idx];

657 static IntersectionResult findclosestPointOnCurve(CurveGeometry& curve, Point& intersectionPoint) {
658 // There are some easy cases where the IP is at the back or at the front of the domain, check these cases first
659 auto [start, end] = curveStartEndPoint(curve);
660
661 const auto domain = curve.domain()[0];
662
663 if (Dune::FloatCmp::eq(start, intersectionPoint, tolerance)) return {domain.left(), start, true};
664 if (Dune::FloatCmp::eq(end, intersectionPoint, tolerance)) return {domain.right(), end, true};
665 Dune::FieldVector<double, 1> uGuess{findGoodStartingPoint(curve, intersectionPoint, 10)};
666 auto [u, Ru, fu, gap] = Dune::IGA::closestPointProjectionByTrustRegion(curve, intersectionPoint, uGuess);
667
668 if (gap > gapTolerance) return {u[0], intersectionPoint, false};
669 return {u[0], curve(u[0]), true};
670 }

672 static std::optional<TraceCurveResult> traceCurveImpl(FindNextBoundaryLoopState* state, CurveGeometry& curveToTrace,
673 const int startEdge) {
674 // Look through the edges and determine the next Intersection Point (the IPs are sorted counter-clockwise)
675 auto pointMapPtr = state->pointMapPtr.get();
676 for (int i = 1; i < 5; ++i) {
677 auto entityIdx = nextEntityIdx(startEdge, i);
678 auto [intersectionPoint, hasIntersectionPoint] = getNextIntersectionPointOnEdge(pointMapPtr, entityIdx);
679 if (hasIntersectionPoint) {
680 // Check if the curveToTrace goes through the intersection point
681 auto [localResult, globalResult, found] = findclosestPointOnCurve(curveToTrace, intersectionPoint);
682 if (found) {
683 markIntersectionPointAsVisited(pointMapPtr, entityIdx, intersectionPoint);
684 return std::make_optional<TraceCurveResult>({localResult, globalResult, entityIdx, false});
685 }
686 }
687 // It may also be on the corresponding Node
688 if (state->nodeCounter[entityIdx] > 0 and entityIdx != state->currentNode) {
689 auto nodePoint = toFloatDomain(state->corners[entityIdx]);
690 auto [localResult, globalResult, found] = findclosestPointOnCurve(curveToTrace, nodePoint);
691 if (found) return std::make_optional<TraceCurveResult>({localResult, globalResult, entityIdx, true});
692 }
693 }
694
695 return std::nullopt;
696 }

698 static bool areLinesParallel(const CurveGeometry& line1, const CurveGeometry& line2) {
699 if (line1.degree()[0] > 1 || line2.degree()[0] > 1) return false;
700
701 auto [p1, p2] = curveStartEndPoint(line1);
702 auto [p3, p4] = curveStartEndPoint(line2);
703
704 // Compute direction vectors of both curves
705 auto d1 = p2 - p1;
706 auto d2 = p4 - p3;
707
708 auto cos_angle = static_cast<double>(Dune::dot(d1, d2) / (d1.two_norm() * d2.two_norm()));
709 auto angle = std::acos(std::min(std::max(cos_angle, -1.0), 1.0));
710
711 // Check if the angle is small enough to consider the curves on the same line (or parallel)
712 return (Dune::FloatCmp::eq(angle, 0.0, tolerance) || Dune::FloatCmp::eq(angle, std::numbers::pi, tolerance));
713 }

715 // Sorts the Intersection Points counter-clockwise around the element (crucial step!)
716 void sortIntersectionPoints(IntersectionPointMap& pointMap) {
717 for (int edge = 0; edge < 2; ++edge)
718 std::ranges::sort(pointMap[edge], [&](const IntersectionPoint& a, const IntersectionPoint& b) {
719 auto orientation = getEdgeOrientation(edge);
720 return a.point[orientation] < b.point[orientation];
721 });
722 for (int edge = 2; edge < 4; ++edge)
723 std::ranges::sort(pointMap[edge], [&](const IntersectionPoint& a, const IntersectionPoint& b) {
724 auto orientation = getEdgeOrientation(edge);
725 return a.point[orientation] > b.point[orientation];
726 });
727 }
