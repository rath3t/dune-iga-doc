278 void globalRefine(int refCount) {
279 for (int i = 0; i < refCount; ++i) {
280 const auto& finestPatchData = patchGeometries_.back().patchData();
281 auto newfinestPatchData = finestPatchData;
282
283 auto& olduniqueKnotVector = patchGeometries_.back().uniqueKnotVector();
284 std::array<std::vector<ctype>, dim> newUniqueKnotVecs;
285 for (int refDirection = 0; refDirection < dim; ++refDirection) {
286 auto additionalKnots = Splines::generateRefinedKnots(finestPatchData.knotSpans, refDirection, 1);
287 newfinestPatchData = Splines::knotRefinement<dim>(newfinestPatchData, additionalKnots, refDirection);
288 auto& newKnotVec = newUniqueKnotVecs[refDirection];
289
290 // compute new unique Knot vectors from refinement
291 newKnotVec.resize(olduniqueKnotVector[refDirection].size() + additionalKnots.size());
292 std::ranges::merge(olduniqueKnotVector[refDirection], additionalKnots, std::begin(newKnotVec));
293 }
294
295 patchGeometries_.emplace_back(newfinestPatchData, newUniqueKnotVecs);
296 patchGeometriesUnElevated.emplace_back(patchGeometries_.back());
297 }
298 // Here it is exploited that the knot refinem above is in sync with the globaRefine of the trimmer
299 trimmer_->globalRefine(refCount);
300 }

317 void globalRefineInDirection(const std::array<int, dim>& refines) {
318 const auto& finestPatchData = patchGeometries_.back().patchData();
319 auto newfinestPatchData = finestPatchData;
320 for (int dir = 0; auto refinesInDirection : refines) {
321 if (refinesInDirection == 0) {
322 ++dir;
323 continue;
324 }
325 auto additionalKnots = Splines::generateRefinedKnots(finestPatchData.knotSpans, dir, refinesInDirection);
326 newfinestPatchData = Splines::knotRefinement<dim>(newfinestPatchData, additionalKnots, dir);
327 ++dir;
328 }
329 auto newGrid = PatchGrid(newfinestPatchData);
330 *this = std::move(newGrid);
331 }

338 void degreeElevate(const std::array<int, dim>& elevationFactors, int lvl) {
339 if (lvl > maxLevel() and lvl >= 0) DUNE_THROW(Dune : RangeError, "This level does not exist");
340 auto& patchData = patchGeometries_[lvl].patchData();
341 patchGeometriesUnElevated[lvl] = GeometryKernel::NURBSPatch<dim, dimworld, ctype>(patchData);
342 for (int dir = 0; auto elevatesInDirection : elevationFactors) {
343 if (elevatesInDirection == 0) {
344 ++dir;
345 continue;
346 }
347 patchData = Splines::degreeElevate(patchData, dir, elevatesInDirection);
348 ++dir;
349 }
350 patchGeometries_[lvl] = GeometryKernel::NURBSPatch<dim, dimworld, ctype>(patchData);
351 }
