47 HostOrTrimmed host_or_trimmed;

50 friend std::ostream& operator<<(std::ostream& stream, const IdType& id) {
51 stream << "Type: " << std::string(id.host_or_trimmed==HostOrTrimmed::host? "Host": "Trimmed") << ", Key: " << id.id << "\n";

56 template<typename HostIdType>
57 bool operator<(const IdType<HostIdType>& lhs, const IdType<HostIdType>& rhs) {
58
59 if(lhs.host_or_trimmed==rhs.host_or_trimmed)
60 return lhs.id < rhs.id;
61 else if(lhs.host_or_trimmed<rhs.host_or_trimmed)
62 return true;
63 else
64 return false;
