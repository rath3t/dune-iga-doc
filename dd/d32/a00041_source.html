18 auto readJson(InputStringType& ibraInputFile) {
19 using json = nlohmann::json;
20
21 std::vector<Ibra::Surface<dimworld>> surfaces;
22 std::vector<Ibra::Curve2D> curves2D;
23
24 std::vector<Ibra::BrepRepresentation> brepRepresentations;
25 std::vector<Ibra::BrepLoopRepresentation> brepLoopRepresentations;
26 std::vector<Ibra::BrepTrimRepresentation> brepTrimRepresentations;
27
28 json ibraJson;
29 try {
30 ibraJson = json::parse(ibraInputFile);
31
32 for (auto& j : ibraJson) {
33 auto geo = j.get<Ibra::IbraBase>();
34
35 switch (geo.type) {
36 case Ibra::Type::NurbsSurfaceGeometry3D:
37 surfaces.push_back(j.get<Ibra::Surface<dimworld>>());
38 break;
39 case Ibra::Type::NurbsCurveGeometry2D:
40 curves2D.push_back(j.get<Ibra::Curve2D>());
41 break;
42 case Ibra::Type::BrepType:
43 brepRepresentations.push_back(j.get<Ibra::BrepRepresentation>());
44 break;
45 case Ibra::Type::BrepLoopType:
46 brepLoopRepresentations.push_back(j.get<Ibra::BrepLoopRepresentation>());
47 break;
48 case Ibra::Type::BrepTrimType:
49 brepTrimRepresentations.push_back(j.get<Ibra::BrepTrimRepresentation>());
50 break;
51 default:
52 // Do nothing -- keep the compiler happy
53 break;
54 }
55 }
56 } catch (json::parse_error& ex) {
57 DUNE_THROW(Dune::IOError, "Error in reading input stream: "
58 << ", parse error at byte " << ex.byte << " What: " << ex.what());
59 }
60
61 // Make Connections
62 auto brepRepresentation = brepRepresentations[0];
63
64 Ibra::Brep brep{brepRepresentation, curves2D, surfaces, brepLoopRepresentations, brepTrimRepresentations};
65 return brep;
66}
