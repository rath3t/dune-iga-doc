11 template <typename ScalarType = double>
12 struct Domain : std::array<ScalarType, 2> {
13 using Base = std::array<ScalarType, 2>;
14 Domain(ScalarType l, ScalarType r) : Base({l, r}) {}
15
16 Domain() : Base({0.0, 1.0}) {}
17
18 ScalarType& left() { return Base::operator[](0); };
19 const ScalarType& left() const { return Base::operator[](0); };
20 ScalarType& right() { return Base::operator[](1); };
21 const ScalarType& right() const { return Base::operator[](1); };
22
23 ScalarType center() const { return (left() + right()) / ScalarType(2.0); }
24 ScalarType size() const { return right() - left(); }
25 };

27 template <typename ScalarType, int dim, size_t dim2>
28 requires(dim == dim2) auto clampToBoundaryAndCheckIfIsAtAllBoundaries(
29 Dune::FieldVector<ScalarType, dim>& x, const std::array<Domain<ScalarType>, dim2>& domain = {}) {
30 int breakDueToBoundaryCounter = 0;
31
32 for (int j = 0; j < dim; ++j) {
33 x[j] = std::clamp(x[j], domain[j].left(), domain[j].right());
34 if (x[j] == domain[j].left() or x[j] == domain[j].right()) ++breakDueToBoundaryCounter;
35 }
36 if (dim == breakDueToBoundaryCounter) return true;
37 return false;
38 }

40 template <std::floating_point T>
41 auto linspace(T a, T b, unsigned int N) {
42 T inc = (b - a) / static_cast<T>(N - 1);
43 auto val = [=](int i) -> T {
44 if (i != 0 and i != N - 1) return a + i * inc;
45 if (i == N - 1) return b;
46 return a;
47 };
48
49 return std::views::iota(0u, N) | std::views::transform(val);
50 }
