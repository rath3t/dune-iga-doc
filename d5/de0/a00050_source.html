9 template <typename Geometry>
10 typename Geometry::LocalCoordinate computeParameterSpaceCoordinate(
11 const Geometry& geo, const typename Geometry::GlobalCoordinate& global) {
12 static constexpr int mydimension = Geometry::mydimension;
13 static constexpr int worlddimension = Geometry::worlddimension;
14 using ctype = typename Geometry::ctype;
15 using LocalCoordinate = typename Geometry::LocalCoordinate;
16 using GlobalCoordinate = typename Geometry::GlobalCoordinate;
17 using MatrixHelper = typename Geometry::MatrixHelper;
18 if constexpr (mydimension == 0)
19 return {};
20 else if constexpr (mydimension != worlddimension) {
21 auto [u, Ru, fu, gap] = Dune::IGA::closestPointProjectionByTrustRegion(geo, global);
22 return u;
23 } else {
24 const ctype tolerance = ctype(16) * std::numeric_limits<ctype>::epsilon();
25 LocalCoordinate x = geo.domainMidPoint();
26
27 LocalCoordinate dx{};
28 do { // from multilinearGeometry
29 const GlobalCoordinate dglobal = geo.global(x) - global;
30 MatrixHelper::template xTRightInvA<mydimension, worlddimension>(geo.jacobianTransposed(x), dglobal, dx);
31 const bool invertible
32 = MatrixHelper::template xTRightInvA<mydimension, worlddimension>(geo.jacobianTransposed(x), dglobal, dx);
33
34 if (!invertible) return LocalCoordinate(std::numeric_limits<ctype>::max());
35 x -= dx;
36 // if local is outside of maximum knot vector span bound, thus we clamp it to it and return
37 // clamp result into boundaries
38 if (IGA::Utilities::clampToBoundaryAndCheckIfIsAtAllBoundaries(x, geo.domain())) {
39 break;
40 }
41
42 } while (dx.two_norm2() > tolerance);
43 return x;
44 }
45 }

47 template <typename Geometry>
48 typename Geometry::Hessian hessian(const typename Geometry::LocalCoordinate& local,const typename Geometry::NurbsLocalView& nurbsLocalView,
49 const typename Geometry::ControlPointCoordinateNetType& localControlPointNet) {
50 typename Geometry::Hessian H;
51 const auto basisFunctionDerivatives = nurbsLocalView.basisFunctionDerivatives(local, 2);
52 static constexpr int mydimension = Geometry::mydimension;
53
54 for (int dir = 0; dir < mydimension; ++dir) {
55 std::array<unsigned int, mydimension> ithVec{};
56 ithVec[dir] = 2; // second derivative in dir direction
57 H[dir] = dot(basisFunctionDerivatives.get(ithVec), localControlPointNet);
58 }
59 if constexpr (mydimension > 1) {
60 std::array<int, mydimension> mixeDerivs;
61 std::ranges::fill(mixeDerivs, 0); // first mixed derivatives
62 if constexpr (mydimension == 2)
63 for (int dir = 0; dir < mydimension; ++dir) {
64 mixeDerivs[dir] = 1;
65 }
66 else
67 std::ranges::fill_n(mixeDerivs.begin()+1, 2, 1); // first mixed derivatives
68 int mixedDireCounter = mydimension;
69 do {
70 H[mixedDireCounter++] = dot(basisFunctionDerivatives.get(mixeDerivs), localControlPointNet);
71
72 if constexpr (mydimension == 2) break;
73
74 } while (std::ranges::next_permutation(mixeDerivs, std::less()).found);
75 }
76 return H;
77 }
