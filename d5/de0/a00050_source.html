71 [[nodiscard]] double volume(int scaleOrder = 1) const {
72 const auto rule = Dune::QuadratureRules<ctype, mydimension>::rule(
73 this->type(), scaleOrder * mydimension * (*std::ranges::max_element(patchData_.degree)));
74 ctype vol = 0.0;
75 for (auto& gp : rule) {
76 vol += integrationElement(gp.position()) * gp.weight();
77 }
78 return vol;
79 }

81 [[nodiscard]] bool affine() const { return false; }

84 [[nodiscard]] int corners() const { return 1 << mydimension; }

87 [[nodiscard]] GlobalCoordinate corner(int k) const {
88 LocalCoordinate localcorner;
89 for (size_t i = 0; i < mydimension; i++) {
90 localcorner[i] = (k & (1 << i)) ? 1 : 0;
91 }
92 return global(localcorner);

101 [[nodiscard]] FieldVector<ctype, worlddimension> global(const LocalCoordinate& u) const {

125 [[nodiscard]] Jacobian jacobian(const LocalCoordinate& local) const { return transpose(jacobianTransposed(local)); }

127 [[nodiscard]] ctype integrationElement(const LocalCoordinate& local) const {
128 auto j = jacobianTransposed(local);

133 [[nodiscard]] GeometryType type() const { return GeometryTypes::cube(mydimension); }

135 // The following are functions are not part of the Geometry Interface
136 [[nodiscard]] std::array<Utilities::Domain<double>, mydimension> domain() const {
137 std::array<Utilities::Domain<double>, mydimension> result{};
138 for (int i = 0; i < mydimension; ++i)
139 result[i] = {patchData_.knotSpans[i].front(), patchData_.knotSpans[i].back()};
140
141 return result;
142 }
143
144 [[nodiscard]] std::array<int, mydimension> degree() const { return patchData_.degree; }
145
146 [[nodiscard]] FieldVector<ctype, mydimension> domainMidPoint() const {
147 auto dom = domain();
148 Dune::FieldVector<ctype, mydimension> result{};
149 for (int i = 0; i < mydimension; ++i)
150 result[i] = dom[i].center();
151
152 return result;
153 }

155 private:
156 auto calculateNurbsAndControlPointNet(const LocalCoordinate& u) const {
157 auto subNetStart = findSpanCorrected(patchData_.degree, u, patchData_.knotSpans);
158
159 auto cpCoordinateNet
160 = netOfSpan(subNetStart, patchData_.degree, extractControlCoordinates(patchData_.controlPoints));
161 auto nurbsLocalView = nurbs_.localView();
162 nurbsLocalView.bind(subNetStart);
163 return std::make_tuple(nurbsLocalView, cpCoordinateNet, subNetStart);
164 }
165
166 static auto zeroFirstAndSecondDerivativeOfPositionImpl(const LocalCoordinate& u,
167 const NurbsLocalView& nurbsLocalView,
168 const ControlPointCoordinateNetType& localControlPointNet) {
169 // TODO the above code is more efficient since there is only on call to the derivatives
170
171 return std::make_tuple(GeometryKernel::position(u, nurbsLocalView, localControlPointNet),
172 GeometryKernel::jacobianTransposed(u, nurbsLocalView, localControlPointNet),
173 GeometryKernel::hessian(u, nurbsLocalView, localControlPointNet));
174 }
175
176 public:
177 NURBSPatchData<mydimension, worlddimension, ScalarType> patchData_;
178
179 private:
180 Nurbs nurbs_;
181 GeometryLocalView<0, Trimming::Disabled> localView_;
182 };
