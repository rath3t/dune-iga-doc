7 inline int giveEdgeIdx(const std::size_t e1, const std::size_t e2) {
8 if ((e1 == 0 and e2 == 1) or (e1 == 1 and e2 == 0)) return 0;
9 if ((e1 == 1 and e2 == 2) or (e1 == 2 and e2 == 1)) return 1;
10 if ((e1 == 2 and e2 == 3) or (e1 == 3 and e2 == 2)) return 2;
11 if ((e1 == 3 and e2 == 0) or (e1 == 0 and e2 == 3)) return 3;

67 void finish(const auto& patchTrimData) {
68 // Sort the points in counter clockwise manner such that the first point is in the lower left
69 const auto minVertex = lowerLeftVertex();
70 std::ranges::sort(vertices_, [&](const VertexVariant& aV, const VertexVariant& bV) {
71 return comparePoints(aV, bV, minVertex);
72 });
73
74 // Resort that we always start on a hostVertex (if there is one)
75 if (const auto it = std::ranges::find_if(
76 vertices_, [](const VertexVariant& vV) { return std::holds_alternative<HostVertex>(vV); });
77 it != vertices_.end())
78 std::ranges::rotate(vertices_, it);
79
80 // Now resort InsideVertices (sometimes they end up on the wrong position)
81 for (const auto i : std::views::iota(0ul, vertices_.size())) {
82 if (const auto insideVertex = std::get_if<InsideVertex>(&vertices_[i])) {
83 auto it = vertices_.begin() + static_cast<long>(i);
84 const size_t indexBeforeIt = it == vertices_.begin() ? vertices_.size() - 1 : std::ranges::distance(vertices_.begin(), it) - 1;
85 const size_t indexAfterIt = it == std::prev(vertices_.end()) ? 0 : std::ranges::distance(vertices_.begin(), it) + 1;
86
87 // Check vertex before if they line up correctly
88 const auto vertexBefore = std::get_if<NewVertex>(&vertices_[indexBeforeIt]);
89 const auto vertexAfter = std::get_if<NewVertex>(&vertices_[indexAfterIt]);
90
91 if (vertexBefore && vertexAfter ) {
92 auto indices = patchTrimData.getIndices(vertexBefore->trimmingCurveZ);
93 if(indices.loop == insideVertex->loopIdx and indices.curve == insideVertex->curveIdxI)
94 continue;
95 }
96 // If we end up here sort the vertex to a correct place
97 auto correctVertexIt = findCorrespondingVertexIt(*insideVertex, patchTrimData);
98 assert(correctVertexIt != vertices_.end() && "No corresponding newVertex found");
99
100 // Add the insideVertex at correctVertexIt+1
101 const auto correctItForInsideVertex = correctVertexIt + 1;
102 if (std::distance(vertices_.begin(), correctItForInsideVertex) < i) {
103 vertices_.insert(correctItForInsideVertex, *insideVertex);
104 vertices_.erase(vertices_.begin() + static_cast<long>(i+1));
105 } else {
106 std::ranges::rotate(it, it + 1, correctItForInsideVertex);
107 }
108 }
109 }

154 // Sort the points based on polar angle with respect to the reference point
155 inline static auto comparePoints
156 = [](const VertexVariant& aV, const VertexVariant& bV, const VertexVariant& referenceV) -> bool {
157 auto polarAngle
158 = [](const auto& p, const auto& reference) -> double { return atan2(p.y - reference.y, p.x - reference.x); };
159
160 const auto a = std::visit(getPoint, aV);
161 const auto b = std::visit(getPoint, bV);
162 const auto reference = std::visit(getPoint, referenceV);
163 const double angleA = polarAngle(a, reference);
164 const double angleB = polarAngle(b, reference);
165
166 if (angleA != angleB) return angleA < angleB;
167
168 // If two points have the same angle, sort based on distance to the reference point
169 return (a.x - reference.x) * (a.x - reference.x) + (a.y - reference.y) * (a.y - reference.y)
170 < (b.x - reference.x) * (b.x - reference.x) + (b.y - reference.y) * (b.y - reference.y);
171 };

174 return std::ranges::find_if(vertices_, [&](const VertexVariant& vV) {
175
176 if(std::holds_alternative<NewVertex>(vV)) {
177 const auto indices = patchTrimData.getIndices(std::get<NewVertex>(vV).trimmingCurveZ);
178 return indices.loop == insideVertex.loopIdx and indices.curve== insideVertex.curveIdxI;
179 }else
180 return false;
181
182 });
