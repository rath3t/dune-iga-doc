71 /* this is for testing purposes only */
72 void drawResult(const std::string& filename, bool inParameterSpace, bool newFig = true) {
73 if (static_cast<int>(flag_) == 1) return;
74 auto eleGeometry = hostEntity_.geometry();
75 auto lowerLeftCorner = inParameterSpace ? eleGeometry.corner(0) : Dune::FieldVector<double, 2>({0, 0});
76 auto upperRightCorner = eleGeometry.corner(3);
77 Dune::DiagonalMatrix<double, 2> scaling(
78 {inParameterSpace ? 1.0 / (upperRightCorner[0] - lowerLeftCorner[0]) : 1,
79 inParameterSpace ? 1.0 / (upperRightCorner[1] - lowerLeftCorner[1]) : 1});
80
81 if (newFig)
82 auto figure = matplot::figure(true);
83 matplot::hold("on");
84
85 constexpr std::array idxLookUp = {0, 1, 3, 2};
86
87 const double scale = eleGeometry.volume() / 5;
88
89 auto plotEllipse = [&](const Vertex& v) {
90 const auto w = 0.025 * scale;
91 const auto c = matplot::ellipse(v[0] - (w / 2), v[1] - (w / 2), w, w);
92 c->color("blue");
93 };
94
95 auto plotLine = [](std::vector<Vertex>& vs, bool thin = false) {
96 std::vector<double> x;
97 std::ranges::transform(vs, std::back_inserter(x), [](auto& v) { return v[0]; });
98
99 std::vector<double> y;
100 std::ranges::transform(vs, std::back_inserter(y), [](auto& v) { return v[1]; });
101 if (thin)
102 matplot::plot(x, y)->line_width(0.5).color("grey");
103 else
104 matplot::plot(x, y)->line_width(2).color("red");
105 };
106
107 for (auto c : edgeLookUp) {
108 Dune::FieldVector<double, 2> firstVertexPos = eleGeometry.corner(c[0]) - lowerLeftCorner;
109 Dune::FieldVector<double, 2> secondVertexPos = eleGeometry.corner(c[1]) - lowerLeftCorner;
110 scaling.mv(firstVertexPos, firstVertexPos);
111 scaling.mv(secondVertexPos, secondVertexPos);
112 std::vector<Vertex> vs{firstVertexPos, secondVertexPos};
113 plotLine(vs, true);
114 }
115
116 for (auto& v : vertices_) {
117 if (v.geometry.has_value()) {
118 Dune::FieldVector<double, 2> pos = (v.geometry.value() - lowerLeftCorner);
119 scaling.mv(pos, pos);
120 plotEllipse(pos);
121 } else {
122 Dune::FieldVector<double, 2> pos = eleGeometry.corner(idxLookUp[v.idx]) - lowerLeftCorner;
123 scaling.mv(pos, pos);
124 plotEllipse(pos);
125 }
126 }
127
128 for (auto& curve : edges_) {
129 if (curve.geometry.has_value()) {
130 std::vector<Vertex> vs;
131 for (double u : Utilities::linspace(curve.geometry.value().domain().front(), 40)) {
132 Dune::FieldVector<double, 2> pos = (curve.geometry.value().global(u) - lowerLeftCorner);
133 scaling.mv(pos, pos);
134 vs.push_back(pos);
135 }
136 plotLine(vs);
137 } else {
138 auto cornerIdx = edgeLookUp[curve.idx];
139 Dune::FieldVector<double, 2> firstVertexPos = (eleGeometry.corner(cornerIdx.front()) - lowerLeftCorner);
140 Dune::FieldVector<double, 2> secondVertexPos = (eleGeometry.corner(cornerIdx.back()) - lowerLeftCorner);
141 scaling.mv(firstVertexPos, firstVertexPos);
142 scaling.mv(secondVertexPos, secondVertexPos);
143 std::vector<Vertex> vs{firstVertexPos, secondVertexPos};
144 plotLine(vs);
145 }
146 }
147
148 matplot::axis(matplot::equal);
149 if (newFig)
150 matplot::save(filename + (inParameterSpace ? "inParameterSpace" : ""), "gif");
151 }
152 bool checkInside(const Dune::FieldVector<double, 2>& local) const {

175 private:
176 auto sampleCurves() {
177 auto eleGeo = hostEntity_.geometry();
178 auto lowerLeftCorner = eleGeo.corner(0);
179 auto upperRightCorner = eleGeo.corner(3);
180 Dune::DiagonalMatrix<double, 2> scaling(
181 {1.0 / (upperRightCorner[0] - lowerLeftCorner[0]), 1.0 / (upperRightCorner[1] - lowerLeftCorner[1])});
182
183 for (auto& edge : edges_) {
184 if (edge.geometry.has_value()) {
185 for (const auto v : Utilities::linspace(edge.geometry->domain()[0], 100)) {
186 auto fV = edge.geometry->global({v}) - lowerLeftCorner;
187 scaling.mv(fV, fV);
188 path.emplace_back(fV[0], fV[1]);
189 }
190 } else {
191 auto cornerIdx = edgeLookUp[edge.idx];
192
193 auto firstVertex = eleGeo.corner(cornerIdx.front()) - lowerLeftCorner;
194 auto secondVertex = eleGeo.corner(cornerIdx.back()) - lowerLeftCorner;
195 scaling.mv(firstVertex, firstVertex);
196 scaling.mv(secondVertex, secondVertex);
197
198 path.emplace_back(firstVertex[0], firstVertex[1]);
199 path.emplace_back(secondVertex[0], secondVertex[1]);
200 }
201 }
202 }

207 Clipper2Lib::PathD path;

210 int newEdgeCounter_ = 4;
211 };
