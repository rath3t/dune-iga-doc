33 // Coordinates in a single knot span differ from coordinates on the B-spline patch
34 // by an affine transformation. This transformation is stored in the diagonal entries.
35 // If trimming is disabled the Jacobian in the parameter space of subentities (edges or surfaces) is a
36 // DiagonalMatrixBlock but is treated as a FieldMatrix, since is also just cubes and an axis-aligned geometry, if
37 // trimming is enabled the Jacobian of the parameterspace for subentities is a potentially fully populated
38 // FieldMatrix, Think about an arbitrary curve in the axis-aligned cube parameter grid
39 // ------->
40 // `____________________C_______D_______`
41 // ::````````````````::``````````'|````::
42 // :: :: `| ::
43 // :: ^F:: `| ::
44 // :: | :: `| ::
45 // :: | :: `| ::
46 // :: |E:: `| ::
47 // :: :: .| x ::
48 // :: :: _: ::
49 // ::................::..........|:....::
50 // ::''''''''''''''''::''''''''''/'''''::
51 // :: :: _:B ::
52 // :: :: .| ::
53 // :: :: `::A ::
54 // :: '::::||::::::' ::
55 // :: ::` :: x ::
56 // :: |' :: ::
57 // :: | x :: ::
58 // ::````````|```````::````````````````::
59 // `____________________________________`
60 // at point A the tangent of the curve (the JacobianInParameterSpace) points into the oblique direction (to B)
61 // for the axis-aligned geometry these tangents are from E to F or from C to D, depending on the orientation of the
62 // subentity, which are aligned with the axis
63 using JacobianTransposedInParameterSpace = typename ParameterSpaceGeometry::JacobianTransposed;
64 using GlobalInParameterSpace = typename ParameterSpaceGeometry::GlobalCoordinate;

107 [[nodiscard]] JacobianTransposed jacobianTransposed(const LocalCoordinate& u) const {
108 assert(parameterSpaceGeometry && "Bind the local view first!");
109 // using JacobianTransposed = FieldMatrix<ctype, mydimension, worlddimension>;
110 const auto ouInPatch = globalInParameterSpace(u);
111 const JacobianTransposedInPatch JTinPatch
112 = patchGeometry_->jacobianTransposedImpl(ouInPatch, nurbsLocalView_, localControlPointNet);
113 const JacobianTransposedInParameterSpace jInparaMeterSpace = jacobianTransposedInParameterSpace(u);
114 JacobianTransposed jacobianOfEntity = jInparaMeterSpace * JTinPatch;
115 return jacobianOfEntity;
116 }
