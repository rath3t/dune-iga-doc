10 template <typename Geometry>
11 typename Geometry::LocalCoordinate computeParameterSpaceCoordinate(
12 const Geometry& geo, const typename Geometry::GlobalCoordinate& global) {
13 static constexpr int mydimension = Geometry::mydimension;
14 static constexpr int worlddimension = Geometry::worlddimension;
15 using ctype = typename Geometry::ctype;
16 using LocalCoordinate = typename Geometry::LocalCoordinate;
17 using GlobalCoordinate = typename Geometry::GlobalCoordinate;
18 using MatrixHelper = typename Geometry::MatrixHelper;
19 if constexpr (mydimension == 0)
20 return {};
21 else if constexpr (mydimension != worlddimension) {
22 auto [u, Ru, fu, gap] = closestPointProjectionByTrustRegion(geo, global);
23 return u;
24 } else {
25 const ctype tolerance = ctype(16) * std::numeric_limits<ctype>::epsilon();
26 LocalCoordinate x = geo.domainMidPoint();
27
28 LocalCoordinate dx{};
29 do { // from multilinearGeometry
30 const GlobalCoordinate dglobal = geo.global(x) - global;
31 MatrixHelper::template xTRightInvA<mydimension, worlddimension>(geo.jacobianTransposed(x), dglobal, dx);
32 const bool invertible
33 = MatrixHelper::template xTRightInvA<mydimension, worlddimension>(geo.jacobianTransposed(x), dglobal, dx);
34
35 if (!invertible) return LocalCoordinate(std::numeric_limits<ctype>::max());
36 x -= dx;
37 // if local is outside of maximum knot vector span bound, thus we clamp it to it and return
38 // clamp result into boundaries
39 if (Dune::IGANEW::Utilities::clampToBoundaryAndCheckIfIsAtAllBoundaries(x, geo.domain())) {
40 break;
41 }
42
43 } while (dx.two_norm2() > tolerance);
44 return x;
45 }
46 }

48 template <typename LocalCoordinate, typename NurbsLocalView, typename ControlPointCoordinateNetType>
49 auto hessian(const LocalCoordinate& local, const NurbsLocalView& nurbsLocalView,
50 const ControlPointCoordinateNetType& localControlPointNet) {
51 static constexpr int mydimension = NurbsLocalView::dimension;
52 static constexpr int worlddimension = ControlPointCoordinateNetType::value_type::dimension;
53 using ctype = typename ControlPointCoordinateNetType::value_type::value_type;
54
55 FieldMatrix<ctype, mydimension*(mydimension + 1) / 2, worlddimension> H;
56
57 const auto basisFunctionDerivatives = nurbsLocalView.basisFunctionDerivatives(local, 2);
58
59 for (int dir = 0; dir < mydimension; ++dir) {
60 std::array<unsigned int, mydimension> ithVec{};
61 ithVec[dir] = 2; // second derivative in dir direction
62 H[dir] = dot(basisFunctionDerivatives.get(ithVec), localControlPointNet);
63 }
64 if constexpr (mydimension > 1) {
65 std::array<int, mydimension> mixeDerivs;
66 std::ranges::fill(mixeDerivs, 0); // first mixed derivatives
67 if constexpr (mydimension == 2)
68 for (int dir = 0; dir < mydimension; ++dir) {
69 mixeDerivs[dir] = 1;
70 }
71 else
72 std::ranges::fill_n(mixeDerivs.begin() + 1, 2, 1); // first mixed derivatives
73 int mixedDireCounter = mydimension;
74 do {
75 H[mixedDireCounter++] = dot(basisFunctionDerivatives.get(mixeDerivs), localControlPointNet);
76 if constexpr (mydimension == 2) break;
77 } while (std::ranges::next_permutation(mixeDerivs, std::less()).found);
78 }
79 return H;
80 }

82 template <typename LocalCoordinate, typename NurbsLocalView, typename ControlPointCoordinateNetType>
83 [[nodiscard]] auto jacobianTransposed(const LocalCoordinate& local, const NurbsLocalView& nurbsLocalView,
84 const ControlPointCoordinateNetType& localControlPointNet) {
85 static constexpr int mydimension = NurbsLocalView::dimension;
86 static constexpr int worlddimension = ControlPointCoordinateNetType::value_type::dimension;
87 using ctype = typename ControlPointCoordinateNetType::value_type::value_type;
88
89 FieldMatrix<ctype, mydimension, worlddimension> result;
90
91 const auto basisFunctionDerivatives = nurbsLocalView.basisFunctionDerivatives(local, 1);
92
93 for (int dir = 0; dir < mydimension; ++dir) {
94 std::array<unsigned int, mydimension> ithVec{};
95 ithVec[dir] = 1;
96 result[dir] = dot(basisFunctionDerivatives.get(ithVec), localControlPointNet);
97 }
98 return result;
99 }
