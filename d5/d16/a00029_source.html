89 [[nodiscard]] GlobalCoordinate outerNormal(const LocalCoordinate& xi) const {
90 if constexpr (mydimension == 0) {
91 const auto xiInElementCoords = geometryInInside().global(xi);
92 const auto insideJacobianTransposed = inside().geometry().jacobianTransposed(xiInElementCoords);
93 return (innerLocalIndex_ == 0) ? -insideJacobianTransposed[0] : insideJacobianTransposed[0];
94 } else if constexpr (mydimension == 1) { // edges
95 if constexpr (dimworld == 2) { // edges in R2
96 const auto innerJacobianTransposed = this->geometry().jacobianTransposed(xi)[0];
97 switch (innerLocalIndex_) {
98 case 0:
99 case 3:
100 return GlobalCoordinate({-innerJacobianTransposed[1], innerJacobianTransposed[0]});
101 case 1:
102 case 2:
103 return GlobalCoordinate({innerJacobianTransposed[1], -innerJacobianTransposed[0]});
104 default:
105 __builtin_unreachable();
106 }
107 } else if constexpr (dimworld == 3) // edges in R3
108 {
109 const auto xiInElementCoords = geometryInInside().global(xi);
110 const auto insideJacobianTransposed = inside().geometry().jacobianTransposed(xiInElementCoords);
111 auto normal = cross(insideJacobianTransposed[0], insideJacobianTransposed[1]);
112 switch (innerLocalIndex_) {
113 case 0:
114 return cross(normal, insideJacobianTransposed[1]);
115 case 1:
116 return cross(insideJacobianTransposed[1], normal);
117 case 2:
118 return cross(insideJacobianTransposed[0], normal);
119 case 3:
120 return cross(normal, insideJacobianTransposed[0]);
121 default:
122 __builtin_unreachable();
123 }
124 }
125 } else if constexpr (mydimension == 2 && dimworld == 3) { // surfaces in R3
126 const auto innerJacobianTransposed = this->geometry().jacobianTransposed(xi);
127 switch (innerLocalIndex_) {
128 case 0:
129 case 3:
130 case 4:
131 return cross(innerJacobianTransposed[1], innerJacobianTransposed[0]);
132 case 1:
133 case 2:
134 case 5:
135 return cross(innerJacobianTransposed[0], innerJacobianTransposed[1]);
136 default:
137 __builtin_unreachable();
138 }
139 }
140 __builtin_unreachable();
141 }
