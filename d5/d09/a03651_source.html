14 template <int dim, int dimworld, typename PatchGrid>
15 requires(dim == 2) && (dimworld == 2 || dimworld == 3) && (dim <= dimworld) class IbraReader {
16 using GridFamily = typename PatchGrid::GridFamily;
17 using PatchData = NURBSPatchData<dim, dimworld, typename GridFamily::ctype>;
18 using PatchTrimData = typename GridFamily::TrimmerTraits::PatchTrimData;
19 using TrimmingCurve = typename GridFamily::TrimmerTraits::TrimmingCurve;
20
21 using ControlPointType = typename PatchData::ControlPointType;
22 using ControlPointNetType = typename PatchData::ControlPointNetType;
23
24 public:
25 static auto read(const std::string& fileName, const bool trim = true, std::array<int, 2> preKnotRefine = {0, 0}) {
26 Ibra::Brep brep = readJson<dimworld>(fileName);
27 // Each surface in a brep is one Patch, as for now only one brep is supported in NURBSGrid
28 assert(brep.surfaces.size() == 1);
29
30 std::cout << "IbraReader successfully read in 1 patch with " << brep.trims.size() << " edges." << std::endl;
31
32 // Reader has done its job, now NURBSGrid can be constructed
33
34 Ibra::Surface<dimworld> _surface = brep.surfaces[0];
35 const std::array<std::vector<double>, dim> knotSpans = _surface.compileKnotVectors();
36 const std::vector<std::vector<ControlPointType>> controlPoints
37 = _surface.template transformControlPoints<ControlPointType>();
38 std::array<int, dim> dimsize = _surface.n_controlPoints;
39
40 auto controlNet = ControlPointNetType(dimsize, controlPoints);
41 PatchData _patchData{knotSpans, controlNet, _surface.degree};
42
43 // Optional preKnot refinement
44 for (const auto i : std::views::iota(0, dim)) {
45 if (preKnotRefine[i] > 0) {
46 auto newKnots = Splines::generateRefinedKnots(knotSpans, i, preKnotRefine[i]);
47 _patchData = Splines::knotRefinement(_patchData, newKnots, i);
48 }
49 }
50
51 PatchTrimData trimData{};
52 if (trim) {
53 constructTrimmingCurves(brep, trimData);
54 }
55
56 return std::make_tuple(_patchData, trimData);
57 }
58
59 private:
60 static void constructTrimmingCurves(const Ibra::Brep<dimworld>& brep, PatchTrimData& trimData) {
61 const std::vector<Ibra::BrepLoop> loops = brep.loops;
62 for (int i = 0; const Ibra::BrepLoop& loop : loops) {
63 trimData.addLoop();
64 for (const Ibra::BrepTrim& trim : loop.trims)
65 trimData.insertTrimCurve(trim.asCurve<TrimmingCurve>(), i);
66 i++;
67 }
68 }
69 };
